     1→"""Anti-detection module for browser automation.
     2→
     3→Implements human-like interaction via Emunium + Patchright:
     4→- Human-like scrolling behavior (chunked, variable speed)
     5→- Human-like clicking via Emunium (natural mouse movement)
     6→- Human-like typing via Emunium (variable keystroke timing)
     7→
     8→Note: Webdriver property hiding and navigator masking are handled
     9→by Patchright automatically in CDP mode.
    10→"""
    11→
    12→import random
    13→from typing import Optional
    14→
    15→from patchright.async_api import Page
    16→from emunium import EmuniumPlaywright
    17→
    18→from src.exceptions.errors import SelectorNotFoundError
    19→
    20→
    21→async def human_scroll(page: Page, direction: str = 'down', distance: Optional[int] = None) -> None:
    22→    """
    23→    Perform human-like scrolling with variable speed.
    24→
    25→    Args:
    26→        page: Playwright page instance
    27→        direction: 'up' or 'down'
    28→        distance: Scroll distance in pixels (random if None)
    29→    """
    30→    if direction not in ('up', 'down'):
    31→        raise ValueError(f"Invalid scroll direction: {direction!r}, expected 'up' or 'down'")
    32→
    33→    if distance is None:
    34→        distance = random.randint(200, 600)
    35→
    36→    if direction == 'up':
    37→        distance = -distance
    38→
    39→    # Break scroll into smaller chunks
    40→    chunks = random.randint(3, 8)
    41→    chunk_size = distance // chunks
    42→
    43→    for i in range(chunks):
    44→        actual_chunk = chunk_size + random.randint(-20, 20)
    45→        await page.mouse.wheel(0, actual_chunk)
    46→        delay = random.uniform(0.05, 0.15)
    47→        await page.wait_for_timeout(delay * 1000)
    48→
    49→    await page.wait_for_timeout(random.uniform(0.2, 0.5) * 1000)
    50→

    52→async def human_click(page: Page, selector: str) -> None:
    53→    """
    54→    Click an element with human-like behavior using Emunium.
    55→
    56→    Args:
    57→        page: Playwright page instance
    58→        selector: CSS selector for element to click
    59→    """
    60→    try:
    61→        element = await page.wait_for_selector(selector, timeout=10000)
    62→    except Exception as e:
    63→        if "timeout" in str(type(e).__name__).lower() or "timeout" in str(e).lower():
    64→            raise SelectorNotFoundError(f"Element not found: {selector}")
    65→        raise
    66→
    67→    if element is None:
    68→        raise SelectorNotFoundError(f"Element resolved to None: {selector}")
    69→
    70→    emunium = EmuniumPlaywright(page)
    71→    await emunium.click_at(element)
    72→
