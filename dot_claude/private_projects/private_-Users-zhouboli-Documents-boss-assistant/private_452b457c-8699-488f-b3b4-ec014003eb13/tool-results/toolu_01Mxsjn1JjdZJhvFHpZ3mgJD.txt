    18→Base = declarative_base()

    21→def _utcnow():
    22→    """Return timezone-aware UTC now."""
    23→    return datetime.now(timezone.utc)

    26→class Candidate(Base):
    27→    """Candidate model - stores candidate profile information."""
    28→
    29→    __tablename__ = 'candidates'
    30→
    31→    id = Column(Integer, primary_key=True, autoincrement=True)
    32→    boss_id = Column(String(50), unique=True, nullable=False, index=True)
    33→    name = Column(String(100))
    34→    title = Column(String(200))
    35→    company = Column(String(200))
    36→    location = Column(String(100))
    37→    salary_expect = Column(String(100))      # Boss直聘: expected salary
    38→    work_years = Column(String(50))           # Boss直聘: work years
    39→    education_level = Column(String(50))      # Boss直聘: education level
    40→    profile_url = Column(Text)
    41→
    42→    # JSON fields for structured data
    43→    basic_info = Column(JSON)
    44→    education = Column(JSON)
    45→    work_experience = Column(JSON)
    46→    skills = Column(JSON)
    47→
    48→    # Metadata
    49→    first_seen = Column(DateTime, default=_utcnow)
    50→    last_updated = Column(DateTime, default=_utcnow, onupdate=_utcnow)
    51→    source_search = Column(Text)
    52→
    53→    # Relationships
    54→    messages = relationship("Message", back_populates="candidate")
    55→
    56→    def __repr__(self):
    57→        return f"<Candidate(boss_id='{self.boss_id}', name='{self.name}', company='{self.company}')>"

    60→class Message(Base):
    61→    """Message model - stores sent messages to candidates."""
    62→
    63→    __tablename__ = 'messages'
    64→
    65→    id = Column(Integer, primary_key=True, autoincrement=True)
    66→    candidate_boss_id = Column(String(50), ForeignKey('candidates.boss_id'), index=True)
    67→    template_name = Column(String(100))
    68→    content = Column(Text, nullable=False)
    69→    status = Column(String(20), CheckConstraint("status IN ('sent', 'failed', 'cancelled')"))
    70→    sent_at = Column(DateTime)
    71→    created_at = Column(DateTime, default=_utcnow)
    72→
    73→    # Relationships
    74→    candidate = relationship("Candidate", back_populates="messages")
    75→
    76→    def __repr__(self):
    77→        return f"<Message(id={self.id}, candidate_id='{self.candidate_boss_id}', status='{self.status}')>"

   135→class DatabaseManager:
   136→    """Database manager for initialization and session management."""
   137→
   138→    def __init__(self, db_path: Optional[str] = None):
   139→        if db_path is None:
   140→            config = get_config()
   141→            db_path = config.get('data.database_path', 'data/boss.db')
   142→
   143→        db_file = Path(db_path)
   144→        db_file.parent.mkdir(parents=True, exist_ok=True)
   145→
   146→        self.db_path = db_path
   147→        # NullPool: each session gets a fresh connection, safe for
   148→        # concurrent Web + background task access to SQLite.
   149→        self.engine = create_engine(
   150→            f'sqlite:///{db_path}',
   151→            echo=False,
   152→            poolclass=NullPool,
   153→            connect_args={"check_same_thread": False},
   154→        )
   155→        self.Session = sessionmaker(bind=self.engine)
   156→
   157→        self.init_db()
   158→        logger.info(f"Database initialized at: {db_path}")
   159→
   160→    def init_db(self):
   161→        """Create all tables and run pending migrations."""
   162→        Base.metadata.create_all(self.engine)
   163→        logger.info("Database tables created successfully")
   164→
   165→        from src.models.migrations import run_migrations
   166→        run_migrations(self.engine)
   167→
   168→    @contextmanager
   169→    def session_scope(self):
   170→        """Provide a transactional scope with auto commit/rollback/close."""
   171→        session = self.Session()
   172→        try:
   173→            yield session
   174→            session.commit()
   175→        except Exception:
   176→            session.rollback()
   177→            raise
   178→        finally:
   179→            session.close()
   180→
   181→    def drop_all(self):
   182→        """Drop all tables (use with caution!)."""
   183→        Base.metadata.drop_all(self.engine)
   184→        logger.warning("All database tables dropped")
   185→
   186→
   187→# Global database manager instance
   188→_db_manager = None
   189→
   190→
   191→def get_db_manager() -> DatabaseManager:
   192→    """Get global database manager instance."""
   193→    global _db_manager
   194→    if _db_manager is None:
   195→        _db_manager = DatabaseManager()
   196→    return _db_manager
   197→
   198→
   199→def reset_db_manager() -> None:
   200→    """Reset global database manager (for testing)."""
   201→    global _db_manager
   202→    _db_manager = None
