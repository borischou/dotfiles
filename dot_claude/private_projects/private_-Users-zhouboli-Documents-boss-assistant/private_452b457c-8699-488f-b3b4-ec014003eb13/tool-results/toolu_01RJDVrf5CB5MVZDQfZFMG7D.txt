     1→#!/usr/bin/env python
     2→"""Database initialization script."""
     3→
     4→import sys
     5→from pathlib import Path
     6→
     7→# Add src to path
     8→sys.path.insert(0, str(Path(__file__).parent.parent))
     9→
    10→from src.utils.config import get_config
    11→from src.utils.logger import setup_logger
    12→from sqlalchemy import text
    13→from src.models.database import get_db_manager
    14→from loguru import logger
    15→
    16→
    17→def main():
    18→    """Initialize database with all tables."""
    19→    try:
    20→        # Setup logger
    21→        setup_logger()
    22→
    23→        logger.info("Starting database initialization...")
    24→
    25→        # Get configuration
    26→        config = get_config()
    27→        db_path = config.get('data.database_path', 'data/boss.db')
    28→
    29→        logger.info(f"Database path: {db_path}")
    30→
    31→        # Initialize database (constructor auto-creates tables)
    32→        db_manager = get_db_manager()

    34→            result = conn.execute(
    35→                text("SELECT name FROM sqlite_master WHERE type='table' ORDER BY name")
    36→            )
    37→            tables = [row[0] for row in result]
    38→
    39→        logger.info(f"Created tables: {', '.join(tables)}")
    40→        logger.success("Database initialized successfully!")
    41→
    42→        return 0
    43→
    44→    except Exception as e:
    45→        logger.error(f"Failed to initialize database: {e}")
    46→        return 1
    47→
    48→
    49→if __name__ == '__main__':
    50→    sys.exit(main())
