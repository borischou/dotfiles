    55→    55→@router.post("/start", response_class=HTMLResponse)
    56→    56→async def start_pipeline(
    57→    57→    request: Request,
    58→    58→    keywords: str = Form(...),
    59→    59→    template_name: str = Form(...),
    60→    60→    location: str = Form(""),
    61→    61→    pages: int = Form(3),
    62→    62→    batch_pages: int = Form(0),
    63→    63→    experience: str = Form(""),
    64→    64→    education: str = Form(""),
    65→    65→    salary: str = Form(""),
    66→    66→    skip_messaged: bool = Form(True),
    67→    67→    dry_run: bool = Form(False),
    68→    68→):
    69→    69→    """Start a pipeline task (queued if another is running)."""
    70→    70→    try:
    71→    71→        params = PipelineParams(
    72→    72→            keywords=keywords,
    73→    73→            template_name=template_name,
    74→    74→            location=location,
    75→    75→            pages=pages,
    76→    76→            batch_pages=batch_pages,
    77→    77→            experience=experience,
    78→    78→            education=education,
    79→    79→            salary=salary,
    80→    80→            skip_messaged=skip_messaged,
    81→    81→            dry_run=dry_run,
    82→    82→        )
    83→    83→    except ValidationError as e:
    84→    84→        msg = e.errors()[0]['msg'] if e.errors() else str(e)
    85→    85→        return HTMLResponse(f'<p><mark>{html.escape(msg)}</mark></p>', status_code=400)
    86→    86→
    87→    87→    import uuid
    88→    88→    task_id = str(uuid.uuid4())[:8]
    89→    89→
    90→    90→    async def _run():
    91→    91→        db = get_db_manager()
    92→    92→
    93→    93→        # Record pipeline run
    94→    94→        with db.session_scope() as session:
    95→    95→            run = PipelineRun(
    96→    96→                task_id=task_id,
    97→    97→                status='running',
    98→    98→                keywords=params.keywords,
    99→    99→                template_name=params.template_name,
   100→   100→                params=params.model_dump(exclude={'keywords', 'template_name'}),
   101→   101→            )
   102→   102→            session.add(run)
   103→   103→
   104→   104→        try:
   105→   105→            browser = get_browser_manager()
   106→   106→            await ensure_logged_in(browser)
   107→   107→
   108→   108→            result = await run_pipeline(
   109→   109→                keywords=tuple(params.keywords),
   110→   110→                template_name=params.template_name,
   111→   111→                variables={},
   112→   112→                location=params.location,
   113→   113→                max_pages=params.pages,
   114→   114→                batch_pages=params.batch_pages,
   115→   115→                experience=params.experience,
   116→   116→                education=params.education,
   117→   117→                salary=params.salary,
   118→   118→                interactive=False,
   119→   119→                skip_messaged=params.skip_messaged,
   120→   120→                dry_run=params.dry_run,
   121→   121→                browser=browser,
   122→   122→            )
   123→   123→
   124→   124→            with db.session_scope() as session:
   125→   125→                run_record = session.query(PipelineRun).filter_by(task_id=task_id).first()
   126→   126→                if run_record:
   127→   127→                    run_record.status = 'completed'
   128→   128→                    run_record.result = result
   129→   129→                    run_record.finished_at = datetime.now(timezone.utc)
   130→   130→
   131→   131→            return result
   132→   132→
   133→   133→        except asyncio.CancelledError:
   134→   134→            with db.session_scope() as session:
   135→   135→                run_record = session.query(PipelineRun).filter_by(task_id=task_id).first()
   136→   136→                if run_record:
   137→   137→                    run_record.status = 'cancelled'
   138→   138→                    run_record.finished_at = datetime.now(timezone.utc)
   139→   139→            raise
   140→   140→
   141→   141→        except Exception as e:
   142→   142→            with db.session_scope() as session:
   143→   143→                run_record = session.query(PipelineRun).filter_by(task_id=task_id).first()
   144→   144→                if run_record:
   145→   145→                    run_record.status = 'failed'
   146→   146→                    run_record.result = {"error": str(e)}
   147→   147→                    run_record.finished_at = datetime.now(timezone.utc)
   148→   148→            raise
   149→   149→
   150→   150→    task = BrowserTask(
   151→   151→        name=f"pipeline:{','.join(params.keywords)}",
   152→   152→        coroutine_factory=_run,
   153→   153→        priority=TaskPriority.LOW,
   154→   154→        task_id=task_id,
   155→   155→    )
   156→   156→    await task_queue.submit(task)
   157→   157→
   158→   158→    queued = task_queue.queue_size > 0 and task_queue.is_busy
   159→   159→    return templates.TemplateResponse("partials/pipeline_progress.html", {
   160→   160→        "request": request,
   161→   161→        "task_id": task.task_id,
   162→   162→        "queued": queued,
   163→   163→    })

   175→   175→@router.get("/progress/{task_id}")
   176→   176→async def pipeline_progress_sse(request: Request, task_id: str):
   177→   177→    """SSE endpoint for pipeline progress events."""
   178→   178→
   179→   179→    async def event_generator():
   180→   180→        async for msg in task_queue.subscribe_progress(task_id):
   181→   181→            if await request.is_disconnected():
   182→   182→                break
   183→   183→            yield {
   184→   184→                "event": msg["event"],
   185→   185→                "data": json.dumps(msg["data"], ensure_ascii=False),
   186→   186→            }
   187→   187→
   188→   188→    return EventSourceResponse(event_generator())
