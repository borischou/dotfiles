     1→"""Rate limiter for preventing account bans."""
     2→
     3→import asyncio
     4→from datetime import datetime, timedelta, timezone
     5→from typing import Tuple, Optional
     6→
     7→from loguru import logger
     8→from sqlalchemy import func
     9→
    10→from src.models.database import ActionLog, get_db_manager, DatabaseManager
    11→from src.utils.config import get_config, Config
    12→from src.exceptions.errors import RateLimitExceededError
    13→
    14→
    15→class RateLimiter:
    16→    """Rate limiter for controlling action frequency.
    17→
    18→    Supports adaptive throttling: when captchas or errors are reported,
    19→    limits are temporarily tightened to reduce detection risk.
    20→    """
    21→
    22→    def __init__(self, config: Optional[Config] = None, db_manager: Optional[DatabaseManager] = None):
    23→        self.config = config or get_config()
    24→        self.db_manager = db_manager or get_db_manager()
    25→
    26→        self._throttle_factor: float = 1.0
    27→        self._captcha_timestamps: list = []
    28→        self._error_timestamps: list = []
    29→        self._pruned: bool = False
    30→
    31→    def report_captcha(self) -> None:
    32→        now = datetime.now(timezone.utc)
    33→        self._captcha_timestamps.append(now)
    34→        cutoff = now - timedelta(minutes=30)
    35→        self._captcha_timestamps = [t for t in self._captcha_timestamps if t >= cutoff]
    36→        self._update_throttle_factor()
    37→
    38→    def report_error(self) -> None:
    39→        """Report an error to adaptive throttling (e.g. page load failure, selector not found)."""
    40→        now = datetime.now(timezone.utc)
    41→        self._error_timestamps.append(now)
    42→        cutoff = now - timedelta(minutes=30)
    43→        self._error_timestamps = [t for t in self._error_timestamps if t >= cutoff]
    44→        self._update_throttle_factor()
    45→
    46→    def _update_throttle_factor(self) -> None:
    47→        captcha_count = len(self._captcha_timestamps)
    48→        error_count = len(self._error_timestamps)
    49→        combined = captcha_count + error_count * 0.5
    50→
    51→        if combined >= 3:
    52→            self._throttle_factor = 0.3
    53→        elif combined >= 2:
    54→            self._throttle_factor = 0.5
    55→        elif combined >= 1:
    56→            self._throttle_factor = 0.7
    57→        else:
    58→            self._throttle_factor = 1.0
    59→
    60→        if self._throttle_factor < 1.0:
    61→            logger.warning(
    62→                f"Adaptive throttle active: factor={self._throttle_factor:.1f} "
    63→                f"(captchas={captcha_count}, errors={error_count} in last 30min)"
    64→            )
    65→
    66→    def _refresh_throttle(self) -> None:
    67→        cutoff = datetime.now(timezone.utc) - timedelta(minutes=30)
    68→        self._captcha_timestamps = [t for t in self._captcha_timestamps if t >= cutoff]
    69→        self._error_timestamps = [t for t in self._error_timestamps if t >= cutoff]
    70→        self._update_throttle_factor()
    71→
    72→    @property
    73→    def throttle_factor(self) -> float:
    74→        return self._throttle_factor
    75→
    76→    def _get_limits(self, action_type: str) -> Tuple[int, int, int]:
    77→        action_map = {
    78→            'search': ('search_per_hour', 'search_per_day'),
    79→            'view_profile': ('profile_view_per_hour', 'profile_view_per_day'),
    80→            'send_message': ('message_per_hour', 'message_per_day'),
    81→        }
    82→
    83→        if action_type not in action_map:
    84→            raise ValueError(f"Unknown action type: {action_type}")
    85→
    86→        hour_key, day_key = action_map[action_type]
    87→
    88→        per_hour = self.config.get(f'rate_limits.{hour_key}', 20)
    89→        per_day = self.config.get(f'rate_limits.{day_key}', 100)
    90→        min_interval = self.config.get('rate_limits.min_interval_seconds', 35)
    91→
    92→        per_hour = max(1, int(per_hour * self._throttle_factor))
    93→        per_day = max(1, int(per_day * self._throttle_factor))
    94→        if self._throttle_factor < 1.0:
    95→            min_interval = int(min_interval / self._throttle_factor)
    96→
    97→        return per_hour, per_day, min_interval
    98→
    99→    def _count_actions(self, action_type: str, since: datetime) -> int:
   100→        with self.db_manager.session_scope() as session:
   101→            count = session.query(func.count(ActionLog.id)).filter(
   102→                ActionLog.action_type == action_type,
   103→                ActionLog.timestamp >= since
   104→            ).scalar()
   105→            return count or 0
   106→
   107→    @staticmethod
   108→    def _ensure_aware(dt: Optional[datetime]) -> Optional[datetime]:
   109→        """Ensure a datetime from SQLite is timezone-aware (UTC)."""
   110→        if dt is None:
   111→            return None
   112→        if dt.tzinfo is None:
   113→            return dt.replace(tzinfo=timezone.utc)
   114→        return dt
   115→
   116→    def _get_last_action_time(self, action_type: Optional[str] = None) -> Optional[datetime]:
   117→        with self.db_manager.session_scope() as session:
   118→            query = session.query(func.max(ActionLog.timestamp))
   119→            if action_type:
   120→                query = query.filter(ActionLog.action_type == action_type)
   121→            last_time = query.scalar()
   122→            return self._ensure_aware(last_time)
   123→
   124→    def check_rate_limit(self, action_type: str) -> None:
   125→        if not self._pruned:
   126→            self.prune_old_entries()
   127→            self._pruned = True
   128→
   129→        self._refresh_throttle()
   130→        per_hour, per_day, min_interval = self._get_limits(action_type)
   131→
   132→        now = datetime.now(timezone.utc)
   133→        one_hour_ago = now - timedelta(hours=1)
   134→        one_day_ago = now - timedelta(days=1)
   135→
   136→        hourly_count = self._count_actions(action_type, one_hour_ago)
   137→        if hourly_count >= per_hour:
   138→            with self.db_manager.session_scope() as session:
   139→                oldest_in_window = session.query(ActionLog.timestamp).filter(
   140→                    ActionLog.action_type == action_type,
   141→                    ActionLog.timestamp >= one_hour_ago
   142→                ).order_by(ActionLog.timestamp.asc()).first()
   143→
   144→                if oldest_in_window:
   145→                    oldest_ts = self._ensure_aware(oldest_in_window[0])
   146→                    wait_until = oldest_ts + timedelta(hours=1)
   147→                    wait_seconds = max(1, int((wait_until - now).total_seconds()))
   148→                else:
   149→                    wait_seconds = 3600
   150→
   151→            raise RateLimitExceededError(
   152→                action_type=action_type,
   153→                wait_time=wait_seconds,
   154→                message=f"Hourly rate limit exceeded for {action_type}. "
   155→                        f"Limit: {per_hour}/hour. Wait {wait_seconds // 60} minutes."
   156→            )
   157→
   158→        daily_count = self._count_actions(action_type, one_day_ago)
   159→        if daily_count >= per_day:
   160→            with self.db_manager.session_scope() as session:
   161→                oldest_in_window = session.query(ActionLog.timestamp).filter(
   162→                    ActionLog.action_type == action_type,
   163→                    ActionLog.timestamp >= one_day_ago
   164→                ).order_by(ActionLog.timestamp.asc()).first()
   165→
   166→                if oldest_in_window:
   167→                    oldest_ts = self._ensure_aware(oldest_in_window[0])
   168→                    wait_until = oldest_ts + timedelta(days=1)
   169→                    wait_seconds = max(1, int((wait_until - now).total_seconds()))
   170→                else:
   171→                    wait_seconds = 86400
   172→
   173→            raise RateLimitExceededError(
   174→                action_type=action_type,
   175→                wait_time=wait_seconds,
   176→                message=f"Daily rate limit exceeded for {action_type}. "
   177→                        f"Limit: {per_day}/day. Wait {wait_seconds // 3600} hours."
   178→            )
   179→
   180→        last_action = self._get_last_action_time(action_type=action_type)
   181→        if last_action:
   182→            elapsed = (now - last_action).total_seconds()
   183→            if elapsed < min_interval:
   184→                wait_seconds = max(1, int(min_interval - elapsed))
   185→                raise RateLimitExceededError(
   186→                    action_type=action_type,
   187→                    wait_time=wait_seconds,
   188→                    message=f"Minimum interval not met for {action_type}. Wait {wait_seconds} seconds."
   189→                )
   190→
   191→        logger.debug(f"Rate limit check passed for {action_type}")
   192→
   193→    def log_action(self, action_type: str, details: Optional[dict] = None) -> None:
   194→        with self.db_manager.session_scope() as session:
   195→            action_log = ActionLog(
   196→                action_type=action_type,
   197→                timestamp=datetime.now(timezone.utc),
   198→                details=details or {}
   199→            )
   200→            session.add(action_log)
   201→        logger.debug(f"Logged action: {action_type}")
   202→
   203→    def prune_old_entries(self, retention_hours: int = 48) -> int:
   204→        cutoff = datetime.now(timezone.utc) - timedelta(hours=retention_hours)
   205→        try:
   206→            with self.db_manager.session_scope() as session:
   207→                count = session.query(ActionLog).filter(
   208→                    ActionLog.timestamp < cutoff
   209→                ).delete()
   210→                logger.debug(f"Pruned {count} old ActionLog entries (older than {retention_hours}h)")
   211→                return count
   212→        except Exception as e:
   213→            logger.error(f"Failed to prune ActionLog entries: {e}")
   214→            return 0
   215→
   216→    def get_quota_status(self, action_type: str) -> dict:
   217→        per_hour, per_day, _ = self._get_limits(action_type)
   218→
   219→        now = datetime.now(timezone.utc)
   220→        one_hour_ago = now - timedelta(hours=1)
   221→        one_day_ago = now - timedelta(days=1)
   222→
   223→        hourly_used = self._count_actions(action_type, one_hour_ago)
   224→        daily_used = self._count_actions(action_type, one_day_ago)
   225→
   226→        return {
   227→            'action_type': action_type,
   228→            'hourly': {
   229→                'used': hourly_used,
   230→                'limit': per_hour,
   231→                'remaining': max(0, per_hour - hourly_used)
   232→            },
   233→            'daily': {
   234→                'used': daily_used,
   235→                'limit': per_day,
   236→                'remaining': max(0, per_day - daily_used)
   237→            }
   238→        }
   239→
   240→
   241→async def async_check_rate_limit(
   242→    action_type: str,
   243→    max_retries: int = 60,
   244→    max_total_wait: int = 1800,
   245→) -> None:
   246→    """Check rate limit, waiting through violations until the action is allowed.
   247→
   248→    Raises RateLimitExceededError if still blocked after max_retries attempts
   249→    or max_total_wait seconds (prevents hanging for hours on daily limits).
   250→    """
   251→    import time
   252→    rate_limiter = get_rate_limiter()
   253→    start = time.monotonic()
   254→    for attempt in range(max_retries):
   255→        try:
   256→            rate_limiter.check_rate_limit(action_type)
   257→            return
   258→        except RateLimitExceededError as e:
   259→            elapsed = time.monotonic() - start
   260→            if attempt >= max_retries - 1 or elapsed + e.wait_time > max_total_wait:
   261→                raise
   262→            wait = e.wait_time + 1  # +1s buffer
   263→            logger.info(
   264→                f"Rate limit hit: {e}. Waiting {wait}s before retrying "
   265→                f"(attempt {attempt + 1}/{max_retries}, elapsed {elapsed:.0f}s)..."
   266→            )
   267→            await asyncio.sleep(wait)
   268→
   269→
   270→_rate_limiter = None
   271→
   272→
   273→def get_rate_limiter() -> RateLimiter:
   274→    global _rate_limiter
   275→    if _rate_limiter is None:
   276→        _rate_limiter = RateLimiter()
   277→    return _rate_limiter
   278→
   279→
   280→def reset_rate_limiter() -> None:
   281→    """Reset global rate limiter instance (for testing)."""
   282→    global _rate_limiter
   283→    _rate_limiter = None
