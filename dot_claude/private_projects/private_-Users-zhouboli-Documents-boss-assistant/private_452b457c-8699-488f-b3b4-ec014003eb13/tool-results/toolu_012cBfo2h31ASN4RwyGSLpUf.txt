    29→<script>
    30→function saveTemplate(event, name) {
    31→    event.preventDefault();
    32→    const content = document.getElementById("content-" + name).value;
    33→    const statusEl = document.getElementById("save-status-" + name);
    34→
    35→    fetch("/templates/" + name, {
    36→        method: "PUT",
    37→        headers: {"Content-Type": "application/json"},
    38→        body: JSON.stringify({content: content})
    39→    })
    40→    .then(r => r.json())
    41→    .then(data => {
    42→        statusEl.textContent = "已保存";
    43→        setTimeout(() => statusEl.textContent = "", 2000);
    44→    })
    45→    .catch(err => {
    46→        statusEl.textContent = "保存失败: " + err;
    47→    });
    48→}
