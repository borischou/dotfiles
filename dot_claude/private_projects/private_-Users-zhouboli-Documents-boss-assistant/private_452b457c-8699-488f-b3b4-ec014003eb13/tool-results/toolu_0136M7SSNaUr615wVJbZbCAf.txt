     7→     7→
     8→     8→from src.web.app import task_queue
     9→     9→from src.core.browser import get_browser_manager
    10→    10→from src.modules.auth import ensure_logged_in

    12→    12→from src.utils.config import get_config
    13→    13→from src.models.database import Candidate, get_db_manager
    14→    14→
    15→    15→router = APIRouter()
    16→    16→
    17→    17→
    18→    18→@router.post("/preview", response_class=HTMLResponse)
    19→    19→async def preview_message(
    20→    20→    request: Request,
    21→    21→    boss_id: str = Form(...),
    22→    22→    template_name: str = Form(...),
    23→    23→):
    24→    24→    """Preview a rendered message for a candidate."""
    25→    25→    if not template_name:
    26→    26→        return HTMLResponse("")
    27→    27→
    28→    28→    try:
    29→    29→        config = get_config()
    30→    30→        template = config.get_template(template_name)
    31→    31→        db = get_db_manager()
    32→    32→
    33→    33→        variables = {}
    34→    34→        with db.session_scope() as session:
    35→    35→            candidate = session.query(Candidate).filter_by(boss_id=boss_id).first()
    36→    36→            if candidate:
    37→    37→                if candidate.name:
    38→    38→                    variables['name'] = candidate.name
    39→    39→                if candidate.company:
    40→    40→                    variables['company'] = candidate.company
    41→    41→                if candidate.title:
    42→    42→                    variables['position'] = candidate.title
    43→    43→
    44→    44→        content = render_template(template, variables)
    45→    45→
    46→    46→        missing = []
    47→    47→        for var in template.get('variables', []):
    48→    48→            if var not in variables:
    49→    49→                missing.append(var)
    50→    50→
    51→    51→        preview_html = f'<blockquote>{html.escape(content)}</blockquote>'
    52→    52→        if missing:
    53→    53→            preview_html += f'<p><small>缺少变量: {", ".join(missing)}</small></p>'
    54→    54→        return HTMLResponse(preview_html)
    55→    55→
    56→    56→    except Exception as e:
    57→    57→        return HTMLResponse(f'<p><mark>预览失败: {html.escape(str(e))}</mark></p>')

    64→    64→    template_name: str = Form(...),

   108→   108→
