    82‚Üí    82‚Üí    82‚Üí                'title': candidate.title,
    83‚Üí    83‚Üí    83‚Üí                'company': candidate.company,
    84‚Üí    84‚Üí    84‚Üí                'location': candidate.location,
    85‚Üí    85‚Üí    85‚Üí                'salary_expect': candidate.salary_expect,
    86‚Üí    86‚Üí    86‚Üí                'work_years': candidate.work_years,
    87‚Üí    87‚Üí    87‚Üí                'education_level': candidate.education_level,
    88‚Üí    88‚Üí    88‚Üí                'profile_url': candidate.profile_url,

    90‚Üí    90‚Üí    90‚Üí                'last_updated': candidate.last_updated,
    91‚Üí    91‚Üí    91‚Üí                'source_search': candidate.source_search,

    93‚Üí    93‚Üí    93‚Üí

   106‚Üí   106‚Üí   106‚Üí                record['latest_experience'] = _format_work_experience(latest_work)
   107‚Üí   107‚Üí   107‚Üí            else:

   109‚Üí   109‚Üí   109‚Üí
   110‚Üí   110‚Üí   110‚Üí            # Format skills

   112‚Üí   112‚Üí   112‚Üí            record['skills'] = ', '.join(skills) if skills else ''

   114‚Üí   114‚Üí   114‚Üí            data.append(record)
   115‚Üí   115‚Üí   115‚Üí
   116‚Üí   116‚Üí   116‚Üí        # Batch-fetch message info to avoid N+1 queries

   118‚Üí   118‚Üí   118‚Üí            from sqlalchemy import func
   119‚Üí   119‚Üí   119‚Üí
   120‚Üí   120‚Üí   120‚Üí            boss_ids = [r['boss_id'] for r in data]

   139‚Üí   139‚Üí   139‚Üí                Message.candidate_boss_id.in_(boss_ids)

   142‚Üí   142‚Üí   142‚Üí            for msg in all_msgs:
   143‚Üí   143‚Üí   143‚Üí                if msg.candidate_boss_id not in latest_msgs:

   147‚Üí   147‚Üí   147‚Üí                bid = record['boss_id']

   149‚Üí   149‚Üí   149‚Üí                latest_msg = latest_msgs.get(bid)
   150‚Üí   150‚Üí   150‚Üí                if latest_msg:

   152‚Üí   152‚Üí   152‚Üí                    record['last_message_date'] = latest_msg.sent_at or latest_msg.created_at
   153‚Üí   153‚Üí   153‚Üí                    record['last_message_template'] = latest_msg.template_name
   154‚Üí   154‚Üí   154‚Üí                else:
   155‚Üí   155‚Üí   155‚Üí                    record['last_message_status'] = ''
   156‚Üí   156‚Üí   156‚Üí                    record['last_message_date'] = None
   157‚Üí   157‚Üí   157‚Üí                    record['last_message_template'] = ''
   158‚Üí   158‚Üí   158‚Üí
   159‚Üí   159‚Üí   159‚Üí        return data
