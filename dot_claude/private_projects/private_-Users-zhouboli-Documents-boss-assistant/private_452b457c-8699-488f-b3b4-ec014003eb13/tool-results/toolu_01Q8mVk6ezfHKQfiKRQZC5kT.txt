     3→from typing import List, Optional
     4→
     5→from pydantic import BaseModel, field_validator

     7→
     8→class PipelineParams(BaseModel):
     9→    """Validated pipeline start parameters."""
    10→    keywords: List[str]
    11→    template_name: str
    12→    location: Optional[str] = None
    13→    pages: int = 3
    14→    batch_pages: int = 0
    15→    experience: Optional[str] = None
    16→    education: Optional[str] = None
    17→    salary: Optional[str] = None
    18→    skip_messaged: bool = True
    19→    dry_run: bool = False
    20→
    21→    @field_validator('keywords', mode='before')
    22→    @classmethod
    23→    def parse_keywords(cls, v):
    24→        if isinstance(v, str):
    25→            return [k.strip() for k in v.split(",") if k.strip()]
    26→        return v
    27→
    28→    @field_validator('keywords')
    29→    @classmethod
    30→    def keywords_not_empty(cls, v):
    31→        if not v:
    32→            raise ValueError("请输入关键词")
    33→        return v
    34→
    35→    @field_validator('template_name')
    36→    @classmethod
    37→    def template_not_empty(cls, v):
    38→        if not v or not v.strip():
    39→            raise ValueError("请选择消息模板")
    40→        return v.strip()
    41→
    42→    @field_validator('pages')
    43→    @classmethod
    44→    def pages_range(cls, v):
    45→        return max(1, min(v, 20))
    46→
    47→    @field_validator('batch_pages')
    48→    @classmethod
    49→    def batch_pages_range(cls, v):
    50→        return max(0, min(v, 20))
    51→
    52→    @field_validator('location', 'experience', 'education', 'salary', mode='before')
    53→    @classmethod
    54→    def empty_str_to_none(cls, v):
    55→        if isinstance(v, str) and not v.strip():
    56→            return None
    57→        return v
