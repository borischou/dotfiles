     3→from contextlib import asynccontextmanager
     4→from pathlib import Path
     5→
     6→from fastapi import FastAPI, Request

     8→from fastapi.templating import Jinja2Templates

    10→
    11→from src.web.services.chrome_manager import ChromeProcessManager
    12→from src.web.task_queue import TaskQueue
    13→
    14→WEB_DIR = Path(__file__).resolve().parent
    15→TEMPLATES_DIR = WEB_DIR / "templates"
    16→STATIC_DIR = WEB_DIR / "static"
    17→
    18→# Shared instances

    24→@asynccontextmanager
    25→async def lifespan(app: FastAPI):
    26→    """Application lifespan: start/stop Chrome and DB."""
    27→    logger.info("Starting Boss Assistant Web service...")
    28→
    29→    # Initialize database
    30→    from src.models.database import get_db_manager
    31→    get_db_manager()
    32→
    33→    # Start Chrome and task queue
    34→    await chrome_manager.start()
    35→    await task_queue.start()
    36→
    37→    yield
    38→
    39→    # Shutdown
    40→    logger.info("Shutting down Boss Assistant Web service...")
    41→    await task_queue.stop()
    42→    await chrome_manager.stop()
    43→
    44→    # Close browser connection
    45→    from src.core.browser import BrowserManager
    46→    await BrowserManager.reset_instance()
    47→
    48→    logger.info("Shutdown complete")
    49→

    74→    app.include_router(dashboard_router)
