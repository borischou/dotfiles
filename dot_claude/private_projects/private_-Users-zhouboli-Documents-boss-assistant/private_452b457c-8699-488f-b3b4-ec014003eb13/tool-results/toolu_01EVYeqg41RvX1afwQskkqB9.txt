    36→# JavaScript extraction for candidate profile from search card (inside searchFrame iframe).
    37→# This extracts all available data from the card since resume detail is wasm-rendered.
    38→EXTRACT_PROFILE_FROM_CARD_JS = """
    39→(expectId) => {
    40→    const link = document.querySelector('a[data-expect="' + expectId + '"]');
    41→    if (!link) return null;
    42→
    43→    const card = link.closest('.geek-info-card');
    44→    if (!card) return null;
    45→
    46→    const result = {
    47→        name: '',
    48→        title: '',
    49→        company: '',
    50→        location: '',
    51→        salary_expect: '',
    52→        work_years: '',
    53→        education_level: '',
    54→        basic_info: {},
    55→        education: [],
    56→        work_experience: [],
    57→        skills: []
    58→    };
    59→
    60→    // Name
    61→    const nameEl = card.querySelector('.name-label');
    62→    result.name = nameEl ? nameEl.innerText.trim() : '';
    63→
    64→    // Info labels: age, work years, education, job status, salary
    65→    const infoLabels = card.querySelectorAll('.info-labels > .label-text');
    66→    const infoTexts = Array.from(infoLabels).map(el => el.innerText.trim());
    67→
    68→    for (const text of infoTexts) {
    69→        if (text.includes('岁')) {
    70→            result.basic_info.age = text;
    71→        } else if (text.includes('年') && !text.includes('岁')) {
    72→            result.work_years = text;
    73→            result.basic_info.work_years = text;
    74→        } else if (['博士', '硕士', '本科', '大专', '中专', '高中'].some(d => text.includes(d))) {
    75→            result.education_level = text;
    76→            result.basic_info.education = text;
    77→        } else if (text.includes('在职') || text.includes('离职') || text.includes('求职')) {
    78→            result.basic_info.job_status = text;
    79→        } else if (text.includes('K') || text.includes('k')) {
    80→            result.salary_expect = text;
    81→        }
    82→    }
    83→
    84→    // Active status
    85→    const activeEl = card.querySelector('.active-desc-text');
    86→    if (activeEl) result.basic_info.active_status = activeEl.innerText.trim();
    87→
    88→    // Labels (e.g. "热搜")
    89→    const labelEls = card.querySelectorAll('.card-label');
    90→    result.basic_info.labels = Array.from(labelEls).map(el => el.innerText.trim()).filter(Boolean);
    91→
    92→    // Self description
    93→    const detailEl = card.querySelector('.info-detail');
    94→    if (detailEl) result.basic_info.self_description = detailEl.innerText.trim();
    95→
    96→    // Expect city
    97→    const expectCityEls = card.querySelectorAll('.expect-exp-box .t-tooltip-slot');
    98→    result.location = Array.from(expectCityEls).map(el => el.innerText.trim()).filter(Boolean).join(' ');
    99→
   100→    // Work experience
   101→    const workExpEls = card.querySelectorAll('.work-exp-box .t-tooltip-slot');
   102→    const workTexts = Array.from(workExpEls).map(el => el.innerText.trim()).filter(Boolean);
   103→    if (workTexts.length >= 1) {
   104→        result.company = workTexts[0];
   105→        result.title = workTexts[1] || '';
   106→        result.work_experience.push({
   107→            company: workTexts[0],
   108→            title: workTexts[1] || ''
   109→        });
   110→    }
   111→
   112→    // Education
   113→    const eduEls = card.querySelectorAll('.edu-exp-box .t-tooltip-slot');
   114→    const eduTexts = Array.from(eduEls).map(el => el.innerText.trim()).filter(Boolean);
   115→    if (eduTexts.length >= 1) {
   116→        result.education.push({
   117→            school: eduTexts[0],
   118→            major: eduTexts[1] || ''
   119→        });
   120→    }
   121→
   122→    // Elite flag
   123→    result.basic_info.is_elite = link.dataset.elitegeek === '1';
   124→
   125→    return result;
   126→}
   127→"""

   133→() => {
   134→    const dialog = document.querySelector('.dialog-wrap.active .dialog-lib-resume');
   135→    if (!dialog) return null;

   137→    const result = {};

   156→async def extract_profile(
   157→    boss_id: str,
   158→    browser: Optional[BrowserManager] = None,
   159→) -> Dict[str, Any]:
   160→    """
   161→    Extract detailed profile for a candidate.
   162→
   163→    Uses search card data as the primary source since the resume
   164→    overlay is wasm-rendered and has empty DOM.
   165→
   166→    Args:
   167→        boss_id: Boss直聘 candidate ID (e.g., "zp_81137184")
   168→        browser: Browser manager instance
   169→
   170→    Returns:
   171→        Dictionary with profile data
   172→
   173→    Raises:
   174→        ExtractionError: If profile extraction fails
   175→    """
   176→    if browser is None:
   177→        browser = get_browser_manager()
   178→
   179→    page = browser.page
   180→    if not page:
   181→        page = await browser.start()
   182→
   183→    rate_limiter = get_rate_limiter()
   184→    db_manager = get_db_manager()
   185→
   186→    # Rate limit check
   187→    await async_check_rate_limit('view_profile')
   188→
   189→    expect_id = boss_id.replace('zp_', '') if boss_id.startswith('zp_') else boss_id
   190→
   191→    try:
   192→        profile_data = None
   193→
   194→        # Strategy 1: Extract from search card in iframe (best source)
   195→        search_frame = page.frame(name='searchFrame')
   196→        if not search_frame:
   197→            for f in page.frames:
   198→                if 'frame/search' in (f.url or ''):
   199→                    search_frame = f
   200→                    break
   201→
   202→        if search_frame:
   203→            profile_data = await search_frame.evaluate(EXTRACT_PROFILE_FROM_CARD_JS, expect_id)
   204→
   205→        # Strategy 2: If not on search page, navigate there
   206→        if not profile_data:
   207→            await retry_navigation(page, 'https://www.zhipin.com/web/chat/search',
   208→                                   wait_until='domcontentloaded')
   209→            await async_random_delay('page_load')
   210→            await check_captcha(page, report_to_rate_limiter=True)
   211→
   212→            search_frame = page.frame(name='searchFrame')
   213→            if not search_frame:
   214→                for f in page.frames:
   215→                    if 'frame/search' in (f.url or ''):
   216→                        search_frame = f
   217→                        break
   218→
   219→            if search_frame:
   220→                profile_data = await search_frame.evaluate(EXTRACT_PROFILE_FROM_CARD_JS, expect_id)
   221→
   222→        # Strategy 3: Use database data as fallback
   223→        if not profile_data:
   224→            with db_manager.session_scope() as session:
   225→                candidate = session.query(Candidate).filter_by(boss_id=boss_id).first()
   226→                if candidate:
   227→                    profile_data = {
   228→                        'name': candidate.name or '',
   229→                        'title': candidate.title or '',
   230→                        'company': candidate.company or '',
   231→                        'location': candidate.location or '',
   232→                        'salary_expect': candidate.salary_expect or '',
   233→                        'work_years': candidate.work_years or '',
   234→                        'education_level': candidate.education_level or '',
   235→                        'basic_info': candidate.basic_info or {},
   236→                        'education': candidate.education or [],
   237→                        'work_experience': candidate.work_experience or [],
   238→                        'skills': candidate.skills or [],
   239→                    }
   240→
   241→        if not profile_data:
   242→            raise ExtractionError(f"No profile data found for {boss_id}")
   243→
   244→        # Try to supplement with overlay data (if resume dialog is open)
   245→        try:
   246→            overlay_data = await page.evaluate(EXTRACT_OVERLAY_JS)
   247→            if overlay_data:
   248→                profile_data.setdefault('basic_info', {})
   249→                if overlay_data.get('chat_position'):
   250→                    profile_data['basic_info']['chat_position'] = overlay_data['chat_position']
   251→                if overlay_data.get('contact_status'):
   252→                    profile_data['basic_info']['contact_status'] = overlay_data['contact_status']
   253→                if overlay_data.get('chat_history'):
   254→                    profile_data['basic_info']['chat_history'] = overlay_data['chat_history']
   255→        except Exception:
   256→            pass  # Overlay data is supplementary, not critical
   257→
   258→        # Log action
   259→        rate_limiter.log_action('view_profile', {'boss_id': boss_id})
   260→
   261→        # Update database
   262→        _update_profile_in_db(boss_id, profile_data, db_manager)
   263→
   264→        # Display profile
   265→        _display_profile(boss_id, profile_data)
   266→
   267→        return profile_data
   268→
   269→    except ExtractionError:
   270→        raise
   271→    except Exception as e:
   272→        logger.error(f"Error extracting profile for {boss_id}: {e}")
   273→        raise ExtractionError(f"Failed to extract profile for {boss_id}: {e}")
   274→

   276→def _update_profile_in_db(boss_id: str, profile_data: Dict, db_manager) -> None:
   277→    """Update candidate profile in the database."""
   278→    with db_manager.session_scope() as session:
   279→        candidate = session.query(Candidate).filter_by(boss_id=boss_id).first()
   280→
   281→        if not candidate:
   282→            candidate = Candidate(boss_id=boss_id)
   283→            session.add(candidate)
   284→
   285→        # Update fields
   286→        candidate.name = profile_data.get('name') or candidate.name
   287→        candidate.title = profile_data.get('title') or candidate.title
   288→        candidate.company = profile_data.get('company') or candidate.company
   289→        candidate.location = profile_data.get('location') or candidate.location
   290→        candidate.salary_expect = profile_data.get('salary_expect') or candidate.salary_expect
   291→        candidate.work_years = profile_data.get('work_years') or candidate.work_years
   292→        candidate.education_level = profile_data.get('education_level') or candidate.education_level
   293→        candidate.basic_info = profile_data.get('basic_info')
   294→        candidate.education = profile_data.get('education')
   295→        candidate.work_experience = profile_data.get('work_experience')
   296→        candidate.skills = profile_data.get('skills')

   299→def _display_profile(boss_id: str, profile_data: Dict) -> None:
   300→    """Display extracted profile in Rich format."""
   301→    name = profile_data.get('name', 'Unknown')
   302→    title = profile_data.get('title', '')
   303→    company = profile_data.get('company', '')
   304→    location = profile_data.get('location', '')
   305→    salary = profile_data.get('salary_expect', '')
   306→    work_years = profile_data.get('work_years', '')
   307→    education_level = profile_data.get('education_level', '')
   308→
   309→    # Header
   310→    header = f"[bold]{name}[/bold]"
   311→    if title:
   312→        header += f" - {title}"
   313→    if company:
   314→        header += f" @ {company}"
   315→
   316→    console.print(Panel(header, title=f"Profile: {boss_id}", border_style="green"))
   317→
   318→    # Basic info
   319→    info_parts = []
   320→    if location:
   321→        info_parts.append(f"Location: {location}")
   322→    if salary:
   323→        info_parts.append(f"Expected Salary: {salary}")
   324→    if work_years:
   325→        info_parts.append(f"Experience: {work_years}")
   326→    if education_level:
   327→        info_parts.append(f"Education: {education_level}")
   328→
   329→    basic_info = profile_data.get('basic_info', {})
   330→    if basic_info.get('age'):
   331→        info_parts.append(f"Age: {basic_info['age']}")
   332→    if basic_info.get('job_status'):
   333→        info_parts.append(f"Status: {basic_info['job_status']}")
   334→    if basic_info.get('active_status'):
   335→        info_parts.append(f"Active: {basic_info['active_status']}")
   336→
   337→    if info_parts:
   338→        console.print("  " + " | ".join(info_parts))
   339→
   340→    # Education
   341→    education = profile_data.get('education', [])
   342→    if education:
   343→        console.print("\n  [cyan]Education:[/cyan]")
   344→        for edu in education:
   345→            parts = []
   346→            if edu.get('school'):
   347→                parts.append(edu['school'])
   348→            if edu.get('degree'):
   349→                parts.append(edu['degree'])
   350→            if edu.get('major'):
   351→                parts.append(edu['major'])
   352→            console.print(f"    - {' | '.join(parts)}")
   353→
   354→    # Work experience
   355→    work_exp = profile_data.get('work_experience', [])
   356→    if work_exp:
   357→        console.print("\n  [cyan]Work Experience:[/cyan]")
   358→        for work in work_exp:
   359→            line = ""
   360→            if work.get('company'):
   361→                line += work['company']
   362→            if work.get('title'):
   363→                line += f" - {work['title']}"
   364→            if work.get('period'):
   365→                line += f" ({work['period']})"
   366→            console.print(f"    - {line}")
   367→
   368→    # Skills
   369→    skills = profile_data.get('skills', [])
   370→    if skills:
   371→        console.print(f"\n  [cyan]Skills:[/cyan] {', '.join(skills)}")
   372→
   373→    # Self description
   374→    if basic_info.get('self_description'):
   375→        desc = basic_info['self_description'][:200]
   376→        console.print(f"\n  [cyan]Self Description:[/cyan]\n    {desc}")
   377→
   378→    console.print()
