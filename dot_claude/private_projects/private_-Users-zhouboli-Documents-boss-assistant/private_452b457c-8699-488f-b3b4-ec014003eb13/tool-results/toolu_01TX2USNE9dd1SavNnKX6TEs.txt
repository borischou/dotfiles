   259‚Üí   259‚Üí   259‚Üídef _tokenize(text: str) -> set:
   260‚Üí   260‚Üí   260‚Üí    """Tokenize text using jieba or fallback to whitespace/punctuation splitting."""
   261‚Üí   261‚Üí   261‚Üí    if not text:
   262‚Üí   262‚Üí   262‚Üí        return set()
   263‚Üí   263‚Üí   263‚Üí    if _HAS_JIEBA:
   264‚Üí   264‚Üí   264‚Üí        return set(w.lower() for w in jieba.cut(text) if len(w.strip()) > 0)
   265‚Üí   265‚Üí   265‚Üí    # Fallback: split on whitespace + punctuation, return lowered tokens
   266‚Üí   266‚Üí   266‚Üí    tokens = re.split(r"[\s,Ôºå;Ôºõ„ÄÅ/|]+", text.lower())
   267‚Üí   267‚Üí   267‚Üí    return set(t for t in tokens if t)

   270‚Üí   270‚Üí   270‚Üídef _score_skills_jieba(
   271‚Üí   271‚Üí   271‚Üí    candidate_text: str,
   272‚Üí   272‚Üí   272‚Üí    candidate_skills: Optional[List[str]],
   273‚Üí   273‚Üí   273‚Üí    required_skills: Optional[List[str]] = None,
   274‚Üí   274‚Üí   274‚Üí    preferred_skills: Optional[List[str]] = None,
   275‚Üí   275‚Üí   275‚Üí) -> float:
   276‚Üí   276‚Üí   276‚Üí    """Score skill match using jieba tokenization. Returns 0.0-1.0."""
   277‚Üí   277‚Üí   277‚Üí    if not required_skills and not preferred_skills:
   278‚Üí   278‚Üí   278‚Üí        return 0.0
   279‚Üí   279‚Üí   279‚Üí
   280‚Üí   280‚Üí   280‚Üí    # Build candidate token set
   281‚Üí   281‚Üí   281‚Üí    tokens = _tokenize(candidate_text)
   282‚Üí   282‚Üí   282‚Üí    if candidate_skills:
   283‚Üí   283‚Üí   283‚Üí        for s in candidate_skills:
   284‚Üí   284‚Üí   284‚Üí            tokens.update(_tokenize(s))
   285‚Üí   285‚Üí   285‚Üí
   286‚Üí   286‚Üí   286‚Üí    def _match_skill(skill: str, token_set: set) -> bool:
   287‚Üí   287‚Üí   287‚Üí        skill_lower = skill.lower()
   288‚Üí   288‚Üí   288‚Üí        # Exact token match
   289‚Üí   289‚Üí   289‚Üí        if skill_lower in token_set:
   290‚Üí   290‚Üí   290‚Üí            return True
   291‚Üí   291‚Üí   291‚Üí        # Substring match in original text
   292‚Üí   292‚Üí   292‚Üí        if skill_lower in candidate_text.lower():
   293‚Üí   293‚Üí   293‚Üí            return True
   294‚Üí   294‚Üí   294‚Üí        # Tokenized skill match
   295‚Üí   295‚Üí   295‚Üí        skill_tokens = _tokenize(skill)
   296‚Üí   296‚Üí   296‚Üí        if skill_tokens and skill_tokens.issubset(token_set):
   297‚Üí   297‚Üí   297‚Üí            return True
   298‚Üí   298‚Üí   298‚Üí        return False
   299‚Üí   299‚Üí   299‚Üí
   300‚Üí   300‚Üí   300‚Üí    score = 0.0
   301‚Üí   301‚Üí   301‚Üí    total_weight = 0.0
   302‚Üí   302‚Üí   302‚Üí
   303‚Üí   303‚Üí   303‚Üí    if required_skills:
   304‚Üí   304‚Üí   304‚Üí        required_weight = 0.7
   305‚Üí   305‚Üí   305‚Üí        matched = sum(1 for s in required_skills if _match_skill(s, tokens))
   306‚Üí   306‚Üí   306‚Üí        score += required_weight * (matched / len(required_skills))
   307‚Üí   307‚Üí   307‚Üí        total_weight += required_weight
   308‚Üí   308‚Üí   308‚Üí
   309‚Üí   309‚Üí   309‚Üí    if preferred_skills:
   310‚Üí   310‚Üí   310‚Üí        preferred_weight = 0.3
   311‚Üí   311‚Üí   311‚Üí        matched = sum(1 for s in preferred_skills if _match_skill(s, tokens))
   312‚Üí   312‚Üí   312‚Üí        score += preferred_weight * (matched / len(preferred_skills))
   313‚Üí   313‚Üí   313‚Üí        total_weight += preferred_weight
   314‚Üí   314‚Üí   314‚Üí
   315‚Üí   315‚Üí   315‚Üí    if total_weight > 0:
   316‚Üí   316‚Üí   316‚Üí        return score / total_weight
   317‚Üí   317‚Üí   317‚Üí
   318‚Üí   318‚Üí   318‚Üí    return 0.0
   319‚Üí   319‚Üí   319‚Üí

   358‚Üí   358‚Üí   358‚Üídef score_candidate(
   359‚Üí   359‚Üí   359‚Üí    name: str,
   360‚Üí   360‚Üí   360‚Üí    title: str,
   361‚Üí   361‚Üí   361‚Üí    company: str,
   362‚Üí   362‚Üí   362‚Üí    search_keywords: Optional[List[str]] = None,
   363‚Üí   363‚Üí   363‚Üí    *,
   364‚Üí   364‚Üí   364‚Üí    education_level: Optional[str] = None,
   365‚Üí   365‚Üí   365‚Üí    work_years: Optional[str] = None,

   368‚Üí   368‚Üí   368‚Üí    basic_info: Optional[Dict] = None,

   370‚Üí   370‚Üí   370‚Üí    weights: Optional[Dict[str, float]] = None,
   371‚Üí   371‚Üí   371‚Üí) -> int:
   372‚Üí   372‚Üí   372‚Üí    """Score a candidate with multi-factor weighted scoring.
   373‚Üí   373‚Üí   373‚Üí
   374‚Üí   374‚Üí   374‚Üí    Without keyword-only args, behavior is backward-compatible with the
   375‚Üí   375‚Üí   375‚Üí    original implementation: seniority + company tier + keyword + rd_bonus
   376‚Üí   376‚Üí   376‚Üí    scored on the original absolute scale.
   377‚Üí   377‚Üí   377‚Üí    """
   378‚Üí   378‚Üí   378‚Üí    # Determine if we should use enhanced scoring
   379‚Üí   379‚Üí   379‚Üí    has_enhanced_info = any([education_level, work_years, skills, work_experience, basic_info])
   380‚Üí   380‚Üí   380‚Üí
   381‚Üí   381‚Üí   381‚Üí    if jd_config is _UNSET:
   382‚Üí   382‚Üí   382‚Üí        # Only auto-load JD config when enhanced candidate data is available
   383‚Üí   383‚Üí   383‚Üí        jd = _get_jd_config() if has_enhanced_info else None
   384‚Üí   384‚Üí   384‚Üí    else:

   386‚Üí   386‚Üí   386‚Üí

   388‚Üí   388‚Üí   388‚Üí        # Legacy mode: original absolute scoring
   389‚Üí   389‚Üí   389‚Üí        return _legacy_score(name, title, company, search_keywords)
   390‚Üí   390‚Üí   390‚Üí
   391‚Üí   391‚Üí   391‚Üí    # Enhanced mode: weighted factor scoring
   392‚Üí   392‚Üí   392‚Üí    w = weights if weights is not None else _get_weights()
   393‚Üí   393‚Üí   393‚Üí    total_weight = sum(v for k, v in w.items() if k != "llm_score")
   394‚Üí   394‚Üí   394‚Üí    if total_weight <= 0:
   395‚Üí   395‚Üí   395‚Üí        total_weight = 1.0
   396‚Üí   396‚Üí   396‚Üí
   397‚Üí   397‚Üí   397‚Üí    # Compute each factor (0.0-1.0)
   398‚Üí   398‚Üí   398‚Üí    seniority = _score_seniority(title)
   399‚Üí   399‚Üí   399‚Üí    company_tier = _score_company(company)

   401‚Üí   401‚Üí   401‚Üí        education_level,
   402‚Üí   402‚Üí   402‚Üí        jd.get("preferred_education") if jd else None,
   403‚Üí   403‚Üí   403‚Üí    )
   404‚Üí   404‚Üí   404‚Üí    experience = _score_experience(
   405‚Üí   405‚Üí   405‚Üí        work_years,
   406‚Üí   406‚Üí   406‚Üí        jd.get("min_experience_years") if jd else None,
   407‚Üí   407‚Üí   407‚Üí    )
   408‚Üí   408‚Üí   408‚Üí

   410‚Üí   410‚Üí   410‚Üí    candidate_text = _build_candidate_text(title, company, skills, work_experience, basic_info)
   411‚Üí   411‚Üí   411‚Üí    skill_match = _score_skills_jieba(

   413‚Üí   413‚Üí   413‚Üí        skills,

   416‚Üí   416‚Üí   416‚Üí    )

   418‚Üí   418‚Üí   418‚Üí    keyword_rel = _score_keyword_relevance(title, company, search_keywords)
   419‚Üí   419‚Üí   419‚Üí    rd_bonus = _score_rd_bonus(title)
   420‚Üí   420‚Üí   420‚Üí
   421‚Üí   421‚Üí   421‚Üí    # Weighted sum (exclude llm_score from rule engine)

   423‚Üí   423‚Üí   423‚Üí        "seniority": seniority,
   424‚Üí   424‚Üí   424‚Üí        "company_tier": company_tier,
   425‚Üí   425‚Üí   425‚Üí        "education": education,
   426‚Üí   426‚Üí   426‚Üí        "experience": experience,

   428‚Üí   428‚Üí   428‚Üí        "keyword_relevance": keyword_rel,
   429‚Üí   429‚Üí   429‚Üí        "rd_bonus": rd_bonus,
   430‚Üí   430‚Üí   430‚Üí    }
   431‚Üí   431‚Üí   431‚Üí
   432‚Üí   432‚Üí   432‚Üí    weighted_sum = 0.0
   433‚Üí   433‚Üí   433‚Üí    for factor_name, factor_value in factors.items():

   436‚Üí   436‚Üí   436‚Üí
   437‚Üí   437‚Üí   437‚Üí    # Normalize to 0-100 scale
   438‚Üí   438‚Üí   438‚Üí    score = int(round(weighted_sum / total_weight * 100))
   439‚Üí   439‚Üí   439‚Üí    return max(0, min(score, 100))

   477‚Üí   477‚Üí   477‚Üí                        score += 20

   479‚Üí   479‚Üí   479‚Üí
   480‚Üí   480‚Üí   480‚Üí    if search_keywords and title:
