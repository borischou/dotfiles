     9→def validate_boss_id(boss_id: str) -> str:
    10→    """Validate Boss直聘 candidate ID format.
    11→
    12→    Boss直聘 candidate IDs stored as zp_{data-expect-id}.
    13→
    14→    Args:
    15→        boss_id: Boss直聘 candidate ID to validate
    16→
    17→    Returns:
    18→        Validated and cleaned ID
    19→
    20→    Raises:
    21→        ValidationError: If ID format is invalid
    22→    """
    23→    if not boss_id:
    24→        raise ValidationError("Boss ID cannot be empty")
    25→
    26→    boss_id = str(boss_id).strip()
    27→
    28→    if not re.match(r'^[a-zA-Z0-9_-]+$', boss_id):
    29→        raise ValidationError(f"Invalid Boss ID format: {boss_id}")
    30→
    31→    return boss_id

    34→def validate_keywords(keywords: str) -> str:
    35→    """Validate search keywords."""
    36→    if not keywords or not keywords.strip():
    37→        raise ValidationError("Search keywords cannot be empty")
    38→
    39→    keywords = keywords.strip()
    40→
    41→    if len(keywords) < 2:
    42→        raise ValidationError("Search keywords must be at least 2 characters")
    43→
    44→    if len(keywords) > 200:
    45→        raise ValidationError("Search keywords cannot exceed 200 characters")
    46→
    47→    return keywords

    50→def validate_template_variables(template: Dict[str, Any], variables: Dict[str, str]) -> None:
    51→    """Validate that all required template variables are provided."""
    52→    required_vars = template.get('variables', [])
    53→    provided_vars = set(variables.keys())
    54→    required_set = set(required_vars)
    55→
    56→    missing_vars = required_set - provided_vars
    57→
    58→    if missing_vars:
    59→        raise ValidationError(
    60→            f"Missing required template variables: {', '.join(missing_vars)}"
    61→        )
