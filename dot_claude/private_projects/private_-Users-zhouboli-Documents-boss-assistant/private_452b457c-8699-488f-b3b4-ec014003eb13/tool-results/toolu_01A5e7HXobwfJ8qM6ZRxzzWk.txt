    65→    def test_sort_candidates(self):
    66→        """Candidates should be sorted by score descending."""
    67→        candidates = [
    68→            {"name": "A", "title": "初级开发", "company": "小公司"},
    69→            {"name": "B", "title": "CTO", "company": "字节跳动"},
    70→            {"name": "C", "title": "高级工程师", "company": "阿里巴巴"},

    73→        assert sorted_list[0]["name"] == "B"
    74→        assert sorted_list[-1]["name"] == "A"

    81→                self.title = title

    86→        ]
    87→        sorted_list = sort_candidates_by_score(candidates, enrich_from_db=False)
    88→        assert sorted_list[0].name == "B"

   258→    def test_enhanced_score_with_jd(self):
   259→        """Enhanced scoring with explicit JD config."""
   260→        jd = {
   261→            "title": "前端工程师",
   262→            "required_skills": ["React", "TypeScript"],
   263→            "preferred_skills": ["Node.js"],
   264→            "min_experience_years": 3,
   265→            "preferred_education": "本科",
   266→        }
   267→        weights = {
   268→            "seniority": 15,
   269→            "company_tier": 15,
   270→            "education": 10,
   271→            "experience": 15,
   272→            "skill_match": 25,
   273→            "keyword_relevance": 10,
   274→            "rd_bonus": 5,
   275→            "llm_score": 5,
   276→        }
   277→        score = score_candidate(
   278→            "张三", "高级前端工程师", "字节跳动",
   279→            search_keywords=["前端"],
   280→            education_level="硕士",
   281→            work_years="5年",
   282→            skills=["React", "TypeScript", "Vue", "Node.js"],
   283→            jd_config=jd,
   284→            weights=weights,
   285→        )
   286→        # Should be a high score
   287→        assert score >= 60
   288→        assert score <= 100
   289→
   290→    def test_enhanced_vs_legacy_different_range(self):
   291→        """Enhanced scoring returns 0-100, legacy can exceed 100."""
   292→        jd = {
   293→            "required_skills": ["React"],
   294→            "min_experience_years": 3,
   295→            "preferred_education": "本科",
   296→        }
   297→        weights = {
   298→            "seniority": 20, "company_tier": 20, "education": 10,
   299→            "experience": 15, "skill_match": 20, "keyword_relevance": 10,
   300→            "rd_bonus": 5,
   301→        }
   302→        score = score_candidate(
   303→            "张三", "CTO", "字节跳动",
   304→            education_level="博士",
   305→            work_years="15年",
   306→            skills=["React"],
   307→            jd_config=jd,
   308→            weights=weights,
   309→        )
   310→        assert 0 <= score <= 100

   312→    def test_no_enhanced_info_uses_legacy(self):
