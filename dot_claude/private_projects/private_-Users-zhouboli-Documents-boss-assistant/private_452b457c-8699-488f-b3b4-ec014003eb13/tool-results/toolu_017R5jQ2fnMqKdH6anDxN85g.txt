    59→# Non-R&D roles to filter out

    61→    "销售", "产品经理", "市场", "运营", "行政", "人事", "财务",

    66→def _is_non_rd_role(title: str) -> bool:
    67→    """Check if title indicates a non-R&D role."""
    68→    if not title:
    69→        return False
    70→    for keyword in NON_RD_KEYWORDS:
    71→        if keyword in title:
    72→            return True
    73→    return False

    76→def _generate_boss_id(name: str, company: str) -> str:
    77→    """Generate a fallback Boss ID from name + company hash."""
    78→    raw = f"{name}_{company}"
    79→    h = hashlib.sha256(raw.encode()).hexdigest()[:20]
    80→    return f"zp_{h}"

    83→async def _get_search_frame(page: Page) -> Optional[Frame]:
    84→    """Get the searchFrame iframe from the main page.

    86→    Boss直聘 search page loads candidates inside iframe[name='searchFrame'].
    87→    Returns the Frame object for executing JS inside the iframe.

    94→    for f in page.frames:
    95→        if 'frame/search' in (f.url or ''):
    96→            return f

   100→
   101→# JavaScript extraction for candidates inside searchFrame iframe.
   102→# Targets: div.card-list > li.geek-info-card > a[data-expect]
   103→EXTRACT_CANDIDATES_JS = """
   104→() => {
   105→    const candidates = [];
   106→    const cards = document.querySelectorAll('.card-list .geek-info-card');
   107→
   108→    for (const card of cards) {
   109→        try {
   110→            const link = card.querySelector('a');
   111→            if (!link) continue;
   112→
   113→            // Candidate ID from data-expect attribute
   114→            const expectId = link.dataset.expect || '';
   115→            if (!expectId) continue;
   116→
   117→            const bossId = 'zp_' + expectId;
   118→
   119→            // Name (anonymized, e.g. "卿**")
   120→            const nameEl = card.querySelector('.name-label');
   121→            const name = nameEl ? nameEl.innerText.trim() : '';
   122→            if (!name) continue;
   123→
   124→            // Labels (e.g. "热搜")
   125→            const labelEls = card.querySelectorAll('.card-label');
   126→            const labels = Array.from(labelEls).map(el => el.innerText.trim()).filter(Boolean);
   127→
   128→            // Active status (e.g. "3日内活跃")
   129→            const activeEl = card.querySelector('.active-desc-text');
   130→            const activeStatus = activeEl ? activeEl.innerText.trim() : '';
   131→
   132→            // Info labels: age, work years, education, job status, salary

   135→

   137→            for (const text of infoTexts) {
   138→                if (text.includes('岁')) age = text;
   139→                else if (text.includes('年') && !text.includes('岁')) workYears = text;
   140→                else if (['博士', '硕士', '本科', '大专', '中专', '高中'].some(d => text.includes(d))) education = text;
   141→                else if (text.includes('在职') || text.includes('离职') || text.includes('求职')) jobStatus = text;
   142→                else if (text.includes('K') || text.includes('k') || text.includes('薪')) salary = text;
   143→            }
   144→
   145→            // Self description / detail
   146→            const detailEl = card.querySelector('.info-detail');
   147→            const selfDesc = detailEl ? detailEl.innerText.trim() : '';

   149→            // Expect city
   150→            const expectCityEls = card.querySelectorAll('.expect-exp-box .t-tooltip-slot');

   155→            const workExpTexts = Array.from(workExpEls).map(el => el.innerText.trim()).filter(Boolean);
   156→            const company = workExpTexts[0] || '';

   158→
   159→            // Education (school + major)
   160→            const eduExpEls = card.querySelectorAll('.edu-exp-box .t-tooltip-slot');
   161→            const eduTexts = Array.from(eduExpEls).map(el => el.innerText.trim()).filter(Boolean);
   162→            const school = eduTexts[0] || '';
   163→            const major = eduTexts[1] || '';
   164→
   165→            // Contact button status
   166→            const contactBtn = card.querySelector('.btn-getcontact');
   167→            const contactText = contactBtn ? contactBtn.innerText.trim() : '';
   168→
   169→            // Elite/hot candidate flag
   170→            const isElite = link.dataset.elitegeek === '1';
   171→
   172→            candidates.push({
   173→                boss_id: bossId,
   174→                name: name,
   175→                title: title,
   176→                company: company,
   177→                location: expectCity,
   178→                salary_expect: salary,
   179→                work_years: workYears,

   190→                profile_url: ''
   191→            });
   192→        } catch (e) {
   193→            console.error('Error extracting candidate:', e);
   194→        }
   195→    }
   196→
   197→    return candidates;
   198→}

   202→async def search_candidates(
   203→    keywords: List[str],
   204→    location: Optional[str] = None,
   205→    max_pages: int = 1,
   206→    experience: Optional[str] = None,
   207→    education: Optional[str] = None,
   208→    salary: Optional[str] = None,
   209→    browser: Optional[BrowserManager] = None,
   210→    start_page: int = 1,
   211→) -> List[CandidateBasic]:
   212→    """
   213→    Search for candidates on Boss直聘.
   214→
   215→    Navigates to search page, types keyword in iframe search input,
   216→    extracts candidates from card list, and optionally refreshes
   217→    via "换一换" button for more results.
   218→
   219→    Args:
   220→        keywords: List of search keywords
   221→        location: Location filter (e.g., "北京")
   222→        max_pages: Number of "换一换" refresh rounds per keyword
   223→        experience: Experience filter
   224→        education: Education filter
   225→        salary: Salary filter
   226→        browser: Browser manager instance
   227→        start_page: Starting refresh round (for resume)
   228→
   229→    Returns:
   230→        List of CandidateBasic instances
   231→    """
   232→    if browser is None:
   233→        browser = get_browser_manager()
   234→
   235→    page = browser.page
   236→    if not page:
   237→        page = await browser.start()
   238→
   239→    rate_limiter = get_rate_limiter()
   240→    db_manager = get_db_manager()
   241→
   242→    all_candidates: List[CandidateBasic] = []
   243→    seen_ids: Set[str] = set()
   244→    seen_dedup_keys: Set[str] = set()
   245→
   246→    for keyword in keywords:
   247→        console.print(f"\n[cyan]Searching for: {keyword}[/cyan]")
   248→
   249→        # Rate limit check
   250→        await async_check_rate_limit('search')
   251→
   252→        try:
   253→            # Navigate to search page (main page, iframe will load)
   254→            search_url = "https://www.zhipin.com/web/chat/search"
   255→            await retry_navigation(page, search_url, wait_until='domcontentloaded')
   256→            await async_random_delay('page_load')
   257→
   258→            # Check for captcha on main page
   259→            await check_captcha(page, report_to_rate_limiter=True)
   260→
   261→            # Get the search iframe
   262→            search_frame = await _get_search_frame(page)
   263→            if not search_frame:
   264→                logger.error("Could not find searchFrame iframe")
   265→                console.print("[red]Search iframe not found. Page structure may have changed.[/red]")
   266→                continue
   267→
   268→            # Wait for search input to be ready inside iframe
   269→            try:
   270→                await search_frame.wait_for_selector('input.search-input', timeout=10000)
   271→            except Exception:
   272→                logger.error("Search input not found in iframe")
   273→                continue
   274→
   275→            # Type keyword in search input
   276→            search_input = await search_frame.query_selector('input.search-input')
   277→            if search_input:
   278→                await search_input.click()
   279→                await page.wait_for_timeout(300)
   280→                # Clear and type
   281→                import sys
   282→                select_all_key = "Meta+a" if sys.platform == "darwin" else "Control+a"
   283→                await page.keyboard.press(select_all_key)
   284→                await search_input.fill(keyword)
   285→                await page.keyboard.press('Enter')
   286→                await async_random_delay('page_load')
   287→            else:
   288→                logger.error("Search input element not found")
   289→                continue
   290→
   291→            # Wait for card list to appear
   292→            try:
   293→                await search_frame.wait_for_selector('.card-list .geek-info-card', timeout=15000)
   294→            except Exception:
   295→                logger.warning(f"No candidate cards loaded for '{keyword}'")
   296→                continue
   297→
   298→            # Check for quota exhaustion
   299→            if await _check_quota_exhausted(search_frame):
   300→                console.print("[red]Daily contact quota exhausted. Stopping search.[/red]")
   301→                break
   302→
   303→            # Extract candidates from current results + refresh rounds
   304→            total_rounds = start_page + max_pages - 1
   305→            for refresh_round in range(start_page, total_rounds + 1):
   306→                if refresh_round > start_page:
   307→                    console.print(f"  [dim]Refresh round {refresh_round}/{total_rounds} (换一换)...[/dim]")
   308→                    await _refresh_search_results(search_frame)
   309→                    await async_random_delay('page_load')
   310→
   311→                # Extract candidates from current card list
   312→                raw_candidates = await search_frame.evaluate(EXTRACT_CANDIDATES_JS)
   313→
   314→                if not raw_candidates:
   315→                    logger.warning(f"No candidates found in round {refresh_round} for '{keyword}'")
   316→                    continue
   317→
   318→                # Process results
   319→                round_count = 0
   320→                for raw in raw_candidates:
   321→                    name = raw.get('name', '').strip()
   322→                    title = raw.get('title', '').strip()
   323→                    company = raw.get('company', '').strip()
   324→
   325→                    if not name:
   326→                        continue
   327→
   328→                    # Filter non-R&D roles
   329→                    if _is_non_rd_role(title):
   330→                        continue
   331→
   332→                    # Use boss_id from data-expect
   333→                    boss_id = raw.get('boss_id', '')
   334→                    if not boss_id:
   335→                        boss_id = _generate_boss_id(name, company)
   336→
   337→                    # Deduplicate
   338→                    if boss_id in seen_ids:
   339→                        continue
   340→
   341→                    dedup_key = f"{name}_{company}"
   342→                    if dedup_key in seen_dedup_keys:
   343→                        continue
   344→
   345→                    seen_ids.add(boss_id)
   346→                    seen_dedup_keys.add(dedup_key)
   347→
   348→                    candidate = CandidateBasic(
   349→                        boss_id=boss_id,
   350→                        name=name,
   351→                        title=title,
   352→                        company=company,
   353→                        location=raw.get('location', ''),
   354→                        salary_expect=raw.get('salary_expect', ''),
   355→                        work_years=raw.get('work_years', ''),
   356→                        education_level=raw.get('education_level', ''),
   357→                        profile_url=raw.get('profile_url', ''),
   358→                        active_status=raw.get('active_status', ''),
   359→                        school=raw.get('school', ''),
   360→                        major=raw.get('major', ''),
   361→                        job_status=raw.get('job_status', ''),
   362→                        self_desc=raw.get('self_desc', ''),
   363→                    )
   364→
   365→                    all_candidates.append(candidate)
   366→                    round_count += 1
   367→
   368→                console.print(f"  [green]Round {refresh_round}: {round_count} new candidates[/green]")
   369→
   370→            # Log action
   371→            rate_limiter.log_action('search', {
   372→                'keyword': keyword,
   373→                'refresh_rounds': max_pages,
   374→                'results': len([c for c in all_candidates]),
   375→            })
   376→
   377→            # Save search history
   378→            keyword_count = len([c for c in all_candidates if c.boss_id in seen_ids])
   379→            with db_manager.session_scope() as session:
   380→                history = SearchHistory(
   381→                    criteria={
   382→                        'keywords': keyword,
   383→                        'location': location,
   384→                        'refresh_rounds': max_pages,
   385→                        'experience': experience,
   386→                        'education': education,
   387→                        'salary': salary,
   388→                    },
   389→                    results_count=keyword_count,
   390→                )
   391→                session.add(history)
   392→
   393→        except Exception as e:
   394→            logger.error(f"Error searching for '{keyword}': {e}")
   395→            console.print(f"  [red]Error: {e}[/red]")
   396→            rate_limiter.report_error()
   397→
   398→    console.print(f"\n[green]Total unique candidates: {len(all_candidates)}[/green]")
   399→
   400→    # Save candidates to database
   401→    _save_candidates_to_db(all_candidates, keywords, db_manager)
   402→
   403→    # Display results summary
   404→    _display_results(all_candidates, keywords)
   405→
   406→    return all_candidates

   409→async def _refresh_search_results(frame: Frame) -> None:
   410→    """Click "换一换" button in the search iframe to load new candidates.
   411→
   412→    Boss直聘 search does NOT use infinite scroll. Instead it shows ~30 cards
   413→    and provides a "换一换" button to refresh the list with new candidates.
   414→    """
   415→    try:
   416→        refresh_btn = await frame.query_selector('.change-btn')
   417→        if refresh_btn and await refresh_btn.is_visible():
   418→            await refresh_btn.click()
   419→            # Wait for new cards to load
   420→            await frame.wait_for_timeout(2000)
   421→            return
   422→    except Exception as e:
   423→        logger.warning(f"Could not click refresh button: {e}")
   424→
   425→    # Fallback: scroll the iframe body to trigger any lazy loading
   426→    await frame.evaluate("window.scrollTo(0, document.body.scrollHeight)")
   427→    await frame.wait_for_timeout(1500)

   430→async def _check_quota_exhausted(frame: Frame) -> bool:
   431→    """Check if Boss直聘 shows a quota exhaustion indicator."""
   432→    try:
   433→        result = await frame.evaluate("""
   434→            () => {
   435→                const bodyText = document.body ? document.body.innerText : '';
   436→                const quotaIndicators = [
   437→                    '今日沟通次数已用完',
   438→                    '沟通次数已达上限',
   439→                    '今日联系人数已满',
   440→                    '升级会员',
   441→                    '开通会员享更多沟通'
   442→                ];
   443→                for (const indicator of quotaIndicators) {
   444→                    if (bodyText.includes(indicator)) {
   445→                        return true;
   446→                    }
   447→                }
   448→                return false;
   449→            }
   450→        """)
   451→        return result
   452→    except Exception:
   453→        return False

   456→def _save_candidates_to_db(
   457→    candidates: List[CandidateBasic],
   458→    keywords: List[str],
   459→    db_manager,
   460→) -> None:
   461→    """Save or update candidates in the database."""
   462→    source = ', '.join(keywords)
   463→
   464→    with db_manager.session_scope() as session:
   465→        # Batch-fetch existing candidates to avoid N+1 queries
   466→        boss_ids = [c.boss_id for c in candidates]
   467→        existing_map = {}
   468→        for i in range(0, len(boss_ids), 500):
   469→            chunk = boss_ids[i:i + 500]
   470→            rows = session.query(Candidate).filter(
   471→                Candidate.boss_id.in_(chunk)
   472→            ).all()
   473→            for row in rows:
   474→                existing_map[row.boss_id] = row
   475→
   476→        for c in candidates:
   477→            existing = existing_map.get(c.boss_id)
   478→            if existing:
   479→                existing.name = c.name or existing.name
   480→                existing.title = c.title or existing.title
   481→                existing.company = c.company or existing.company
   482→                existing.location = c.location or existing.location
   483→                existing.salary_expect = c.salary_expect or existing.salary_expect
   484→                existing.work_years = c.work_years or existing.work_years
   485→                existing.education_level = c.education_level or existing.education_level
   486→                existing.profile_url = c.profile_url or existing.profile_url
   487→                if source and source not in (existing.source_search or ''):
   488→                    existing.source_search = f"{existing.source_search or ''}, {source}".strip(', ')
   489→            else:
   490→                new_candidate = Candidate(
   491→                    boss_id=c.boss_id,
   492→                    name=c.name,
   493→                    title=c.title,
   494→                    company=c.company,
   495→                    location=c.location,
   496→                    salary_expect=c.salary_expect,
   497→                    work_years=c.work_years,
   498→                    education_level=c.education_level,
   499→                    profile_url=c.profile_url,
   500→                    source_search=source,
   501→                )
   502→                session.add(new_candidate)

   505→def _display_results(candidates: List[CandidateBasic], keywords: List[str]) -> None:
   506→    """Display search results in a Rich table."""
   507→    if not candidates:
   508→        console.print("[yellow]No candidates found.[/yellow]")

   510→
   511→    table = Table(title=f"Search Results ({len(candidates)} candidates)")
   512→    table.add_column("Boss ID", style="dim", max_width=20)
   513→    table.add_column("Name", style="bold")
   514→    table.add_column("Title")
   515→    table.add_column("Company")
   516→    table.add_column("Education")
   517→    table.add_column("Salary")
   518→    table.add_column("Experience")
   519→    table.add_column("Active", style="green")
   520→
   521→    for c in candidates[:50]:

   524→            c.name,
   525→            c.title[:30] if c.title else "",
   526→            c.company[:20] if c.company else "",
   527→            c.education_level or "",
   528→            c.salary_expect or "",
   529→            c.work_years or "",
   530→            c.active_status or "",
   531→        )
   532→
   533→    console.print(table)
   534→
   535→    if len(candidates) > 50:
   536→        console.print(f"[dim]...and {len(candidates) - 50} more candidates[/dim]")
   537→
