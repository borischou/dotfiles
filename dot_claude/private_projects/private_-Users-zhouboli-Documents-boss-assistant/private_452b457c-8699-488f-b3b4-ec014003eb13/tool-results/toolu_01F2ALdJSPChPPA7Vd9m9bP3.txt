    60вЖТ    60вЖТ    60вЖТ

    63вЖТ    63вЖТ    63вЖТ    def replacer(match):
    64вЖТ    64вЖТ    64вЖТ        var_name = match.group(1)
    65вЖТ    65вЖТ    65вЖТ        if var_name in variables:
    66вЖТ    66вЖТ    66вЖТ            return str(variables[var_name])

    68вЖТ    68вЖТ    68вЖТ
