   203→   203→   203→    keywords: List[str],

   205→   205→   205→    max_pages: int = 1,

   212→   212→   212→    """
   213→   213→   213→    Search for candidates on Boss直聘.
   214→   214→   214→
   215→   215→   215→    Navigates to search page, types keyword in iframe search input,
   216→   216→   216→    extracts candidates from card list, and optionally refreshes

   218→   218→   218→

   223→   223→   223→        experience: Experience filter

   226→   226→   226→        browser: Browser manager instance

   242→   242→   242→    all_candidates: List[CandidateBasic] = []

   244→   244→   244→    seen_dedup_keys: Set[str] = set()
   245→   245→   245→

   264→   264→   264→                logger.error("Could not find searchFrame iframe")

   266→   266→   266→                continue

   268→   268→   268→            # Wait for search input to be ready inside iframe
   269→   269→   269→            try:
   270→   270→   270→                await search_frame.wait_for_selector('input.search-input', timeout=10000)
   271→   271→   271→            except Exception:
   272→   272→   272→                logger.error("Search input not found in iframe")

   275→   275→   275→            # Type keyword in search input

   277→   277→   277→            if search_input:
   278→   278→   278→                await search_input.click()
   279→   279→   279→                await page.wait_for_timeout(300)
   280→   280→   280→                # Clear and type

   282→   282→   282→                select_all_key = "Meta+a" if sys.platform == "darwin" else "Control+a"
   283→   283→   283→                await page.keyboard.press(select_all_key)

   287→   287→   287→            else:

   289→   289→   289→                continue

   291→   291→   291→            # Wait for card list to appear

   293→   293→   293→                await search_frame.wait_for_selector('.card-list .geek-info-card', timeout=15000)

   323→   323→   323→                    company = raw.get('company', '').strip()

   339→   339→   339→                        continue
   340→   340→   340→
   341→   341→   341→                    dedup_key = f"{name}_{company}"
   342→   342→   342→                    if dedup_key in seen_dedup_keys:
   343→   343→   343→                        continue
   344→   344→   344→
   345→   345→   345→                    seen_ids.add(boss_id)
   346→   346→   346→                    seen_dedup_keys.add(dedup_key)
   347→   347→   347→
   348→   348→   348→                    candidate = CandidateBasic(
   349→   349→   349→                        boss_id=boss_id,
   350→   350→   350→                        name=name,
   351→   351→   351→                        title=title,
   352→   352→   352→                        company=company,
   353→   353→   353→                        location=raw.get('location', ''),
   354→   354→   354→                        salary_expect=raw.get('salary_expect', ''),
   355→   355→   355→                        work_years=raw.get('work_years', ''),

   361→   361→   361→                        job_status=raw.get('job_status', ''),

   371→   371→   371→            rate_limiter.log_action('search', {
   372→   372→   372→                'keyword': keyword,
   373→   373→   373→                'refresh_rounds': max_pages,
   374→   374→   374→                'results': len([c for c in all_candidates]),

   376→   376→   376→
   377→   377→   377→            # Save search history
   378→   378→   378→            keyword_count = len([c for c in all_candidates if c.boss_id in seen_ids])
   379→   379→   379→            with db_manager.session_scope() as session:

   382→   382→   382→                        'keywords': keyword,

   456→   456→   456→def _save_candidates_to_db(
   457→   457→   457→    candidates: List[CandidateBasic],
   458→   458→   458→    keywords: List[str],
   459→   459→   459→    db_manager,
   460→   460→   460→) -> None:
   461→   461→   461→    """Save or update candidates in the database."""
   462→   462→   462→    source = ', '.join(keywords)
   463→   463→   463→
   464→   464→   464→    with db_manager.session_scope() as session:
   465→   465→   465→        # Batch-fetch existing candidates to avoid N+1 queries
   466→   466→   466→        boss_ids = [c.boss_id for c in candidates]
   467→   467→   467→        existing_map = {}
   468→   468→   468→        for i in range(0, len(boss_ids), 500):
   469→   469→   469→            chunk = boss_ids[i:i + 500]
   470→   470→   470→            rows = session.query(Candidate).filter(
   471→   471→   471→                Candidate.boss_id.in_(chunk)
   472→   472→   472→            ).all()
   473→   473→   473→            for row in rows:
   474→   474→   474→                existing_map[row.boss_id] = row
   475→   475→   475→
   476→   476→   476→        for c in candidates:
   477→   477→   477→            existing = existing_map.get(c.boss_id)
   478→   478→   478→            if existing:
   479→   479→   479→                existing.name = c.name or existing.name
   480→   480→   480→                existing.title = c.title or existing.title
   481→   481→   481→                existing.company = c.company or existing.company
   482→   482→   482→                existing.location = c.location or existing.location
   483→   483→   483→                existing.salary_expect = c.salary_expect or existing.salary_expect
   484→   484→   484→                existing.work_years = c.work_years or existing.work_years
   485→   485→   485→                existing.education_level = c.education_level or existing.education_level
   486→   486→   486→                existing.profile_url = c.profile_url or existing.profile_url
   487→   487→   487→                if source and source not in (existing.source_search or ''):
   488→   488→   488→                    existing.source_search = f"{existing.source_search or ''}, {source}".strip(', ')
   489→   489→   489→            else:
   490→   490→   490→                new_candidate = Candidate(
   491→   491→   491→                    boss_id=c.boss_id,
   492→   492→   492→                    name=c.name,
   493→   493→   493→                    title=c.title,
   494→   494→   494→                    company=c.company,
   495→   495→   495→                    location=c.location,
   496→   496→   496→                    salary_expect=c.salary_expect,
   497→   497→   497→                    work_years=c.work_years,
   498→   498→   498→                    education_level=c.education_level,
   499→   499→   499→                    profile_url=c.profile_url,
   500→   500→   500→                    source_search=source,
   501→   501→   501→                )
   502→   502→   502→                session.add(new_candidate)
