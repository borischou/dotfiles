     1→"""Lightweight database migration system."""
     2→
     3→from typing import Callable, List, Tuple
     4→
     5→from sqlalchemy import text, inspect

     7→
     8→_migrations: List[Tuple[int, str, Callable]] = []
     9→
    10→
    11→def migration(version: int, description: str):
    12→    """Decorator to register a migration function."""
    13→    def decorator(fn: Callable):
    14→        _migrations.append((version, description, fn))
    15→        return fn
    16→    return decorator

    25→@migration(2, "Add retry_count column to messages table")
    26→def _m002_messages_retry_count(conn):
    27→    inspector = inspect(conn)
    28→    columns = [c["name"] for c in inspector.get_columns("messages")]
    29→    if "retry_count" not in columns:

    33→def _ensure_version_table(conn) -> None:
    34→    conn.execute(text("""
    35→        CREATE TABLE IF NOT EXISTS schema_version (
    36→            version INTEGER PRIMARY KEY,
    37→            description TEXT NOT NULL,
    38→            applied_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    39→        )
    40→    """))
    41→    conn.commit()
    42→
    43→
    44→def _current_version(conn) -> int:
    45→    result = conn.execute(text("SELECT MAX(version) FROM schema_version"))
    46→    row = result.fetchone()
    47→    return row[0] or 0 if row else 0
    48→
    49→
    50→def run_migrations(engine) -> None:
    51→    """Apply all pending migrations."""
    52→    with engine.connect() as conn:
    53→        _ensure_version_table(conn)
    54→        current = _current_version(conn)
    55→
    56→        pending = [(v, d, fn) for v, d, fn in _migrations if v > current]
    57→        if not pending:
    58→            logger.debug(f"Database schema is up-to-date (version {current})")
    59→            return
    60→
    61→        for version, description, fn in sorted(pending):
    62→            logger.info(f"Applying migration v{version}: {description}")
    63→            fn(conn)
    64→            conn.execute(
    65→                text("INSERT INTO schema_version (version, description) VALUES (:v, :d)"),
    66→                {"v": version, "d": description},
    67→            )
    68→            conn.commit()
    69→
    70→        final = _current_version(conn)
    71→        logger.info(f"Migrations complete — schema version {final}")
