     1→"""Auth routes — login status, check, logout."""
     2→
     3→from fastapi import APIRouter, Request
     4→from fastapi.responses import HTMLResponse
     5→
     6→from src.core.browser import get_browser_manager
     7→from src.modules.auth import check_login_status
     8→
     9→router = APIRouter()
    10→
    11→
    12→@router.get("/status", response_class=HTMLResponse)
    13→async def auth_status(request: Request):
    14→    """Return login status as an HTML fragment (for HTMX)."""
    15→    try:
    16→        browser = get_browser_manager()
    17→        if not browser.is_running:
    18→            await browser.start()
    19→        page = browser.page
    20→        if page:
    21→            logged_in = await check_login_status(page)
    22→        else:
    23→            logged_in = False
    24→    except Exception:
    25→        logged_in = False
    26→
    27→    if logged_in:
    28→        return '<p><span class="status-dot online"></span>Boss直聘: 已登录</p>'
    29→    else:
    30→        return (
    31→            '<p><span class="status-dot offline"></span>Boss直聘: 未登录</p>'
    32→            '<p><small>请在 Chrome 浏览器中手动登录 Boss直聘</small></p>'
    33→            '<button hx-post="/auth/check" hx-swap="outerHTML">重新检测</button>'
    34→        )
    35→
    36→
    37→@router.post("/check", response_class=HTMLResponse)

    46→            logged_in = await check_login_status(page)
    47→        else:
    48→            logged_in = False
    49→    except Exception:
    50→        logged_in = False
    51→
    52→    if logged_in:
    53→        return '<p><span class="status-dot online"></span>Boss直聘: 已登录</p>'
    54→    else:
    55→        return (
    56→            '<p><span class="status-dot offline"></span>Boss直聘: 未登录</p>'

    58→            '<button hx-post="/auth/check" hx-swap="outerHTML">重新检测</button>'
    59→        )
    60→
    61→
    62→@router.post("/logout")
    63→async def auth_logout(request: Request):
    64→    """Clear saved session."""
    65→    from src.core.session import get_session_manager
    66→    session_mgr = get_session_manager()
    67→    if session_mgr.has_saved_session():
    68→        await session_mgr.clear()
    69→    return {"status": "ok", "message": "Session cleared"}
