   213→   213→   213→            if not search_frame:
   214→   214→   214→                for f in page.frames:
   215→   215→   215→                    if 'frame/search' in (f.url or ''):
   216→   216→   216→                        search_frame = f
   217→   217→   217→                        break
   218→   218→   218→
   219→   219→   219→            if search_frame:
   220→   220→   220→                profile_data = await search_frame.evaluate(EXTRACT_PROFILE_FROM_CARD_JS, expect_id)
   221→   221→   221→
   222→   222→   222→        # Strategy 3: Use database data as fallback

   241→   241→   241→        if not profile_data:

   268→   268→   268→

   288→   288→   288→        candidate.company = profile_data.get('company') or candidate.company
   289→   289→   289→        candidate.location = profile_data.get('location') or candidate.location
   290→   290→   290→        candidate.salary_expect = profile_data.get('salary_expect') or candidate.salary_expect
   291→   291→   291→        candidate.work_years = profile_data.get('work_years') or candidate.work_years
   292→   292→   292→        candidate.education_level = profile_data.get('education_level') or candidate.education_level
