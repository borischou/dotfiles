     8→     8→     8→

    10→    10→    10→from src.web.schemas import SearchParams

    14→    14→    14→
    15→    15→    15→router = APIRouter()

    17→    17→    17→
    18→    18→    18→@router.get("/", response_class=HTMLResponse)
    19→    19→    19→async def search_page(request: Request):

    30→    30→    30→    request: Request,
    31→    31→    31→    keywords: str = Form(...),
    32→    32→    32→    location: str = Form(""),
    33→    33→    33→    pages: int = Form(1),
    34→    34→    34→    experience: str = Form(""),
    35→    35→    35→    education: str = Form(""),
    36→    36→    36→    salary: str = Form(""),
    37→    37→    37→):

    70→    70→    70→        )
    71→    71→    71→
    72→    72→    72→        return templates.TemplateResponse("partials/search_results.html", {
    73→    73→    73→            "request": request,
    74→    74→    74→            "candidates": candidates,
    75→    75→    75→            "keywords": keywords,

    77→    77→    77→
    78→    78→    78→    except Exception as e:
