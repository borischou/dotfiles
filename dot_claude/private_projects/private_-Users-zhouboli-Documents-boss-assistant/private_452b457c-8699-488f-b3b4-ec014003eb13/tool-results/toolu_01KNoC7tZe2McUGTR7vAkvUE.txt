    45→    45→def render_template(template: Dict, variables: Dict[str, str]) -> str:
    46→    46→    """
    47→    47→    Render a message template with variable substitution.
    48→    48→
    49→    49→    Uses a single-pass approach to avoid recursive replacement issues
    50→    50→    (e.g., variable values containing {other_var} patterns).
    51→    51→
    52→    52→    Args:
    53→    53→        template: Template dict with 'content' and 'variables' keys
    54→    54→        variables: Variable name-value pairs
    55→    55→
    56→    56→    Returns:
    57→    57→        Rendered message string
    58→    58→    """
    59→    59→    import re
    60→    60→
    61→    61→    content = template['content']
    62→    62→
    63→    63→    def replacer(match):
    64→    64→        var_name = match.group(1)
    65→    65→        if var_name in variables:
    66→    66→            return str(variables[var_name])
    67→    67→        return match.group(0)  # Keep original if no matching variable
    68→    68→
    69→    69→    content = re.sub(r'\{(\w+)\}', replacer, content)
    70→    70→    return content

    73→    73→async def send_message(
    74→    74→    boss_id: str,
    75→    75→    template_name: str,

    86→    86→        template_name: Name of the message template
    87→    87→        variables: Template variable overrides

   100→   100→
   101→   101→    config = get_config()

   103→   103→    db_manager = get_db_manager()
   104→   104→
   105→   105→    # Load template
   106→   106→    template = config.get_template(template_name)
   107→   107→

   116→   116→                final_vars['name'] = candidate.name

   118→   118→                final_vars['company'] = candidate.company
   119→   119→            if 'position' not in final_vars and candidate.title:

   138→   138→        console.print("[dim]Preview only — message not sent.[/dim]")
   139→   139→        return False
   140→   140→
   141→   141→    # Confirm send — skip confirmation in non-interactive mode (Web)
   142→   142→    import sys
   143→   143→    if not interactive:
   144→   144→        confirm = True
   145→   145→    elif sys.stdin.isatty():
   146→   146→        confirm = questionary.confirm(
   147→   147→            "Send this message?",
   148→   148→            default=False,
   149→   149→        ).ask()
   150→   150→    else:
   151→   151→        confirm = False
   152→   152→

   156→   156→        return False
   157→   157→
   158→   158→    # Rate limit check
   159→   159→    await async_check_rate_limit('send_message')
   160→   160→
   161→   161→    try:
   162→   162→        # Navigate to chat with the candidate
   163→   163→        success = await _navigate_to_chat(page, boss_id, db_manager)

   172→   172→        _save_message_to_db(boss_id, template_name, message_content, 'sent', db_manager)
   173→   173→
   174→   174→        console.print(f"[green]Message sent to {boss_id}![/green]")
   175→   175→        return True
   176→   176→
   177→   177→    except PermanentFailureError as e:

   180→   180→        rate_limiter.report_error()

   182→   182→        return False
   183→   183→    except Exception as e:
   184→   184→        logger.error(f"Error sending message to {boss_id}: {e}")

   186→   186→        rate_limiter.report_error()
   187→   187→        console.print(f"[red]Error: {e}[/red]")

   191→   191→async def send_message_batch(
   192→   192→    candidates: list,
   193→   193→    template_name: str,
   194→   194→    variables: Optional[Dict[str, str]] = None,
   195→   195→    interactive: bool = True,
   196→   196→    skip_messaged: bool = True,
   197→   197→    dry_run: bool = False,
   198→   198→    browser: Optional[BrowserManager] = None,
   199→   199→) -> Dict[str, int]:
   200→   200→    """
   201→   201→    Send messages to multiple candidates.
   202→   202→
   203→   203→    Args:
   204→   204→        candidates: List of CandidateBasic or dicts with boss_id
   205→   205→        template_name: Template name
   206→   206→        variables: Shared template variables
   207→   207→        interactive: Confirm each message individually
   208→   208→        skip_messaged: Skip candidates already messaged
   209→   209→        dry_run: Simulate without actually sending
   210→   210→        browser: Browser manager instance
   211→   211→
   212→   212→    Returns:
   213→   213→        Dict with counts: sent, failed, skipped, cancelled
   214→   214→    """
   215→   215→    if browser is None:
   216→   216→        browser = get_browser_manager()
   217→   217→
   218→   218→    config = get_config()
   219→   219→    db_manager = get_db_manager()
   220→   220→    rate_limiter = get_rate_limiter()
   221→   221→
   222→   222→    stats = {'sent': 0, 'failed': 0, 'skipped': 0, 'cancelled': 0}
   223→   223→
   224→   224→    for i, candidate in enumerate(candidates):
   225→   225→        if isinstance(candidate, dict):
   226→   226→            boss_id = candidate.get('boss_id', '')
   227→   227→            name = candidate.get('name', '')
   228→   228→            title = candidate.get('title', '')
   229→   229→            company = candidate.get('company', '')
   230→   230→        else:
   231→   231→            boss_id = getattr(candidate, 'boss_id', '')
   232→   232→            name = getattr(candidate, 'name', '')
   233→   233→            title = getattr(candidate, 'title', '')
   234→   234→            company = getattr(candidate, 'company', '')
   235→   235→
   236→   236→        if not boss_id:
   237→   237→            stats['skipped'] += 1
   238→   238→            continue
   239→   239→
   240→   240→        console.print(f"\n[cyan]({i+1}/{len(candidates)})[/cyan] {name} - {title} @ {company}")
   241→   241→
   242→   242→        # Check if already messaged
   243→   243→        if skip_messaged:
   244→   244→            with db_manager.session_scope() as session:
   245→   245→                existing_msg = session.query(Message).filter_by(
   246→   246→                    candidate_boss_id=boss_id, status='sent'
   247→   247→                ).first()
   248→   248→                if existing_msg:
   249→   249→                    console.print(f"  [dim]Already messaged, skipping[/dim]")
   250→   250→                    stats['skipped'] += 1
   251→   251→                    continue
   252→   252→
   253→   253→        if dry_run:
   254→   254→            console.print(f"  [dim]Dry run — would send '{template_name}' template[/dim]")
   255→   255→            stats['sent'] += 1
   256→   256→            continue
   257→   257→
   258→   258→        # Build per-candidate variables
   259→   259→        per_vars = dict(variables or {})
   260→   260→        if name:
   261→   261→            per_vars.setdefault('name', name)
   262→   262→        if company:
   263→   263→            per_vars.setdefault('company', company)
   264→   264→        if title:
   265→   265→            per_vars.setdefault('position', title)
   266→   266→
   267→   267→        try:
   268→   268→            if interactive:
   269→   269→                result = await send_message(
   270→   270→                    boss_id=boss_id,
   271→   271→                    template_name=template_name,
   272→   272→                    variables=per_vars,
   273→   273→                    browser=browser,
   274→   274→                )
   275→   275→                if result:
   276→   276→                    stats['sent'] += 1
   277→   277→                else:
   278→   278→                    stats['cancelled'] += 1
   279→   279→            else:
   280→   280→                # Non-interactive: send without confirmation
   281→   281→                await async_check_rate_limit('send_message')
   282→   282→
   283→   283→                template = config.get_template(template_name)
   284→   284→                validate_template_variables(template, per_vars)
   285→   285→                message_content = render_template(template, per_vars)
   286→   286→
   287→   287→                page = browser.page
   288→   288→                success = await _navigate_to_chat(page, boss_id, db_manager)
   289→   289→                if success:
   290→   290→                    await _type_and_send_message(page, message_content)
   291→   291→                    rate_limiter.log_action('send_message', {'boss_id': boss_id})
   292→   292→                    _save_message_to_db(boss_id, template_name, message_content, 'sent', db_manager)
   293→   293→                    stats['sent'] += 1
   294→   294→                    console.print(f"  [green]Sent![/green]")
   295→   295→                else:
   296→   296→                    _save_message_to_db(boss_id, template_name, message_content, 'failed', db_manager)
   297→   297→                    stats['failed'] += 1
   298→   298→                    console.print(f"  [red]Failed to open chat[/red]")
   299→   299→
   300→   300→        except Exception as e:
   301→   301→            logger.error(f"Error sending to {boss_id}: {e}")
   302→   302→            stats['failed'] += 1
   303→   303→            console.print(f"  [red]Error: {e}[/red]")
   304→   304→
   305→   305→        # Delay between messages
   306→   306→        if i < len(candidates) - 1:
   307→   307→            await async_random_delay('message')
   308→   308→
   309→   309→    # Summary
   310→   310→    console.print(Panel(
   311→   311→        f"[green]Sent: {stats['sent']}[/green] | "
   312→   312→        f"[red]Failed: {stats['failed']}[/red] | "
   313→   313→        f"[yellow]Skipped: {stats['skipped']}[/yellow] | "
   314→   314→        f"[dim]Cancelled: {stats['cancelled']}[/dim]",
   315→   315→        title="Batch Message Summary",
   316→   316→    ))
   317→   317→
   318→   318→    return stats

   321→   321→async def _navigate_to_chat(page: Page, boss_id: str, db_manager) -> bool:
   322→   322→    """
   323→   323→    Navigate to the chat window for a candidate.
   324→   324→
   325→   325→    Strategy 1: Go to /web/chat/index and find conversation by name
   326→   326→    Strategy 2: Use search page contact button (requires 搜索畅聊卡)
   327→   327→
   328→   328→    Args:
   329→   329→        page: Playwright page instance
   330→   330→        boss_id: Candidate Boss ID (e.g., "zp_81137184")
   331→   331→        db_manager: Database manager
   332→   332→
   333→   333→    Returns:
   334→   334→        True if chat window is ready for input
   335→   335→    """
   336→   336→    # Get candidate info from database
   337→   337→    candidate_name = None
   338→   338→    with db_manager.session_scope() as session:
   339→   339→        candidate = session.query(Candidate).filter_by(boss_id=boss_id).first()
   340→   340→        if candidate:
   341→   341→            candidate_name = candidate.name
   342→   342→
   343→   343→    # Strategy 1: Navigate to chat page and find existing conversation
   344→   344→    try:
   345→   345→        await retry_navigation(page, 'https://www.zhipin.com/web/chat/index',
   346→   346→                               wait_until='domcontentloaded')
   347→   347→        await async_random_delay('page_load')
   348→   348→        await check_captcha(page, report_to_rate_limiter=True)
   349→   349→
   350→   350→        # Wait for conversation list to load
   351→   351→        try:
   352→   352→            await page.wait_for_selector('.geek-item', timeout=10000)
   353→   353→        except Exception:
   354→   354→            logger.warning("Conversation list did not load")
   355→   355→
   356→   356→        # Try to find the candidate in the conversation list
   357→   357→        # Match by data-id (which is the user ID, different from data-expect)
   358→   358→        # or by name text match
   359→   359→        if candidate_name:
   360→   360→            found = await page.evaluate("""
   361→   361→                (name) => {
   362→   362→                    const items = document.querySelectorAll('.geek-item');
   363→   363→                    for (const item of items) {
   364→   364→                        if (item.innerText.includes(name)) {
   365→   365→                            item.click();
   366→   366→                            return true;
   367→   367→                        }
   368→   368→                    }
   369→   369→                    return false;
   370→   370→                }
   371→   371→            """, candidate_name)
   372→   372→
   373→   373→            if found:
   374→   374→                await page.wait_for_timeout(2000)
   375→   375→                # Verify chat input is visible
   376→   376→                chat_input = await page.query_selector('#boss-chat-editor-input')
   377→   377→                if chat_input and await chat_input.is_visible():
   378→   378→                    return True
   379→   379→                logger.warning("Chat input not visible after clicking conversation")
   380→   380→
   381→   381→    except Exception as e:
   382→   382→        logger.warning(f"Strategy 1 (chat page) failed for {boss_id}: {e}")
   383→   383→
   384→   384→    # Strategy 2: Navigate to search page and use contact button
   385→   385→    # This requires 搜索畅聊卡 and may trigger purchase dialog
   386→   386→    expect_id = boss_id.replace('zp_', '') if boss_id.startswith('zp_') else boss_id
   387→   387→    try:
   388→   388→        await retry_navigation(page, 'https://www.zhipin.com/web/chat/search',
   389→   389→                               wait_until='domcontentloaded')
   390→   390→        await async_random_delay('page_load')
   391→   391→
   392→   392→        # Get search iframe
   393→   393→        search_frame = page.frame(name='searchFrame')
   394→   394→        if not search_frame:
   395→   395→            for f in page.frames:
   396→   396→                if 'frame/search' in (f.url or ''):
   397→   397→                    search_frame = f
   398→   398→                    break
   399→   399→
   400→   400→        if search_frame:
   401→   401→            # Find the candidate card by data-expect and click contact
   402→   402→            clicked = await search_frame.evaluate("""
   403→   403→                (expectId) => {
   404→   404→                    const link = document.querySelector('a[data-expect="' + expectId + '"]');
   405→   405→                    if (!link) return false;
   406→   406→                    const card = link.closest('.geek-info-card');
   407→   407→                    if (!card) return false;
   408→   408→                    const btn = card.querySelector('.btn-getcontact');
   409→   409→                    if (btn) {
   410→   410→                        btn.click();
   411→   411→                        return true;
   412→   412→                    }
   413→   413→                    return false;
   414→   414→                }
   415→   415→            """, expect_id)
   416→   416→
   417→   417→            if clicked:
   418→   418→                await page.wait_for_timeout(3000)
   419→   419→                # Check if chat opened (not a purchase dialog)
   420→   420→                chat_input = await page.query_selector('#boss-chat-editor-input')
   421→   421→                if chat_input and await chat_input.is_visible():
   422→   422→                    return True
   423→   423→
   424→   424→                # Close any purchase dialogs
   425→   425→                await _close_purchase_dialogs(page)
   426→   426→
   427→   427→    except Exception as e:
   428→   428→        logger.warning(f"Strategy 2 (search contact) failed for {boss_id}: {e}")
   429→   429→
   430→   430→    logger.error(f"All chat navigation strategies failed for {boss_id}")
   431→   431→    return False

   451→   451→async def _type_and_send_message(page: Page, message: str) -> None:
   452→   452→    """
   453→   453→    Type a message into the chat input and send it.
   454→   454→
   455→   455→    Chat input: #boss-chat-editor-input (contenteditable div)
   456→   456→    Send button: div.submit (text "发送")
   457→   457→
   458→   458→    Args:
   459→   459→        page: Playwright page instance
   460→   460→        message: Message text to send
   461→   461→    """
   462→   462→    # Find the message input
   463→   463→    # Verified: #boss-chat-editor-input is the chat input (contenteditable div)
   464→   464→    input_selectors = [
   465→   465→        '#boss-chat-editor-input',
   466→   466→        '[contenteditable="true"]',
   467→   467→    ]
   468→   468→
   469→   469→    input_element = None
   470→   470→    for selector in input_selectors:
   471→   471→        try:
   472→   472→            element = await page.query_selector(selector)
   473→   473→            if element and await element.is_visible():
   474→   474→                input_element = element
   475→   475→                break
   476→   476→        except Exception:
   477→   477→            continue
   478→   478→
   479→   479→    if not input_element:
   480→   480→        raise SelectorNotFoundError("Chat message input not found")
   481→   481→
   482→   482→    # Click to focus
   483→   483→    await input_element.click()
   484→   484→    await page.wait_for_timeout(300)
   485→   485→
   486→   486→    # Clear any existing content
   487→   487→    import sys
   488→   488→    select_all_key = "Meta+a" if sys.platform == "darwin" else "Control+a"
