    53→def _is_non_rd_title(title: str) -> bool:
    54→    """Check if a title indicates a non-R&D role.
    55→
    56→    Strict mode: if any non-R&D keyword appears, filter out regardless of
    57→    whether R&D keywords are also present (e.g. "技术产品经理" is filtered).
    58→    Returns False (keep) only if title is empty or contains no non-R&D keywords.
    59→    """
    60→    if not title:
    61→        return False
    62→
    63→    # Strict: any non-R&D keyword present → filter out
    64→    if any(kw in title for kw in NON_RD_KEYWORDS):
    65→        return True
    66→
    67→    return False

    73→    maimai_id: str
    74→    name: str

    82→async def search_candidates(
    83→    keywords: str,
    84→    company: Optional[tuple] = None,
    85→    location: Optional[str] = None,
    86→    pages: int = 1,
    87→    start_page: int = 1,
    88→    browser: Optional[BrowserManager] = None
    89→) -> List[CandidateBasic]:
    90→    """
    91→    Search for candidates on Maimai.
    92→
    93→    Args:
    94→        keywords: Search keywords (e.g., "高级前端工程师")
    95→        company: Filter by company names, tuple of strings (optional)
    96→        location: Filter by location (optional)
    97→        pages: Number of pages to scrape (default 1)
    98→        start_page: First page number to scrape (default 1)
    99→        browser: Browser manager instance
   100→
   101→    Returns:
   102→        List of CandidateBasic objects
   103→
   104→    Raises:
   105→        RateLimitExceededError: If rate limit exceeded
   106→        SelectorNotFoundError: If page elements not found
   107→    """
   108→    # Validate inputs
   109→    validate_keywords(keywords)
   110→
   111→    if browser is None:
   112→        browser = get_browser_manager()
   113→
   114→    config = get_config()
   115→    rate_limiter = get_rate_limiter()
   116→    db_manager = get_db_manager()
   117→
   118→    # Ensure logged in
   119→    await ensure_logged_in(browser)
   120→
   121→    all_candidates: List[CandidateBasic] = []
   122→    seen_ids: set = set()  # Deduplicate across infinite-scroll extractions
   123→
   124→    with Progress(
   125→        SpinnerColumn(),
   126→        TextColumn("[progress.description]{task.description}"),
   127→        console=console
   128→    ) as progress:
   129→
   130→        end_page = start_page + pages
   131→        for page_num in range(start_page, end_page):
   132→            task = progress.add_task(f"Searching page {page_num}/{end_page - 1}...", total=None)
   133→
   134→            try:
   135→                # Check rate limit (waits through min-interval)
   136→                await async_check_rate_limit('search')
   137→
   138→                page = browser.page
   139→
   140→                if page_num == start_page:
   141→                    # First page: interactive search (maimai SPA ignores URL query params)
   142→                    await retry_navigation(page, MAIMAI_SEARCH_URL, wait_until='domcontentloaded')
   143→                    await asyncio.sleep(1)
   144→
   145→                    # Find search input and type keyword
   146→                    search_input = (
   147→                        await page.query_selector('input[placeholder="搜索"]')
   148→                        or await page.query_selector('input[type="search"]')
   149→                        or await page.query_selector('input[type="text"]')
   150→                    )
   151→                    if not search_input:
   152→                        raise SelectorNotFoundError("Search input not found on search center page")
   153→
   154→                    await search_input.click()
   155→                    await search_input.fill(keywords)
   156→                    await asyncio.sleep(0.5)
   157→                    await page.keyboard.press('Enter')
   158→                    await asyncio.sleep(2)
   159→
   160→                    # Default tab is 实名动态(feed); switch to 人脉(contacts)
   161→                    renmai_tab = await page.query_selector(':text-is("人脉")')
   162→                    if renmai_tab:
   163→                        await renmai_tab.click()
   164→                        await asyncio.sleep(2)
   165→                    else:
   166→                        logger.warning("Could not find '人脉' tab on search results page")
   167→                else:
   168→                    # Subsequent pages: scroll to load more (infinite scroll)
   169→                    prev_count = await page.evaluate(
   170→                        'document.querySelectorAll(\'[id^="contactcard_"]\').length'
   171→                    )
   172→                    await human_scroll(page, 'down', distance=1500)
   173→                    # Wait for new cards to load
   174→                    try:
   175→                        await page.wait_for_function(
   176→                            f'''() => {{
   177→                                const cards = document.querySelectorAll('[id^="contactcard_"]');
   178→                                return cards.length > {prev_count};
   179→                            }}''',
   180→                            timeout=8000,
   181→                        )
   182→                    except Exception:
   183→                        logger.debug(f"No new cards after scroll (had {prev_count})")
   184→
   185→                # Wait for contactcard elements
   186→                try:
   187→                    await page.wait_for_function(
   188→                        '''() => {
   189→                            const cards = document.querySelectorAll('[id^="contactcard_"]');
   190→                            return cards.length >= 1;
   191→                        }''',
   192→                        timeout=8000,
   193→                    )
   194→                except TimeoutError:
   195→                    logger.debug("Timed out waiting for contactcard elements, proceeding anyway")
   196→                except Exception as e:
   197→                    logger.warning(f"Error waiting for contactcard elements: {e}")
   198→
   199→                # Extract candidates from this page
   200→                candidates = await _extract_search_results(page, config)
   201→
   202→                # Deduplicate: infinite scroll re-extracts all visible cards
   203→                candidates = [c for c in candidates if c.maimai_id not in seen_ids]
   204→                for c in candidates:
   205→                    seen_ids.add(c.maimai_id)
   206→
   207→                # Anti-detection: human behavior AFTER extraction
   208→                await check_captcha(page, report_to_rate_limiter=True)
   209→                await human_scroll(page, 'down', distance=300)
   210→                await async_random_delay('action')
   211→
   212→                if not candidates:
   213→                    logger.info(f"No results found on page {page_num}")
   214→                    await browser.screenshot(f"search_debug_page{page_num}")
   215→                    progress.update(task, description=f"Page {page_num}: No results")
   216→                    # Allow one empty page (could be transient), break on second consecutive
   217→                    if page_num > start_page and not all_candidates:
   218→                        # First page had no results either — genuinely empty
   219→                        break
   220→                    continue
   221→
   222→                # Filter out non-R&D roles
   223→                before_rd_filter = len(candidates)
   224→                candidates = [c for c in candidates if not _is_non_rd_title(c.title)]
   225→                rd_filtered = before_rd_filter - len(candidates)
   226→                if rd_filtered > 0:
   227→                    logger.info(f"Filtered out {rd_filtered} non-R&D candidates")
   228→
   229→                # Apply filters if specified
   230→                if company:
   231→                    company_lower = [cn.lower() for cn in company]
   232→                    candidates = [c for c in candidates if any(cn in c.company.lower() for cn in company_lower)]
   233→
   234→                if location:
   235→                    candidates = [c for c in candidates if c.location and location.lower() in c.location.lower()]
   236→
   237→                all_candidates.extend(candidates)
   238→                progress.update(task, description=f"Page {page_num}: Found {len(candidates)} candidates")
   239→
   240→                # Log the action
   241→                rate_limiter.log_action('search', {
   242→                    'keywords': keywords,
   243→                    'page': page_num,
   244→                    'results': len(candidates)
   245→                })
   246→
   247→                # Save candidates to database
   248→                try:
   249→                    _save_candidates_to_db(db_manager, candidates, keywords)
   250→                except Exception as e:
   251→                    console.print(f"[yellow]Warning: Failed to save candidates to DB: {e}[/yellow]")
   252→
   253→                # Delay before next page
   254→                if page_num < end_page - 1:
   255→                    await async_random_delay('page_load')
   256→
   257→            except (BrowserError, SessionExpiredError):
   258→                raise  # Critical errors must propagate
   259→            except Exception as e:
   260→                logger.error(f"Error on page {page_num}: {e}")
   261→                progress.update(task, description=f"Page {page_num}: Error - {str(e)[:30]}")
   262→                # Take screenshot for debugging
   263→                await browser.screenshot(f"search_error_page{page_num}")
   264→                continue
   265→
   266→    # Log search to history
   267→    _log_search_history(db_manager, keywords, company, location, pages, len(all_candidates))
   268→
   269→    console.print(f"[green]Search complete. Found {len(all_candidates)} candidates.[/green]")
   270→    return all_candidates

   272→
   273→async def _extract_search_results(page: Page, config) -> List[CandidateBasic]:
   274→    """
   275→    Extract candidate information from search results page.
   276→
   277→    Args:
   278→        page: Playwright page instance
   279→        config: Config instance
   280→
   281→    Returns:
   282→        List of CandidateBasic objects
   283→    """
   284→    candidates = []
   285→
   286→    # NOTE: The caller (search_candidates) already waits for contactcard_
   287→    # elements before calling this function.  Maimai's SPA strips URL
   288→    # params within ~5-8s and clears results, so we must NOT add long
   289→    # waits here (no networkidle, no loading-text polling).
   290→
   291→    try:
   292→        # Strategy: Extract candidates by finding avatar images and their associated text
   293→        candidates_data = await page.evaluate('''() => {
   294→            const candidates = [];
   295→            const seenNames = new Set();
   296→
   297→            // Find all images that look like avatars
   298→            const images = document.querySelectorAll('img');
   299→
   300→            for (const img of images) {
   301→                // Skip large images (probably not avatars)
   302→                const rect = img.getBoundingClientRect();
   303→                if (rect.width > 100 || rect.height > 100) continue;
   304→                if (rect.width < 20 || rect.height < 20) continue;
   305→
   306→                // Find parent container (the card)
   307→                let card = img.parentElement;
   308→                for (let i = 0; i < 8 && card; i++) {
   309→                    const cardRect = card.getBoundingClientRect();
   310→                    // Look for a reasonable card size
   311→                    if (cardRect.width > 200 && cardRect.height > 50 && cardRect.height < 200) {
   312→                        break;
   313→                    }
   314→                    card = card.parentElement;
   315→                }
   316→
   317→                if (!card) continue;
   318→
   319→                // Get all text nodes from card, preserving order
   320→                const textParts = [];
   321→                const walker = document.createTreeWalker(card, NodeFilter.SHOW_TEXT, null, false);
   322→                while (walker.nextNode()) {
   323→                    const text = walker.currentNode.textContent.trim();
   324→                    if (text && text.length > 0 && text.length < 100) {
   325→                        textParts.push(text);
   326→                    }
   327→                }
   328→
   329→                // Parse candidate info from text parts
   330→                // Pattern: Name, Activity time, Company+Position, Industry|Department
   331→                let name = '';
   332→                let company = '';
   333→                let title = '';
   334→                let industry = '';
   335→
   336→                // Compound job title patterns (longer patterns first for greedy match)
   337→                const jobPatterns = [
   338→                    // Compound titles with prefix modifiers
   339→                    /(?:高级|资深|首席|初级|中级|senior|junior|staff|principal|lead)?\s*(?:前端|后端|全栈|web|Web|iOS|Android|Java|Python|Go|C\+\+|PHP|.NET|算法|数据|AI|机器学习|深度学习|NLP|大数据|云计算|DevOps|SRE|安全|测试|嵌入式|移动端|客户端|服务端|区块链|具身智能|自动驾驶|机器人)?\s*(?:工程师|开发|架构师|专家|技术经理|技术总监|技术负责人|Tech\s*Lead)/i,
   340→                    // Management titles
   341→                    /(?:技术|产品|运营|市场|销售|人力|财务|行政|项目|研发)?\s*(?:总监|经理|主管|负责人|VP|总裁|副总|合伙人|创始人|董事|CEO|CTO|COO|CFO|CMO|CHO|CIO)/i,
   342→                    // Other roles
   343→                    /(?:产品|UI|UX|交互|视觉)?\s*(?:设计师|分析师|顾问|运营|助理|实习生?)/i,
   344→                ];
   345→
   346→                // Patterns for text that should be skipped
   347→                const skipPatterns = [
   348→                    /^[+＋]\s*好友$/,          // "+ 好友" button
   349→                    /^\d+个共同好友$/,           // "1个共同好友" social label
   350→                    /^共同好友/,                 // "共同好友..." variants
   351→                    /^已关注$/,                  // "已关注" button
   352→                    /^添加好友$/,                // "添加好友" button
   353→                    /^关注$/,                    // "关注" button
   354→                    /^好友$/,                    // "好友" label
   355→                    /^[|｜·]$/,                  // separators
   356→                ];
   357→
   358→                const isSkippable = (text) => {
   359→                    if (skipPatterns.some(p => p.test(text))) return true;
   360→                    // Activity time
   361→                    if (/天前|小时前|分钟前|刚刚|活跃|在线/.test(text)) return true;
   362→                    // Social relationship labels
   363→                    if (/同行|前同事|校友|共同好友/.test(text)) return true;
   364→                    return false;
   365→                };
   366→
   367→                for (let i = 0; i < textParts.length; i++) {
   368→                    const part = textParts[i];
   369→
   370→                    if (isSkippable(part)) continue;
   371→
   372→                    // First short text (2-15 chars, pure CJK/letters) is likely the name
   373→                    if (!name && part.length >= 2 && part.length <= 15) {
   374→                        if (/^[\u4e00-\u9fa5a-zA-Z·\s()（）]+$/.test(part)) {
   375→                            name = part.replace(/[（(][^)）]*[）)]/g, '').trim();
   376→                            continue;
   377→                        }
   378→                    }
   379→
   380→                    // After name, next substantial text is company+position
   381→                    if (name && !company && part.length > 2) {
   382→                        // Try to split company and position using regex patterns
   383→                        let splitDone = false;
   384→                        for (const pattern of jobPatterns) {
   385→                            const match = part.match(pattern);
   386→                            if (match && match.index > 0) {
   387→                                company = part.substring(0, match.index).trim();
   388→                                title = match[0].trim();
   389→                                splitDone = true;
   390→                                break;
   391→                            }
   392→                        }
   393→                        if (!splitDone) {
   394→                            company = part;
   395→                        }
   396→                        continue;
   397→                    }
   398→
   399→                    // Next part is industry|department info
   400→                    if (name && company && !industry && part.length > 2) {
   401→                        industry = part;
   402→                        // If title wasn't split from company, use industry as fallback context
   403→                        if (!title) {
   404→                            title = industry;
   405→                        }
   406→                        break;
   407→                    }
   408→                }
   409→
   410→                // Skip if no name found or premium placeholder
   411→                if (!name) continue;
   412→                if (name === '脉脉高价值人脉') continue;
   413→
   414→                // Extract user ID from DOM patterns on card and ancestor elements.
   415→                // Two patterns exist:
   416→                //   1. contactcard_{numeric_id} — on ancestors (colleagues layout)
   417→                //   2. member_entry_{numeric_id} — on descendants (standard layout)
   418→                let userId = '';
   419→
   420→                // Check ancestors for numeric contactcard_{id}
   421→                let el = card;
   422→                while (el) {
   423→                    const elId = el.id || '';
   424→                    const ccMatch = elId.match(/^contactcard_(\d+)$/);
   425→                    if (ccMatch) {
   426→                        userId = ccMatch[1];
   427→                        break;
   428→                    }
   429→                    el = el.parentElement;
   430→                }
   431→
   432→                // Check descendants for member_entry_{id} (most common pattern)
   433→                if (!userId) {
   434→                    const meEl = card.querySelector('[id^="member_entry_"]');
   435→                    if (meEl) {
   436→                        const meMatch = meEl.id.match(/^member_entry_(\d+)$/);
   437→                        if (meMatch) {
   438→                            userId = meMatch[1];
   439→                        }
   440→                    }
   441→                }
   442→
   443→                // Also check ancestor descendants for contactcard with numeric ID
   444→                if (!userId) {
   445→                    const ccEl = card.querySelector('[id^="contactcard_"]');
   446→                    if (ccEl) {
   447→                        const ccMatch = ccEl.id.match(/^contactcard_(\d+)$/);
   448→                        if (ccMatch) {
   449→                            userId = ccMatch[1];
   450→                        }
   451→                    }
   452→                }
   453→
   454→                // Use real user_id as maimai_id when available, fall back to hash
   455→                let maimaiId;
   456→                if (userId) {
   457→                    maimaiId = 'mm_' + userId;
   458→                } else {
   459→                    const idSource = (name || '') + '|' + (company || '') + '|' + (title || '');
   460→                    let hash = 0;
   461→                    for (let j = 0; j < idSource.length; j++) {
   462→                        const ch = idSource.charCodeAt(j);
   463→                        hash = ((hash << 5) - hash) + ch;
   464→                        hash = hash & hash; // Convert to 32bit integer
   465→                    }
   466→                    maimaiId = 'mm_' + Math.abs(hash).toString(16).padStart(8, '0');
   467→                }
   468→
   469→                // Skip duplicates based on the computed ID
   470→                if (seenNames.has(maimaiId)) continue;
   471→                seenNames.add(maimaiId);
   472→
   473→                // Build profile URL from user ID when available
   474→                let profileUrl = '';
   475→                if (userId) {
   476→                    profileUrl = 'https://maimai.cn/contact/detail/' + userId + '?dstu=' + userId;
   477→                }
   478→
   479→                // Fallback: extract profile URL from <a> tags inside the card
   480→                if (!profileUrl) {
   481→                    let fallbackUrl = '';
   482→
   483→                    // First: check <a> ancestors between img and card
   484→                    let ancestor = img.parentElement;
   485→                    while (ancestor && ancestor !== card) {
   486→                        if (ancestor.tagName === 'A' && ancestor.href) {
   487→                            const href = ancestor.getAttribute('href') || '';
   488→                            if (href.includes('/profile/detail') || href.includes('/contact/detail/') || href.includes('/u/') || href.includes('egid=') || href.includes('dstu=')) {
   489→                                profileUrl = href.startsWith('http') ? href : 'https://maimai.cn' + href;
   490→                                break;
   491→                            }
   492→                        }
   493→                        ancestor = ancestor.parentElement;
   494→                    }
   495→
   496→                    // Second: search all <a> descendants inside the card
   497→                    if (!profileUrl) {
   498→                        const links = card.querySelectorAll('a[href]');
   499→                        for (const link of links) {
   500→                            const href = link.getAttribute('href') || '';
   501→                            if (href.includes('/profile/detail') || href.includes('/contact/detail/') || href.includes('/u/') || href.includes('egid=') || href.includes('dstu=')) {
   502→                                profileUrl = href.startsWith('http') ? href : 'https://maimai.cn' + href;
   503→                                break;
   504→                            }
   505→                            if (!fallbackUrl && href && !href.startsWith('javascript:') && (href.startsWith('/') || href.includes('maimai.cn'))) {
   506→                                fallbackUrl = href.startsWith('http') ? href : 'https://maimai.cn' + href;
   507→                            }
   508→                        }
   509→                    }
   510→                    if (!profileUrl && fallbackUrl) {
   511→                        profileUrl = fallbackUrl;
   512→                    }
   513→                }
   514→
   515→                // Extract location: look for remaining short text parts after name/company/title
   516→                // Common patterns: city names like "北京", "上海", "深圳" etc.
   517→                let location = '';
   518→                const locationPatterns = /^(北京|上海|广州|深圳|杭州|成都|南京|武汉|西安|苏州|长沙|天津|重庆|郑州|东莞|青岛|大连|宁波|厦门|合肥|佛山|福州|昆明|济南|珠海|无锡|常州|海外|远程|香港|台北|新加坡|美国|日本|韩国|英国|全国|其他)/;
   519→                for (let i = 0; i < textParts.length; i++) {
   520→                    const part = textParts[i];
   521→                    if (part === name || part === company || part === title || part === industry) continue;
   522→                    if (isSkippable(part)) continue;
   523→                    if (locationPatterns.test(part) && part.length <= 10) {
   524→                        location = part;
   525→                        break;
   526→                    }
   527→                }
   528→
   529→                candidates.push({
   530→                    name: name,
   531→                    title: title || '',
   532→                    company: company || '',
   533→                    maimai_id: maimaiId,
   534→                    profile_url: profileUrl,
   535→                    location: location,
   536→                });
   537→            }
   538→
   539→            return candidates;
   540→        }''')
   541→
   542→        logger.info(f"JavaScript extraction found {len(candidates_data)} candidates")
   543→
   544→        for data in candidates_data:
   545→            if data.get('name'):
   546→                candidates.append(CandidateBasic(
   547→                    maimai_id=data.get('maimai_id', ''),
   548→                    name=data.get('name', ''),
   549→                    title=data.get('title', ''),
   550→                    company=data.get('company', ''),
   551→                    location=data.get('location', ''),
   552→                    profile_url=data.get('profile_url', '')
   553→                ))
   554→
   555→        # For candidates missing profile_url, try extracting from contactcard DOM
   556→        missing_url = [c for c in candidates if not c.profile_url]
   557→        if missing_url:
   558→            logger.info(f"{len(missing_url)} candidates missing profile_url, searching contactcard DOM")
   559→            for c in missing_url:
   560→                try:
   561→                    profile_url = await _click_avatar_for_profile_url(page, c.name, c.company)
   562→                    if profile_url:
   563→                        c.profile_url = profile_url
   564→                except Exception as e:
   565→                    logger.debug(f"Contactcard lookup failed for '{c.name}': {e}")
   566→
   567→        # Fallback: Strategy 1 - Find all profile links directly
   568→        if not candidates:
   569→            profile_links = await page.query_selector_all('a[href*="/contact/"], a[href*="/u/"], a[href*="egid="], a[href*="profile"]')
   570→        else:
   571→            profile_links = []
   572→
   573→        if profile_links and not candidates:
   574→            logger.debug(f"Found {len(profile_links)} profile links")
   575→
   576→            seen_ids = set()
   577→            for link in profile_links:
   578→                try:
   579→                    href = await link.get_attribute('href')
   580→                    if not href:
   581→                        continue
   582→
   583→                    # Extract maimai_id from URL
   584→                    maimai_id = None
   585→                    if '/contact/detail/' in href:
   586→                        maimai_id = href.split('/contact/detail/')[-1].split('?')[0].split('/')[0]
   587→                    elif '/u/' in href:
   588→                        maimai_id = href.split('/u/')[-1].split('?')[0].split('/')[0]
   589→
   590→                    if not maimai_id or maimai_id in seen_ids:
   591→                        continue
   592→                    seen_ids.add(maimai_id)
   593→
   594→                    # Get name from link text
   595→                    name = await link.inner_text()
   596→                    name = name.strip()
   597→                    if not name or len(name) > 50:  # Skip if no name or too long (probably not a name)
   598→                        continue
   599→
   600→                    # Try to get parent container for more info
   601→                    parent = await link.evaluate_handle('el => el.closest("div, li, article") || el.parentElement.parentElement')
   602→
   603→                    title = ""
   604→                    company = ""
   605→
   606→                    if parent:
   607→                        # Get all text content and try to parse
   608→                        try:
   609→                            full_text = await parent.inner_text()
   610→                            lines = [l.strip() for l in full_text.split('\n') if l.strip()]
   611→
   612→                            # Usually: name, then title/company info
   613→                            for line in lines:
   614→                                if line == name:
   615→                                    continue
   616→                                if '好友' in line or '关注' in line:
   617→                                    continue
   618→                                if not title and len(line) < 50:
   619→                                    title = line
   620→                                elif not company and len(line) < 50:
   621→                                    company = line
   622→                                    break
   623→                        except Exception:
   624→                            pass  # Best-effort text parsing; proceed with partial data
   625→
   626→                    profile_url = href if href.startswith('http') else f"https://maimai.cn{href}"
   627→
   628→                    candidates.append(CandidateBasic(
   629→                        maimai_id=maimai_id,
   630→                        name=name,
   631→                        title=title,
   632→                        company=company,
   633→                        profile_url=profile_url
   634→                    ))
   635→
   636→                except Exception as e:
   637→                    logger.debug(f"Failed to extract from link: {e}")
   638→                    continue
   639→
   640→        # Strategy 2: If no links found, try button-based detection
   641→        if not candidates:
   642→            logger.debug("Trying button-based detection")
   643→            buttons = await page.query_selector_all('button:has-text("好友"), button:has-text("关注")')
   644→
   645→            for button in buttons:
   646→                try:
   647→                    parent = await button.evaluate_handle('el => el.closest("div, li, article")')
   648→                    if not parent:
   649→                        continue
   650→
   651→                    # Find link in parent
   652→                    link = await parent.query_selector('a[href*="/contact"], a[href*="/u/"]')
   653→                    if link:
   654→                        href = await link.get_attribute('href')
   655→                        name = await link.inner_text()
   656→
   657→                        maimai_id = None
   658→                        if '/contact/detail/' in href:
   659→                            maimai_id = href.split('/contact/detail/')[-1].split('?')[0]
   660→                        elif '/u/' in href:
   661→                            maimai_id = href.split('/u/')[-1].split('?')[0]
   662→
   663→                        if maimai_id and name:
   664→                            candidates.append(CandidateBasic(
   665→                                maimai_id=maimai_id,
   666→                                name=name.strip(),
   667→                                title="",
   668→                                company="",
   669→                                profile_url=href
   670→                            ))
   671→                except Exception:
   672→                    continue
   673→
   674→        # Filter out premium placeholder profiles
   675→        before_filter = len(candidates)
   676→        candidates = [c for c in candidates if c.name not in PREMIUM_PLACEHOLDER_NAMES]
   677→        filtered_count = before_filter - len(candidates)
   678→        if filtered_count > 0:
   679→            logger.info(f"Filtered out {filtered_count} premium placeholder profiles")
   680→
   681→        logger.info(f"Extracted {len(candidates)} candidates from page")
   682→
   683→    except Exception as e:
   684→        logger.error(f"Error extracting search results: {e}")
   685→        console.print(f"[yellow]Warning: Search result extraction failed: {e}[/yellow]")
   686→
   687→    return candidates

   690→def _save_candidates_to_db(db_manager, candidates: List[CandidateBasic], source_search: str) -> None:
   691→    """
   692→    Save or update candidates in database.
   693→
   694→    Args:
   695→        db_manager: Database manager instance
   696→        candidates: List of candidates to save
   697→        source_search: Search keywords that found these candidates
   698→    """
   699→    try:
   700→        with db_manager.session_scope() as session:
   701→            # Batch pre-fetch existing candidates to avoid N+1 queries
   702→            maimai_ids = [c.maimai_id for c in candidates]
   703→            existing_by_id = {
   704→                row.maimai_id: row
   705→                for row in session.query(Candidate).filter(
   706→                    Candidate.maimai_id.in_(maimai_ids)
   707→                ).all()
   708→            }
   709→
   710→            # Also batch-fetch by name+company for dedup
   711→            name_company_pairs = [(c.name, c.company) for c in candidates]
   712→            from sqlalchemy import tuple_
   713→            existing_by_name = {}
   714→            if name_company_pairs:
   715→                for row in session.query(Candidate).filter(
   716→                    tuple_(Candidate.name, Candidate.company).in_(name_company_pairs)
   717→                ).all():
   718→                    existing_by_name[(row.name, row.company)] = row
   719→
   720→            for c in candidates:
   721→                existing = existing_by_id.get(c.maimai_id)
   722→                if not existing:
   723→                    existing = existing_by_name.get((c.name, c.company))
   724→
   725→                if existing:
   726→                    existing.name = c.name
   727→                    existing.title = c.title
   728→                    existing.company = c.company
   729→                    existing.location = c.location
   730→                    existing.profile_url = c.profile_url
   731→                else:
   732→                    candidate = Candidate(
   733→                        maimai_id=c.maimai_id,
   734→                        name=c.name,
   735→                        title=c.title,
   736→                        company=c.company,
   737→                        location=c.location,
   738→                        profile_url=c.profile_url,
   739→                        source_search=source_search
   740→                    )
   741→                    session.add(candidate)
   742→
   743→            logger.debug(f"Saved {len(candidates)} candidates to database")
   744→
   745→    except Exception as e:
   746→        logger.error(f"Failed to save candidates to database: {e}")
   747→        raise

   775→async def _click_avatar_for_profile_url(page: Page, name: str, company: str) -> Optional[str]:
   776→    """Try to extract profile URL for a candidate from the DOM contactcard pattern.
   777→
   778→    Searches the current page for a contactcard element whose parent card
   779→    contains the candidate's name (and optionally company), then constructs
   780→    the profile URL from the extracted user ID.
   781→
   782→    Args:
   783→        page: Playwright page on the search results
   784→        name: Candidate name to match
   785→        company: Candidate company to help disambiguate
   786→
   787→    Returns:
   788→        Profile URL string, or None if not found
   789→    """
   790→    try:
   791→        user_id = await page.evaluate('''(args) => {
   792→            const { name, company } = args;
   793→            // Check contactcard_{numeric_id} ancestors
   794→            const ccElements = document.querySelectorAll('[id^="contactcard_"]');
   795→            for (const el of ccElements) {
   796→                const text = el.textContent || '';
   797→                if (!text.includes(name)) continue;
   798→                if (company && !text.includes(company)) continue;
   799→                const match = el.id.match(/^contactcard_(\\d+)$/);
   800→                if (match) return match[1];
   801→            }
   802→            // Check member_entry_{numeric_id} descendants
   803→            const meElements = document.querySelectorAll('[id^="member_entry_"]');
   804→            for (const el of meElements) {
   805→                // Walk up to find the card containing this member_entry
   806→                let card = el.parentElement;
   807→                for (let i = 0; i < 8 && card; i++) {
   808→                    const text = card.textContent || '';
   809→                    if (text.includes(name) && (!company || text.includes(company))) {
   810→                        const match = el.id.match(/^member_entry_(\\d+)$/);
   811→                        if (match) return match[1];
   812→                    }
   813→                    card = card.parentElement;
   814→                }
   815→            }
   816→            return null;
   817→        }''', {'name': name, 'company': company})
   818→
   819→        if user_id:
   820→            profile_url = f"https://maimai.cn/contact/detail/{user_id}?dstu={user_id}"
   821→            logger.info(f"Resolved profile_url for '{name}' via contactcard: {profile_url}")
   822→            return profile_url
   823→
   824→        logger.debug(f"No contactcard found for '{name}'")
   825→        return None
   826→
   827→    except Exception as e:
   828→        logger.debug(f"_click_avatar_for_profile_url error for '{name}': {e}")
   829→        return None

  1028→async def _extract_enterprise_results(page: Page) -> List[CandidateBasic]:
  1029→    """Extract candidate information from enterprise recruitment results page.
  1030→
  1031→    The enterprise UI has a different card layout from search_center.
  1032→    Uses JS-based extraction tailored to the enterprise talent cards.
  1033→
  1034→    Args:
  1035→        page: Playwright page instance on the enterprise results page
  1036→
  1037→    Returns:
  1038→        List of CandidateBasic objects
  1039→    """
  1040→    candidates = []
  1041→
  1042→    try:
  1043→        await page.wait_for_load_state("networkidle", timeout=10000)
  1044→    except Exception:
  1045→        pass  # Continue even if networkidle times out
  1046→
  1047→    try:
  1048→        candidates_data = await page.evaluate('''() => {
  1049→            const candidates = [];
  1050→            const seenIds = new Set();
  1051→
  1052→            // Enterprise talent cards typically have a structured layout
  1053→            // Try multiple selectors for the card containers
  1054→            const cardSelectors = [
  1055→                '[class*="talent-card"]',
  1056→                '[class*="candidate-card"]',
  1057→                '[class*="recommend-card"]',
  1058→                '[class*="talent-item"]',
  1059→                '[class*="card-wrap"]',
  1060→            ];
  1061→
  1062→            let cards = [];
  1063→            for (const sel of cardSelectors) {
  1064→                cards = document.querySelectorAll(sel);
  1065→                if (cards.length > 0) break;
  1066→            }
  1067→
  1068→            // Fallback: find cards by avatar image pattern (same as search_center)
  1069→            if (cards.length === 0) {
  1070→                const images = document.querySelectorAll('img');
  1071→                const foundCards = [];
  1072→
  1073→                for (const img of images) {
  1074→                    const rect = img.getBoundingClientRect();
  1075→                    if (rect.width > 100 || rect.height > 100) continue;
  1076→                    if (rect.width < 30 || rect.height < 30) continue;
  1077→
  1078→                    let card = img.parentElement;
  1079→                    for (let i = 0; i < 8 && card; i++) {
  1080→                        const cardRect = card.getBoundingClientRect();
  1081→                        if (cardRect.width > 200 && cardRect.height > 80) {
  1082→                            break;
  1083→                        }
  1084→                        card = card.parentElement;
  1085→                    }
  1086→                    if (card) foundCards.push(card);
  1087→                }
  1088→                cards = foundCards;
  1089→            }
  1090→
  1091→            for (const card of cards) {
  1092→                const textParts = [];
  1093→                const walker = document.createTreeWalker(card, NodeFilter.SHOW_TEXT, null, false);
  1094→                while (walker.nextNode()) {
  1095→                    const text = walker.currentNode.textContent.trim();
  1096→                    if (text && text.length > 0 && text.length < 100) {
  1097→                        textParts.push(text);
  1098→                    }
  1099→                }
