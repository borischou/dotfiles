     1→"""Rate limiter for preventing account bans."""
     2→
     3→import asyncio
     4→from datetime import datetime, timedelta, timezone
     5→from typing import Tuple, Optional
     6→
     7→from loguru import logger
     8→from sqlalchemy import func
     9→
    10→from src.models.database import ActionLog, get_db_manager, DatabaseManager
    11→from src.utils.config import get_config, Config
    12→from src.exceptions.errors import RateLimitExceededError
    13→
    14→
    15→class RateLimiter:
    16→    """Rate limiter for controlling action frequency.
    17→
    18→    Supports adaptive throttling: when captchas or errors are reported,
    19→    limits are temporarily tightened to reduce detection risk.
    20→    """
    21→
    22→    def __init__(self, config: Optional[Config] = None, db_manager: Optional[DatabaseManager] = None):
    23→        """Initialize rate limiter with configuration.
    24→
    25→        Args:
    26→            config: Config instance (uses global singleton if None)
    27→            db_manager: DatabaseManager instance (uses global singleton if None)
    28→        """
    29→        self.config = config or get_config()
    30→        self.db_manager = db_manager or get_db_manager()
    31→
    32→        # Adaptive throttle state
    33→        self._throttle_factor: float = 1.0  # 1.0 = normal, <1.0 = stricter
    34→        self._captcha_timestamps: list = []
    35→        self._pruned: bool = False  # one-time prune flag
    36→
    37→    def report_captcha(self) -> None:
    38→        """Report a captcha occurrence to trigger adaptive throttling."""
    39→        now = datetime.now(timezone.utc)
    40→        self._captcha_timestamps.append(now)
    41→        # Keep only last 30 minutes of captcha events
    42→        cutoff = now - timedelta(minutes=30)
    43→        self._captcha_timestamps = [t for t in self._captcha_timestamps if t >= cutoff]
    44→        self._update_throttle_factor()
    45→
    46→    def _update_throttle_factor(self) -> None:
    47→        """Recalculate throttle factor based on recent captcha frequency.
    48→
    49→        Throttle factor ranges from 1.0 (normal) to 0.3 (heavily throttled).
    50→        - 1 captcha in 30min: reduce to 0.7x (30% fewer operations allowed)
    51→        - 2 captchas: reduce to 0.5x
    52→        - 3+ captchas: reduce to 0.3x
    53→        """
    54→        captcha_count = len(self._captcha_timestamps)
    55→
    56→        if captcha_count >= 3:
    57→            self._throttle_factor = 0.3
    58→        elif captcha_count >= 2:
    59→            self._throttle_factor = 0.5
    60→        elif captcha_count >= 1:
    61→            self._throttle_factor = 0.7
    62→        else:
    63→            self._throttle_factor = 1.0
    64→
    65→        if self._throttle_factor < 1.0:
    66→            logger.warning(
    67→                f"Adaptive throttle active: factor={self._throttle_factor:.1f} "
    68→                f"(captchas={captcha_count} in last 30min)"
    69→            )
    70→
    71→    def _refresh_throttle(self) -> None:
    72→        """Prune expired timestamps and recalculate throttle factor."""
    73→        cutoff = datetime.now(timezone.utc) - timedelta(minutes=30)
    74→        self._captcha_timestamps = [t for t in self._captcha_timestamps if t >= cutoff]
    75→        self._update_throttle_factor()
    76→
    77→    @property
    78→    def throttle_factor(self) -> float:
    79→        """Current adaptive throttle factor (1.0 = normal, <1.0 = throttled)."""
    80→        return self._throttle_factor
    81→
    82→    def _get_limits(self, action_type: str) -> Tuple[int, int, int]:
    83→        """
    84→        Get rate limits for an action type, adjusted by adaptive throttle.
    85→
    86→        Args:
    87→            action_type: Type of action ('search', 'view_profile', 'send_message')
    88→
    89→        Returns:
    90→            Tuple of (per_hour_limit, per_day_limit, min_interval_seconds)
    91→        """
    92→        action_map = {
    93→            'search': ('search_per_hour', 'search_per_day'),
    94→            'view_profile': ('profile_view_per_hour', 'profile_view_per_day'),
    95→            'send_message': ('message_per_hour', 'message_per_day'),
    96→        }
    97→
    98→        if action_type not in action_map:
    99→            raise ValueError(f"Unknown action type: {action_type}")
   100→
   101→        hour_key, day_key = action_map[action_type]
   102→
   103→        per_hour = self.config.get(f'rate_limits.{hour_key}', 20)
   104→        per_day = self.config.get(f'rate_limits.{day_key}', 100)
   105→        min_interval = self.config.get('rate_limits.min_interval_seconds', 30)
   106→
   107→        # Apply adaptive throttle: reduce limits and increase interval
   108→        per_hour = max(1, int(per_hour * self._throttle_factor))
   109→        per_day = max(1, int(per_day * self._throttle_factor))
   110→        if self._throttle_factor < 1.0:
   111→            min_interval = int(min_interval / self._throttle_factor)
   112→
   113→        return per_hour, per_day, min_interval
   114→
   115→    def _count_actions(self, action_type: str, since: datetime) -> int:
   116→        """
   117→        Count actions of a type since a given time.
   118→
   119→        Args:
   120→            action_type: Type of action
   121→            since: Count actions since this time
   122→
   123→        Returns:
   124→            Number of actions
   125→        """
   126→        with self.db_manager.session_scope() as session:
   127→            count = session.query(func.count(ActionLog.id)).filter(
   128→                ActionLog.action_type == action_type,
   129→                ActionLog.timestamp >= since
   130→            ).scalar()
   131→
   132→            return count or 0
   133→
   134→    @staticmethod
   135→    def _ensure_aware(dt: Optional[datetime]) -> Optional[datetime]:
   136→        """Attach UTC timezone to naive datetimes returned by SQLite."""
   137→        if dt is not None and dt.tzinfo is None:
   138→            return dt.replace(tzinfo=timezone.utc)
   139→        return dt
   140→
   141→    def _get_last_action_time(self, action_type: Optional[str] = None) -> Optional[datetime]:
   142→        """
   143→        Get timestamp of last action.
   144→
   145→        Args:
   146→            action_type: Type of action, or None for any action
   147→
   148→        Returns:
   149→            Timestamp of last action (timezone-aware), or None if no actions
   150→        """
   151→        with self.db_manager.session_scope() as session:
   152→            query = session.query(func.max(ActionLog.timestamp))
   153→
   154→            if action_type:
   155→                query = query.filter(ActionLog.action_type == action_type)
   156→
   157→            last_time = query.scalar()
   158→            return self._ensure_aware(last_time)
   159→
   160→    def check_rate_limit(self, action_type: str) -> None:
   161→        """
   162→        Check if action is allowed under rate limits.
   163→
   164→        Args:
   165→            action_type: Type of action to check
   166→
   167→        Raises:
   168→            RateLimitExceededError: If rate limit is exceeded
   169→        """
   170→        # Prune stale entries once per session to keep counts accurate
   171→        if not self._pruned:
   172→            self.prune_old_entries()
   173→            self._pruned = True
   174→
   175→        self._refresh_throttle()
   176→        per_hour, per_day, min_interval = self._get_limits(action_type)
   177→
   178→        now = datetime.now(timezone.utc)
   179→        one_hour_ago = now - timedelta(hours=1)
   180→        one_day_ago = now - timedelta(days=1)
   181→
   182→        # Check hourly limit
   183→        hourly_count = self._count_actions(action_type, one_hour_ago)
   184→        if hourly_count >= per_hour:
   185→            # Calculate wait time until oldest action expires
   186→            with self.db_manager.session_scope() as session:
   187→                oldest_in_window = session.query(ActionLog.timestamp).filter(
   188→                    ActionLog.action_type == action_type,
   189→                    ActionLog.timestamp >= one_hour_ago
   190→                ).order_by(ActionLog.timestamp.asc()).first()
   191→
   192→                if oldest_in_window:
   193→                    oldest_ts = self._ensure_aware(oldest_in_window[0])
   194→                    wait_until = oldest_ts + timedelta(hours=1)
   195→                    wait_seconds = max(1, int((wait_until - now).total_seconds()))
   196→                else:
   197→                    wait_seconds = 3600
   198→
   199→            raise RateLimitExceededError(
   200→                action_type=action_type,
   201→                wait_time=wait_seconds,
   202→                message=f"Hourly rate limit exceeded for {action_type}. "
   203→                        f"Limit: {per_hour}/hour. Wait {wait_seconds // 60} minutes."
   204→            )
   205→
   206→        # Check daily limit
   207→        daily_count = self._count_actions(action_type, one_day_ago)
   208→        if daily_count >= per_day:
   209→            with self.db_manager.session_scope() as session:
   210→                oldest_in_window = session.query(ActionLog.timestamp).filter(
   211→                    ActionLog.action_type == action_type,
   212→                    ActionLog.timestamp >= one_day_ago
   213→                ).order_by(ActionLog.timestamp.asc()).first()
   214→
   215→                if oldest_in_window:
   216→                    oldest_ts = self._ensure_aware(oldest_in_window[0])
   217→                    wait_until = oldest_ts + timedelta(days=1)
   218→                    wait_seconds = max(1, int((wait_until - now).total_seconds()))
   219→                else:
   220→                    wait_seconds = 86400
   221→
   222→            raise RateLimitExceededError(
   223→                action_type=action_type,
   224→                wait_time=wait_seconds,
   225→                message=f"Daily rate limit exceeded for {action_type}. "
   226→                        f"Limit: {per_day}/day. Wait {wait_seconds // 3600} hours."
   227→            )
   228→
   229→        # Check minimum interval (per-action-type to avoid cross-type blocking)
   230→        last_action = self._get_last_action_time(action_type)
   231→        if last_action:
   232→            elapsed = (now - last_action).total_seconds()
   233→            if elapsed < min_interval:
   234→                wait_seconds = max(1, int(min_interval - elapsed))
   235→                raise RateLimitExceededError(
   236→                    action_type=action_type,
   237→                    wait_time=wait_seconds,
   238→                    message=f"Minimum interval not met for {action_type}. Wait {wait_seconds} seconds."
   239→                )
   240→
   241→        logger.debug(f"Rate limit check passed for {action_type}")
   242→
   243→    def log_action(self, action_type: str, details: Optional[dict] = None) -> None:
   244→        """
   245→        Log an action to the database.
   246→
   247→        Args:
   248→            action_type: Type of action
   249→            details: Optional action details
   250→        """
   251→        with self.db_manager.session_scope() as session:
   252→            action_log = ActionLog(
   253→                action_type=action_type,
   254→                timestamp=datetime.now(timezone.utc),
   255→                details=details or {}
   256→            )
   257→            session.add(action_log)
   258→
   259→        logger.debug(f"Logged action: {action_type}")
   260→
   261→    def prune_old_entries(self, retention_hours: int = 48) -> int:
   262→        """Remove ActionLog entries older than the retention window.
   263→
   264→        This prevents stale counts from accumulating across sessions and
   265→        keeps the ActionLog table from growing indefinitely.
   266→
   267→        Args:
   268→            retention_hours: Delete entries older than this many hours (default 48)
   269→
   270→        Returns:
   271→            Number of entries deleted
   272→        """
   273→        cutoff = datetime.now(timezone.utc) - timedelta(hours=retention_hours)
   274→        try:
   275→            with self.db_manager.session_scope() as session:
   276→                count = session.query(ActionLog).filter(
   277→                    ActionLog.timestamp < cutoff
   278→                ).delete()
   279→                logger.debug(f"Pruned {count} old ActionLog entries (older than {retention_hours}h)")
   280→                return count
   281→        except Exception as e:
   282→            logger.error(f"Failed to prune ActionLog entries: {e}")
   283→            return 0
   284→
   285→    def get_quota_status(self, action_type: str) -> dict:
   286→        """
   287→        Get current quota status for an action type.
   288→
   289→        Args:
   290→            action_type: Type of action
   291→
   292→        Returns:
   293→            Dictionary with quota information
   294→        """
   295→        per_hour, per_day, _ = self._get_limits(action_type)
   296→
   297→        now = datetime.now(timezone.utc)
   298→        one_hour_ago = now - timedelta(hours=1)
   299→        one_day_ago = now - timedelta(days=1)
   300→
   301→        hourly_used = self._count_actions(action_type, one_hour_ago)
   302→        daily_used = self._count_actions(action_type, one_day_ago)
   303→
   304→        return {
   305→            'action_type': action_type,
   306→            'hourly': {
   307→                'used': hourly_used,
   308→                'limit': per_hour,
   309→                'remaining': max(0, per_hour - hourly_used)
   310→            },
   311→            'daily': {
   312→                'used': daily_used,
   313→                'limit': per_day,
   314→                'remaining': max(0, per_day - daily_used)
   315→            }
   316→        }

   319→async def async_check_rate_limit(
   320→    action_type: str,
   321→    max_retries: int = 60,
   322→    max_total_wait: int = 1800,
   323→) -> None:
   324→    """
   325→    Check rate limit, waiting through any violations until the action is allowed.
   326→
   327→    Blocks until the rate limit clears. Raises RateLimitExceededError if
   328→    still blocked after max_retries attempts or max_total_wait seconds
   329→    (prevents hanging for hours when daily limits are hit).
   330→
   331→    Args:
   332→        action_type: Type of action to check
   333→        max_retries: Maximum wait-retry cycles before giving up
   334→        max_total_wait: Maximum total seconds to wait before giving up (default 30 min)
   335→    """
   336→    import time
   337→    rate_limiter = get_rate_limiter()
   338→    start = time.monotonic()
   339→    for attempt in range(max_retries):
   340→        try:
   341→            rate_limiter.check_rate_limit(action_type)
   342→            return
   343→        except RateLimitExceededError as e:
   344→            elapsed = time.monotonic() - start
   345→            if attempt >= max_retries - 1 or elapsed + e.wait_time > max_total_wait:
   346→                raise
   347→            wait = e.wait_time + 1  # +1s buffer
   348→            logger.info(f"Rate limit hit: {e}. Waiting {wait}s before retrying (attempt {attempt + 1}/{max_retries}, elapsed {elapsed:.0f}s)...")
   349→            await asyncio.sleep(wait)
   350→
   351→
   352→# Global rate limiter instance
   353→_rate_limiter = None
   354→
   355→
   356→def get_rate_limiter() -> RateLimiter:
   357→    """Get global rate limiter instance."""
   358→    global _rate_limiter
   359→    if _rate_limiter is None:
   360→        _rate_limiter = RateLimiter()
   361→    return _rate_limiter
