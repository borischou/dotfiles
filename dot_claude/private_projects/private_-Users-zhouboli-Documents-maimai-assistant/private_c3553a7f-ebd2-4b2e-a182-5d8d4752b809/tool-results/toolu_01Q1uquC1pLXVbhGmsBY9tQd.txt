    74→def _extract_user_id(profile_url: str) -> Optional[str]:
    75→    """Extract dstu/user_id from a Maimai profile URL for direct chat navigation.
    76→
    77→    Only returns IDs that are known to work with the chat URL (u2= parameter).
    78→    Query-param IDs (dstu, u2, egid) are reliable for chat navigation.
    79→    Path-based IDs (/contact/detail/xxx) are NOT the same as chat user IDs
    80→    and should not be used for direct chat URL construction.
    81→    """
    82→    from urllib.parse import urlparse, parse_qs
    83→
    84→    parsed = urlparse(profile_url)
    85→    params = parse_qs(parsed.query)
    86→    # Only use query parameter IDs — these are confirmed to work with uh_chat_pc
    87→    for key in ("dstu", "u2", "egid"):
    88→        if key in params:
    89→            return params[key][0]
    90→    return None

    93→async def _navigate_to_chat(
    94→    page: Page,
    95→    profile_url: Optional[str],
    96→    config,
    97→    browser: BrowserManager,
    98→    maimai_id: str,
    99→    interactive: bool = False,
   100→    discovered_selectors=None,
   101→) -> bool:
   102→    """Navigate to the chat page for a candidate.
   103→
   104→    Tries in order:
   105→    1. Extract user_id from profile_url query params → direct chat URL
   106→    2. Navigate to profile_url, extract user_id from redirected URL → direct chat URL
   107→    3. Click chat button on profile page as fallback
   108→
   109→    If no profile_url is provided, looks up the candidate's profile_url from DB.
   110→
   111→    Args:
   112→        page: Playwright page instance
   113→        profile_url: Candidate's profile URL (may be None)
   114→        config: Config instance
   115→        browser: Browser manager instance
   116→        maimai_id: Candidate's Maimai ID
   117→        interactive: Enable interactive selector discovery
   118→        discovered_selectors: Shared DiscoveredSelectors instance
   119→
   120→    Returns:
   121→        True if successfully navigated to chat page
   122→    """
   123→    db_manager = get_db_manager()
   124→
   125→    # Resolve profile URL from DB if not provided
   126→    if not profile_url:
   127→        with db_manager.session_scope() as session:
   128→            db_candidate = session.query(Candidate).filter_by(maimai_id=maimai_id).first()
   129→            profile_url = db_candidate.profile_url if db_candidate else None
   130→
   131→        if not profile_url:
   132→            raise PermanentFailureError(f"No profile URL for {maimai_id}")
   133→
   134→    # Try extracting user_id directly from profile URL
   135→    user_id = _extract_user_id(profile_url)
   136→    logger.debug(f"_navigate_to_chat: profile_url={profile_url!r}, extracted user_id={user_id}")
   137→
   138→    if user_id:
   139→        chat_url = MAIMAI_CHAT_URL.format(user_id=user_id)
   140→        await retry_navigation(page, chat_url, wait_until="domcontentloaded")
   141→        await async_random_delay("page_load")
   142→        return True
   143→
   144→    # Navigate to profile page, then try to extract user_id from resulting URL
   145→    logger.debug(f"_navigate_to_chat: navigating to profile {profile_url}")
   146→    await retry_navigation(page, profile_url, wait_until="domcontentloaded")
   147→    await async_random_delay("page_load")
   148→
   149→    current_url = page.url
   150→    logger.debug(f"_navigate_to_chat: landed on {current_url}")
   151→    redirected_user_id = _extract_user_id(current_url)
   152→    if redirected_user_id:
   153→        logger.info(f"Extracted user_id={redirected_user_id} from page URL, redirecting to chat")
   154→        chat_url = MAIMAI_CHAT_URL.format(user_id=redirected_user_id)
   155→        await retry_navigation(page, chat_url, wait_until="domcontentloaded")
   156→        await async_random_delay("page_load")
   157→        return True
   158→
   159→    # Fallback: click chat button on profile page
   160→    chat_button_found = await _click_chat_button(page, config)
   161→    if chat_button_found:
   162→        await async_random_delay("action")
   163→        return True
   164→
   165→    # If interactive mode, trigger selector discovery
   166→    if interactive and discovered_selectors is not None and not discovered_selectors.is_populated:
   167→        from src.modules.selector_discovery import discover_chat_page
   168→
   169→        console.print(
   170→            "[yellow]Chat button not found. Entering interactive discovery mode...[/yellow]"
   171→        )
   172→        result = await discover_chat_page(page, maimai_id, browser)
   173→        if result.is_populated:
   174→            discovered_selectors.chat_button = result.chat_button
   175→            discovered_selectors.message_input = result.message_input
   176→            discovered_selectors.send_button = result.send_button
   177→            discovered_selectors.chat_url_pattern = result.chat_url_pattern
   178→            discovered_selectors.skip_chat_button = result.skip_chat_button
   179→            discovered_selectors.is_populated = True
   180→            return True
   181→
   182→    console.print("[red]Could not find chat button[/red]")
   183→    await browser.screenshot(f"message_error_{maimai_id}")
   184→    return False

   187→async def extract_contact_quota(page: Page) -> Optional[dict]:
   188→    """Extract contact quota from Quick Contact page's embedded share_data.
   189→
   190→    The Quick Contact page server-renders ``window.share_data`` which includes
   191→    ``mem_status.mem_info`` with remaining quotas for various member actions.
   192→
   193→    Args:
   194→        page: Playwright page on a Quick Contact (极速联系) page
   195→
   196→    Returns:
   197→        Dict with quota info, or None if data is unavailable::
   198→
   199→            {
   200→                'uh_left': int,        # 极速联系 remaining
   201→                'addfr_left': int,     # 加好友 remaining
   202→                'dc_left': int,        # 直聊 remaining
   203→                'expire_time': str,    # membership expiry
   204→                'cost': int,           # vouchers consumed per contact
   205→            }
   206→    """
   207→    try:
   208→        quota = await page.evaluate(
   209→            """() => {
   210→            if (window.share_data && window.share_data.mem_status &&
   211→                window.share_data.mem_status.mem_info) {
   212→                const info = window.share_data.mem_status.mem_info;
   213→                return {
   214→                    uh_left: info.uh_left,
   215→                    addfr_left: info.addfr_left,
   216→                    dc_left: info.dc_left,
   217→                    expire_time: info.expire_time || null,
   218→                    cost: (window.share_data.fast_contact_pay_num || {}).final_num || 1,
   219→                };
   220→            }
   221→            return null;
   222→        }"""
   223→        )
   224→        return quota
   225→    except Exception as e:
   226→        logger.debug(f"Failed to extract contact quota: {e}")
   227→        return None

   272→async def _handle_quick_contact(page: Page, content: str) -> Optional[bool]:
   273→    """Handle Maimai's "极速联系" (Quick Contact) page.
   274→
   275→    This page appears instead of regular chat for some candidates. It has a
   276→    category selector, a textarea, and a "提交" (Submit) button.
   277→
   278→    Submission works by manipulating the React component directly: setting
   279→    the message via setState and calling the debounced send method (with
   280→    flush) to trigger the POST to /bizmember/uh-chat/api/send_v2.
   281→
   282→    DOM-level textarea.fill() does NOT update React's controlled state, and
   283→    DOM click on the submit button does NOT reliably trigger the send API.
   284→
   285→    Args:
   286→        page: Playwright page instance
   287→        content: Message content to send
   288→
   289→    Returns:
   290→        True if message was sent via quick contact form,
   291→        False if quick contact page was detected but sending failed,
   292→        None if this is not a quick contact page (caller should continue normal flow)
   293→    """
   294→    try:
   295→        # Detect quick contact page via page title or page content
   296→        is_quick_contact = await page.evaluate(
   297→            """() => {
   298→            const title = document.title || '';
   299→            const bodyText = document.body ? document.body.innerText : '';
   300→            return title.includes('极速联系') || bodyText.includes('极速联系');
   301→        }"""
   302→        )
   303→
   304→        if not is_quick_contact:
   305→            return None
   306→
   307→        logger.info("Detected '极速联系' (Quick Contact) page")
   308→
   309→        # Validate message length (Quick Contact enforces 10-300 chars)
   310→        if len(content) < 10:
   311→            logger.error(
   312→                f"Quick contact: message too short ({len(content)} chars, "
   313→                "minimum 10). Quick Contact requires 10-300 characters."
   314→            )
   315→            return False
   316→        if len(content) > 300:
   317→            logger.error(
   318→                f"Quick contact: message too long ({len(content)} chars, "
   319→                "maximum 300). Quick Contact requires 10-300 characters."
   320→            )
   321→            return False
   322→
   323→        # Extract quota info before submitting (share_data is available on page load)
   324→        quota = await extract_contact_quota(page)
   325→
   326→        # Check if quota is exhausted
   327→        if quota and quota.get("uh_left", 0) <= 0:
   328→            raise QuotaExhaustedError(remaining=quota.get("uh_left", 0))
   329→
   330→        # Install XHR interceptor to capture the send_v2 response
   331→        await page.evaluate(
   332→            """() => {
   333→            window.__sendV2Result = null;
   334→            const origOpen = XMLHttpRequest.prototype.open;
   335→            const origSend = XMLHttpRequest.prototype.send;
   336→            XMLHttpRequest.prototype.open = function(m, u) {
   337→                this.__url = u; this.__method = m;
   338→                return origOpen.apply(this, arguments);
   339→            };
   340→            XMLHttpRequest.prototype.send = function(b) {
   341→                if (this.__url && this.__url.includes('send_v2')) {
   342→                    const xhr = this;
   343→                    xhr.addEventListener('load', function() {
   344→                        try {
   345→                            window.__sendV2Result = {
   346→                                status: xhr.status,
   347→                                response: xhr.responseText,
   348→                            };
   349→                        } catch(e) {
   350→                            window.__sendV2Result = {status: xhr.status, error: e.message};
   351→                        }
   352→                    });
   353→                    xhr.addEventListener('error', function() {
   354→                        window.__sendV2Result = {status: 0, error: 'network_error'};
   355→                    });
   356→                }
   357→                return origSend.apply(this, arguments);
   358→            };
   359→        }"""
   360→        )
   361→
   362→        # Submit via React component internals:
   363→        # 1. Find the class component that owns the form state
   364→        # 2. Set message + unlock push_type via setState
   365→        # 3. Call the debounced send() method and flush() to execute immediately
   366→        send_triggered = await page.evaluate(
   367→            """(msg) => {
   368→            const textarea = document.querySelector('textarea');
   369→            if (!textarea) return {success: false, error: 'no textarea'};
   370→            const keys = Object.keys(textarea);
   371→            const fiberKey = keys.find(
   372→                k => k.startsWith('__reactInternalInstance') || k.startsWith('__reactFiber')
   373→            );
   374→            if (!fiberKey) return {success: false, error: 'no React fiber'};
   375→
   376→            let fiber = textarea[fiberKey];
   377→            while (fiber) {
   378→                if (fiber.stateNode && fiber.stateNode.state
   379→                    && 'message' in fiber.stateNode.state
   380→                    && typeof fiber.stateNode.send === 'function') {
   381→                    const component = fiber.stateNode;
   382→                    // Set message and ensure push_type is not locked (true = sending lock)
   383→                    component.setState({message: msg, push_type: false}, () => {
   384→                        component.send();
   385→                        if (typeof component.send.flush === 'function') {
   386→                            component.send.flush();
   387→                        }
   388→                    });
   389→                    return {success: true};
   390→                }
   391→                fiber = fiber.return;
   392→            }
   393→            return {success: false, error: 'React component not found'};
   394→        }""",
   395→            content,
   396→        )
   397→
   398→        if not send_triggered.get("success"):
   399→            logger.warning(
   400→                f"Quick contact: failed to trigger send via React: "
   401→                f"{send_triggered.get('error', 'unknown')}"
   402→            )
   403→            return False
   404→
   405→        # Wait for the API response (send_v2 XHR)
   406→        send_result = None
   407→        for _ in range(10):
   408→            await asyncio.sleep(0.5)
   409→            send_result = await page.evaluate("() => window.__sendV2Result")
   410→            if send_result is not None:
   411→                break
   412→
   413→        if send_result is None:
   414→            logger.warning("Quick contact: send_v2 API call was not captured (timeout)")
   415→            return False
   416→
   417→        # Parse the API response
   418→        response_text = send_result.get("response", "")
   419→        try:
   420→            resp_data = json.loads(response_text) if response_text else {}
   421→        except (json.JSONDecodeError, TypeError):
   422→            resp_data = {}
   423→
   424→        ret = resp_data.get("ret")
   425→        error_msg = resp_data.get("error", "")
   426→
   427→        if ret == -1:
   428→            logger.warning(f"Quick contact: send failed — {error_msg}")
   429→            if error_msg:
   430→                console.print(f"  [red]Quick Contact 失败: {error_msg}[/red]")
   431→            return False
   432→
   433→        logger.info("Quick contact: message sent successfully")
   434→        await async_random_delay("action")
   435→
   436→        # Display remaining quota
   437→        if quota:
   438→            _display_quota(quota, just_sent=True)
   439→
   440→        return True
   441→
   442→    except PermanentFailureError:
   443→        raise  # Let quota exhaustion etc. propagate to abort the batch
   444→    except Exception as e:
   445→        logger.error(f"Quick contact handling failed: {e}\n{traceback.format_exc()}")
   446→        return False

   449→async def _detect_vip_paywall(page: Page) -> bool:
   450→    """Detect if a VIP/membership paywall is blocking the current page.
   451→
   452→    Non-VIP accounts may be redirected to a recharge/upgrade page when
   453→    trying to message certain candidates. This checks the page URL and
   454→    visible text for common paywall indicators.
   455→
   456→    Args:
   457→        page: Playwright page instance
   458→
   459→    Returns:
   460→        True if a VIP paywall is detected
   461→    """
   462→    try:
   463→        return await page.evaluate(
   464→            """() => {
   465→            const url = window.location.href;
   466→            const urlKeywords = ['vip', 'member', 'pay', 'recharge', 'upgrade', 'subscribe'];
   467→            if (urlKeywords.some(kw => url.toLowerCase().includes(kw))) {
   468→                return true;
   469→            }
   470→
   471→            const title = document.title || '';
   472→            const bodyText = document.body ? document.body.innerText.substring(0, 3000) : '';
   473→            const text = title + ' ' + bodyText;
   474→
   475→            const paywallPatterns = [
   476→                '开通会员', '升级会员', '会员专享', '会员权益',
   477→                '充值', '付费', '购买会员', '开通VIP', '升级VIP',
   478→                '立即开通', '立即升级', '立即购买',
   479→                '会员到期', '续费', '解锁', '次数已用完',
   480→                '今日沟通次数', '沟通人数已达上限',
   481→            ];
   482→
   483→            return paywallPatterns.some(p => text.includes(p));
   484→        }"""
   485→        )
   486→    except Exception as e:
   487→        logger.debug(f"VIP paywall detection failed: {e}")
   488→        return False

   491→async def send_message(
   492→    maimai_id: str,
   493→    template_name: str,
   494→    variables: Dict[str, str],
   495→    browser: Optional[BrowserManager] = None,
   496→    skip_confirmation: bool = False,
   497→    interactive: bool = False,
   498→    discovered_selectors: Optional["DiscoveredSelectors"] = None,
   499→    _skip_rate_check: bool = False,
   500→    profile_url: Optional[str] = None,
   501→) -> bool:
   502→    """
   503→    Send a message to a candidate.
   504→
   505→    Args:
   506→        maimai_id: Candidate's Maimai ID
   507→        template_name: Name of message template to use
   508→        variables: Variables to substitute in template
   509→        browser: Browser manager instance
   510→        skip_confirmation: Skip confirmation (only for testing)
   511→        interactive: Enable interactive selector discovery on failure
   512→        discovered_selectors: Shared DiscoveredSelectors instance for reuse
   513→
   514→    Returns:
   515→        True if message was sent successfully
   516→
   517→    Raises:
   518→        RateLimitExceededError: If rate limit exceeded
   519→        ValidationError: If template or variables invalid
   520→    """
   521→    # Validate and clean inputs
   522→    maimai_id = validate_maimai_id(maimai_id)
   523→
   524→    if browser is None:
   525→        browser = get_browser_manager()
   526→
   527→    config = get_config()
   528→    rate_limiter = get_rate_limiter()
   529→    db_manager = get_db_manager()
   530→
   531→    # Get template
   532→    try:
   533→        template = config.get_template(template_name)
   534→    except Exception as e:
   535→        raise ValidationError(f"Template not found: {template_name}")
   536→
   537→    # Validate template variables
   538→    validate_template_variables(template, variables)
   539→
   540→    # Render message content
   541→    content = render_template(template["content"], variables)
   542→
   543→    # Get candidate info for display
   544→    candidate_info = _get_candidate_info(db_manager, maimai_id)
   545→
   546→    # Preview and confirm
   547→    if not skip_confirmation:
   548→        action = await interactive_confirm(content, candidate_info, template_name)
   549→
   550→        if action == "skip":
   551→            console.print("[yellow]Message skipped[/yellow]")
   552→            _log_message(db_manager, maimai_id, template_name, content, "cancelled")
   553→            return False
   554→
   555→        elif action == "edit":
   556→            # Allow user to edit content
   557→            content = await edit_message(content)
   558→            if content is None:
   559→                console.print("[yellow]Message cancelled[/yellow]")
   560→                return False
   561→
   562→            # Re-confirm after edit
   563→            action = await interactive_confirm(content, candidate_info, template_name)
   564→            if action != "send":
   565→                console.print("[yellow]Message skipped[/yellow]")
   566→                return False
   567→
   568→        elif action != "send":
   569→            console.print("[yellow]Message cancelled[/yellow]")
   570→            return False
   571→
   572→    # Check rate limit (waits through min-interval) - skip when called from batch
   573→    if not _skip_rate_check:
   574→        await async_check_rate_limit("send_message")
   575→
   576→    # Ensure logged in
   577→    await ensure_logged_in(browser)
   578→
   579→    try:
   580→        page = browser.page
   581→
   582→        # If we have populated discovered selectors, use them directly

   751→async def _send_message_batch_common(
   752→    candidates: List[dict],
   753→    template_name: str,
   754→    common_variables: Dict[str, str],
   755→    send_one: Callable[[dict, str], Awaitable[CandidateSendResult]],
   756→    batch_config: BatchConfig,
   757→    skip_messaged: bool = True,
   758→    skip_confirmation: bool = False,
   759→) -> Dict[str, Any]:
   760→    """Common orchestration for all three batch-send paths.
   761→
   762→    Handles: template loading, filtering already-messaged, preview panel,
   763→    confirmation, progress bar, rate limiting, template rendering, and
   764→    result collection.  The actual per-candidate send logic is delegated
   765→    to the ``send_one`` callback.
   766→
   767→    Args:
   768→        candidates: List of dicts with maimai_id, name, company, title
   769→        template_name: Name of message template
   770→        common_variables: Shared template variables
   771→        send_one: async callback(candidate_dict, rendered_content) → CandidateSendResult
   772→        batch_config: Display / behaviour config for this path
   773→        skip_messaged: Skip already-messaged candidates
   774→        skip_confirmation: Skip interactive confirmation prompt
   775→
   776→    Returns:
   777→        Dict with total, sent, failed, skipped, rate_limited, details
   778→    """
   779→    config = get_config()
   780→    db_manager = get_db_manager()
   781→
   782→    try:
   783→        template = config.get_template(template_name)
   784→    except Exception:
   785→        raise ValidationError(f"Template not found: {template_name}")
   786→
   787→    results = {
   788→        "total": len(candidates),
   789→        "sent": 0,
   790→        "failed": 0,
   791→        "skipped": 0,
   792→        "rate_limited": 0,
   793→        "details": [],
   794→    }
   795→
   796→    # --- Filter already-messaged candidates ---
   797→    send_list = []
   798→    if skip_messaged:
   799→        all_ids = [c["maimai_id"] for c in candidates]
   800→        with db_manager.session_scope() as session:
   801→            rows = (
   802→                session.query(Message.candidate_maimai_id)
   803→                .filter(Message.candidate_maimai_id.in_(all_ids), Message.status == "sent")
   804→                .distinct()
   805→                .all()
   806→            )
   807→            messaged_ids = {row[0] for row in rows}
   808→
   809→        for c in candidates:
   810→            if c["maimai_id"] in messaged_ids:
   811→                results["skipped"] += 1
   812→                results["details"].append(
   813→                    {
   814→                        "maimai_id": c["maimai_id"],
   815→                        "name": c["name"],
   816→                        "status": "skipped",
   817→                        "reason": "already messaged",
   818→                    }
   819→                )
   820→            else:
   821→                send_list.append(c)
   822→    else:
   823→        send_list = list(candidates)
   824→
   825→    if not send_list:
   826→        if results["skipped"] > 0:
   827→            console.print(
   828→                f"[yellow]All {results['skipped']} candidates already messaged. Skipping send.[/yellow]"
   829→            )
   830→        else:
   831→            console.print("[yellow]No candidates to send messages to.[/yellow]")
   832→        return results
   833→
   834→    # --- Optional pre-validation (Path A) ---
   835→    if batch_config.pre_validate:
   836→        console.print("[dim]Validating template for all candidates...[/dim]")
   837→        for c in send_list:
   838→            test_vars = {
   839→                **common_variables,
   840→                "name": c.get("name", ""),
   841→                "company": c.get("company", ""),
   842→                "position": c.get("title", ""),
   843→            }
   844→            is_valid, missing = validate_template_rendering(template["content"], test_vars, c)
   845→            if not is_valid:
   846→                raise ValidationError(
   847→                    f"候选人 {c.get('name', 'Unknown')} 的模板变量缺失或为空:\n"
   848→                    f"  缺失: {', '.join(missing)}\n"
   849→                    f"  候选人信息: {c}\n"
   850→                    f"  请检查模板或补充变量"
   851→                )
   852→
   853→    # --- Build preview ---
   854→    first = send_list[0]
   855→    first_vars = {
   856→        **common_variables,
   857→        "name": first["name"],
   858→        "company": first["company"],
   859→        "position": first["title"],
   860→    }
   861→    # If pre-validated, skip redundant validation in render
   862→    first_content = render_template(
   863→        template["content"], first_vars, validate=not batch_config.pre_validate
   864→    )
   865→
   866→    summary_text = Text()
   867→    summary_text.append("Template: ", style="bold")
   868→    summary_text.append(f"{template_name}\n")
   869→    if batch_config.mode_label:
   870→        summary_text.append("Mode: ", style="bold")
   871→        summary_text.append(f"{batch_config.mode_label}\n")
   872→    summary_text.append("Candidates: ", style="bold")
   873→    summary_text.append(f"{len(candidates)} found, {results['skipped']} already messaged\n")
   874→    summary_text.append("Will send to: ", style="bold")
   875→    summary_text.append(f"{len(send_list)} candidates\n\n")
   876→    summary_text.append("Message Preview ", style="bold cyan")
   877→    summary_text.append("(first candidate):\n", style="dim")
   878→    summary_text.append("To: ", style="bold cyan")
   879→    recipient = first["name"]
   880→    if first.get("title") and first.get("company"):
   881→        recipient += f" ({first['title']} @ {first['company']})"
   882→    summary_text.append(f"{recipient}\n\n")
   883→    summary_text.append(first_content)
   884→
   885→    console.print(
   886→        Panel(
   887→            summary_text,
   888→            title=batch_config.panel_title,
   889→            border_style="green",
   890→            padding=(1, 2),
   891→        )
   892→    )
   893→
   894→    # --- Confirmation ---
   895→    if skip_confirmation:
   896→        console.print(f"[green]Auto-confirmed: sending to {len(send_list)} candidates[/green]")
   897→    else:
   898→        loop = asyncio.get_running_loop()
   899→        confirm = await loop.run_in_executor(
   900→            None,
   901→            lambda: questionary.confirm(
   902→                f"Confirm batch send to {len(send_list)} candidates?"
   903→            ).unsafe_ask(),
   904→        )
   905→        if not confirm:
   906→            console.print("[yellow]Batch send cancelled.[/yellow]")
   907→            return results
   908→
   909→    # --- Send loop ---
   910→    from rich.progress import (
   911→        Progress,
   912→        BarColumn,
   913→        TextColumn as ProgressTextColumn,
   914→        MofNCompleteColumn,
   915→    )
   916→
   917→    with Progress(
   918→        ProgressTextColumn("[progress.description]{task.description}"),
   919→        BarColumn(),
   920→        MofNCompleteColumn(),
   921→        console=console,
   922→    ) as progress:
   923→        task = progress.add_task(batch_config.progress_description, total=len(send_list))
   924→
   925→        for c in send_list:
   926→            # Rate limit check
   927→            try:
   928→                await async_check_rate_limit("send_message")
   929→            except RateLimitExceededError as e:
   930→                console.print(f"\n[yellow]Rate limit reached: {e}[/yellow]")
   931→                remaining = len(send_list) - (results["sent"] + results["failed"])
   932→                results["rate_limited"] = remaining
   933→                results["details"].append(
   934→                    {
   935→                        "maimai_id": c["maimai_id"],
   936→                        "name": c["name"],
   937→                        "status": "rate_limited",
   938→                        "reason": str(e),
   939→                    }
   940→                )
   941→                break
   942→
   943→            # Build per-candidate variables and render template
   944→            variables = {
   945→                **common_variables,
   946→                "name": c["name"],
   947→                "company": c["company"],
   948→                "position": c["title"],
   949→            }
   950→            try:
   951→                content = render_template(
   952→                    template["content"],
   953→                    variables,
   954→                    validate=not batch_config.pre_validate,
   955→                )
   956→            except ValidationError as e:
   957→                logger.error(f"Template rendering failed for {c['name']}: {e}")
   958→                results["failed"] += 1
   959→                results["details"].append(
   960→                    {
   961→                        "maimai_id": c["maimai_id"],
   962→                        "name": c["name"],
   963→                        "status": "failed",
   964→                        "reason": f"模板渲染失败: {e}",
   965→                    }
   966→                )
   967→                progress.advance(task)
   968→                continue
   969→
   970→            # Delegate to path-specific sender
   971→            result = await send_one(c, content)
   972→
   973→            # Record result
   974→            if result.status == "sent":
   975→                results["sent"] += 1
   976→            elif result.status in ("quota_exhausted", "rate_limited"):
   977→                if result.status == "rate_limited":
   978→                    remaining = len(send_list) - (results["sent"] + results["failed"])
   979→                    results["rate_limited"] = remaining
   980→                else:
   981→                    results["failed"] += 1
   982→            else:
   983→                results["failed"] += 1
   984→
   985→            detail = {
   986→                "maimai_id": c["maimai_id"],
   987→                "name": c["name"],
   988→                "status": result.status,
   989→            }
   990→            if result.reason:
   991→                detail["reason"] = result.reason
   992→            if result.attempts > 1:
   993→                detail["attempts"] = result.attempts
   994→            if result.content:
   995→                detail["content"] = result.content
   996→            elif content:
   997→                detail["content"] = content
   998→            results["details"].append(detail)
   999→
  1000→            if result.abort_batch:
  1001→                remaining = len(send_list) - (results["sent"] + results["failed"])
  1002→                if remaining > 0:
  1003→                    console.print(f"[yellow]跳过剩余 {remaining} 位候选人[/yellow]")
  1004→                break
  1005→
  1006→            progress.advance(task)
  1007→
  1008→            await async_random_delay("message")
  1009→
  1010→    return results

  1018→def _make_search_center_sender(
  1019→    browser: BrowserManager,
  1020→    template_name: str,
  1021→    interactive: bool,
  1022→    discovered_selectors,
  1023→) -> Callable[[dict, str], Awaitable[CandidateSendResult]]:
  1024→    """Create a sender callback for search-center (Path A) with retry + timeout."""
  1025→
  1026→    max_retries = 2
  1027→    per_candidate_timeout = 90
  1028→
  1029→    async def _send(c: dict, content: str) -> CandidateSendResult:
  1030→        variables = {
  1031→            "name": c["name"],
  1032→            "company": c["company"],
  1033→            "position": c["title"],
  1034→        }
  1035→        last_error = None
  1036→        for attempt in range(1 + max_retries):
  1037→            try:
  1038→                success = await asyncio.wait_for(
  1039→                    send_message(
  1040→                        maimai_id=c["maimai_id"],
  1041→                        template_name=template_name,
  1042→                        variables=variables,
  1043→                        browser=browser,
  1044→                        skip_confirmation=True,
  1045→                        interactive=interactive,
  1046→                        discovered_selectors=discovered_selectors,
  1047→                        _skip_rate_check=True,
  1048→                        profile_url=c.get("profile_url"),
  1049→                    ),
  1050→                    timeout=per_candidate_timeout,
  1051→                )
  1052→                if success:
  1053→                    return CandidateSendResult(
  1054→                        success=True,
  1055→                        status="sent",
  1056→                        attempts=attempt + 1,
  1057→                        content=content,
  1058→                    )
  1059→                last_error = "send returned False"
  1060→
  1061→            except asyncio.TimeoutError:
  1062→                console.print(
  1063→                    f"[yellow]Timeout for {c['name']} ({c['maimai_id']}), skipping.[/yellow]"
  1064→                )
  1065→                return CandidateSendResult(
  1066→                    success=False,
  1067→                    status="failed",
  1068→                    reason=f"Timed out after {per_candidate_timeout}s",
  1069→                    attempts=attempt + 1,
  1070→                    content=content,
  1071→                )
  1072→            except QuotaExhaustedError as e:
  1073→                console.print(f"\n[red]联系券已用尽，终止批量发送[/red]")
  1074→                return CandidateSendResult(
  1075→                    success=False,
  1076→                    status="quota_exhausted",
  1077→                    reason=str(e),
  1078→                    attempts=attempt + 1,
  1079→                    content=content,
  1080→                    abort_batch=True,
  1081→                )
  1082→            except PermanentFailureError as e:
  1083→                console.print(f"[yellow]{e}. Skipping.[/yellow]")
  1084→                return CandidateSendResult(
  1085→                    success=False,
  1086→                    status="failed",
  1087→                    reason=str(e),
  1088→                    attempts=attempt + 1,
  1089→                    content=content,
  1090→                )
  1091→            except RateLimitExceededError as e:
  1092→                console.print(f"\n[yellow]Rate limit reached: {e}[/yellow]")
  1093→                return CandidateSendResult(
  1094→                    success=False,
  1095→                    status="rate_limited",
  1096→                    reason=str(e),
  1097→                    attempts=attempt + 1,
  1098→                    content=content,
  1099→                    abort_batch=True,
  1100→                )
  1101→            except Exception as e:
  1102→                last_error = str(e)
  1103→
  1104→            # Retry with exponential backoff
  1105→            if attempt < max_retries:
  1106→                from src.utils.delays import get_retry_delay
  1107→
  1108→                delay = get_retry_delay(attempt, base_delay=3.0, max_delay=30.0)
  1109→                logger.info(f"Retry {attempt + 1}/{max_retries} for {c['name']} in {delay:.1f}s")
  1110→                await asyncio.sleep(delay)
  1111→
  1112→        # All retries exhausted
  1113→        actual_attempts = max_retries + 1
  1114→        logger.error(
  1115→            f"Failed to send message to {c['name']} after {actual_attempts} attempt(s): {last_error}"
  1116→        )
  1117→        return CandidateSendResult(
  1118→            success=False,
  1119→            status="failed",
  1120→            reason=last_error or "unknown error",
  1121→            attempts=actual_attempts,
  1122→            content=content,
  1123→        )
  1124→
  1125→    return _send

  1302→    """
  1303→    Render template with variables.
  1304→

  1310→        variables: Variable values to substitute

  1324→
  1325→    content = template_content
  1326→    for key, value in variables.items():
  1327→        content = content.replace(f"{{{key}}}", str(value))

  1615→async def _verify_message_sent(page: Page, content: str, timeout: int = 5000) -> bool:
  1616→    """
  1617→    Verify that a message was actually sent by checking the conversation area.
  1618→
  1619→    Polls up to `timeout` ms with short intervals to handle async rendering.
  1620→
  1621→    Checks for:
  1622→    1. The sent message content appearing in the conversation
  1623→    2. The message input being cleared (indicating successful send)
  1624→    3. A sent/delivered indicator element
  1625→
  1626→    Args:
  1627→        page: Playwright page instance
  1628→        content: The message content that was sent
  1629→        timeout: Max time to wait for verification in ms
  1630→
  1631→    Returns:
  1632→        True if message send was verified
  1633→    """
  1634→    import time
  1635→
  1636→    # Short snippet for matching (first 20 chars, avoid template artifacts)
  1637→    snippet = content[:20].strip()
  1638→
  1639→    poll_interval_ms = 500
  1640→    deadline = time.monotonic() + timeout / 1000.0
  1641→
  1642→    while True:
  1643→        try:
  1644→            # Strategy 1: Check if message content appears in conversation area
  1645→            verified = await page.evaluate(
  1646→                """(snippet) => {
  1647→                const chatSelectors = [
  1648→                    '[class*="message-list"]',
  1649→                    '[class*="chat-content"]',
  1650→                    '[class*="conversation"]',
  1651→                    '[class*="msg-list"]',
  1652→                    '[class*="im-message"]',
  1653→                    '[class*="chat-msg"]',
  1654→                ];
  1655→
  1656→                for (const sel of chatSelectors) {
  1657→                    const container = document.querySelector(sel);
  1658→                    if (container && container.textContent.includes(snippet)) {
  1659→                        return true;
  1660→                    }
  1661→                }
  1662→
  1663→                // Fallback: scan leaf-ish elements in the lower portion of the page
  1664→                const viewportHeight = window.innerHeight;
  1665→                const elements = document.querySelectorAll('div, p, span');
  1666→                for (const el of elements) {
  1667→                    const rect = el.getBoundingClientRect();
  1668→                    if (rect.top > viewportHeight * 0.3 && el.textContent.includes(snippet)) {
  1669→                        if (el.textContent.length < snippet.length + 200) {
  1670→                            return true;
  1671→                        }
  1672→                    }
  1673→                }
  1674→
  1675→                return false;
  1676→            }""",
  1677→                snippet,
  1678→            )
  1679→
  1680→            if verified:
  1681→                logger.debug("Message send verified: content found in conversation")
  1682→                return True
  1683→
  1684→            # Strategy 2: Check if input was cleared (indicates successful send)
  1685→            input_cleared = await page.evaluate(
  1686→                """() => {
  1687→                const inputs = document.querySelectorAll(
  1688→                    'textarea, [contenteditable="true"], input[type="text"]'
  1689→                );
  1690→                for (const input of inputs) {
  1691→                    const rect = input.getBoundingClientRect();
  1692→                    if (rect.width > 0 && rect.height > 0) {
  1693→                        const value = input.value || input.textContent || '';
  1694→                        if (value.trim() === '') {
  1695→                            return true;
  1696→                        }
  1697→                    }
  1698→                }
  1699→                return false;
  1700→            }"""
  1701→            )
  1702→
  1703→            if input_cleared:
  1704→                logger.debug("Message send verified: input field was cleared")
  1705→                return True
  1706→
  1707→            # Strategy 3: Check for sent/delivered indicators
  1708→            sent_indicators = [
  1709→                '[class*="sent"]',
  1710→                '[class*="delivered"]',
  1711→                '[class*="success"]',
  1712→                '[class*="msg-status"]',
  1713→            ]
  1714→            for selector in sent_indicators:
  1715→                try:
  1716→                    indicator = await page.query_selector(selector)
  1717→                    if indicator and await indicator.is_visible():
  1718→                        logger.debug(f"Message send verified: indicator found ({selector})")
  1719→                        return True
  1720→                except Exception:
  1721→                    continue
  1722→
  1723→        except Exception as e:
  1724→            logger.warning(f"Message verification poll failed: {e}")
  1725→
  1726→        # Check if we've exceeded the deadline
  1727→        if time.monotonic() >= deadline:
  1728→            break
  1729→
  1730→        await asyncio.sleep(poll_interval_ms / 1000.0)
  1731→
  1732→    logger.warning("Could not verify message was sent after polling")
  1733→    return False

  1737→    """
  1738→    Check if there's already an existing conversation with this candidate.

  1740→    Checks for actual user message bubbles (not system prompts/tips) in the
  1741→    conversation area.
  1742→
  1743→    Args:
  1744→        page: Playwright page instance
  1745→        maimai_id: Candidate's Maimai ID
  1746→
  1747→    Returns:
  1748→        True if an existing conversation was detected
  1749→    """
  1750→    try:
  1751→        has_conversation = await page.evaluate(
  1752→            """() => {
  1753→            // Look for conversation message containers
  1754→            const chatSelectors = [
  1755→                '[class*="message-list"]',
  1756→                '[class*="chat-content"]',
  1757→                '[class*="conversation"]',
  1758→                '[class*="msg-list"]',
  1759→                '[class*="im-message"]',
  1760→                '[class*="chat-msg"]',
  1761→                '[class*="chat-record"]',
  1762→            ];
  1763→
  1764→            // System/tip patterns that should NOT count as existing conversation
  1765→            const systemPatterns = [
  1766→                '还不是好友', '添加好友', '发送好友申请', '对方未回复',
  1767→                '以上是打招呼内容', '暂无消息', '开始聊天', '你们已经是好友',
  1768→                '系统消息', '温馨提示', '安全提示', '注意事项',
  1769→                '开启招呼语', '求职意向', '目前情况', '竞业点',
  1770→                '欢迎联系', '不感兴趣',
  1771→            ];
  1772→
  1773→            const isSystemText = (text) => {
  1774→                const t = text.trim();
  1775→                if (t.length < 5) return true;  // Too short to be a real message
  1776→                return systemPatterns.some(p => t.includes(p));
  1777→            };
  1778→
  1779→            for (const sel of chatSelectors) {
  1780→                const container = document.querySelector(sel);
  1781→                if (!container) continue;
  1782→
  1783→                // Strategy 1: Look for message bubble elements
  1784→                const bubbleSelectors = [
  1785→                    '[class*="msg-item"]', '[class*="message-item"]',
  1786→                    '[class*="bubble"]', '[class*="chat-item"]',
  1787→                    '[class*="msg-bubble"]', '[class*="chat-bubble"]',
  1788→                ];
  1789→                for (const bSel of bubbleSelectors) {
  1790→                    const bubbles = container.querySelectorAll(bSel);
  1791→                    // Filter out system/tip bubbles
  1792→                    let realMessageCount = 0;
  1793→                    for (const bubble of bubbles) {
  1794→                        const text = bubble.textContent.trim();
  1795→                        if (text.length > 5 && !isSystemText(text)) {
  1796→                            realMessageCount++;
  1797→                        }
  1798→                    }
  1799→                    if (realMessageCount > 0) {
  1800→                        return true;
  1801→                    }
  1802→                }
  1803→
  1804→                // Strategy 2: If no bubble selectors matched, check direct children
  1805→                // that look like message blocks (not system tips)
  1806→                const children = container.children;
  1807→                let realChildCount = 0;
  1808→                for (const child of children) {
  1809→                    const text = child.textContent.trim();
  1810→                    // Must have meaningful content and not be a system message
  1811→                    if (text.length > 10 && !isSystemText(text)) {
  1812→                        // Check it's not a timestamp-only or status-only element
  1813→                        if (child.querySelector('img[class*="avatar"]') || text.length > 20) {
  1814→                            realChildCount++;
  1815→                        }
  1816→                    }
  1817→                }
  1818→                if (realChildCount > 0) {
  1819→                    return true;
  1820→                }
  1821→            }
  1822→            return false;
  1823→        }"""
  1824→        )
  1825→
  1826→        if has_conversation:
  1827→            logger.info(f"Existing conversation detected for {maimai_id}")
  1828→        return has_conversation
  1829→
  1830→    except Exception as e:
  1831→        # On error, assume conversation exists (safe side: avoid duplicate messages)
  1832→        logger.warning(
  1833→            f"Conversation check failed for {maimai_id}: {e} — assuming existing conversation"
  1834→        )
  1835→        return True
