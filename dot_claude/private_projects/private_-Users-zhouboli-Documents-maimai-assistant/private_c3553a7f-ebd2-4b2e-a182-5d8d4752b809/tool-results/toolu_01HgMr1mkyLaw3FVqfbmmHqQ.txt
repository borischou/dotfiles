    97→@cli.command()
    98→@click.option('--keywords', '-k', required=True, multiple=True, help='Search keywords (can specify multiple times)')
    99→@click.option('--company', '-c', multiple=True, help='Filter by company name (can specify multiple)')
   100→@click.option('--location', '-l', help='Filter by location')
   101→@click.option('--max-pages', '-p', type=int, default=1, help='Maximum pages to scrape per keyword')
   102→@click.option('--mode', '-m', type=click.Choice(['search_center', 'enterprise', 'talent_bank']), default='search_center',
   103→              help='Search mode: search_center (default), enterprise, or talent_bank')
   104→@click.option('--export', '-e', help='Export results to file (Excel/CSV)')
   105→@click.option('--auto-filter/--no-auto-filter', default=True, help='Auto filter low-quality candidates')
   106→@click.pass_context
   107→def search(ctx, keywords, company, location, max_pages, mode, export, auto_filter):
   108→    """
   109→    Search for candidates with specified criteria.
   110→
   111→    Supports multiple keywords - each will be searched separately and results merged.
   112→
   113→    Use --mode enterprise for the enterprise recruitment tab.
   114→    Use --mode talent_bank for the talent bank path.
   115→
   116→    Examples:\n
   117→        maimai search -k "高级前端工程师" -c "阿里巴巴" --max-pages 3\n
   118→        maimai search -k "前端工程师" -k "React开发" -k "Vue工程师" -p 2\n
   119→        maimai search -k "研发" -m enterprise\n
   120→        maimai search -k "算法" -m talent_bank
   121→    """
   122→    from src.modules.export import export_candidates
   123→    from src.core.browser import get_browser_manager
   124→    from src.services.pipeline import search_all_keywords, search_all_keywords_enterprise, search_all_keywords_talent_bank
   125→
   126→    _apply_cdp_endpoint(ctx)
   127→
   128→    async def _search():
   129→        reset_delay_session()
   130→        browser = get_browser_manager()
   131→        try:
   132→            if mode == 'enterprise':
   133→                all_candidates, seen_ids = await search_all_keywords_enterprise(
   134→                    browser, keywords, company, max_pages
   135→                )
   136→            elif mode == 'talent_bank':
   137→                all_candidates, seen_ids = await search_all_keywords_talent_bank(
   138→                    browser, keywords, company, max_pages
   139→                )
   140→            else:
   141→                all_candidates, seen_ids = await search_all_keywords(
   142→                    browser, keywords, company, location, max_pages
   143→                )
   144→
   145→            # Apply auto-filter
   146→            if auto_filter:
   147→                from src.utils.filters import filter_candidates
   148→                before_count = len(all_candidates)
   149→                # Convert to dict for filtering
   150→                candidates_dict = [
   151→                    {
   152→                        'maimai_id': c.maimai_id,
   153→                        'name': c.name,
   154→                        'title': c.title,
   155→                        'company': c.company,
   156→                        'location': c.location,
   157→                    }
   158→                    for c in all_candidates
   159→                ]
   160→                filtered_dict = filter_candidates(candidates_dict, auto_filter=True)
   161→
   162→                # Filter original list
   163→                filtered_ids = {c['maimai_id'] for c in filtered_dict}
   164→                all_candidates = [c for c in all_candidates if c.maimai_id in filtered_ids]
   165→
   166→                if len(all_candidates) < before_count:
   167→                    console.print(f"[dim]自动过滤: {before_count} → {len(all_candidates)} 位候选人[/dim]\n")
   168→
   169→            # Display combined results
   170→            if all_candidates:
   171→                table = Table(title=f"Search Results ({len(all_candidates)} candidates total)")
   172→                table.add_column("Name", style="cyan")
   173→                table.add_column("Title", style="green")
   174→                table.add_column("Company", style="yellow")
   175→                table.add_column("Location", style="blue")
   176→                table.add_column("ID", style="dim")
   177→
   178→                for c in all_candidates[:30]:  # Show first 30
   179→                    table.add_row(c.name, c.title, c.company, c.location or '', c.maimai_id)
   180→
   181→                console.print(table)
   182→
   183→                if len(all_candidates) > 30:
   184→                    console.print(f"[dim]... and {len(all_candidates) - 30} more[/dim]")
   185→            else:
   186→                console.print("[yellow]No candidates found for any keyword.[/yellow]")
   187→
   188→            # Export if requested
   189→            if export and all_candidates:
   190→                fmt = 'csv' if export.endswith('.csv') else 'excel'
   191→                export_candidates(format=fmt, output_path=export)
   192→
   193→        finally:
   194→            await browser.close()
   195→
   196→    run_async(_search())

   277→@cli.command()
   278→@click.option('--keywords', '-k', multiple=True, help='Search keywords (can specify multiple)')
   279→@click.option('--template', '-t', help='Message template name')
   280→@click.option('--var', '-v', multiple=True, help='Template variable (format: key=value)')
   281→@click.option('--company', '-c', multiple=True, help='Filter by company name (can specify multiple)')
   282→@click.option('--location', '-l', help='Filter by location')
   283→@click.option('--max-pages', '-p', type=int, default=1, help='Max pages to scrape per keyword')
   284→@click.option('--mode', '-m', type=click.Choice(['search_center', 'enterprise', 'talent_bank']), default='search_center',
   285→              help='Search mode: search_center (default), enterprise, or talent_bank')
   286→@click.option('--dry-run', is_flag=True, help='Search only, do not send messages')
   287→@click.option('--skip-messaged/--no-skip-messaged', default=True, help='Skip already messaged candidates')
   288→@click.option('--batch-pages', '-b', type=int, default=0, help='Send messages every N pages (0 = send after all pages)')
   289→@click.option('--interactive', '-i', is_flag=True, help='Enable interactive selector discovery when auto-detection fails')
   290→@click.option('--resume', is_flag=True, help='Resume from last checkpoint')
   291→@click.option('--yes', '-y', is_flag=True, help='Skip confirmation prompt (auto-confirm send)')
   292→@click.option('--preview', is_flag=True, help='Preview first 3 messages without sending (exits after preview)')
   293→@click.option('--auto-filter/--no-auto-filter', default=True, help='Auto filter low-quality candidates')
   294→@click.option('--exchange-phone/--no-exchange-phone', default=False, help='Exchange phone number after sending (default ON in talent_bank mode)')
   295→@click.option('--exchange-wechat/--no-exchange-wechat', default=False, help='Exchange WeChat after sending (default ON in talent_bank mode)')
   296→@click.option('--min-years', type=int, default=None, help='Minimum work years filter (talent_bank mode only)')
   297→@click.pass_context
   298→def pipeline(ctx, keywords, template, var, company, location, max_pages, mode, dry_run, skip_messaged, batch_pages, interactive, resume, yes, preview, auto_filter, exchange_phone, exchange_wechat, min_years):
   299→    """
   300→    Search candidates and batch send messages in one command.
   301→
   302→    Combines multi-keyword search with batch messaging:
   303→    search → preview → confirm → send.
   304→
   305→    Use --mode enterprise for the enterprise recruitment path with inline messaging.
   306→    Use --mode talent_bank for the talent bank path with inline messaging.
   307→
   308→    Use --batch-pages to interleave search and messaging every N pages.
   309→    Example: -p 10 --batch-pages 2 searches pages 1-2, sends messages,
   310→    then pages 3-4, sends messages, etc.
   311→
   312→    Use --resume to continue from the last interrupted pipeline run.
   313→
   314→    Examples:\n
   315→        maimai pipeline -k "前端工程师" -t greeting -v "recruiter_name=张三" -p 2\n
   316→        maimai pipeline -k "前端" -k "后端" -t greeting -v "recruiter_name=张三" --dry-run\n
   317→        maimai pipeline -k "前端" -t greeting -p 10 --batch-pages 2\n
   318→        maimai pipeline -k "前端" -t greeting -p 1 -i\n
   319→        maimai pipeline -k "研发" -t greeting -m enterprise -p 3\n
   320→        maimai pipeline -k "算法" -t greeting -m talent_bank -p 2\n
   321→        maimai pipeline --resume
   322→    """
   323→    from src.core.browser import get_browser_manager
   324→    from src.utils.checkpoint import CheckpointManager
   325→    from src.services.pipeline import run_pipeline_default, run_pipeline_batched, SearchMode
   326→
   327→    # Handle resume mode
   328→    if resume:
   329→        ckpt = CheckpointManager.load_latest()
   330→        if ckpt is None:
   331→            console.print("[red]No checkpoint found to resume from.[/red]")
   332→            return
   333→
   334→        from rich.panel import Panel
   335→        console.print(Panel(ckpt.summary, title="Resuming Pipeline", border_style="cyan"))
   336→
   337→        p = ckpt.params
   338→        keywords = tuple(p.get('keywords', []))
   339→        template = p.get('template', template)
   340→        variables = p.get('variables', {})
   341→        company = tuple(p.get('company', []))
   342→        location = p.get('location', location)
   343→        max_pages = p.get('max_pages', max_pages)
   344→        batch_pages = p.get('batch_pages', batch_pages)
   345→        skip_messaged = p.get('skip_messaged', skip_messaged)
   346→        interactive = p.get('interactive', interactive)
   347→        mode = p.get('mode', mode)
   348→        dry_run = p.get('dry_run', dry_run)
   349→        exchange_phone = p.get('exchange_phone', exchange_phone)
   350→        exchange_wechat = p.get('exchange_wechat', exchange_wechat)
   351→        min_years = p.get('min_years', min_years)
   352→    else:
   353→        ckpt = None
   354→        variables = _parse_variables(var)
   355→
   356→    if not keywords or not template:
   357→        console.print("[red]--keywords (-k) and --template (-t) are required (or use --resume).[/red]")
   358→        return
   359→
   360→    # Resolve SearchMode enum
   361→    search_mode = SearchMode(mode)
   362→
   363→    # Talent bank defaults to exchanging phone & WeChat
   364→    if search_mode == SearchMode.TALENT_BANK:
   365→        exchange_phone = True
   366→        exchange_wechat = True
   367→
   368→    if search_mode in (SearchMode.ENTERPRISE, SearchMode.TALENT_BANK) and batch_pages > 0:
   369→        console.print(f"[yellow]Note: --batch-pages with {search_mode.value} mode uses infinite scroll; "
   370→                      f"each batch opens a fresh search. Deduplication prevents re-messaging.[/yellow]")
   371→
   372→    _apply_cdp_endpoint(ctx)
   373→
   374→    async def _pipeline():
   375→        from src.modules.auth import ensure_logged_in
   376→        from src.services.pipeline import search_all_keywords, search_all_keywords_enterprise, search_all_keywords_talent_bank
   377→        from src.modules.message import render_template
   378→        from src.utils.config import get_config
   379→        from rich.panel import Panel
   380→
   381→        reset_delay_session()
   382→        browser = get_browser_manager()
   383→        try:
   384→            await ensure_logged_in(browser)
   385→
   386→            # Preview mode: search and show first 3 messages
   387→            if preview:
   388→                console.print("[cyan]Preview 模式: 搜索并预览前 3 条消息[/cyan]\n")
   389→
   390→                if mode == 'enterprise':
   391→                    all_candidates, _ = await search_all_keywords_enterprise(
   392→                        browser, keywords, company, min(max_pages, 1)
   393→                    )
   394→                elif mode == 'talent_bank':
   395→                    all_candidates, _ = await search_all_keywords_talent_bank(
   396→                        browser, keywords, company, min(max_pages, 1),
   397→                        min_years=min_years,
   398→                    )
   399→                else:
   400→                    all_candidates, _ = await search_all_keywords(
   401→                        browser, keywords, company, location, min(max_pages, 1)
   402→                    )
   403→
   404→                if not all_candidates:
   405→                    console.print("[yellow]没有找到候选人[/yellow]")
   406→                    return
   407→
   408→                # Apply auto-filter
   409→                if auto_filter:
   410→                    from src.utils.filters import filter_candidates
   411→                    before_count = len(all_candidates)
   412→                    candidates_dict = [
   413→                        {
   414→                            'maimai_id': c.maimai_id,
   415→                            'name': c.name,
   416→                            'title': c.title,
   417→                            'company': c.company,
   418→                        }
   419→                        for c in all_candidates
   420→                    ]
   421→                    filtered_dict = filter_candidates(candidates_dict, auto_filter=True)
   422→                    filtered_ids = {c['maimai_id'] for c in filtered_dict}
   423→                    all_candidates = [c for c in all_candidates if c.maimai_id in filtered_ids]
   424→
   425→                    if len(all_candidates) < before_count:
   426→                        console.print(f"[dim]自动过滤: {before_count} → {len(all_candidates)} 位候选人[/dim]\n")
   427→
   428→                    if not all_candidates:
   429→                        console.print("[yellow]过滤后没有符合条件的候选人[/yellow]")
   430→                        return
   431→
   432→                # Load template
   433→                config = get_config()
   434→                tmpl = config.get_template(template)
   435→
   436→                # Show first 3 candidates with rendered messages
   437→                preview_count = min(3, len(all_candidates))
   438→                console.print(f"[bold]预览前 {preview_count} 位候选人的消息:[/bold]\n")
   439→
   440→                for i, c in enumerate(all_candidates[:preview_count], 1):
   441→                    msg_vars = {
   442→                        **variables,
   443→                        'name': c.name,
   444→                        'company': c.company,
   445→                        'position': c.title,
   446→                    }
   447→                    try:
   448→                        rendered = render_template(tmpl['content'], msg_vars)
   449→                        console.print(f"[bold cyan]#{i} {c.name}[/bold cyan] - {c.title} @ {c.company}")
   450→                        console.print(Panel(rendered, title=f"消息预览", border_style="green"))
   451→                        console.print()
   452→                    except Exception as e:
   453→                        console.print(f"[red]渲染失败: {e}[/red]\n")
   454→
   455→                console.print(f"[dim]总共找到 {len(all_candidates)} 位候选人[/dim]")
   456→                console.print("[yellow]Preview 完成，使用 --dry-run 搜索所有页面但不发送[/yellow]")
   457→                return
   458→
   459→            if batch_pages > 0:
   460→                await run_pipeline_batched(
   461→                    browser, keywords, company, location,
   462→                    max_pages, batch_pages, template, variables,
   463→                    dry_run, skip_messaged, interactive, ckpt,
   464→                    mode=search_mode,
   465→                    skip_confirmation=yes,
   466→                    exchange_phone=exchange_phone,
   467→                    exchange_wechat=exchange_wechat,
   468→                    min_years=min_years,
   469→                )
   470→            else:
   471→                await run_pipeline_default(
   472→                    browser, keywords, company, location,
   473→                    max_pages, template, variables,
   474→                    dry_run, skip_messaged, interactive,
   475→                    mode=search_mode,
   476→                    skip_confirmation=yes,
   477→                    exchange_phone=exchange_phone,
   478→                    exchange_wechat=exchange_wechat,
   479→                    min_years=min_years,
   480→                )
   481→        finally:
   482→            await browser.close()
   483→
   484→    run_async(_pipeline())

   715→@cli.command()
   716→def duplicates():
   717→    """
   718→    查找重复的候选人.
   719→
   720→    按姓名+公司分组，显示重复项。
   721→
   722→    Example:
   723→        maimai duplicates
   724→    """
   725→    from src.models.database import Candidate, get_db_manager
   726→    from sqlalchemy import func
   727→    from rich.table import Table
   728→
   729→    db_manager = get_db_manager()
   730→
   731→    with db_manager.session_scope() as session:
   732→        # 按 name + company 分组，找出重复项
   733→        duplicates_query = session.query(
   734→            Candidate.name,
   735→            Candidate.company,
   736→            func.count(Candidate.id).label('count')
   737→        ).group_by(Candidate.name, Candidate.company)\
   738→         .having(func.count(Candidate.id) > 1)\
   739→         .all()
   740→
   741→        if not duplicates_query:
   742→            console.print("[green]✓ 没有发现重复的候选人[/green]")
   743→            return
   744→
   745→        console.print(f"\n[yellow]发现 {len(duplicates_query)} 组重复候选人:[/yellow]\n")
   746→
   747→        table = Table()
   748→        table.add_column("姓名", style="cyan")
   749→        table.add_column("公司", style="yellow")
   750→        table.add_column("重复次数", style="red", justify="right")
   751→
   752→        for name, company, count in duplicates_query:
   753→            table.add_row(name, company or '[dim]未知[/dim]', f"×{count}")
   754→
   755→        console.print(table)
   756→        console.print()
   757→        console.print("[dim]提示: 使用数据库工具手动清理重复项[/dim]")

   760→def _parse_variables(var: tuple) -> dict:
   761→    """Parse key=value variable pairs from CLI option."""
   762→    variables = {}
   763→    for v in var:
   764→        if '=' in v:
   765→            key, value = v.split('=', 1)
   766→            variables[key.strip()] = value.strip()
   767→        else:
   768→            console.print(f"[yellow]Warning: ignoring malformed variable '{v}' (expected key=value format)[/yellow]")
   769→    return variables
