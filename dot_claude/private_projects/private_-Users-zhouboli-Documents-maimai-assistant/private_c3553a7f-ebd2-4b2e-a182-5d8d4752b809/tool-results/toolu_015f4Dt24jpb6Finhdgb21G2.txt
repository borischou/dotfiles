    53→def _is_non_rd_title(title: str) -> bool:
    54→    """Check if a title indicates a non-R&D role.
    55→
    56→    Strict mode: if any non-R&D keyword appears, filter out regardless of
    57→    whether R&D keywords are also present (e.g. "技术产品经理" is filtered).
    58→    Returns False (keep) only if title is empty or contains no non-R&D keywords.
    59→    """
    60→    if not title:
    61→        return False
    62→
    63→    # Strict: any non-R&D keyword present → filter out (case-insensitive)
    64→    title_lower = title.lower()
    65→    if any(kw.lower() in title_lower for kw in NON_RD_KEYWORDS):
    66→        return True
    67→
    68→    return False

    83→async def search_candidates(
    84→    keywords: str,
    85→    company: Optional[tuple] = None,
    86→    location: Optional[str] = None,
    87→    pages: int = 1,
    88→    start_page: int = 1,
    89→    browser: Optional[BrowserManager] = None
    90→) -> List[CandidateBasic]:
    91→    """
    92→    Search for candidates on Maimai.
    93→
    94→    Args:
    95→        keywords: Search keywords (e.g., "高级前端工程师")
    96→        company: Filter by company names, tuple of strings (optional)
    97→        location: Filter by location (optional)
    98→        pages: Number of pages to scrape (default 1)
    99→        start_page: First page number to scrape (default 1)
   100→        browser: Browser manager instance
   101→
   102→    Returns:
   103→        List of CandidateBasic objects
   104→
   105→    Raises:
   106→        RateLimitExceededError: If rate limit exceeded
   107→        SelectorNotFoundError: If page elements not found
   108→    """
   109→    # Validate inputs
   110→    validate_keywords(keywords)
   111→
   112→    if browser is None:
   113→        browser = get_browser_manager()
   114→
   115→    config = get_config()
   116→    rate_limiter = get_rate_limiter()
   117→    db_manager = get_db_manager()
   118→
   119→    # Ensure logged in
   120→    await ensure_logged_in(browser)
   121→
   122→    all_candidates: List[CandidateBasic] = []
   123→    seen_ids: set = set()  # Deduplicate across infinite-scroll extractions
   124→
   125→    with Progress(
   126→        SpinnerColumn(),
   127→        TextColumn("[progress.description]{task.description}"),
   128→        console=console
   129→    ) as progress:
   130→
   131→        end_page = start_page + pages
   132→        for page_num in range(start_page, end_page):
   133→            task = progress.add_task(f"Searching page {page_num}/{end_page - 1}...", total=None)
   134→
   135→            try:
   136→                # Check rate limit (waits through min-interval)
   137→                await async_check_rate_limit('search')
   138→
   139→                page = browser.page
   140→
   141→                if page_num == start_page:
   142→                    # First page: interactive search (maimai SPA ignores URL query params)
   143→                    await retry_navigation(page, MAIMAI_SEARCH_URL, wait_until='domcontentloaded')
   144→                    await asyncio.sleep(1)
   145→
   146→                    # Find search input and type keyword
   147→                    search_input = (
   148→                        await page.query_selector('input[placeholder="搜索"]')
   149→                        or await page.query_selector('input[type="search"]')
   150→                        or await page.query_selector('.search-input input, .header input[type="text"]')
   151→                    )
   152→                    if not search_input:
   153→                        raise SelectorNotFoundError("Search input not found on search center page")
   154→
   155→                    await search_input.click()
   156→                    await search_input.fill(keywords)
   157→                    await asyncio.sleep(0.5)
   158→                    await page.keyboard.press('Enter')
   159→                    await asyncio.sleep(2)
   160→
   161→                    # Default tab is 实名动态(feed); switch to 人脉(contacts)
   162→                    renmai_tab = await page.query_selector(':text-is("人脉")')
   163→                    if renmai_tab:
   164→                        await renmai_tab.click()
   165→                        await asyncio.sleep(2)
   166→                    else:
   167→                        logger.warning("Could not find '人脉' tab on search results page")
   168→                else:
   169→                    # Subsequent pages: scroll to load more (infinite scroll)
   170→                    prev_count = await page.evaluate(
   171→                        'document.querySelectorAll(\'[id^="contactcard_"]\').length'
   172→                    )
   173→                    await human_scroll(page, 'down', distance=1500)
   174→                    # Wait for new cards to load
   175→                    try:
   176→                        await page.wait_for_function(
   177→                            f'''() => {{
   178→                                const cards = document.querySelectorAll('[id^="contactcard_"]');
   179→                                return cards.length > {prev_count};
   180→                            }}''',
   181→                            timeout=8000,
   182→                        )
   183→                    except TimeoutError:
   184→                        logger.debug(f"No new cards after scroll (had {prev_count})")
   185→                    except Exception as e:
   186→                        logger.warning(f"Error waiting for new cards after scroll: {e}")
   187→
   188→                # Wait for contactcard elements
   189→                try:
   190→                    await page.wait_for_function(
   191→                        '''() => {
   192→                            const cards = document.querySelectorAll('[id^="contactcard_"]');
   193→                            return cards.length >= 1;
   194→                        }''',
   195→                        timeout=8000,
   196→                    )
   197→                except TimeoutError:
   198→                    logger.debug("Timed out waiting for contactcard elements, proceeding anyway")
   199→                except Exception as e:
   200→                    logger.warning(f"Error waiting for contactcard elements: {e}")
   201→
   202→                # Extract candidates from this page
   203→                candidates = await _extract_search_results(page, config)
   204→
   205→                # Deduplicate: infinite scroll re-extracts all visible cards
   206→                candidates = [c for c in candidates if c.maimai_id not in seen_ids]
   207→                for c in candidates:
   208→                    seen_ids.add(c.maimai_id)
   209→
   210→                # Anti-detection: human behavior AFTER extraction
   211→                await check_captcha(page, report_to_rate_limiter=True)
   212→                await human_scroll(page, 'down', distance=300)
   213→                await async_random_delay('action')
   214→
   215→                if not candidates:
   216→                    logger.info(f"No results found on page {page_num}")
   217→                    await browser.screenshot(f"search_debug_page{page_num}")
   218→                    progress.update(task, description=f"Page {page_num}: No results")
   219→                    # Allow one empty page (could be transient), break on second consecutive
   220→                    if page_num > start_page and not all_candidates:
   221→                        # First page had no results either — genuinely empty
   222→                        break
   223→                    continue
   224→
   225→                # Filter out non-R&D roles
   226→                before_rd_filter = len(candidates)
   227→                candidates = [c for c in candidates if not _is_non_rd_title(c.title)]
   228→                rd_filtered = before_rd_filter - len(candidates)
   229→                if rd_filtered > 0:
   230→                    logger.info(f"Filtered out {rd_filtered} non-R&D candidates")
   231→
   232→                # Apply filters if specified
   233→                if company:
   234→                    company_lower = [cn.lower() for cn in company]
   235→                    candidates = [c for c in candidates if c.company and any(cn in c.company.lower() for cn in company_lower)]
   236→
   237→                if location:
   238→                    candidates = [c for c in candidates if c.location and location.lower() in c.location.lower()]
   239→
   240→                all_candidates.extend(candidates)
   241→                progress.update(task, description=f"Page {page_num}: Found {len(candidates)} candidates")
   242→
   243→                # Log the action
   244→                rate_limiter.log_action('search', {
   245→                    'keywords': keywords,
   246→                    'page': page_num,
   247→                    'results': len(candidates)
   248→                })
   249→
   250→                # Save candidates to database
   251→                try:
   252→                    _save_candidates_to_db(db_manager, candidates, keywords)
   253→                except Exception as e:
   254→                    console.print(f"[yellow]Warning: Failed to save candidates to DB: {e}[/yellow]")
   255→
   256→                # Delay before next page
   257→                if page_num < end_page - 1:
   258→                    await async_random_delay('page_load')
   259→
   260→            except (BrowserError, SessionExpiredError):
   261→                raise  # Critical errors must propagate
   262→            except Exception as e:
   263→                logger.error(f"Error on page {page_num}: {e}")
   264→                progress.update(task, description=f"Page {page_num}: Error - {str(e)[:30]}")
   265→                # Take screenshot for debugging
   266→                await browser.screenshot(f"search_error_page{page_num}")
   267→                continue
   268→
   269→    # Log search to history
   270→    _log_search_history(db_manager, keywords, company, location, pages, len(all_candidates))
   271→
   272→    console.print(f"[green]Search complete. Found {len(all_candidates)} candidates.[/green]")
   273→    return all_candidates

   276→async def _extract_search_results(page: Page, config) -> List[CandidateBasic]:
   277→    """
   278→    Extract candidate information from search results page.
   279→
   280→    Args:
   281→        page: Playwright page instance
   282→        config: Config instance
   283→
   284→    Returns:
   285→        List of CandidateBasic objects
   286→    """
   287→    candidates = []
   288→
   289→    # NOTE: The caller (search_candidates) already waits for contactcard_
   290→    # elements before calling this function.  Maimai's SPA strips URL
   291→    # params within ~5-8s and clears results, so we must NOT add long
   292→    # waits here (no networkidle, no loading-text polling).
   293→
   294→    try:
   295→        # Strategy: Extract candidates by finding avatar images and their associated text
   296→        candidates_data = await page.evaluate('''() => {
   297→            const candidates = [];
   298→            const seenNames = new Set();
   299→
   300→            // Find all images that look like avatars
   301→            const images = document.querySelectorAll('img');
   302→
   303→            for (const img of images) {
   304→                // Skip large images (probably not avatars)
   305→                const rect = img.getBoundingClientRect();
   306→                if (rect.width > 100 || rect.height > 100) continue;
   307→                if (rect.width < 20 || rect.height < 20) continue;
   308→
   309→                // Find parent container (the card)
   310→                let card = img.parentElement;
   311→                for (let i = 0; i < 8 && card; i++) {
   312→                    const cardRect = card.getBoundingClientRect();
   313→                    // Look for a reasonable card size
   314→                    if (cardRect.width > 200 && cardRect.height > 50 && cardRect.height < 200) {
   315→                        break;
   316→                    }
   317→                    card = card.parentElement;
   318→                }
   319→
   320→                if (!card) continue;
   321→
   322→                // Get all text nodes from card, preserving order
   323→                const textParts = [];
   324→                const walker = document.createTreeWalker(card, NodeFilter.SHOW_TEXT, null, false);
   325→                while (walker.nextNode()) {
   326→                    const text = walker.currentNode.textContent.trim();
   327→                    if (text && text.length > 0 && text.length < 100) {
   328→                        textParts.push(text);
   329→                    }
   330→                }
   331→
   332→                // Parse candidate info from text parts
   333→                // Pattern: Name, Activity time, Company+Position, Industry|Department
   334→                let name = '';
   335→                let company = '';
   336→                let title = '';
   337→                let industry = '';
   338→
   339→                // Compound job title patterns (longer patterns first for greedy match)
   340→                const jobPatterns = [
   341→                    // Compound titles with prefix modifiers
   342→                    /(?:高级|资深|首席|初级|中级|senior|junior|staff|principal|lead)?\s*(?:前端|后端|全栈|web|Web|iOS|Android|Java|Python|Go|C\+\+|PHP|.NET|算法|数据|AI|机器学习|深度学习|NLP|大数据|云计算|DevOps|SRE|安全|测试|嵌入式|移动端|客户端|服务端|区块链|具身智能|自动驾驶|机器人)?\s*(?:工程师|开发|架构师|专家|技术经理|技术总监|技术负责人|Tech\s*Lead)/i,
   343→                    // Management titles
   344→                    /(?:技术|产品|运营|市场|销售|人力|财务|行政|项目|研发)?\s*(?:总监|经理|主管|负责人|VP|总裁|副总|合伙人|创始人|董事|CEO|CTO|COO|CFO|CMO|CHO|CIO)/i,
   345→                    // Other roles
   346→                    /(?:产品|UI|UX|交互|视觉)?\s*(?:设计师|分析师|顾问|运营|助理|实习生?)/i,
   347→                ];
   348→
   349→                // Patterns for text that should be skipped
   350→                const skipPatterns = [
   351→                    /^[+＋]\s*好友$/,          // "+ 好友" button
   352→                    /^\d+个共同好友$/,           // "1个共同好友" social label
   353→                    /^共同好友/,                 // "共同好友..." variants
   354→                    /^已关注$/,                  // "已关注" button
   355→                    /^添加好友$/,                // "添加好友" button
   356→                    /^关注$/,                    // "关注" button
   357→                    /^好友$/,                    // "好友" label
   358→                    /^[|｜·]$/,                  // separators
   359→                ];
   360→
   361→                const isSkippable = (text) => {
   362→                    if (skipPatterns.some(p => p.test(text))) return true;
   363→                    // Activity time
   364→                    if (/天前|小时前|分钟前|刚刚|活跃|在线/.test(text)) return true;
   365→                    // Social relationship labels
   366→                    if (/同行|前同事|校友|共同好友/.test(text)) return true;
   367→                    return false;
   368→                };
   369→
   370→                for (let i = 0; i < textParts.length; i++) {
   371→                    const part = textParts[i];
   372→
   373→                    if (isSkippable(part)) continue;
   374→
   375→                    // First short text (2-15 chars, pure CJK/letters) is likely the name
   376→                    if (!name && part.length >= 2 && part.length <= 15) {
   377→                        if (/^[\u4e00-\u9fa5a-zA-Z·\s()（）]+$/.test(part)) {
   378→                            name = part.replace(/[（(][^)）]*[）)]/g, '').trim();
   379→                            continue;
   380→                        }
   381→                    }
   382→
   383→                    // After name, next substantial text is company+position
   384→                    if (name && !company && part.length > 2) {
   385→                        // Try to split company and position using regex patterns
   386→                        let splitDone = false;
   387→                        for (const pattern of jobPatterns) {
   388→                            const match = part.match(pattern);
   389→                            if (match && match.index > 0) {
   390→                                company = part.substring(0, match.index).trim();
   391→                                title = match[0].trim();
   392→                                splitDone = true;
   393→                                break;
   394→                            }
   395→                        }
   396→                        if (!splitDone) {
   397→                            company = part;
   398→                        }
   399→                        continue;
   400→                    }
   401→
   402→                    // Next part is industry|department info
   403→                    if (name && company && !industry && part.length > 2) {
   404→                        industry = part;
   405→                        // If title wasn't split from company, use industry as fallback context
   406→                        if (!title) {
   407→                            title = industry;
   408→                        }
   409→                        break;
   410→                    }
   411→                }
   412→
   413→                // Skip if no name found or premium placeholder
   414→                if (!name) continue;
   415→                if (name === '脉脉高价值人脉') continue;
   416→
   417→                // Extract user ID from DOM patterns on card and ancestor elements.
   418→                // Two patterns exist:
   419→                //   1. contactcard_{numeric_id} — on ancestors (colleagues layout)
   420→                //   2. member_entry_{numeric_id} — on descendants (standard layout)
   421→                let userId = '';
   422→
   423→                // Check ancestors for numeric contactcard_{id}
   424→                let el = card;
   425→                while (el) {
   426→                    const elId = el.id || '';
   427→                    const ccMatch = elId.match(/^contactcard_(\d+)$/);
   428→                    if (ccMatch) {
   429→                        userId = ccMatch[1];
   430→                        break;
   431→                    }
   432→                    el = el.parentElement;
   433→                }
   434→
   435→                // Check descendants for member_entry_{id} (most common pattern)
   436→                if (!userId) {
   437→                    const meEl = card.querySelector('[id^="member_entry_"]');
   438→                    if (meEl) {
   439→                        const meMatch = meEl.id.match(/^member_entry_(\d+)$/);
   440→                        if (meMatch) {
   441→                            userId = meMatch[1];
   442→                        }
   443→                    }
   444→                }
   445→
   446→                // Also check ancestor descendants for contactcard with numeric ID
   447→                if (!userId) {
   448→                    const ccEl = card.querySelector('[id^="contactcard_"]');
   449→                    if (ccEl) {
   450→                        const ccMatch = ccEl.id.match(/^contactcard_(\d+)$/);
   451→                        if (ccMatch) {
   452→                            userId = ccMatch[1];
   453→                        }
   454→                    }
   455→                }
   456→
   457→                // Use real user_id as maimai_id when available, fall back to hash
   458→                let maimaiId;
   459→                if (userId) {
   460→                    maimaiId = 'mm_' + userId;
   461→                } else {
   462→                    const idSource = (name || '') + '|' + (company || '') + '|' + (title || '');
   463→                    let hash = 0;
   464→                    for (let j = 0; j < idSource.length; j++) {
   465→                        const ch = idSource.charCodeAt(j);
   466→                        hash = ((hash << 5) - hash) + ch;
   467→                        hash = hash & hash; // Convert to 32bit integer
   468→                    }
   469→                    maimaiId = 'mm_' + Math.abs(hash).toString(16).padStart(8, '0');
   470→                }
   471→
   472→                // Skip duplicates based on the computed ID
   473→                if (seenNames.has(maimaiId)) continue;
   474→                seenNames.add(maimaiId);
   475→
   476→                // Build profile URL from user ID when available
   477→                let profileUrl = '';
   478→                if (userId) {
   479→                    profileUrl = 'https://maimai.cn/contact/detail/' + userId + '?dstu=' + userId;
   480→                }
   481→
   482→                // Fallback: extract profile URL from <a> tags inside the card
   483→                if (!profileUrl) {
   484→                    let fallbackUrl = '';
   485→
   486→                    // First: check <a> ancestors between img and card
   487→                    let ancestor = img.parentElement;
   488→                    while (ancestor && ancestor !== card) {
   489→                        if (ancestor.tagName === 'A' && ancestor.href) {
   490→                            const href = ancestor.getAttribute('href') || '';
   491→                            if (href.includes('/profile/detail') || href.includes('/contact/detail/') || href.includes('/u/') || href.includes('egid=') || href.includes('dstu=')) {
   492→                                profileUrl = href.startsWith('http') ? href : 'https://maimai.cn' + href;
   493→                                break;
   494→                            }
   495→                        }
   496→                        ancestor = ancestor.parentElement;
   497→                    }
   498→
   499→                    // Second: search all <a> descendants inside the card
   500→                    if (!profileUrl) {
   501→                        const links = card.querySelectorAll('a[href]');
   502→                        for (const link of links) {
   503→                            const href = link.getAttribute('href') || '';
   504→                            if (href.includes('/profile/detail') || href.includes('/contact/detail/') || href.includes('/u/') || href.includes('egid=') || href.includes('dstu=')) {
   505→                                profileUrl = href.startsWith('http') ? href : 'https://maimai.cn' + href;
   506→                                break;
   507→                            }
   508→                            if (!fallbackUrl && href && !href.startsWith('javascript:') && (href.startsWith('/') || href.includes('maimai.cn'))) {
   509→                                fallbackUrl = href.startsWith('http') ? href : 'https://maimai.cn' + href;
   510→                            }
   511→                        }
   512→                    }
   513→                    if (!profileUrl && fallbackUrl) {
   514→                        profileUrl = fallbackUrl;
   515→                    }
   516→                }
   517→
   518→                // Extract location: look for remaining short text parts after name/company/title
   519→                // Common patterns: city names like "北京", "上海", "深圳" etc.
   520→                let location = '';
   521→                const locationPatterns = /^(北京|上海|广州|深圳|杭州|成都|南京|武汉|西安|苏州|长沙|天津|重庆|郑州|东莞|青岛|大连|宁波|厦门|合肥|佛山|福州|昆明|济南|珠海|无锡|常州|海外|远程|香港|台北|新加坡|美国|日本|韩国|英国|全国|其他)/;
   522→                for (let i = 0; i < textParts.length; i++) {
   523→                    const part = textParts[i];
   524→                    if (part === name || part === company || part === title || part === industry) continue;
   525→                    if (isSkippable(part)) continue;
   526→                    if (locationPatterns.test(part) && part.length <= 10) {
   527→                        location = part;
   528→                        break;
   529→                    }
   530→                }
   531→
   532→                candidates.push({
   533→                    name: name,
   534→                    title: title || '',
   535→                    company: company || '',
   536→                    maimai_id: maimaiId,
   537→                    profile_url: profileUrl,
   538→                    location: location,
   539→                });
   540→            }
   541→
   542→            return candidates;
   543→        }''')
   544→
   545→        logger.info(f"JavaScript extraction found {len(candidates_data)} candidates")
   546→
   547→        for data in candidates_data:
   548→            if data.get('name'):
   549→                candidates.append(CandidateBasic(
   550→                    maimai_id=data.get('maimai_id', ''),
   551→                    name=data.get('name', ''),
   552→                    title=data.get('title', ''),
   553→                    company=data.get('company', ''),
   554→                    location=data.get('location', ''),
   555→                    profile_url=data.get('profile_url', '')
   556→                ))
   557→
   558→        # For candidates missing profile_url, try extracting from contactcard DOM
   559→        missing_url = [c for c in candidates if not c.profile_url]
   560→        if missing_url:
   561→            logger.info(f"{len(missing_url)} candidates missing profile_url, searching contactcard DOM")
   562→            for c in missing_url:
   563→                try:
   564→                    profile_url = await _click_avatar_for_profile_url(page, c.name, c.company)
   565→                    if profile_url:
   566→                        c.profile_url = profile_url
   567→                except Exception as e:
   568→                    logger.debug(f"Contactcard lookup failed for '{c.name}': {e}")
   569→
   570→        # Fallback: Strategy 1 - Find all profile links directly
   571→        if not candidates:
   572→            profile_links = await page.query_selector_all('a[href*="/contact/"], a[href*="/u/"], a[href*="egid="], a[href*="profile"]')
   573→        else:
   574→            profile_links = []
   575→
   576→        if profile_links and not candidates:
   577→            logger.debug(f"Found {len(profile_links)} profile links")
   578→
   579→            seen_ids = set()
   580→            for link in profile_links:
   581→                try:
   582→                    href = await link.get_attribute('href')
   583→                    if not href:
   584→                        continue
   585→
   586→                    # Extract maimai_id from URL
   587→                    maimai_id = None
   588→                    if '/contact/detail/' in href:
   589→                        maimai_id = href.split('/contact/detail/')[-1].split('?')[0].split('/')[0]
   590→                    elif '/u/' in href:
   591→                        maimai_id = href.split('/u/')[-1].split('?')[0].split('/')[0]
   592→
   593→                    if not maimai_id or maimai_id in seen_ids:
   594→                        continue
   595→                    seen_ids.add(maimai_id)
   596→
   597→                    # Get name from link text
   598→                    name = await link.inner_text()
   599→                    name = name.strip()
   600→                    if not name or len(name) > 50:  # Skip if no name or too long (probably not a name)
   601→                        continue
   602→
   603→                    # Try to get parent container for more info
   604→                    parent = await link.evaluate_handle('el => el.closest("div, li, article") || el.parentElement.parentElement')
   605→
   606→                    title = ""
   607→                    company = ""
   608→
   609→                    if parent:
   610→                        # Get all text content and try to parse
   611→                        try:
   612→                            full_text = await parent.inner_text()
   613→                            lines = [l.strip() for l in full_text.split('\n') if l.strip()]
   614→
   615→                            # Usually: name, then title/company info
   616→                            for line in lines:
   617→                                if line == name:
   618→                                    continue
   619→                                if '好友' in line or '关注' in line:
   620→                                    continue
   621→                                if not title and len(line) < 50:
   622→                                    title = line
   623→                                elif not company and len(line) < 50:
   624→                                    company = line
   625→                                    break
   626→                        except Exception:
   627→                            pass  # Best-effort text parsing; proceed with partial data
   628→
   629→                    profile_url = href if href.startswith('http') else f"https://maimai.cn{href}"
   630→
   631→                    candidates.append(CandidateBasic(
   632→                        maimai_id=maimai_id,
   633→                        name=name,
   634→                        title=title,
   635→                        company=company,
   636→                        profile_url=profile_url
   637→                    ))
   638→
   639→                except Exception as e:
   640→                    logger.debug(f"Failed to extract from link: {e}")
   641→                    continue
   642→
   643→        # Strategy 2: If no links found, try button-based detection
   644→        if not candidates:
   645→            logger.debug("Trying button-based detection")
   646→            buttons = await page.query_selector_all('button:has-text("好友"), button:has-text("关注")')
   647→
   648→            for button in buttons:
   649→                try:
   650→                    parent = await button.evaluate_handle('el => el.closest("div, li, article")')
   651→                    if not parent:
   652→                        continue
   653→
   654→                    # Find link in parent
   655→                    link = await parent.query_selector('a[href*="/contact"], a[href*="/u/"]')
   656→                    if link:
   657→                        href = await link.get_attribute('href')
   658→                        name = await link.inner_text()
   659→
   660→                        maimai_id = None
   661→                        if '/contact/detail/' in href:
   662→                            maimai_id = href.split('/contact/detail/')[-1].split('?')[0]
   663→                        elif '/u/' in href:
   664→                            maimai_id = href.split('/u/')[-1].split('?')[0]
   665→
   666→                        if maimai_id and name:
   667→                            candidates.append(CandidateBasic(
   668→                                maimai_id=maimai_id,
   669→                                name=name.strip(),
   670→                                title="",
   671→                                company="",
   672→                                profile_url=href
   673→                            ))
   674→                except Exception:
   675→                    continue
   676→
   677→        # Filter out premium placeholder profiles
   678→        before_filter = len(candidates)
   679→        candidates = [c for c in candidates if c.name not in PREMIUM_PLACEHOLDER_NAMES]
   680→        filtered_count = before_filter - len(candidates)
   681→        if filtered_count > 0:
   682→            logger.info(f"Filtered out {filtered_count} premium placeholder profiles")
   683→
   684→        logger.info(f"Extracted {len(candidates)} candidates from page")
   685→
   686→    except Exception as e:
   687→        logger.error(f"Error extracting search results: {e}")
   688→        console.print(f"[yellow]Warning: Search result extraction failed: {e}[/yellow]")
   689→
   690→    return candidates

   693→def _save_candidates_to_db(db_manager, candidates: List[CandidateBasic], source_search: str) -> None:
   694→    """
   695→    Save or update candidates in database.
   696→
   697→    Args:
   698→        db_manager: Database manager instance
   699→        candidates: List of candidates to save
   700→        source_search: Search keywords that found these candidates
   701→    """
   702→    try:
   703→        with db_manager.session_scope() as session:
   704→            # Batch pre-fetch existing candidates to avoid N+1 queries
   705→            maimai_ids = [c.maimai_id for c in candidates]
   706→            existing_by_id = {
   707→                row.maimai_id: row
   708→                for row in session.query(Candidate).filter(
   709→                    Candidate.maimai_id.in_(maimai_ids)
   710→                ).all()
   711→            }
   712→
   713→            # Also batch-fetch by name+company for dedup
   714→            name_company_pairs = [(c.name, c.company) for c in candidates]
   715→            from sqlalchemy import tuple_
   716→            existing_by_name = {}
   717→            if name_company_pairs:
   718→                for row in session.query(Candidate).filter(
   719→                    tuple_(Candidate.name, Candidate.company).in_(name_company_pairs)
   720→                ).all():
   721→                    existing_by_name[(row.name, row.company)] = row
   722→
   723→            for c in candidates:
   724→                existing = existing_by_id.get(c.maimai_id)
   725→                if not existing:
   726→                    existing = existing_by_name.get((c.name, c.company))
   727→
   728→                if existing:
   729→                    existing.name = c.name
   730→                    existing.title = c.title
   731→                    existing.company = c.company
   732→                    existing.location = c.location
   733→                    existing.profile_url = c.profile_url
   734→                else:
   735→                    candidate = Candidate(
   736→                        maimai_id=c.maimai_id,
   737→                        name=c.name,
   738→                        title=c.title,
   739→                        company=c.company,
   740→                        location=c.location,
   741→                        profile_url=c.profile_url,
   742→                        source_search=source_search
   743→                    )
   744→                    session.add(candidate)
   745→
   746→            logger.debug(f"Saved {len(candidates)} candidates to database")
   747→
   748→    except Exception as e:
   749→        logger.error(f"Failed to save candidates to database: {e}")
   750→        raise

   778→async def _click_avatar_for_profile_url(page: Page, name: str, company: str) -> Optional[str]:
   779→    """Try to extract profile URL for a candidate from the DOM contactcard pattern.
   780→
   781→    Searches the current page for a contactcard element whose parent card
   782→    contains the candidate's name (and optionally company), then constructs
   783→    the profile URL from the extracted user ID.
   784→
   785→    Args:
   786→        page: Playwright page on the search results
   787→        name: Candidate name to match
   788→        company: Candidate company to help disambiguate
   789→
   790→    Returns:
   791→        Profile URL string, or None if not found
   792→    """
   793→    try:
   794→        user_id = await page.evaluate('''(args) => {
   795→            const { name, company } = args;
   796→            // Check contactcard_{numeric_id} ancestors
   797→            const ccElements = document.querySelectorAll('[id^="contactcard_"]');
   798→            for (const el of ccElements) {
   799→                const text = el.textContent || '';
   800→                if (!text.includes(name)) continue;
   801→                if (company && !text.includes(company)) continue;
   802→                const match = el.id.match(/^contactcard_(\\d+)$/);
   803→                if (match) return match[1];
   804→            }
   805→            // Check member_entry_{numeric_id} descendants
   806→            const meElements = document.querySelectorAll('[id^="member_entry_"]');
   807→            for (const el of meElements) {
   808→                // Walk up to find the card containing this member_entry
   809→                let card = el.parentElement;
   810→                for (let i = 0; i < 8 && card; i++) {
   811→                    const text = card.textContent || '';
   812→                    if (text.includes(name) && (!company || text.includes(company))) {
   813→                        const match = el.id.match(/^member_entry_(\\d+)$/);
   814→                        if (match) return match[1];
   815→                    }
   816→                    card = card.parentElement;
   817→                }
   818→            }
   819→            return null;
   820→        }''', {'name': name, 'company': company})
   821→
   822→        if user_id:
   823→            profile_url = f"https://maimai.cn/contact/detail/{user_id}?dstu={user_id}"
   824→            logger.info(f"Resolved profile_url for '{name}' via contactcard: {profile_url}")
   825→            return profile_url
   826→
   827→        logger.debug(f"No contactcard found for '{name}'")
   828→        return None
   829→
   830→    except Exception as e:
   831→        logger.debug(f"_click_avatar_for_profile_url error for '{name}': {e}")
   832→        return None

  1031→async def _extract_enterprise_results(page: Page) -> List[CandidateBasic]:
  1032→    """Extract candidate information from enterprise recruitment results page.
  1033→
  1034→    The enterprise UI has a different card layout from search_center.
  1035→    Uses JS-based extraction tailored to the enterprise talent cards.
  1036→
  1037→    Args:
  1038→        page: Playwright page instance on the enterprise results page
  1039→
  1040→    Returns:
  1041→        List of CandidateBasic objects
  1042→    """
  1043→    candidates = []
  1044→
  1045→    try:
  1046→        await page.wait_for_load_state("networkidle", timeout=10000)
  1047→    except Exception:
  1048→        pass  # Continue even if networkidle times out
  1049→
  1050→    try:
  1051→        candidates_data = await page.evaluate('''() => {
  1052→            const candidates = [];
  1053→            const seenIds = new Set();
  1054→
  1055→            // Enterprise talent cards typically have a structured layout
  1056→            // Try multiple selectors for the card containers
  1057→            const cardSelectors = [
  1058→                '[class*="talent-card"]',
  1059→                '[class*="candidate-card"]',
  1060→                '[class*="recommend-card"]',
  1061→                '[class*="talent-item"]',
  1062→                '[class*="card-wrap"]',
  1063→            ];
  1064→
  1065→            let cards = [];
  1066→            for (const sel of cardSelectors) {
  1067→                cards = document.querySelectorAll(sel);
  1068→                if (cards.length > 0) break;
  1069→            }
  1070→
  1071→            // Fallback: find cards by avatar image pattern (same as search_center)
  1072→            if (cards.length === 0) {
  1073→                const images = document.querySelectorAll('img');
  1074→                const foundCards = [];
  1075→
  1076→                for (const img of images) {
  1077→                    const rect = img.getBoundingClientRect();
  1078→                    if (rect.width > 100 || rect.height > 100) continue;
  1079→                    if (rect.width < 30 || rect.height < 30) continue;
  1080→
  1081→                    let card = img.parentElement;
  1082→                    for (let i = 0; i < 8 && card; i++) {
  1083→                        const cardRect = card.getBoundingClientRect();
  1084→                        if (cardRect.width > 200 && cardRect.height > 80) {
  1085→                            break;
  1086→                        }
  1087→                        card = card.parentElement;
  1088→                    }
  1089→                    if (card) foundCards.push(card);
  1090→                }
  1091→                cards = foundCards;
  1092→            }
  1093→
  1094→            for (const card of cards) {
