    53	def _is_non_rd_title(title: str) -> bool:
    54	    """Check if a title indicates a non-R&D role.
    55	
    56	    Strict mode: if any non-R&D keyword appears, filter out regardless of
    57	    whether R&D keywords are also present (e.g. "技术产品经理" is filtered).
    58	    Returns False (keep) only if title is empty or contains no non-R&D keywords.
    59	    """
    60	    if not title:
    61	        return False
    62	
    63	    # Strict: any non-R&D keyword present → filter out
    64	    if any(kw in title for kw in NON_RD_KEYWORDS):
    65	        return True
    66	
    67	    return False

    82	async def search_candidates(
    83	    keywords: str,
    84	    company: Optional[tuple] = None,
    85	    location: Optional[str] = None,
    86	    pages: int = 1,
    87	    start_page: int = 1,
    88	    browser: Optional[BrowserManager] = None
    89	) -> List[CandidateBasic]:
    90	    """
    91	    Search for candidates on Maimai.
    92	
    93	    Args:
    94	        keywords: Search keywords (e.g., "高级前端工程师")
    95	        company: Filter by company names, tuple of strings (optional)
    96	        location: Filter by location (optional)
    97	        pages: Number of pages to scrape (default 1)
    98	        start_page: First page number to scrape (default 1)
    99	        browser: Browser manager instance
   100	
   101	    Returns:
   102	        List of CandidateBasic objects
   103	
   104	    Raises:
   105	        RateLimitExceededError: If rate limit exceeded
   106	        SelectorNotFoundError: If page elements not found
   107	    """
   108	    # Validate inputs
   109	    validate_keywords(keywords)
   110	
   111	    if browser is None:
   112	        browser = get_browser_manager()
   113	
   114	    config = get_config()
   115	    rate_limiter = get_rate_limiter()
   116	    db_manager = get_db_manager()
   117	
   118	    # Ensure logged in
   119	    await ensure_logged_in(browser)
   120	
   121	    all_candidates: List[CandidateBasic] = []
   122	    seen_ids: set = set()  # Deduplicate across infinite-scroll extractions
   123	
   124	    with Progress(
   125	        SpinnerColumn(),
   126	        TextColumn("[progress.description]{task.description}"),
   127	        console=console
   128	    ) as progress:
   129	
   130	        end_page = start_page + pages
   131	        for page_num in range(start_page, end_page):
   132	            task = progress.add_task(f"Searching page {page_num}/{end_page - 1}...", total=None)
   133	
   134	            try:
   135	                # Check rate limit (waits through min-interval)
   136	                await async_check_rate_limit('search')
   137	
   138	                page = browser.page
   139	
   140	                if page_num == start_page:
   141	                    # First page: interactive search (maimai SPA ignores URL query params)
   142	                    await retry_navigation(page, MAIMAI_SEARCH_URL, wait_until='domcontentloaded')
   143	                    await asyncio.sleep(1)
   144	
   145	                    # Find search input and type keyword
   146	                    search_input = (
   147	                        await page.query_selector('input[placeholder="搜索"]')
   148	                        or await page.query_selector('input[type="search"]')
   149	                        or await page.query_selector('.search-input input, .header input[type="text"]')
   150	                    )
   151	                    if not search_input:
   152	                        raise SelectorNotFoundError("Search input not found on search center page")
   153	
   154	                    await search_input.click()
   155	                    await search_input.fill(keywords)
   156	                    await asyncio.sleep(0.5)
   157	                    await page.keyboard.press('Enter')
   158	                    await asyncio.sleep(2)
   159	
   160	                    # Default tab is 实名动态(feed); switch to 人脉(contacts)
   161	                    renmai_tab = await page.query_selector(':text-is("人脉")')
   162	                    if renmai_tab:
   163	                        await renmai_tab.click()
   164	                        await asyncio.sleep(2)
   165	                    else:
   166	                        logger.warning("Could not find '人脉' tab on search results page")
   167	                else:
   168	                    # Subsequent pages: scroll to load more (infinite scroll)
   169	                    prev_count = await page.evaluate(
   170	                        'document.querySelectorAll(\'[id^="contactcard_"]\').length'
   171	                    )
   172	                    await human_scroll(page, 'down', distance=1500)
   173	                    # Wait for new cards to load
   174	                    try:
   175	                        await page.wait_for_function(
   176	                            f'''() => {{
   177	                                const cards = document.querySelectorAll('[id^="contactcard_"]');
   178	                                return cards.length > {prev_count};
   179	                            }}''',
   180	                            timeout=8000,
   181	                        )
   182	                    except TimeoutError:
   183	                        logger.debug(f"No new cards after scroll (had {prev_count})")
   184	                    except Exception as e:
   185	                        logger.warning(f"Error waiting for new cards after scroll: {e}")
   186	
   187	                # Wait for contactcard elements
   188	                try:
   189	                    await page.wait_for_function(
   190	                        '''() => {
   191	                            const cards = document.querySelectorAll('[id^="contactcard_"]');
   192	                            return cards.length >= 1;
   193	                        }''',
   194	                        timeout=8000,
   195	                    )
   196	                except TimeoutError:
   197	                    logger.debug("Timed out waiting for contactcard elements, proceeding anyway")
   198	                except Exception as e:
   199	                    logger.warning(f"Error waiting for contactcard elements: {e}")
   200	
   201	                # Extract candidates from this page
   202	                candidates = await _extract_search_results(page, config)
   203	
   204	                # Deduplicate: infinite scroll re-extracts all visible cards
   205	                candidates = [c for c in candidates if c.maimai_id not in seen_ids]
   206	                for c in candidates:
   207	                    seen_ids.add(c.maimai_id)
   208	
   209	                # Anti-detection: human behavior AFTER extraction
   210	                await check_captcha(page, report_to_rate_limiter=True)
   211	                await human_scroll(page, 'down', distance=300)
   212	                await async_random_delay('action')
   213	
   214	                if not candidates:
   215	                    logger.info(f"No results found on page {page_num}")
   216	                    await browser.screenshot(f"search_debug_page{page_num}")
   217	                    progress.update(task, description=f"Page {page_num}: No results")
   218	                    # Allow one empty page (could be transient), break on second consecutive
   219	                    if page_num > start_page and not all_candidates:
   220	                        # First page had no results either — genuinely empty
   221	                        break
   222	                    continue
   223	
   224	                # Filter out non-R&D roles
   225	                before_rd_filter = len(candidates)
   226	                candidates = [c for c in candidates if not _is_non_rd_title(c.title)]
   227	                rd_filtered = before_rd_filter - len(candidates)
   228	                if rd_filtered > 0:
   229	                    logger.info(f"Filtered out {rd_filtered} non-R&D candidates")
   230	
   231	                # Apply filters if specified
   232	                if company:
   233	                    company_lower = [cn.lower() for cn in company]
   234	                    candidates = [c for c in candidates if c.company and any(cn in c.company.lower() for cn in company_lower)]
   235	
   236	                if location:
   237	                    candidates = [c for c in candidates if c.location and location.lower() in c.location.lower()]
   238	
   239	                all_candidates.extend(candidates)
   240	                progress.update(task, description=f"Page {page_num}: Found {len(candidates)} candidates")
   241	
   242	                # Log the action
   243	                rate_limiter.log_action('search', {
   244	                    'keywords': keywords,
   245	                    'page': page_num,
   246	                    'results': len(candidates)
   247	                })
   248	
   249	                # Save candidates to database
   250	                try:
   251	                    _save_candidates_to_db(db_manager, candidates, keywords)
   252	                except Exception as e:
   253	                    console.print(f"[yellow]Warning: Failed to save candidates to DB: {e}[/yellow]")
   254	
   255	                # Delay before next page
   256	                if page_num < end_page - 1:
   257	                    await async_random_delay('page_load')
   258	
   259	            except (BrowserError, SessionExpiredError):
   260	                raise  # Critical errors must propagate
   261	            except Exception as e:
   262	                logger.error(f"Error on page {page_num}: {e}")
   263	                progress.update(task, description=f"Page {page_num}: Error - {str(e)[:30]}")
   264	                # Take screenshot for debugging
   265	                await browser.screenshot(f"search_error_page{page_num}")
   266	                continue
   267	
   268	    # Log search to history
   269	    _log_search_history(db_manager, keywords, company, location, pages, len(all_candidates))
   270	
   271	    console.print(f"[green]Search complete. Found {len(all_candidates)} candidates.[/green]")
   272	    return all_candidates

   275	async def _extract_search_results(page: Page, config) -> List[CandidateBasic]:
   276	    """
   277	    Extract candidate information from search results page.
   278	
   279	    Args:
   280	        page: Playwright page instance
   281	        config: Config instance
   282	
   283	    Returns:
   284	        List of CandidateBasic objects
   285	    """
   286	    candidates = []
   287	
   288	    # NOTE: The caller (search_candidates) already waits for contactcard_
   289	    # elements before calling this function.  Maimai's SPA strips URL
   290	    # params within ~5-8s and clears results, so we must NOT add long
   291	    # waits here (no networkidle, no loading-text polling).
   292	
   293	    try:
   294	        # Strategy: Extract candidates by finding avatar images and their associated text
   295	        candidates_data = await page.evaluate('''() => {
   296	            const candidates = [];
   297	            const seenNames = new Set();
   298	
   299	            // Find all images that look like avatars
   300	            const images = document.querySelectorAll('img');
   301	
   302	            for (const img of images) {
   303	                // Skip large images (probably not avatars)
   304	                const rect = img.getBoundingClientRect();
   305	                if (rect.width > 100 || rect.height > 100) continue;
   306	                if (rect.width < 20 || rect.height < 20) continue;
   307	
   308	                // Find parent container (the card)
   309	                let card = img.parentElement;
   310	                for (let i = 0; i < 8 && card; i++) {
   311	                    const cardRect = card.getBoundingClientRect();
   312	                    // Look for a reasonable card size
   313	                    if (cardRect.width > 200 && cardRect.height > 50 && cardRect.height < 200) {
   314	                        break;
   315	                    }
   316	                    card = card.parentElement;
   317	                }
   318	
   319	                if (!card) continue;
   320	
   321	                // Get all text nodes from card, preserving order
   322	                const textParts = [];
   323	                const walker = document.createTreeWalker(card, NodeFilter.SHOW_TEXT, null, false);
   324	                while (walker.nextNode()) {
   325	                    const text = walker.currentNode.textContent.trim();
   326	                    if (text && text.length > 0 && text.length < 100) {
   327	                        textParts.push(text);
   328	                    }
   329	                }
   330	
   331	                // Parse candidate info from text parts
   332	                // Pattern: Name, Activity time, Company+Position, Industry|Department
   333	                let name = '';
   334	                let company = '';
   335	                let title = '';
   336	                let industry = '';
   337	
   338	                // Compound job title patterns (longer patterns first for greedy match)
   339	                const jobPatterns = [
   340	                    // Compound titles with prefix modifiers
   341	                    /(?:高级|资深|首席|初级|中级|senior|junior|staff|principal|lead)?\s*(?:前端|后端|全栈|web|Web|iOS|Android|Java|Python|Go|C\+\+|PHP|.NET|算法|数据|AI|机器学习|深度学习|NLP|大数据|云计算|DevOps|SRE|安全|测试|嵌入式|移动端|客户端|服务端|区块链|具身智能|自动驾驶|机器人)?\s*(?:工程师|开发|架构师|专家|技术经理|技术总监|技术负责人|Tech\s*Lead)/i,
   342	                    // Management titles
   343	                    /(?:技术|产品|运营|市场|销售|人力|财务|行政|项目|研发)?\s*(?:总监|经理|主管|负责人|VP|总裁|副总|合伙人|创始人|董事|CEO|CTO|COO|CFO|CMO|CHO|CIO)/i,
   344	                    // Other roles
   345	                    /(?:产品|UI|UX|交互|视觉)?\s*(?:设计师|分析师|顾问|运营|助理|实习生?)/i,
   346	                ];
   347	
   348	                // Patterns for text that should be skipped
   349	                const skipPatterns = [
   350	                    /^[+＋]\s*好友$/,          // "+ 好友" button
   351	                    /^\d+个共同好友$/,           // "1个共同好友" social label
   352	                    /^共同好友/,                 // "共同好友..." variants
   353	                    /^已关注$/,                  // "已关注" button
   354	                    /^添加好友$/,                // "添加好友" button
   355	                    /^关注$/,                    // "关注" button
   356	                    /^好友$/,                    // "好友" label
   357	                    /^[|｜·]$/,                  // separators
   358	                ];
   359	
   360	                const isSkippable = (text) => {
   361	                    if (skipPatterns.some(p => p.test(text))) return true;
   362	                    // Activity time
   363	                    if (/天前|小时前|分钟前|刚刚|活跃|在线/.test(text)) return true;
   364	                    // Social relationship labels
   365	                    if (/同行|前同事|校友|共同好友/.test(text)) return true;
   366	                    return false;
   367	                };
   368	
   369	                for (let i = 0; i < textParts.length; i++) {
   370	                    const part = textParts[i];
   371	
   372	                    if (isSkippable(part)) continue;
   373	
   374	                    // First short text (2-15 chars, pure CJK/letters) is likely the name
   375	                    if (!name && part.length >= 2 && part.length <= 15) {
   376	                        if (/^[\u4e00-\u9fa5a-zA-Z·\s()（）]+$/.test(part)) {
   377	                            name = part.replace(/[（(][^)）]*[）)]/g, '').trim();
   378	                            continue;
   379	                        }
   380	                    }
   381	
   382	                    // After name, next substantial text is company+position
   383	                    if (name && !company && part.length > 2) {
   384	                        // Try to split company and position using regex patterns
   385	                        let splitDone = false;
   386	                        for (const pattern of jobPatterns) {
   387	                            const match = part.match(pattern);
   388	                            if (match && match.index > 0) {
   389	                                company = part.substring(0, match.index).trim();
   390	                                title = match[0].trim();
   391	                                splitDone = true;
   392	                                break;
   393	                            }
   394	                        }
   395	                        if (!splitDone) {
   396	                            company = part;
   397	                        }
   398	                        continue;
   399	                    }
   400	
   401	                    // Next part is industry|department info
   402	                    if (name && company && !industry && part.length > 2) {
   403	                        industry = part;
   404	                        // If title wasn't split from company, use industry as fallback context
   405	                        if (!title) {
   406	                            title = industry;
   407	                        }
   408	                        break;
   409	                    }
   410	                }
   411	
   412	                // Skip if no name found or premium placeholder
   413	                if (!name) continue;
   414	                if (name === '脉脉高价值人脉') continue;
   415	
   416	                // Extract user ID from DOM patterns on card and ancestor elements.
   417	                // Two patterns exist:
   418	                //   1. contactcard_{numeric_id} — on ancestors (colleagues layout)
   419	                //   2. member_entry_{numeric_id} — on descendants (standard layout)
   420	                let userId = '';
   421	
   422	                // Check ancestors for numeric contactcard_{id}
   423	                let el = card;
   424	                while (el) {
   425	                    const elId = el.id || '';
   426	                    const ccMatch = elId.match(/^contactcard_(\d+)$/);
   427	                    if (ccMatch) {
   428	                        userId = ccMatch[1];
   429	                        break;
   430	                    }
   431	                    el = el.parentElement;
   432	                }
   433	
   434	                // Check descendants for member_entry_{id} (most common pattern)
   435	                if (!userId) {
   436	                    const meEl = card.querySelector('[id^="member_entry_"]');
   437	                    if (meEl) {
   438	                        const meMatch = meEl.id.match(/^member_entry_(\d+)$/);
   439	                        if (meMatch) {
   440	                            userId = meMatch[1];
   441	                        }
   442	                    }
   443	                }
   444	
   445	                // Also check ancestor descendants for contactcard with numeric ID
   446	                if (!userId) {
   447	                    const ccEl = card.querySelector('[id^="contactcard_"]');
   448	                    if (ccEl) {
   449	                        const ccMatch = ccEl.id.match(/^contactcard_(\d+)$/);
   450	                        if (ccMatch) {
   451	                            userId = ccMatch[1];
   452	                        }
   453	                    }
   454	                }
   455	
   456	                // Use real user_id as maimai_id when available, fall back to hash
   457	                let maimaiId;
   458	                if (userId) {
   459	                    maimaiId = 'mm_' + userId;
   460	                } else {
   461	                    const idSource = (name || '') + '|' + (company || '') + '|' + (title || '');
   462	                    let hash = 0;
   463	                    for (let j = 0; j < idSource.length; j++) {
   464	                        const ch = idSource.charCodeAt(j);
   465	                        hash = ((hash << 5) - hash) + ch;
   466	                        hash = hash & hash; // Convert to 32bit integer
   467	                    }
   468	                    maimaiId = 'mm_' + Math.abs(hash).toString(16).padStart(8, '0');
   469	                }
   470	
   471	                // Skip duplicates based on the computed ID
   472	                if (seenNames.has(maimaiId)) continue;
   473	                seenNames.add(maimaiId);
   474	
   475	                // Build profile URL from user ID when available
   476	                let profileUrl = '';
   477	                if (userId) {
   478	                    profileUrl = 'https://maimai.cn/contact/detail/' + userId + '?dstu=' + userId;
   479	                }
   480	
   481	                // Fallback: extract profile URL from <a> tags inside the card
   482	                if (!profileUrl) {
   483	                    let fallbackUrl = '';
   484	
   485	                    // First: check <a> ancestors between img and card
   486	                    let ancestor = img.parentElement;
   487	                    while (ancestor && ancestor !== card) {
   488	                        if (ancestor.tagName === 'A' && ancestor.href) {
   489	                            const href = ancestor.getAttribute('href') || '';
   490	                            if (href.includes('/profile/detail') || href.includes('/contact/detail/') || href.includes('/u/') || href.includes('egid=') || href.includes('dstu=')) {
   491	                                profileUrl = href.startsWith('http') ? href : 'https://maimai.cn' + href;
   492	                                break;
   493	                            }
   494	                        }
   495	                        ancestor = ancestor.parentElement;
   496	                    }
   497	
   498	                    // Second: search all <a> descendants inside the card
   499	                    if (!profileUrl) {
   500	                        const links = card.querySelectorAll('a[href]');
   501	                        for (const link of links) {
   502	                            const href = link.getAttribute('href') || '';
   503	                            if (href.includes('/profile/detail') || href.includes('/contact/detail/') || href.includes('/u/') || href.includes('egid=') || href.includes('dstu=')) {
   504	                                profileUrl = href.startsWith('http') ? href : 'https://maimai.cn' + href;
   505	                                break;
   506	                            }
   507	                            if (!fallbackUrl && href && !href.startsWith('javascript:') && (href.startsWith('/') || href.includes('maimai.cn'))) {
   508	                                fallbackUrl = href.startsWith('http') ? href : 'https://maimai.cn' + href;
   509	                            }
   510	                        }
   511	                    }
   512	                    if (!profileUrl && fallbackUrl) {
   513	                        profileUrl = fallbackUrl;
   514	                    }
   515	                }
   516	
   517	                // Extract location: look for remaining short text parts after name/company/title
   518	                // Common patterns: city names like "北京", "上海", "深圳" etc.
   519	                let location = '';
   520	                const locationPatterns = /^(北京|上海|广州|深圳|杭州|成都|南京|武汉|西安|苏州|长沙|天津|重庆|郑州|东莞|青岛|大连|宁波|厦门|合肥|佛山|福州|昆明|济南|珠海|无锡|常州|海外|远程|香港|台北|新加坡|美国|日本|韩国|英国|全国|其他)/;
   521	                for (let i = 0; i < textParts.length; i++) {
   522	                    const part = textParts[i];
   523	                    if (part === name || part === company || part === title || part === industry) continue;
   524	                    if (isSkippable(part)) continue;
   525	                    if (locationPatterns.test(part) && part.length <= 10) {
   526	                        location = part;
   527	                        break;
   528	                    }
   529	                }
   530	
   531	                candidates.push({
   532	                    name: name,
   533	                    title: title || '',
   534	                    company: company || '',
   535	                    maimai_id: maimaiId,
   536	                    profile_url: profileUrl,
   537	                    location: location,
   538	                });
   539	            }
   540	
   541	            return candidates;
   542	        }''')
   543	
   544	        logger.info(f"JavaScript extraction found {len(candidates_data)} candidates")
   545	
   546	        for data in candidates_data:
   547	            if data.get('name'):
   548	                candidates.append(CandidateBasic(
   549	                    maimai_id=data.get('maimai_id', ''),
   550	                    name=data.get('name', ''),
   551	                    title=data.get('title', ''),
   552	                    company=data.get('company', ''),
   553	                    location=data.get('location', ''),
   554	                    profile_url=data.get('profile_url', '')
   555	                ))
   556	
   557	        # For candidates missing profile_url, try extracting from contactcard DOM
   558	        missing_url = [c for c in candidates if not c.profile_url]
   559	        if missing_url:
   560	            logger.info(f"{len(missing_url)} candidates missing profile_url, searching contactcard DOM")
   561	            for c in missing_url:
   562	                try:
   563	                    profile_url = await _click_avatar_for_profile_url(page, c.name, c.company)
   564	                    if profile_url:
   565	                        c.profile_url = profile_url
   566	                except Exception as e:
   567	                    logger.debug(f"Contactcard lookup failed for '{c.name}': {e}")
   568	
   569	        # Fallback: Strategy 1 - Find all profile links directly
   570	        if not candidates:
   571	            profile_links = await page.query_selector_all('a[href*="/contact/"], a[href*="/u/"], a[href*="egid="], a[href*="profile"]')
   572	        else:
   573	            profile_links = []
   574	
   575	        if profile_links and not candidates:
   576	            logger.debug(f"Found {len(profile_links)} profile links")
   577	
   578	            seen_ids = set()
   579	            for link in profile_links:
   580	                try:
   581	                    href = await link.get_attribute('href')
   582	                    if not href:
   583	                        continue
   584	
   585	                    # Extract maimai_id from URL
   586	                    maimai_id = None
   587	                    if '/contact/detail/' in href:
   588	                        maimai_id = href.split('/contact/detail/')[-1].split('?')[0].split('/')[0]
   589	                    elif '/u/' in href:
   590	                        maimai_id = href.split('/u/')[-1].split('?')[0].split('/')[0]
   591	
   592	                    if not maimai_id or maimai_id in seen_ids:
   593	                        continue
   594	                    seen_ids.add(maimai_id)
   595	
   596	                    # Get name from link text
   597	                    name = await link.inner_text()
   598	                    name = name.strip()
   599	                    if not name or len(name) > 50:  # Skip if no name or too long (probably not a name)
   600	                        continue
   601	
   602	                    # Try to get parent container for more info
   603	                    parent = await link.evaluate_handle('el => el.closest("div, li, article") || el.parentElement.parentElement')
   604	
   605	                    title = ""
   606	                    company = ""
   607	
   608	                    if parent:
   609	                        # Get all text content and try to parse
   610	                        try:
   611	                            full_text = await parent.inner_text()
   612	                            lines = [l.strip() for l in full_text.split('\n') if l.strip()]
   613	
   614	                            # Usually: name, then title/company info
   615	                            for line in lines:
   616	                                if line == name:
   617	                                    continue
   618	                                if '好友' in line or '关注' in line:
   619	                                    continue
   620	                                if not title and len(line) < 50:
   621	                                    title = line
   622	                                elif not company and len(line) < 50:
   623	                                    company = line
   624	                                    break
   625	                        except Exception:
   626	                            pass  # Best-effort text parsing; proceed with partial data
   627	
   628	                    profile_url = href if href.startswith('http') else f"https://maimai.cn{href}"
   629	
   630	                    candidates.append(CandidateBasic(
   631	                        maimai_id=maimai_id,
   632	                        name=name,
   633	                        title=title,
   634	                        company=company,
   635	                        profile_url=profile_url
   636	                    ))
   637	
   638	                except Exception as e:
   639	                    logger.debug(f"Failed to extract from link: {e}")
   640	                    continue
   641	
   642	        # Strategy 2: If no links found, try button-based detection
   643	        if not candidates:
   644	            logger.debug("Trying button-based detection")
   645	            buttons = await page.query_selector_all('button:has-text("好友"), button:has-text("关注")')
   646	
   647	            for button in buttons:
   648	                try:
   649	                    parent = await button.evaluate_handle('el => el.closest("div, li, article")')
   650	                    if not parent:
   651	                        continue
   652	
   653	                    # Find link in parent
   654	                    link = await parent.query_selector('a[href*="/contact"], a[href*="/u/"]')
   655	                    if link:
   656	                        href = await link.get_attribute('href')
   657	                        name = await link.inner_text()
   658	
   659	                        maimai_id = None
   660	                        if '/contact/detail/' in href:
   661	                            maimai_id = href.split('/contact/detail/')[-1].split('?')[0]
   662	                        elif '/u/' in href:
   663	                            maimai_id = href.split('/u/')[-1].split('?')[0]
   664	
   665	                        if maimai_id and name:
   666	                            candidates.append(CandidateBasic(
   667	                                maimai_id=maimai_id,
   668	                                name=name.strip(),
   669	                                title="",
   670	                                company="",
   671	                                profile_url=href
   672	                            ))
   673	                except Exception:
   674	                    continue
   675	
   676	        # Filter out premium placeholder profiles
   677	        before_filter = len(candidates)
   678	        candidates = [c for c in candidates if c.name not in PREMIUM_PLACEHOLDER_NAMES]
   679	        filtered_count = before_filter - len(candidates)
   680	        if filtered_count > 0:
   681	            logger.info(f"Filtered out {filtered_count} premium placeholder profiles")
   682	
   683	        logger.info(f"Extracted {len(candidates)} candidates from page")
   684	
   685	    except Exception as e:
   686	        logger.error(f"Error extracting search results: {e}")
   687	        console.print(f"[yellow]Warning: Search result extraction failed: {e}[/yellow]")
   688	
   689	    return candidates

   692	def _save_candidates_to_db(db_manager, candidates: List[CandidateBasic], source_search: str) -> None:
   693	    """
   694	    Save or update candidates in database.
   695	
   696	    Args:
   697	        db_manager: Database manager instance
   698	        candidates: List of candidates to save
   699	        source_search: Search keywords that found these candidates
   700	    """
   701	    try:
   702	        with db_manager.session_scope() as session:
   703	            # Batch pre-fetch existing candidates to avoid N+1 queries
   704	            maimai_ids = [c.maimai_id for c in candidates]
   705	            existing_by_id = {
   706	                row.maimai_id: row
   707	                for row in session.query(Candidate).filter(
   708	                    Candidate.maimai_id.in_(maimai_ids)
   709	                ).all()
   710	            }
   711	
   712	            # Also batch-fetch by name+company for dedup
   713	            name_company_pairs = [(c.name, c.company) for c in candidates]
   714	            from sqlalchemy import tuple_
   715	            existing_by_name = {}
   716	            if name_company_pairs:
   717	                for row in session.query(Candidate).filter(
   718	                    tuple_(Candidate.name, Candidate.company).in_(name_company_pairs)
   719	                ).all():
   720	                    existing_by_name[(row.name, row.company)] = row
   721	
   722	            for c in candidates:
   723	                existing = existing_by_id.get(c.maimai_id)
   724	                if not existing:
   725	                    existing = existing_by_name.get((c.name, c.company))
   726	
   727	                if existing:
   728	                    existing.name = c.name
   729	                    existing.title = c.title
   730	                    existing.company = c.company
   731	                    existing.location = c.location
   732	                    existing.profile_url = c.profile_url
   733	                else:
   734	                    candidate = Candidate(
   735	                        maimai_id=c.maimai_id,
   736	                        name=c.name,
   737	                        title=c.title,
   738	                        company=c.company,
   739	                        location=c.location,
   740	                        profile_url=c.profile_url,
   741	                        source_search=source_search
   742	                    )
   743	                    session.add(candidate)
   744	
   745	            logger.debug(f"Saved {len(candidates)} candidates to database")
   746	
   747	    except Exception as e:
   748	        logger.error(f"Failed to save candidates to database: {e}")
   749	        raise

   777	async def _click_avatar_for_profile_url(page: Page, name: str, company: str) -> Optional[str]:
   778	    """Try to extract profile URL for a candidate from the DOM contactcard pattern.
   779	
   780	    Searches the current page for a contactcard element whose parent card
   781	    contains the candidate's name (and optionally company), then constructs
   782	    the profile URL from the extracted user ID.
   783	
   784	    Args:
   785	        page: Playwright page on the search results
   786	        name: Candidate name to match
   787	        company: Candidate company to help disambiguate
   788	
   789	    Returns:
   790	        Profile URL string, or None if not found
   791	    """
   792	    try:
   793	        user_id = await page.evaluate('''(args) => {
   794	            const { name, company } = args;
   795	            // Check contactcard_{numeric_id} ancestors
   796	            const ccElements = document.querySelectorAll('[id^="contactcard_"]');
   797	            for (const el of ccElements) {
   798	                const text = el.textContent || '';
   799	                if (!text.includes(name)) continue;
   800	                if (company && !text.includes(company)) continue;
   801	                const match = el.id.match(/^contactcard_(\\d+)$/);
   802	                if (match) return match[1];
   803	            }
   804	            // Check member_entry_{numeric_id} descendants
   805	            const meElements = document.querySelectorAll('[id^="member_entry_"]');
   806	            for (const el of meElements) {
   807	                // Walk up to find the card containing this member_entry
   808	                let card = el.parentElement;
   809	                for (let i = 0; i < 8 && card; i++) {
   810	                    const text = card.textContent || '';
   811	                    if (text.includes(name) && (!company || text.includes(company))) {
   812	                        const match = el.id.match(/^member_entry_(\\d+)$/);
   813	                        if (match) return match[1];
   814	                    }
   815	                    card = card.parentElement;
   816	                }
   817	            }
   818	            return null;
   819	        }''', {'name': name, 'company': company})
   820	
   821	        if user_id:
   822	            profile_url = f"https://maimai.cn/contact/detail/{user_id}?dstu={user_id}"
   823	            logger.info(f"Resolved profile_url for '{name}' via contactcard: {profile_url}")
   824	            return profile_url
   825	
   826	        logger.debug(f"No contactcard found for '{name}'")
   827	        return None
   828	
   829	    except Exception as e:
   830	        logger.debug(f"_click_avatar_for_profile_url error for '{name}': {e}")
   831	        return None

  1030	async def _extract_enterprise_results(page: Page) -> List[CandidateBasic]:
  1031	    """Extract candidate information from enterprise recruitment results page.
  1032	
  1033	    The enterprise UI has a different card layout from search_center.
  1034	    Uses JS-based extraction tailored to the enterprise talent cards.
  1035	
  1036	    Args:
  1037	        page: Playwright page instance on the enterprise results page
  1038	
  1039	    Returns:
  1040	        List of CandidateBasic objects
  1041	    """
  1042	    candidates = []
  1043	
  1044	    try:
  1045	        await page.wait_for_load_state("networkidle", timeout=10000)
  1046	    except Exception:
  1047	        pass  # Continue even if networkidle times out
  1048	
  1049	    try:
  1050	        candidates_data = await page.evaluate('''() => {
  1051	            const candidates = [];
  1052	            const seenIds = new Set();
  1053	
  1054	            // Enterprise talent cards typically have a structured layout
  1055	            // Try multiple selectors for the card containers
  1056	            const cardSelectors = [
  1057	                '[class*="talent-card"]',
  1058	                '[class*="candidate-card"]',
  1059	                '[class*="recommend-card"]',
  1060	                '[class*="talent-item"]',
  1061	                '[class*="card-wrap"]',
  1062	            ];
  1063	
  1064	            let cards = [];
  1065	            for (const sel of cardSelectors) {
  1066	                cards = document.querySelectorAll(sel);
  1067	                if (cards.length > 0) break;
  1068	            }
  1069	
  1070	            // Fallback: find cards by avatar image pattern (same as search_center)
  1071	            if (cards.length === 0) {
  1072	                const images = document.querySelectorAll('img');
  1073	                const foundCards = [];
  1074	
  1075	                for (const img of images) {
  1076	                    const rect = img.getBoundingClientRect();
  1077	                    if (rect.width > 100 || rect.height > 100) continue;
  1078	                    if (rect.width < 30 || rect.height < 30) continue;
  1079	
  1080	                    let card = img.parentElement;
  1081	                    for (let i = 0; i < 8 && card; i++) {
  1082	                        const cardRect = card.getBoundingClientRect();
  1083	                        if (cardRect.width > 200 && cardRect.height > 80) {
  1084	                            break;
  1085	                        }
  1086	                        card = card.parentElement;
  1087	                    }
  1088	                    if (card) foundCards.push(card);
  1089	                }
  1090	                cards = foundCards;
  1091	            }
  1092	
  1093	            for (const card of cards) {
  1094	                const textParts = [];
  1095	                const walker = document.createTreeWalker(card, NodeFilter.SHOW_TEXT, null, false);
