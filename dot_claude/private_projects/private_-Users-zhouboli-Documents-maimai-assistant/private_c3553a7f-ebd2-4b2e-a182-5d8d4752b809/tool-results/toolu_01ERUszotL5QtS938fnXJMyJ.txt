    53→    53→def _is_non_rd_title(title: str) -> bool:
    54→    54→    """Check if a title indicates a non-R&D role.
    55→    55→
    56→    56→    Strict mode: if any non-R&D keyword appears, filter out regardless of
    57→    57→    whether R&D keywords are also present (e.g. "技术产品经理" is filtered).
    58→    58→    Returns False (keep) only if title is empty or contains no non-R&D keywords.
    59→    59→    """
    60→    60→    if not title:
    61→    61→        return False
    62→    62→
    63→    63→    # Strict: any non-R&D keyword present → filter out (case-insensitive)
    64→    64→    title_lower = title.lower()
    65→    65→    if any(kw.lower() in title_lower for kw in NON_RD_KEYWORDS):
    66→    66→        return True
    67→    67→
    68→    68→    return False

    83→    83→async def search_candidates(
    84→    84→    keywords: str,
    85→    85→    company: Optional[tuple] = None,
    86→    86→    location: Optional[str] = None,
    87→    87→    pages: int = 1,
    88→    88→    start_page: int = 1,
    89→    89→    browser: Optional[BrowserManager] = None
    90→    90→) -> List[CandidateBasic]:
    91→    91→    """
    92→    92→    Search for candidates on Maimai.
    93→    93→
    94→    94→    Args:
    95→    95→        keywords: Search keywords (e.g., "高级前端工程师")
    96→    96→        company: Filter by company names, tuple of strings (optional)
    97→    97→        location: Filter by location (optional)
    98→    98→        pages: Number of pages to scrape (default 1)
    99→    99→        start_page: First page number to scrape (default 1)
   100→   100→        browser: Browser manager instance
   101→   101→
   102→   102→    Returns:
   103→   103→        List of CandidateBasic objects
   104→   104→
   105→   105→    Raises:
   106→   106→        RateLimitExceededError: If rate limit exceeded
   107→   107→        SelectorNotFoundError: If page elements not found
   108→   108→    """
   109→   109→    # Validate inputs
   110→   110→    validate_keywords(keywords)
   111→   111→
   112→   112→    if browser is None:
   113→   113→        browser = get_browser_manager()
   114→   114→
   115→   115→    config = get_config()
   116→   116→    rate_limiter = get_rate_limiter()
   117→   117→    db_manager = get_db_manager()
   118→   118→
   119→   119→    # Ensure logged in
   120→   120→    await ensure_logged_in(browser)
   121→   121→
   122→   122→    all_candidates: List[CandidateBasic] = []
   123→   123→    seen_ids: set = set()  # Deduplicate across infinite-scroll extractions
   124→   124→
   125→   125→    with Progress(
   126→   126→        SpinnerColumn(),
   127→   127→        TextColumn("[progress.description]{task.description}"),
   128→   128→        console=console
   129→   129→    ) as progress:
   130→   130→
   131→   131→        end_page = start_page + pages
   132→   132→        for page_num in range(start_page, end_page):
   133→   133→            task = progress.add_task(f"Searching page {page_num}/{end_page - 1}...", total=None)
   134→   134→
   135→   135→            try:
   136→   136→                # Check rate limit (waits through min-interval)
   137→   137→                await async_check_rate_limit('search')
   138→   138→
   139→   139→                page = browser.page
   140→   140→
   141→   141→                if page_num == start_page:
   142→   142→                    # First page: interactive search (maimai SPA ignores URL query params)
   143→   143→                    await retry_navigation(page, MAIMAI_SEARCH_URL, wait_until='domcontentloaded')
   144→   144→                    await asyncio.sleep(1)
   145→   145→
   146→   146→                    # Find search input and type keyword
   147→   147→                    search_input = (
   148→   148→                        await page.query_selector('input[placeholder="搜索"]')
   149→   149→                        or await page.query_selector('input[type="search"]')
   150→   150→                        or await page.query_selector('.search-input input, .header input[type="text"]')
   151→   151→                    )
   152→   152→                    if not search_input:
   153→   153→                        raise SelectorNotFoundError("Search input not found on search center page")
   154→   154→
   155→   155→                    await search_input.click()
   156→   156→                    await search_input.fill(keywords)
   157→   157→                    await asyncio.sleep(0.5)
   158→   158→                    await page.keyboard.press('Enter')
   159→   159→                    await asyncio.sleep(2)
   160→   160→
   161→   161→                    # Default tab is 实名动态(feed); switch to 人脉(contacts)
   162→   162→                    renmai_tab = await page.query_selector(':text-is("人脉")')
   163→   163→                    if renmai_tab:
   164→   164→                        await renmai_tab.click()
   165→   165→                        await asyncio.sleep(2)
   166→   166→                    else:
   167→   167→                        logger.warning("Could not find '人脉' tab on search results page")
   168→   168→                else:
   169→   169→                    # Subsequent pages: scroll to load more (infinite scroll)
   170→   170→                    prev_count = await page.evaluate(
   171→   171→                        'document.querySelectorAll(\'[id^="contactcard_"]\').length'
   172→   172→                    )
   173→   173→                    await human_scroll(page, 'down', distance=1500)
   174→   174→                    # Wait for new cards to load
   175→   175→                    try:
   176→   176→                        await page.wait_for_function(
   177→   177→                            f'''() => {{
   178→   178→                                const cards = document.querySelectorAll('[id^="contactcard_"]');
   179→   179→                                return cards.length > {prev_count};
   180→   180→                            }}''',
   181→   181→                            timeout=8000,
   182→   182→                        )
   183→   183→                    except TimeoutError:
   184→   184→                        logger.debug(f"No new cards after scroll (had {prev_count})")
   185→   185→                    except Exception as e:
   186→   186→                        logger.warning(f"Error waiting for new cards after scroll: {e}")
   187→   187→
   188→   188→                # Wait for contactcard elements
   189→   189→                try:
   190→   190→                    await page.wait_for_function(
   191→   191→                        '''() => {
   192→   192→                            const cards = document.querySelectorAll('[id^="contactcard_"]');
   193→   193→                            return cards.length >= 1;
   194→   194→                        }''',
   195→   195→                        timeout=8000,
   196→   196→                    )
   197→   197→                except TimeoutError:
   198→   198→                    logger.debug("Timed out waiting for contactcard elements, proceeding anyway")
   199→   199→                except Exception as e:
   200→   200→                    logger.warning(f"Error waiting for contactcard elements: {e}")
   201→   201→
   202→   202→                # Extract candidates from this page
   203→   203→                candidates = await _extract_search_results(page, config)
   204→   204→
   205→   205→                # Deduplicate: infinite scroll re-extracts all visible cards
   206→   206→                candidates = [c for c in candidates if c.maimai_id not in seen_ids]
   207→   207→                for c in candidates:
   208→   208→                    seen_ids.add(c.maimai_id)
   209→   209→
   210→   210→                # Anti-detection: human behavior AFTER extraction
   211→   211→                await check_captcha(page, report_to_rate_limiter=True)
   212→   212→                await human_scroll(page, 'down', distance=300)
   213→   213→                await async_random_delay('action')
   214→   214→
   215→   215→                if not candidates:
   216→   216→                    logger.info(f"No results found on page {page_num}")
   217→   217→                    await browser.screenshot(f"search_debug_page{page_num}")
   218→   218→                    progress.update(task, description=f"Page {page_num}: No results")
   219→   219→                    # Allow one empty page (could be transient), break on second consecutive
   220→   220→                    if page_num > start_page and not all_candidates:
   221→   221→                        # First page had no results either — genuinely empty
   222→   222→                        break
   223→   223→                    continue
   224→   224→
   225→   225→                # Filter out non-R&D roles
   226→   226→                before_rd_filter = len(candidates)
   227→   227→                candidates = [c for c in candidates if not _is_non_rd_title(c.title)]
   228→   228→                rd_filtered = before_rd_filter - len(candidates)
   229→   229→                if rd_filtered > 0:
   230→   230→                    logger.info(f"Filtered out {rd_filtered} non-R&D candidates")
   231→   231→
   232→   232→                # Apply filters if specified
   233→   233→                if company:
   234→   234→                    company_lower = [cn.lower() for cn in company]
   235→   235→                    candidates = [c for c in candidates if c.company and any(cn in c.company.lower() for cn in company_lower)]
   236→   236→
   237→   237→                if location:
   238→   238→                    candidates = [c for c in candidates if c.location and location.lower() in c.location.lower()]
   239→   239→
   240→   240→                all_candidates.extend(candidates)
   241→   241→                progress.update(task, description=f"Page {page_num}: Found {len(candidates)} candidates")
   242→   242→
   243→   243→                # Log the action
   244→   244→                rate_limiter.log_action('search', {
   245→   245→                    'keywords': keywords,
   246→   246→                    'page': page_num,
   247→   247→                    'results': len(candidates)
   248→   248→                })
   249→   249→
   250→   250→                # Save candidates to database
   251→   251→                try:
   252→   252→                    _save_candidates_to_db(db_manager, candidates, keywords)
   253→   253→                except Exception as e:
   254→   254→                    console.print(f"[yellow]Warning: Failed to save candidates to DB: {e}[/yellow]")
   255→   255→
   256→   256→                # Delay before next page
   257→   257→                if page_num < end_page - 1:
   258→   258→                    await async_random_delay('page_load')
   259→   259→
   260→   260→            except (BrowserError, SessionExpiredError):
   261→   261→                raise  # Critical errors must propagate
   262→   262→            except Exception as e:
   263→   263→                logger.error(f"Error on page {page_num}: {e}")
   264→   264→                progress.update(task, description=f"Page {page_num}: Error - {str(e)[:30]}")
   265→   265→                # Take screenshot for debugging
   266→   266→                await browser.screenshot(f"search_error_page{page_num}")
   267→   267→                continue
   268→   268→
   269→   269→    # Log search to history
   270→   270→    _log_search_history(db_manager, keywords, company, location, pages, len(all_candidates))
   271→   271→
   272→   272→    console.print(f"[green]Search complete. Found {len(all_candidates)} candidates.[/green]")
   273→   273→    return all_candidates

   276→   276→async def _extract_search_results(page: Page, config) -> List[CandidateBasic]:
   277→   277→    """
   278→   278→    Extract candidate information from search results page.
   279→   279→
   280→   280→    Args:
   281→   281→        page: Playwright page instance
   282→   282→        config: Config instance
   283→   283→
   284→   284→    Returns:
   285→   285→        List of CandidateBasic objects
   286→   286→    """
   287→   287→    candidates = []
   288→   288→
   289→   289→    # NOTE: The caller (search_candidates) already waits for contactcard_
   290→   290→    # elements before calling this function.  Maimai's SPA strips URL
   291→   291→    # params within ~5-8s and clears results, so we must NOT add long
   292→   292→    # waits here (no networkidle, no loading-text polling).
   293→   293→
   294→   294→    try:
   295→   295→        # Strategy: Extract candidates by finding avatar images and their associated text
   296→   296→        candidates_data = await page.evaluate('''() => {
   297→   297→            const candidates = [];
   298→   298→            const seenNames = new Set();
   299→   299→
   300→   300→            // Find all images that look like avatars
   301→   301→            const images = document.querySelectorAll('img');
   302→   302→
   303→   303→            for (const img of images) {
   304→   304→                // Skip large images (probably not avatars)
   305→   305→                const rect = img.getBoundingClientRect();
   306→   306→                if (rect.width > 100 || rect.height > 100) continue;
   307→   307→                if (rect.width < 20 || rect.height < 20) continue;
   308→   308→
   309→   309→                // Find parent container (the card)
   310→   310→                let card = img.parentElement;
   311→   311→                for (let i = 0; i < 8 && card; i++) {
   312→   312→                    const cardRect = card.getBoundingClientRect();
   313→   313→                    // Look for a reasonable card size
   314→   314→                    if (cardRect.width > 200 && cardRect.height > 50 && cardRect.height < 200) {
   315→   315→                        break;
   316→   316→                    }
   317→   317→                    card = card.parentElement;
   318→   318→                }
   319→   319→
   320→   320→                if (!card) continue;
   321→   321→
   322→   322→                // Get all text nodes from card, preserving order
   323→   323→                const textParts = [];
   324→   324→                const walker = document.createTreeWalker(card, NodeFilter.SHOW_TEXT, null, false);
   325→   325→                while (walker.nextNode()) {
   326→   326→                    const text = walker.currentNode.textContent.trim();
   327→   327→                    if (text && text.length > 0 && text.length < 100) {
   328→   328→                        textParts.push(text);
   329→   329→                    }
   330→   330→                }
   331→   331→
   332→   332→                // Parse candidate info from text parts
   333→   333→                // Pattern: Name, Activity time, Company+Position, Industry|Department
   334→   334→                let name = '';
   335→   335→                let company = '';
   336→   336→                let title = '';
   337→   337→                let industry = '';
   338→   338→
   339→   339→                // Compound job title patterns (longer patterns first for greedy match)
   340→   340→                const jobPatterns = [
   341→   341→                    // Compound titles with prefix modifiers
   342→   342→                    /(?:高级|资深|首席|初级|中级|senior|junior|staff|principal|lead)?\s*(?:前端|后端|全栈|web|Web|iOS|Android|Java|Python|Go|C\+\+|PHP|.NET|算法|数据|AI|机器学习|深度学习|NLP|大数据|云计算|DevOps|SRE|安全|测试|嵌入式|移动端|客户端|服务端|区块链|具身智能|自动驾驶|机器人)?\s*(?:工程师|开发|架构师|专家|技术经理|技术总监|技术负责人|Tech\s*Lead)/i,
   343→   343→                    // Management titles
   344→   344→                    /(?:技术|产品|运营|市场|销售|人力|财务|行政|项目|研发)?\s*(?:总监|经理|主管|负责人|VP|总裁|副总|合伙人|创始人|董事|CEO|CTO|COO|CFO|CMO|CHO|CIO)/i,
   345→   345→                    // Other roles
   346→   346→                    /(?:产品|UI|UX|交互|视觉)?\s*(?:设计师|分析师|顾问|运营|助理|实习生?)/i,
   347→   347→                ];
   348→   348→
   349→   349→                // Patterns for text that should be skipped
   350→   350→                const skipPatterns = [
   351→   351→                    /^[+＋]\s*好友$/,          // "+ 好友" button
   352→   352→                    /^\d+个共同好友$/,           // "1个共同好友" social label
   353→   353→                    /^共同好友/,                 // "共同好友..." variants
   354→   354→                    /^已关注$/,                  // "已关注" button
   355→   355→                    /^添加好友$/,                // "添加好友" button
   356→   356→                    /^关注$/,                    // "关注" button
   357→   357→                    /^好友$/,                    // "好友" label
   358→   358→                    /^[|｜·]$/,                  // separators
   359→   359→                ];
   360→   360→
   361→   361→                const isSkippable = (text) => {
   362→   362→                    if (skipPatterns.some(p => p.test(text))) return true;
   363→   363→                    // Activity time
   364→   364→                    if (/天前|小时前|分钟前|刚刚|活跃|在线/.test(text)) return true;
   365→   365→                    // Social relationship labels
   366→   366→                    if (/同行|前同事|校友|共同好友/.test(text)) return true;
   367→   367→                    return false;
   368→   368→                };
   369→   369→
   370→   370→                for (let i = 0; i < textParts.length; i++) {
   371→   371→                    const part = textParts[i];
   372→   372→
   373→   373→                    if (isSkippable(part)) continue;
   374→   374→
   375→   375→                    // First short text (2-15 chars, pure CJK/letters) is likely the name
   376→   376→                    if (!name && part.length >= 2 && part.length <= 15) {
   377→   377→                        if (/^[\u4e00-\u9fa5a-zA-Z·\s()（）]+$/.test(part)) {
   378→   378→                            name = part.replace(/[（(][^)）]*[）)]/g, '').trim();
   379→   379→                            continue;
   380→   380→                        }
   381→   381→                    }
   382→   382→
   383→   383→                    // After name, next substantial text is company+position
   384→   384→                    if (name && !company && part.length > 2) {
   385→   385→                        // Try to split company and position using regex patterns
   386→   386→                        let splitDone = false;
   387→   387→                        for (const pattern of jobPatterns) {
   388→   388→                            const match = part.match(pattern);
   389→   389→                            if (match && match.index > 0) {
   390→   390→                                company = part.substring(0, match.index).trim();
   391→   391→                                title = match[0].trim();
   392→   392→                                splitDone = true;
   393→   393→                                break;
   394→   394→                            }
   395→   395→                        }
   396→   396→                        if (!splitDone) {
   397→   397→                            company = part;
   398→   398→                        }
   399→   399→                        continue;
   400→   400→                    }
   401→   401→
   402→   402→                    // Next part is industry|department info
   403→   403→                    if (name && company && !industry && part.length > 2) {
   404→   404→                        industry = part;
   405→   405→                        // If title wasn't split from company, use industry as fallback context
   406→   406→                        if (!title) {
   407→   407→                            title = industry;
   408→   408→                        }
   409→   409→                        break;
   410→   410→                    }
   411→   411→                }
   412→   412→
   413→   413→                // Skip if no name found or premium placeholder
   414→   414→                if (!name) continue;
   415→   415→                if (name === '脉脉高价值人脉') continue;
   416→   416→
   417→   417→                // Extract user ID from DOM patterns on card and ancestor elements.
   418→   418→                // Two patterns exist:
   419→   419→                //   1. contactcard_{numeric_id} — on ancestors (colleagues layout)
   420→   420→                //   2. member_entry_{numeric_id} — on descendants (standard layout)
   421→   421→                let userId = '';
   422→   422→
   423→   423→                // Check ancestors for numeric contactcard_{id}
   424→   424→                let el = card;
   425→   425→                while (el) {
   426→   426→                    const elId = el.id || '';
   427→   427→                    const ccMatch = elId.match(/^contactcard_(\d+)$/);
   428→   428→                    if (ccMatch) {
   429→   429→                        userId = ccMatch[1];
   430→   430→                        break;
   431→   431→                    }
   432→   432→                    el = el.parentElement;
   433→   433→                }
   434→   434→
   435→   435→                // Check descendants for member_entry_{id} (most common pattern)
   436→   436→                if (!userId) {
   437→   437→                    const meEl = card.querySelector('[id^="member_entry_"]');
   438→   438→                    if (meEl) {
   439→   439→                        const meMatch = meEl.id.match(/^member_entry_(\d+)$/);
   440→   440→                        if (meMatch) {
   441→   441→                            userId = meMatch[1];
   442→   442→                        }
   443→   443→                    }
   444→   444→                }
   445→   445→
   446→   446→                // Also check ancestor descendants for contactcard with numeric ID
   447→   447→                if (!userId) {
   448→   448→                    const ccEl = card.querySelector('[id^="contactcard_"]');
   449→   449→                    if (ccEl) {
   450→   450→                        const ccMatch = ccEl.id.match(/^contactcard_(\d+)$/);
   451→   451→                        if (ccMatch) {
   452→   452→                            userId = ccMatch[1];
   453→   453→                        }

   693→   693→def _save_candidates_to_db(db_manager, candidates: List[CandidateBasic], source_search: str) -> None:
   694→   694→    """
   695→   695→    Save or update candidates in database.
   696→   696→
   697→   697→    Args:
   698→   698→        db_manager: Database manager instance
   699→   699→        candidates: List of candidates to save
   700→   700→        source_search: Search keywords that found these candidates
   701→   701→    """
   702→   702→    try:
   703→   703→        with db_manager.session_scope() as session:
   704→   704→            # Batch pre-fetch existing candidates to avoid N+1 queries
   705→   705→            maimai_ids = [c.maimai_id for c in candidates]
   706→   706→            existing_by_id = {
   707→   707→                row.maimai_id: row

   709→   709→                    Candidate.maimai_id.in_(maimai_ids)

   711→   711→            }
   712→   712→
   713→   713→            # Also batch-fetch by name+company for dedup
   714→   714→            name_company_pairs = [(c.name, c.company) for c in candidates]
   715→   715→            from sqlalchemy import tuple_
   716→   716→            existing_by_name = {}
   717→   717→            if name_company_pairs:
   718→   718→                for row in session.query(Candidate).filter(
   719→   719→                    tuple_(Candidate.name, Candidate.company).in_(name_company_pairs)
   720→   720→                ).all():
   721→   721→                    existing_by_name[(row.name, row.company)] = row
   722→   722→
   723→   723→            for c in candidates:
   724→   724→                existing = existing_by_id.get(c.maimai_id)
   725→   725→                if not existing:
   726→   726→                    existing = existing_by_name.get((c.name, c.company))
   727→   727→
   728→   728→                if existing:
   729→   729→                    existing.name = c.name
   730→   730→                    existing.title = c.title
   731→   731→                    existing.company = c.company
   732→   732→                    existing.location = c.location
   733→   733→                    existing.profile_url = c.profile_url
   734→   734→                else:
   735→   735→                    candidate = Candidate(
   736→   736→                        maimai_id=c.maimai_id,
   737→   737→                        name=c.name,
   738→   738→                        title=c.title,
   739→   739→                        company=c.company,
   740→   740→                        location=c.location,
   741→   741→                        profile_url=c.profile_url,
   742→   742→                        source_search=source_search

   778→   778→async def _click_avatar_for_profile_url(page: Page, name: str, company: str) -> Optional[str]:
   779→   779→    """Try to extract profile URL for a candidate from the DOM contactcard pattern.
   780→   780→
   781→   781→    Searches the current page for a contactcard element whose parent card
   782→   782→    contains the candidate's name (and optionally company), then constructs
   783→   783→    the profile URL from the extracted user ID.
   784→   784→
   785→   785→    Args:
   786→   786→        page: Playwright page on the search results
   787→   787→        name: Candidate name to match
   788→   788→        company: Candidate company to help disambiguate
   789→   789→
   790→   790→    Returns:
   791→   791→        Profile URL string, or None if not found
   792→   792→    """
   793→   793→    try:
   794→   794→        user_id = await page.evaluate('''(args) => {
   795→   795→            const { name, company } = args;
   796→   796→            // Check contactcard_{numeric_id} ancestors
   797→   797→            const ccElements = document.querySelectorAll('[id^="contactcard_"]');
   798→   798→            for (const el of ccElements) {
   799→   799→                const text = el.textContent || '';
   800→   800→                if (!text.includes(name)) continue;
   801→   801→                if (company && !text.includes(company)) continue;
   802→   802→                const match = el.id.match(/^contactcard_(\\d+)$/);
   803→   803→                if (match) return match[1];
   804→   804→            }
   805→   805→            // Check member_entry_{numeric_id} descendants
   806→   806→            const meElements = document.querySelectorAll('[id^="member_entry_"]');
   807→   807→            for (const el of meElements) {
   808→   808→                // Walk up to find the card containing this member_entry
   809→   809→                let card = el.parentElement;
   810→   810→                for (let i = 0; i < 8 && card; i++) {
   811→   811→                    const text = card.textContent || '';
   812→   812→                    if (text.includes(name) && (!company || text.includes(company))) {
   813→   813→                        const match = el.id.match(/^member_entry_(\\d+)$/);
   814→   814→                        if (match) return match[1];
   815→   815→                    }
   816→   816→                    card = card.parentElement;
   817→   817→                }
   818→   818→            }
   819→   819→            return null;
   820→   820→        }''', {'name': name, 'company': company})
   821→   821→
   822→   822→        if user_id:
   823→   823→            profile_url = f"https://maimai.cn/contact/detail/{user_id}?dstu={user_id}"
   824→   824→            logger.info(f"Resolved profile_url for '{name}' via contactcard: {profile_url}")
   825→   825→            return profile_url
   826→   826→
   827→   827→        logger.debug(f"No contactcard found for '{name}'")
   828→   828→        return None
   829→   829→
   830→   830→    except Exception as e:
   831→   831→        logger.debug(f"_click_avatar_for_profile_url error for '{name}': {e}")
   832→   832→        return None
