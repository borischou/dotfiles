    74→def _extract_user_id(profile_url: str) -> Optional[str]:
    75→    """Extract dstu/user_id from a Maimai profile URL for direct chat navigation.
    76→
    77→    Only returns IDs that are known to work with the chat URL (u2= parameter).
    78→    Query-param IDs (dstu, u2, egid) are reliable for chat navigation.
    79→    Path-based IDs (/contact/detail/xxx) are NOT the same as chat user IDs
    80→    and should not be used for direct chat URL construction.
    81→    """
    82→    from urllib.parse import urlparse, parse_qs
    83→
    84→    parsed = urlparse(profile_url)
    85→    params = parse_qs(parsed.query)
    86→    # Only use query parameter IDs — these are confirmed to work with uh_chat_pc
    87→    for key in ("dstu", "u2", "egid"):
    88→        if key in params:
    89→            return params[key][0]
    90→    return None

    93→async def _navigate_to_chat(
    94→    page: Page,
    95→    profile_url: Optional[str],
    96→    config,
    97→    browser: BrowserManager,
    98→    maimai_id: str,
    99→    interactive: bool = False,
   100→    discovered_selectors=None,
   101→) -> bool:
   102→    """Navigate to the chat page for a candidate.
   103→
   104→    Tries in order:
   105→    1. Extract user_id from profile_url query params → direct chat URL
   106→    2. Navigate to profile_url, extract user_id from redirected URL → direct chat URL
   107→    3. Click chat button on profile page as fallback
   108→
   109→    If no profile_url is provided, looks up the candidate's profile_url from DB.
   110→
   111→    Args:
   112→        page: Playwright page instance
   113→        profile_url: Candidate's profile URL (may be None)
   114→        config: Config instance
   115→        browser: Browser manager instance
   116→        maimai_id: Candidate's Maimai ID
   117→        interactive: Enable interactive selector discovery
   118→        discovered_selectors: Shared DiscoveredSelectors instance
   119→
   120→    Returns:
   121→        True if successfully navigated to chat page
   122→    """
   123→    db_manager = get_db_manager()
   124→
   125→    # Resolve profile URL from DB if not provided
   126→    if not profile_url:
   127→        with db_manager.session_scope() as session:
   128→            db_candidate = session.query(Candidate).filter_by(maimai_id=maimai_id).first()
   129→            profile_url = db_candidate.profile_url if db_candidate else None
   130→
   131→        if not profile_url:
   132→            raise PermanentFailureError(f"No profile URL for {maimai_id}")
   133→
   134→    # Try extracting user_id directly from profile URL
   135→    user_id = _extract_user_id(profile_url)
   136→    logger.debug(f"_navigate_to_chat: profile_url={profile_url!r}, extracted user_id={user_id}")
   137→
   138→    if user_id:
   139→        chat_url = MAIMAI_CHAT_URL.format(user_id=user_id)
   140→        await retry_navigation(page, chat_url, wait_until="domcontentloaded")
   141→        await async_random_delay("page_load")
   142→        return True
   143→
   144→    # Navigate to profile page, then try to extract user_id from resulting URL
   145→    logger.debug(f"_navigate_to_chat: navigating to profile {profile_url}")
   146→    await retry_navigation(page, profile_url, wait_until="domcontentloaded")
   147→    await async_random_delay("page_load")
   148→
   149→    current_url = page.url
   150→    logger.debug(f"_navigate_to_chat: landed on {current_url}")
   151→    redirected_user_id = _extract_user_id(current_url)
   152→    if redirected_user_id:
   153→        logger.info(f"Extracted user_id={redirected_user_id} from page URL, redirecting to chat")
   154→        chat_url = MAIMAI_CHAT_URL.format(user_id=redirected_user_id)
   155→        await retry_navigation(page, chat_url, wait_until="domcontentloaded")
   156→        await async_random_delay("page_load")
   157→        return True
   158→
   159→    # Fallback: click chat button on profile page
   160→    chat_button_found = await _click_chat_button(page, config)
   161→    if chat_button_found:
   162→        await async_random_delay("action")
   163→        return True
   164→
   165→    # If interactive mode, trigger selector discovery
   166→    if interactive and discovered_selectors is not None and not discovered_selectors.is_populated:
   167→        from src.modules.selector_discovery import discover_chat_page
   168→
   169→        console.print(
   170→            "[yellow]Chat button not found. Entering interactive discovery mode...[/yellow]"
   171→        )
   172→        result = await discover_chat_page(page, maimai_id, browser)
   173→        if result.is_populated:
   174→            discovered_selectors.chat_button = result.chat_button
   175→            discovered_selectors.message_input = result.message_input
   176→            discovered_selectors.send_button = result.send_button
   177→            discovered_selectors.chat_url_pattern = result.chat_url_pattern
   178→            discovered_selectors.skip_chat_button = result.skip_chat_button
   179→            discovered_selectors.is_populated = True
   180→            return True
   181→
   182→    console.print("[red]Could not find chat button[/red]")
   183→    await browser.screenshot(f"message_error_{maimai_id}")
   184→    return False

   187→async def extract_contact_quota(page: Page) -> Optional[dict]:
   188→    """Extract contact quota from Quick Contact page's embedded share_data.
   189→
   190→    The Quick Contact page server-renders ``window.share_data`` which includes
   191→    ``mem_status.mem_info`` with remaining quotas for various member actions.
   192→
   193→    Args:
   194→        page: Playwright page on a Quick Contact (极速联系) page
   195→
   196→    Returns:
   197→        Dict with quota info, or None if data is unavailable::
   198→
   199→            {
   200→                'uh_left': int,        # 极速联系 remaining
   201→                'addfr_left': int,     # 加好友 remaining
   202→                'dc_left': int,        # 直聊 remaining
   203→                'expire_time': str,    # membership expiry
   204→                'cost': int,           # vouchers consumed per contact
   205→            }
   206→    """
   207→    try:
   208→        quota = await page.evaluate(
   209→            """() => {
   210→            if (window.share_data && window.share_data.mem_status &&
   211→                window.share_data.mem_status.mem_info) {
   212→                const info = window.share_data.mem_status.mem_info;
   213→                return {
   214→                    uh_left: info.uh_left,
   215→                    addfr_left: info.addfr_left,
   216→                    dc_left: info.dc_left,
   217→                    expire_time: info.expire_time || null,
   218→                    cost: (window.share_data.fast_contact_pay_num || {}).final_num || 1,
   219→                };
   220→            }
   221→            return null;
   222→        }"""
   223→        )
   224→        return quota
   225→    except Exception as e:
   226→        logger.debug(f"Failed to extract contact quota: {e}")
   227→        return None

   272→async def _handle_quick_contact(page: Page, content: str) -> Optional[bool]:
   273→    """Handle Maimai's "极速联系" (Quick Contact) page.
   274→
   275→    This page appears instead of regular chat for some candidates. It has a
   276→    category selector, a textarea, and a "提交" (Submit) button.
   277→
   278→    Submission works by manipulating the React component directly: setting
   279→    the message via setState and calling the debounced send method (with
   280→    flush) to trigger the POST to /bizmember/uh-chat/api/send_v2.
   281→
   282→    DOM-level textarea.fill() does NOT update React's controlled state, and
   283→    DOM click on the submit button does NOT reliably trigger the send API.
   284→
   285→    Args:
   286→        page: Playwright page instance
   287→        content: Message content to send
   288→
   289→    Returns:
   290→        True if message was sent via quick contact form,
   291→        False if quick contact page was detected but sending failed,
   292→        None if this is not a quick contact page (caller should continue normal flow)
   293→    """
   294→    try:
   295→        # Detect quick contact page via page title or page content
   296→        is_quick_contact = await page.evaluate(
   297→            """() => {
   298→            const title = document.title || '';
   299→            const bodyText = document.body ? document.body.innerText : '';
   300→            return title.includes('极速联系') || bodyText.includes('极速联系');
   301→        }"""
   302→        )
   303→
   304→        if not is_quick_contact:
   305→            return None
   306→
   307→        logger.info("Detected '极速联系' (Quick Contact) page")
   308→
   309→        # Validate message length (Quick Contact enforces 10-300 chars)
   310→        if len(content) < 10:
   311→            logger.error(
   312→                f"Quick contact: message too short ({len(content)} chars, "
   313→                "minimum 10). Quick Contact requires 10-300 characters."
   314→            )
   315→            return False
   316→        if len(content) > 300:
   317→            logger.error(
   318→                f"Quick contact: message too long ({len(content)} chars, "
   319→                "maximum 300). Quick Contact requires 10-300 characters."
   320→            )
   321→            return False
   322→
   323→        # Extract quota info before submitting (share_data is available on page load)
   324→        quota = await extract_contact_quota(page)
   325→
   326→        # Check if quota is exhausted
   327→        if quota and quota.get("uh_left", 0) <= 0:
   328→            raise QuotaExhaustedError(remaining=quota.get("uh_left", 0))
   329→
   330→        # Install XHR interceptor to capture the send_v2 response
   331→        await page.evaluate(
   332→            """() => {
   333→            window.__sendV2Result = null;
   334→            const origOpen = XMLHttpRequest.prototype.open;
   335→            const origSend = XMLHttpRequest.prototype.send;
   336→            XMLHttpRequest.prototype.open = function(m, u) {
   337→                this.__url = u; this.__method = m;
   338→                return origOpen.apply(this, arguments);
   339→            };
   340→            XMLHttpRequest.prototype.send = function(b) {
   341→                if (this.__url && this.__url.includes('send_v2')) {
   342→                    const xhr = this;
   343→                    xhr.addEventListener('load', function() {
   344→                        try {
   345→                            window.__sendV2Result = {
   346→                                status: xhr.status,
   347→                                response: xhr.responseText,
   348→                            };
   349→                        } catch(e) {
   350→                            window.__sendV2Result = {status: xhr.status, error: e.message};
   351→                        }
   352→                    });
   353→                    xhr.addEventListener('error', function() {
   354→                        window.__sendV2Result = {status: 0, error: 'network_error'};
   355→                    });
   356→                }
   357→                return origSend.apply(this, arguments);
   358→            };
   359→        }"""
   360→        )
   361→
   362→        # Submit via React component internals:
   363→        # 1. Find the class component that owns the form state
   364→        # 2. Set message + unlock push_type via setState
   365→        # 3. Call the debounced send() method and flush() to execute immediately
   366→        send_triggered = await page.evaluate(
   367→            """(msg) => {
   368→            const textarea = document.querySelector('textarea');
   369→            if (!textarea) return {success: false, error: 'no textarea'};
   370→            const keys = Object.keys(textarea);
   371→            const fiberKey = keys.find(
   372→                k => k.startsWith('__reactInternalInstance') || k.startsWith('__reactFiber')
   373→            );
   374→            if (!fiberKey) return {success: false, error: 'no React fiber'};
   375→
   376→            let fiber = textarea[fiberKey];
   377→            while (fiber) {
   378→                if (fiber.stateNode && fiber.stateNode.state
   379→                    && 'message' in fiber.stateNode.state
   380→                    && typeof fiber.stateNode.send === 'function') {
   381→                    const component = fiber.stateNode;
   382→                    // Set message and ensure push_type is not locked (true = sending lock)
   383→                    component.setState({message: msg, push_type: false}, () => {
   384→                        component.send();
   385→                        if (typeof component.send.flush === 'function') {
   386→                            component.send.flush();
   387→                        }
   388→                    });
   389→                    return {success: true};
   390→                }
   391→                fiber = fiber.return;
   392→            }
   393→            return {success: false, error: 'React component not found'};
   394→        }""",
   395→            content,
   396→        )
   397→
   398→        if not send_triggered.get("success"):
   399→            logger.warning(
   400→                f"Quick contact: failed to trigger send via React: "
   401→                f"{send_triggered.get('error', 'unknown')}"
   402→            )
   403→            return False
   404→
   405→        # Wait for the API response (send_v2 XHR)
   406→        send_result = None
   407→        for _ in range(10):
   408→            await asyncio.sleep(0.5)
   409→            send_result = await page.evaluate("() => window.__sendV2Result")
   410→            if send_result is not None:
   411→                break
   412→
   413→        if send_result is None:
   414→            logger.warning("Quick contact: send_v2 API call was not captured (timeout)")
   415→            return False
   416→
   417→        # Parse the API response
   418→        response_text = send_result.get("response", "")
   419→        try:
   420→            resp_data = json.loads(response_text) if response_text else {}
   421→        except (json.JSONDecodeError, TypeError):
   422→            resp_data = {}
   423→
   424→        ret = resp_data.get("ret")
   425→        error_msg = resp_data.get("error", "")
   426→
   427→        if ret == -1:
   428→            logger.warning(f"Quick contact: send failed — {error_msg}")
   429→            if error_msg:
   430→                console.print(f"  [red]Quick Contact 失败: {error_msg}[/red]")
   431→            return False
   432→
   433→        logger.info("Quick contact: message sent successfully")
   434→        await async_random_delay("action")
   435→
   436→        # Display remaining quota
   437→        if quota:
   438→            _display_quota(quota, just_sent=True)
   439→
   440→        return True
   441→
   442→    except PermanentFailureError:
   443→        raise  # Let quota exhaustion etc. propagate to abort the batch
   444→    except Exception as e:
   445→        logger.error(f"Quick contact handling failed: {e}\n{traceback.format_exc()}")
   446→        return False

   449→async def _detect_vip_paywall(page: Page) -> bool:
   450→    """Detect if a VIP/membership paywall is blocking the current page.
   451→
   452→    Non-VIP accounts may be redirected to a recharge/upgrade page when
   453→    trying to message certain candidates. This checks the page URL and
   454→    visible text for common paywall indicators.
   455→
   456→    Args:
   457→        page: Playwright page instance
   458→
   459→    Returns:
   460→        True if a VIP paywall is detected
   461→    """
   462→    try:
   463→        return await page.evaluate(
   464→            """() => {
   465→            const url = window.location.href;
   466→            const urlKeywords = ['vip', 'member', 'pay', 'recharge', 'upgrade', 'subscribe'];
   467→            if (urlKeywords.some(kw => url.toLowerCase().includes(kw))) {
   468→                return true;
   469→            }
   470→
   471→            const title = document.title || '';
   472→            const bodyText = document.body ? document.body.innerText.substring(0, 3000) : '';
   473→            const text = title + ' ' + bodyText;
   474→
   475→            const paywallPatterns = [
   476→                '开通会员', '升级会员', '会员专享', '会员权益',
   477→                '充值', '付费', '购买会员', '开通VIP', '升级VIP',
   478→                '立即开通', '立即升级', '立即购买',
   479→                '会员到期', '续费', '解锁', '次数已用完',
   480→                '今日沟通次数', '沟通人数已达上限',
   481→            ];
   482→
   483→            return paywallPatterns.some(p => text.includes(p));
   484→        }"""
   485→        )
   486→    except Exception as e:
   487→        logger.debug(f"VIP paywall detection failed: {e}")
   488→        return False

   491→async def send_message(
   492→    maimai_id: str,
   493→    template_name: str,
   494→    variables: Dict[str, str],
   495→    browser: Optional[BrowserManager] = None,
   496→    skip_confirmation: bool = False,
   497→    interactive: bool = False,
   498→    discovered_selectors: Optional["DiscoveredSelectors"] = None,
   499→    _skip_rate_check: bool = False,
   500→    profile_url: Optional[str] = None,
   501→) -> bool:
   502→    """
   503→    Send a message to a candidate.
   504→
   505→    Args:
   506→        maimai_id: Candidate's Maimai ID
   507→        template_name: Name of message template to use
   508→        variables: Variables to substitute in template
   509→        browser: Browser manager instance
   510→        skip_confirmation: Skip confirmation (only for testing)
   511→        interactive: Enable interactive selector discovery on failure
   512→        discovered_selectors: Shared DiscoveredSelectors instance for reuse
   513→
   514→    Returns:
   515→        True if message was sent successfully
   516→
   517→    Raises:
   518→        RateLimitExceededError: If rate limit exceeded
   519→        ValidationError: If template or variables invalid
   520→    """
   521→    # Validate and clean inputs
   522→    maimai_id = validate_maimai_id(maimai_id)
   523→
   524→    if browser is None:
   525→        browser = get_browser_manager()
   526→
   527→    config = get_config()
   528→    rate_limiter = get_rate_limiter()
   529→    db_manager = get_db_manager()
   530→
   531→    # Get template
   532→    try:
   533→        template = config.get_template(template_name)
   534→    except Exception as e:
   535→        raise ValidationError(f"Template not found: {template_name}")
   536→
   537→    # Validate template variables
   538→    validate_template_variables(template, variables)
   539→
   540→    # Render message content
   541→    content = render_template(template["content"], variables)
   542→
   543→    # Get candidate info for display
   544→    candidate_info = _get_candidate_info(db_manager, maimai_id)
   545→
   546→    # Preview and confirm
   547→    if not skip_confirmation:
   548→        action = await interactive_confirm(content, candidate_info, template_name)
   549→
   550→        if action == "skip":
   551→            console.print("[yellow]Message skipped[/yellow]")
   552→            _log_message(db_manager, maimai_id, template_name, content, "cancelled")
   553→            return False
   554→
   555→        elif action == "edit":
   556→            # Allow user to edit content
   557→            content = await edit_message(content)
   558→            if content is None:
   559→                console.print("[yellow]Message cancelled[/yellow]")
   560→                return False
   561→
   562→            # Re-confirm after edit
   563→            action = await interactive_confirm(content, candidate_info, template_name)
   564→            if action != "send":
   565→                console.print("[yellow]Message skipped[/yellow]")
   566→                return False
   567→
   568→        elif action != "send":
   569→            console.print("[yellow]Message cancelled[/yellow]")
   570→            return False
   571→
   572→    # Check rate limit (waits through min-interval) - skip when called from batch
   573→    if not _skip_rate_check:
   574→        await async_check_rate_limit("send_message")
   575→
   576→    # Ensure logged in
   577→    await ensure_logged_in(browser)
   578→
   579→    try:
   580→        page = browser.page
   581→
   582→        # If we have populated discovered selectors, use them directly

   751→async def _send_message_batch_common(
   752→    candidates: List[dict],
   753→    template_name: str,
   754→    common_variables: Dict[str, str],
   755→    send_one: Callable[[dict, str], Awaitable[CandidateSendResult]],
   756→    batch_config: BatchConfig,
   757→    skip_messaged: bool = True,
   758→    skip_confirmation: bool = False,
   759→) -> Dict[str, Any]:
   760→    """Common orchestration for all three batch-send paths.
   761→
   762→    Handles: template loading, filtering already-messaged, preview panel,
   763→    confirmation, progress bar, rate limiting, template rendering, and
   764→    result collection.  The actual per-candidate send logic is delegated
   765→    to the ``send_one`` callback.
   766→
   767→    Args:
   768→        candidates: List of dicts with maimai_id, name, company, title
   769→        template_name: Name of message template
   770→        common_variables: Shared template variables
   771→        send_one: async callback(candidate_dict, rendered_content) → CandidateSendResult
   772→        batch_config: Display / behaviour config for this path
   773→        skip_messaged: Skip already-messaged candidates
   774→        skip_confirmation: Skip interactive confirmation prompt
   775→
   776→    Returns:
   777→        Dict with total, sent, failed, skipped, rate_limited, details
   778→    """
   779→    config = get_config()
   780→    db_manager = get_db_manager()
   781→
   782→    try:
   783→        template = config.get_template(template_name)
   784→    except Exception:
   785→        raise ValidationError(f"Template not found: {template_name}")
   786→
   787→    results = {
   788→        "total": len(candidates),
   789→        "sent": 0,
   790→        "failed": 0,
   791→        "skipped": 0,
   792→        "rate_limited": 0,
   793→        "quota_exhausted": 0,
   794→        "details": [],
   795→    }
   796→
   797→    # --- Filter already-messaged candidates ---
   798→    send_list = []
   799→    if skip_messaged:
   800→        all_ids = [c["maimai_id"] for c in candidates]
   801→        with db_manager.session_scope() as session:
   802→            rows = (
   803→                session.query(Message.candidate_maimai_id)
   804→                .filter(Message.candidate_maimai_id.in_(all_ids), Message.status == "sent")
   805→                .distinct()
   806→                .all()
   807→            )
   808→            messaged_ids = {row[0] for row in rows}
   809→
   810→        for c in candidates:
   811→            if c["maimai_id"] in messaged_ids:
   812→                results["skipped"] += 1
   813→                results["details"].append(
   814→                    {
   815→                        "maimai_id": c["maimai_id"],
   816→                        "name": c["name"],
   817→                        "status": "skipped",
   818→                        "reason": "already messaged",
   819→                    }
   820→                )
   821→            else:
   822→                send_list.append(c)
   823→    else:
   824→        send_list = list(candidates)
   825→
   826→    if not send_list:
   827→        if results["skipped"] > 0:
   828→            console.print(
   829→                f"[yellow]All {results['skipped']} candidates already messaged. Skipping send.[/yellow]"
   830→            )
   831→        else:
   832→            console.print("[yellow]No candidates to send messages to.[/yellow]")
   833→        return results
   834→
   835→    # --- Optional pre-validation (Path A) ---
   836→    if batch_config.pre_validate:
   837→        console.print("[dim]Validating template for all candidates...[/dim]")
   838→        for c in send_list:
   839→            test_vars = {
   840→                **common_variables,
   841→                "name": c.get("name", ""),
   842→                "company": c.get("company", ""),
   843→                "position": c.get("title", ""),
   844→            }
   845→            is_valid, missing = validate_template_rendering(template["content"], test_vars, c)
   846→            if not is_valid:
   847→                raise ValidationError(
   848→                    f"候选人 {c.get('name', 'Unknown')} 的模板变量缺失或为空:\n"
   849→                    f"  缺失: {', '.join(missing)}\n"
   850→                    f"  候选人信息: {c}\n"
   851→                    f"  请检查模板或补充变量"
   852→                )
   853→
   854→    # --- Build preview ---
   855→    first = send_list[0]
   856→    first_vars = {
   857→        **common_variables,
   858→        "name": first["name"],
   859→        "company": first["company"],
   860→        "position": first["title"],
   861→    }
   862→    # If pre-validated, skip redundant validation in render
   863→    first_content = render_template(
   864→        template["content"], first_vars, validate=not batch_config.pre_validate
   865→    )
   866→
   867→    summary_text = Text()
   868→    summary_text.append("Template: ", style="bold")
   869→    summary_text.append(f"{template_name}\n")
   870→    if batch_config.mode_label:
   871→        summary_text.append("Mode: ", style="bold")
   872→        summary_text.append(f"{batch_config.mode_label}\n")
   873→    summary_text.append("Candidates: ", style="bold")
   874→    summary_text.append(f"{len(candidates)} found, {results['skipped']} already messaged\n")
   875→    summary_text.append("Will send to: ", style="bold")
   876→    summary_text.append(f"{len(send_list)} candidates\n\n")
   877→    summary_text.append("Message Preview ", style="bold cyan")
   878→    summary_text.append("(first candidate):\n", style="dim")
   879→    summary_text.append("To: ", style="bold cyan")
   880→    recipient = first["name"]
   881→    if first.get("title") and first.get("company"):
   882→        recipient += f" ({first['title']} @ {first['company']})"
   883→    summary_text.append(f"{recipient}\n\n")
   884→    summary_text.append(first_content)
   885→
   886→    console.print(
   887→        Panel(
   888→            summary_text,
   889→            title=batch_config.panel_title,
   890→            border_style="green",
   891→            padding=(1, 2),
   892→        )
   893→    )
   894→
   895→    # --- Confirmation ---
   896→    if skip_confirmation:
   897→        console.print(f"[green]Auto-confirmed: sending to {len(send_list)} candidates[/green]")
   898→    else:
   899→        loop = asyncio.get_running_loop()
   900→        confirm = await loop.run_in_executor(
   901→            None,
   902→            lambda: questionary.confirm(
   903→                f"Confirm batch send to {len(send_list)} candidates?"
   904→            ).unsafe_ask(),
   905→        )
   906→        if not confirm:
   907→            console.print("[yellow]Batch send cancelled.[/yellow]")
   908→            return results
   909→
   910→    # --- Send loop ---
   911→    from rich.progress import (
   912→        Progress,
   913→        BarColumn,
   914→        TextColumn as ProgressTextColumn,
   915→        MofNCompleteColumn,
   916→    )
   917→
   918→    with Progress(
   919→        ProgressTextColumn("[progress.description]{task.description}"),
   920→        BarColumn(),
   921→        MofNCompleteColumn(),
   922→        console=console,
   923→    ) as progress:
   924→        task = progress.add_task(batch_config.progress_description, total=len(send_list))
   925→
   926→        for c in send_list:
   927→            # Rate limit check
   928→            try:
   929→                await async_check_rate_limit("send_message")
   930→            except RateLimitExceededError as e:
   931→                console.print(f"\n[yellow]Rate limit reached: {e}[/yellow]")
   932→                remaining = len(send_list) - (results["sent"] + results["failed"])
   933→                results["rate_limited"] = remaining
   934→                results["details"].append(
   935→                    {
   936→                        "maimai_id": c["maimai_id"],
   937→                        "name": c["name"],
   938→                        "status": "rate_limited",
   939→                        "reason": str(e),
   940→                    }
   941→                )
   942→                break
   943→
   944→            # Build per-candidate variables and render template
   945→            variables = {
   946→                **common_variables,
   947→                "name": c["name"],
   948→                "company": c["company"],
   949→                "position": c["title"],
   950→            }
   951→            try:
   952→                content = render_template(
   953→                    template["content"],
   954→                    variables,
   955→                    validate=not batch_config.pre_validate,
   956→                )
   957→            except ValidationError as e:
   958→                logger.error(f"Template rendering failed for {c['name']}: {e}")
   959→                results["failed"] += 1
   960→                results["details"].append(
   961→                    {
   962→                        "maimai_id": c["maimai_id"],
   963→                        "name": c["name"],
   964→                        "status": "failed",
   965→                        "reason": f"模板渲染失败: {e}",
   966→                    }
   967→                )
   968→                progress.advance(task)
   969→                continue
   970→
   971→            # Delegate to path-specific sender
   972→            result = await send_one(c, content)
   973→
   974→            # Record result
   975→            if result.status == "sent":
   976→                results["sent"] += 1
   977→            elif result.status in ("quota_exhausted", "rate_limited"):
   978→                if result.status == "rate_limited":
   979→                    remaining = len(send_list) - (results["sent"] + results["failed"])
   980→                    results["rate_limited"] = remaining
   981→                else:
   982→                    results["quota_exhausted"] += 1
   983→            else:
   984→                results["failed"] += 1
   985→
   986→            detail = {
   987→                "maimai_id": c["maimai_id"],
   988→                "name": c["name"],
   989→                "status": result.status,
   990→            }
   991→            if result.reason:
   992→                detail["reason"] = result.reason
   993→            if result.attempts > 1:
   994→                detail["attempts"] = result.attempts
   995→            if result.content:
   996→                detail["content"] = result.content
   997→            elif content:
   998→                detail["content"] = content
   999→            results["details"].append(detail)
  1000→
  1001→            if result.abort_batch:
  1002→                remaining = len(send_list) - (results["sent"] + results["failed"])
  1003→                if remaining > 0:
  1004→                    console.print(f"[yellow]跳过剩余 {remaining} 位候选人[/yellow]")
  1005→                break
  1006→
  1007→            progress.advance(task)
  1008→
  1009→            await async_random_delay("message")
  1010→
  1011→    return results

  1019→def _make_search_center_sender(
  1020→    browser: BrowserManager,
  1021→    template_name: str,
  1022→    interactive: bool,
  1023→    discovered_selectors,
  1024→) -> Callable[[dict, str], Awaitable[CandidateSendResult]]:
  1025→    """Create a sender callback for search-center (Path A) with retry + timeout."""
  1026→
  1027→    max_retries = 2
  1028→    per_candidate_timeout = 90
  1029→
  1030→    async def _send(c: dict, content: str) -> CandidateSendResult:
  1031→        variables = {
  1032→            "name": c["name"],
  1033→            "company": c["company"],
  1034→            "position": c["title"],
  1035→        }
  1036→        last_error = None
  1037→        for attempt in range(1 + max_retries):
  1038→            try:
  1039→                success = await asyncio.wait_for(
  1040→                    send_message(
  1041→                        maimai_id=c["maimai_id"],
  1042→                        template_name=template_name,
  1043→                        variables=variables,
  1044→                        browser=browser,
  1045→                        skip_confirmation=True,
  1046→                        interactive=interactive,
  1047→                        discovered_selectors=discovered_selectors,
  1048→                        _skip_rate_check=True,
  1049→                        profile_url=c.get("profile_url"),
  1050→                    ),
  1051→                    timeout=per_candidate_timeout,
  1052→                )
  1053→                if success:
  1054→                    return CandidateSendResult(
  1055→                        success=True,
  1056→                        status="sent",
  1057→                        attempts=attempt + 1,
  1058→                        content=content,
  1059→                    )
  1060→                last_error = "send returned False"
  1061→
  1062→            except asyncio.TimeoutError:
  1063→                console.print(
  1064→                    f"[yellow]Timeout for {c['name']} ({c['maimai_id']}), skipping.[/yellow]"
  1065→                )
  1066→                return CandidateSendResult(
  1067→                    success=False,
  1068→                    status="failed",
  1069→                    reason=f"Timed out after {per_candidate_timeout}s",
  1070→                    attempts=attempt + 1,
  1071→                    content=content,
  1072→                )
  1073→            except QuotaExhaustedError as e:
  1074→                console.print(f"\n[red]联系券已用尽，终止批量发送[/red]")
  1075→                return CandidateSendResult(
  1076→                    success=False,
  1077→                    status="quota_exhausted",
  1078→                    reason=str(e),
  1079→                    attempts=attempt + 1,
  1080→                    content=content,
  1081→                    abort_batch=True,
  1082→                )
  1083→            except PermanentFailureError as e:
  1084→                console.print(f"[yellow]{e}. Skipping.[/yellow]")
  1085→                return CandidateSendResult(
  1086→                    success=False,
  1087→                    status="failed",
  1088→                    reason=str(e),
  1089→                    attempts=attempt + 1,
  1090→                    content=content,
  1091→                )
  1092→            except RateLimitExceededError as e:
  1093→                console.print(f"\n[yellow]Rate limit reached: {e}[/yellow]")
  1094→                return CandidateSendResult(
  1095→                    success=False,
  1096→                    status="rate_limited",
  1097→                    reason=str(e),
  1098→                    attempts=attempt + 1,
  1099→                    content=content,
  1100→                    abort_batch=True,
  1101→                )
  1102→            except Exception as e:
  1103→                last_error = str(e)
  1104→
  1105→            # Retry with exponential backoff
  1106→            if attempt < max_retries:
  1107→                from src.utils.delays import get_retry_delay
  1108→
  1109→                delay = get_retry_delay(attempt, base_delay=3.0, max_delay=30.0)
  1110→                logger.info(f"Retry {attempt + 1}/{max_retries} for {c['name']} in {delay:.1f}s")
  1111→                await asyncio.sleep(delay)
  1112→
  1113→        # All retries exhausted
  1114→        actual_attempts = max_retries + 1
  1115→        logger.error(
  1116→            f"Failed to send message to {c['name']} after {actual_attempts} attempt(s): {last_error}"
  1117→        )
  1118→        return CandidateSendResult(
  1119→            success=False,
  1120→            status="failed",
  1121→            reason=last_error or "unknown error",
  1122→            attempts=actual_attempts,
  1123→            content=content,
  1124→        )
  1125→
  1126→    return _send

  1303→    """

  1616→async def _verify_message_sent(page: Page, content: str, timeout: int = 5000) -> bool:
  1617→    """
  1618→    Verify that a message was actually sent by checking the conversation area.
  1619→
  1620→    Polls up to `timeout` ms with short intervals to handle async rendering.
  1621→
  1622→    Checks for:
  1623→    1. The sent message content appearing in the conversation
  1624→    2. The message input being cleared (indicating successful send)
  1625→    3. A sent/delivered indicator element
  1626→
  1627→    Args:
  1628→        page: Playwright page instance
  1629→        content: The message content that was sent
  1630→        timeout: Max time to wait for verification in ms
  1631→
  1632→    Returns:
  1633→        True if message send was verified
  1634→    """
  1635→    import time
  1636→
  1637→    # Short snippet for matching (first 20 chars, avoid template artifacts)
  1638→    snippet = content[:20].strip()
  1639→
  1640→    poll_interval_ms = 500
  1641→    deadline = time.monotonic() + timeout / 1000.0
  1642→
  1643→    while True:
  1644→        try:
  1645→            # Strategy 1: Check if message content appears in conversation area
  1646→            verified = await page.evaluate(
  1647→                """(snippet) => {
  1648→                const chatSelectors = [
  1649→                    '[class*="message-list"]',
  1650→                    '[class*="chat-content"]',
  1651→                    '[class*="conversation"]',
  1652→                    '[class*="msg-list"]',
  1653→                    '[class*="im-message"]',
  1654→                    '[class*="chat-msg"]',
  1655→                ];
  1656→
  1657→                for (const sel of chatSelectors) {
  1658→                    const container = document.querySelector(sel);
  1659→                    if (container && container.textContent.includes(snippet)) {
  1660→                        return true;
  1661→                    }
  1662→                }
  1663→
  1664→                // Fallback: scan leaf-ish elements in the lower portion of the page
  1665→                const viewportHeight = window.innerHeight;
  1666→                const elements = document.querySelectorAll('div, p, span');
  1667→                for (const el of elements) {
  1668→                    const rect = el.getBoundingClientRect();
  1669→                    if (rect.top > viewportHeight * 0.3 && el.textContent.includes(snippet)) {
  1670→                        if (el.textContent.length < snippet.length + 200) {
  1671→                            return true;
  1672→                        }
  1673→                    }
  1674→                }
  1675→
  1676→                return false;
  1677→            }""",
  1678→                snippet,
  1679→            )
  1680→
  1681→            if verified:
  1682→                logger.debug("Message send verified: content found in conversation")
  1683→                return True
  1684→
  1685→            # Strategy 2: Check if the chat input was cleared (indicates successful send)
  1686→            # Only check textarea/contenteditable inside chat containers to avoid
  1687→            # false positives from search boxes or other form inputs.
  1688→            input_cleared = await page.evaluate(
  1689→                """() => {
  1690→                const chatContainers = document.querySelectorAll(
  1691→                    '[class*="chat"], [class*="im-"], [class*="message"], [class*="editor"]'
  1692→                );
  1693→                for (const container of chatContainers) {
  1694→                    const inputs = container.querySelectorAll(
  1695→                        'textarea, [contenteditable="true"]'
  1696→                    );
  1697→                    for (const input of inputs) {
  1698→                        const rect = input.getBoundingClientRect();
  1699→                        if (rect.width > 0 && rect.height > 0) {
  1700→                            const value = input.value || input.textContent || '';
  1701→                            if (value.trim() === '') {
  1702→                                return true;
  1703→                            }
  1704→                        }
  1705→                    }
  1706→                }
  1707→                return false;
  1708→            }"""
  1709→            )
  1710→
  1711→            if input_cleared:
  1712→                logger.debug("Message send verified: input field was cleared")
  1713→                return True
  1714→
  1715→            # Strategy 3: Check for sent/delivered indicators
  1716→            sent_indicators = [
  1717→                '[class*="sent"]',
  1718→                '[class*="delivered"]',
  1719→                '[class*="success"]',
  1720→                '[class*="msg-status"]',
  1721→            ]
  1722→            for selector in sent_indicators:
  1723→                try:
  1724→                    indicator = await page.query_selector(selector)
  1725→                    if indicator and await indicator.is_visible():
  1726→                        logger.debug(f"Message send verified: indicator found ({selector})")
  1727→                        return True
  1728→                except Exception:
  1729→                    continue
  1730→
  1731→        except Exception as e:
  1732→            logger.warning(f"Message verification poll failed: {e}")
  1733→

  1744→async def _check_existing_conversation(page: Page, maimai_id: str) -> bool:
  1745→    """
  1746→    Check if there's already an existing conversation with this candidate.
  1747→
  1748→    Checks for actual user message bubbles (not system prompts/tips) in the
  1749→    conversation area.
  1750→
  1751→    Args:
  1752→        page: Playwright page instance
  1753→        maimai_id: Candidate's Maimai ID
  1754→
  1755→    Returns:
  1756→        True if an existing conversation was detected
  1757→    """
  1758→    try:
  1759→        has_conversation = await page.evaluate(
  1760→            """() => {
  1761→            // Look for conversation message containers
  1762→            const chatSelectors = [
  1763→                '[class*="message-list"]',
  1764→                '[class*="chat-content"]',
  1765→                '[class*="conversation"]',
  1766→                '[class*="msg-list"]',
  1767→                '[class*="im-message"]',
  1768→                '[class*="chat-msg"]',
  1769→                '[class*="chat-record"]',
  1770→            ];
  1771→
  1772→            // System/tip patterns that should NOT count as existing conversation
  1773→            const systemPatterns = [
  1774→                '还不是好友', '添加好友', '发送好友申请', '对方未回复',
  1775→                '以上是打招呼内容', '暂无消息', '开始聊天', '你们已经是好友',
  1776→                '系统消息', '温馨提示', '安全提示', '注意事项',
  1777→                '开启招呼语', '求职意向', '目前情况', '竞业点',
  1778→                '欢迎联系', '不感兴趣',
  1779→            ];
  1780→
  1781→            const isSystemText = (text) => {
  1782→                const t = text.trim();
  1783→                if (t.length < 5) return true;  // Too short to be a real message
  1784→                return systemPatterns.some(p => t.includes(p));
  1785→            };
  1786→
  1787→            for (const sel of chatSelectors) {
  1788→                const container = document.querySelector(sel);
  1789→                if (!container) continue;
  1790→
  1791→                // Strategy 1: Look for message bubble elements
  1792→                const bubbleSelectors = [
  1793→                    '[class*="msg-item"]', '[class*="message-item"]',
  1794→                    '[class*="bubble"]', '[class*="chat-item"]',
  1795→                    '[class*="msg-bubble"]', '[class*="chat-bubble"]',
  1796→                ];
  1797→                for (const bSel of bubbleSelectors) {
  1798→                    const bubbles = container.querySelectorAll(bSel);
  1799→                    // Filter out system/tip bubbles
  1800→                    let realMessageCount = 0;
  1801→                    for (const bubble of bubbles) {
  1802→                        const text = bubble.textContent.trim();
  1803→                        if (text.length > 5 && !isSystemText(text)) {
  1804→                            realMessageCount++;
  1805→                        }
  1806→                    }
  1807→                    if (realMessageCount > 0) {
  1808→                        return true;
  1809→                    }
  1810→                }
  1811→
  1812→                // Strategy 2: If no bubble selectors matched, check direct children
  1813→                // that look like message blocks (not system tips)
  1814→                const children = container.children;
  1815→                let realChildCount = 0;
  1816→                for (const child of children) {
  1817→                    const text = child.textContent.trim();
  1818→                    // Must have meaningful content and not be a system message
  1819→                    if (text.length > 10 && !isSystemText(text)) {
  1820→                        // Check it's not a timestamp-only or status-only element
  1821→                        if (child.querySelector('img[class*="avatar"]') || text.length > 20) {
  1822→                            realChildCount++;
  1823→                        }
  1824→                    }
  1825→                }
  1826→                if (realChildCount > 0) {
  1827→                    return true;
  1828→                }
  1829→            }
  1830→            return false;
  1831→        }"""
  1832→        )
  1833→
  1834→        if has_conversation:
  1835→            logger.info(f"Existing conversation detected for {maimai_id}")
  1836→        return has_conversation
  1837→
  1838→    except Exception as e:
  1839→        # Conservative: assume conversation exists to avoid duplicate messages.
  1840→        # The caller logs this as "cancelled" in the audit DB.
  1841→        logger.warning(
  1842→            f"Conversation check failed for {maimai_id}: {e} — skipping to avoid potential duplicate"
  1843→        )
  1844→        return True
