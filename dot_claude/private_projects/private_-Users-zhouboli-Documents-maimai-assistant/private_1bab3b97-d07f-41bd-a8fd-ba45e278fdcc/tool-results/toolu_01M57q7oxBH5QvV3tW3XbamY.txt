     1→"""Browser manager for Playwright automation.
     2→
     3→Provides a singleton browser manager that connects to an existing Chrome
     4→instance via CDP (Chrome DevTools Protocol). Supports multi-tab management
     5→for workflows that open new tabs (e.g., enterprise recruitment).
     6→"""
     7→
     8→import asyncio
     9→from pathlib import Path
    10→from typing import List, Optional
    11→from datetime import datetime
    12→
    13→from patchright.async_api import async_playwright, Browser, BrowserContext, Page, Playwright
    14→from loguru import logger
    15→
    16→from src.utils.config import get_config, Config
    17→from src.core.session import SessionManager, get_session_manager
    18→from src.exceptions.errors import BrowserError
    19→
    20→
    21→class BrowserManager:
    22→    """
    23→    Singleton browser manager for Maimai automation.
    24→
    25→    Connects to a user-managed Chrome instance via CDP endpoint.
    26→    The user starts Chrome with --remote-debugging-port=9222 and logs in
    27→    manually. This manager connects/disconnects without closing Chrome.
    28→    """
    29→
    30→    _instance: Optional['BrowserManager'] = None
    31→    _lock: Optional[asyncio.Lock] = None
    32→
    33→    def __new__(cls, **kwargs):
    34→        if cls._instance is None:
    35→            cls._instance = super().__new__(cls)
    36→            cls._instance._initialized = False
    37→        return cls._instance
    38→
    39→    def __init__(
    40→        self,
    41→        config: Optional[Config] = None,
    42→        session_manager: Optional[SessionManager] = None,
    43→    ):
    44→        if self._initialized:
    45→            return
    46→
    47→        self._playwright: Optional[Playwright] = None
    48→        self._browser: Optional[Browser] = None
    49→        self._context: Optional[BrowserContext] = None
    50→        self._page: Optional[Page] = None
    51→        self._primary_page: Optional[Page] = None  # Original tab to return to
    52→        self._opened_tabs: List[Page] = []  # Tabs opened during our session
    53→        self._session_manager: SessionManager = session_manager or get_session_manager()
    54→        self._config: Config = config or get_config()
    55→        self._cdp_endpoint: Optional[str] = None
    56→        self._owned_page: bool = False  # True if we created the page (vs reusing existing)
    57→        self._initialized = True
    58→
    59→    @property
    60→    def is_running(self) -> bool:
    61→        """Check if browser is currently connected."""
    62→        return self._browser is not None and self._browser.is_connected()
    63→
    64→    @property
    65→    def page(self) -> Optional[Page]:
    66→        """Get current page instance."""
    67→        return self._page
    68→
    69→    @property
    70→    def context(self) -> Optional[BrowserContext]:
    71→        """Get current browser context."""
    72→        return self._context
    73→
    74→    @classmethod
    75→    def _ensure_lock(cls):
    76→        """Lazily create the asyncio.Lock inside a running event loop."""
    77→        if cls._lock is None:
    78→            cls._lock = asyncio.Lock()
    79→
    80→    def set_cdp_endpoint(self, endpoint: str) -> None:
    81→        """Override CDP endpoint (e.g. from CLI --cdp flag)."""
    82→        self._cdp_endpoint = endpoint
    83→
    84→    async def start(self, headless: Optional[bool] = None) -> Page:
    85→        """
    86→        Connect to Chrome via CDP and return a page.
    87→
    88→        The headless parameter is accepted for API compatibility but ignored
    89→        in CDP mode (the user's Chrome is always visible).
    90→
    91→        Returns:
    92→            Configured Playwright page instance
    93→
    94→        Raises:
    95→            BrowserError: If connection fails
    96→        """
    97→        self._ensure_lock()
    98→        async with self._lock:
    99→            if self.is_running:
   100→                logger.debug("Browser already connected, returning existing page")
   101→                return self._page
   102→
   103→            try:
   104→                # Determine CDP endpoint
   105→                endpoint = self._cdp_endpoint or self._config.get(
   106→                    'browser.cdp_endpoint', 'http://localhost:9222'
   107→                )
   108→                timeout = self._config.get('browser.timeout', 30000)
   109→
   110→                logger.info(f"Connecting to Chrome via CDP: {endpoint}")
   111→
   112→                # Start Playwright
   113→                self._playwright = await async_playwright().start()
   114→
   115→                # Connect to existing Chrome via CDP
   116→                self._browser = await self._playwright.chromium.connect_over_cdp(endpoint)
   117→
   118→                # Get the default context (the user's browser context with cookies)
   119→                contexts = self._browser.contexts
   120→                if contexts:
   121→                    self._context = contexts[0]
   122→                else:
   123→                    # Fallback: create a new context (shouldn't happen with CDP)
   124→                    logger.warning("No existing context found, creating new one")
   125→                    self._context = await self._browser.new_context()
   126→
   127→                # Set default timeout
   128→                self._context.set_default_timeout(timeout)
   129→
   130→                # Reuse existing page or create one
   131→                pages = self._context.pages
   132→                if pages:
   133→                    self._page = pages[0]
   134→                    self._owned_page = False
   135→                    logger.info(f"Reusing existing tab: {self._page.url}")
   136→                else:
   137→                    self._page = await self._context.new_page()
   138→                    self._owned_page = True
   139→                    logger.info("Created new tab")
   140→
   141→                # Remember the primary tab
   142→                self._primary_page = self._page
   143→                self._opened_tabs = []
   144→
   145→                # Note: Stealth scripts (add_init_script) are NOT applied in CDP
   146→                # mode. On Chrome 144+, add_init_script on a CDP-connected page
   147→                # breaks the page's network resolver (ERR_NAME_NOT_RESOLVED).
   148→                # Patchright already provides undetectable automation without
   149→                # manual stealth injection.
   150→
   151→                logger.info("Connected to Chrome via CDP successfully")
   152→                return self._page
   153→
   154→            except Exception as e:
   155→                logger.error(f"Failed to connect to Chrome via CDP: {e}")
   156→                await self.close()
   157→                raise BrowserError(
   158→                    f"Failed to connect to Chrome via CDP at {endpoint}. "
   159→                    f"Make sure Chrome is running with --remote-debugging-port=9222. "
   160→                    f"Error: {e}"
   161→                )
   162→
   163→    async def close(self) -> None:
   164→        """Disconnect from Chrome via CDP. Does NOT close the user's Chrome."""
   165→        self._ensure_lock()
   166→        async with self._lock:
   167→            try:
   168→                # Close tabs we opened during the session
   169→                for tab in self._opened_tabs:
   170→                    try:
   171→                        if not tab.is_closed():
   172→                            await tab.close()
   173→                    except Exception:
   174→                        pass
   175→                self._opened_tabs = []
   176→
   177→                # Only close pages we created, not the user's existing tabs
   178→                if self._page and self._owned_page:
   179→                    try:
   180→                        await self._page.close()
   181→                    except Exception:
   182→                        pass
   183→                self._page = None
   184→                self._primary_page = None
   185→                self._owned_page = False
   186→
   187→                # Don't close the context — it belongs to the user's Chrome
   188→                self._context = None
   189→
   190→                # browser.close() in CDP mode just disconnects, doesn't kill Chrome
   191→                if self._browser:
   192→                    await self._browser.close()
   193→                    self._browser = None
   194→
   195→                if self._playwright:
   196→                    await self._playwright.stop()
   197→                    self._playwright = None
   198→
   199→                logger.info("Disconnected from Chrome (browser stays open)")
   200→
   201→            except Exception as e:
   202→                logger.error(f"Error disconnecting from Chrome: {e}")
   203→
   204→    async def wait_for_new_tab(self, timeout: int = 10000) -> Page:
   205→        """Wait for a click-triggered new tab, apply stealth, and switch to it.
   206→
   207→        Call this right after performing an action that opens a new tab
   208→        (e.g., clicking a link with target=_blank).
   209→
   210→        Args:
   211→            timeout: Max wait time in milliseconds
   212→
   213→        Returns:
   214→            The new Page instance (also set as the active page)
   215→
   216→        Raises:
   217→            BrowserError: If no new tab appears within timeout
   218→        """
   219→        if not self._context:
   220→            raise BrowserError("No active context for wait_for_new_tab")
   221→
   222→        try:
   223→            new_page = await self._context.wait_for_event(
   224→                "page", timeout=timeout
   225→            )
   226→            await new_page.wait_for_load_state("domcontentloaded")
   227→            self._opened_tabs.append(new_page)
   228→            self._page = new_page
   229→            logger.info(f"New tab captured: {new_page.url}")
   230→            return new_page
   231→        except Exception as e:
   232→            raise BrowserError(f"No new tab opened within {timeout}ms: {e}")
   233→
   234→    def switch_to_latest_tab(self) -> Page:
   235→        """Switch _page to the most recently opened tab in the context.
   236→
   237→        Returns:
   238→            The latest Page instance
   239→        """
   240→        if not self._context or not self._context.pages:
   241→            raise BrowserError("No tabs available")
   242→
   243→        self._page = self._context.pages[-1]
   244→        logger.debug(f"Switched to latest tab: {self._page.url}")
   245→        return self._page
   246→
   247→    def switch_to_primary_tab(self) -> Page:
   248→        """Switch back to the original (primary) tab.
   249→
   250→        Returns:
   251→            The primary Page instance
   252→        """
   253→        if not self._primary_page or self._primary_page.is_closed():
   254→            raise BrowserError("Primary tab is not available")
   255→
   256→        self._page = self._primary_page
   257→        logger.debug(f"Switched to primary tab: {self._page.url}")
   258→        return self._page
   259→
   260→    async def close_current_tab(self) -> Page:
   261→        """Close the current active tab and switch to the previous one.
   262→
   263→        Will not close the primary tab. After closing, switches to the
   264→        most recent remaining tab or falls back to the primary tab.
   265→
   266→        Returns:
   267→            The page that became active after closing
   268→
   269→        Raises:
   270→            BrowserError: If trying to close the primary tab
   271→        """
   272→        if not self._page:
   273→            raise BrowserError("No active page to close")
   274→
   275→        if self._page == self._primary_page:
   276→            raise BrowserError("Cannot close the primary tab")
   277→
   278→        closing = self._page
   279→        try:
   280→            # Remove from opened_tabs tracking
   281→            if closing in self._opened_tabs:
   282→                self._opened_tabs.remove(closing)
   283→            await closing.close()
   284→        except Exception as e:
   285→            logger.warning(f"Error closing tab: {e}")
   286→
   287→        # Switch to previous tab
   288→        if self._context and self._context.pages:
   289→            self._page = self._context.pages[-1]
   290→        else:
   291→            self._page = self._primary_page
   292→
   293→        logger.debug(f"Closed tab, switched to: {self._page.url if self._page else 'none'}")
   294→        return self._page
   295→
   296→    @property
   297→    def tab_count(self) -> int:
   298→        """Return the number of open tabs in the context."""
   299→        if not self._context:
   300→            return 0
   301→        return len(self._context.pages)
   302→
   303→    async def save_session(self) -> None:
   304→        """Save current session to disk (backup of cookies)."""
   305→        if self._context:
   306→            await self._session_manager.save(self._context)
   307→        else:
   308→            logger.warning("No active context to save session from")
   309→
   310→    async def screenshot(self, name: str = None) -> str:
   311→        """
   312→        Take a screenshot of current page.
   313→
   314→        Args:
   315→            name: Screenshot name (timestamp used if None)
   316→
   317→        Returns:
   318→            Path to saved screenshot
   319→        """
   320→        if not self._page:
   321→            raise BrowserError("No active page for screenshot")
   322→
   323→        screenshot_dir = Path(self._config.get('data.screenshot_dir', 'data/screenshots'))
   324→        screenshot_dir.mkdir(parents=True, exist_ok=True)
   325→
   326→        if name is None:
   327→            name = datetime.now().strftime('%Y%m%d_%H%M%S')
   328→
   329→        filepath = screenshot_dir / f"{name}.png"
   330→
   331→        await self._page.screenshot(path=str(filepath), full_page=True)
   332→        logger.info(f"Screenshot saved: {filepath}")
   333→
   334→        return str(filepath)
   335→
   336→
   337→
   338→# Global browser manager instance
   339→_browser_manager: Optional[BrowserManager] = None
   340→
   341→
   342→def get_browser_manager() -> BrowserManager:
   343→    """Get global browser manager instance."""
   344→    global _browser_manager
   345→    if _browser_manager is None:
   346→        _browser_manager = BrowserManager()
   347→    return _browser_manager
   348→
   349→
   350→async def ensure_browser() -> Page:
   351→    """
   352→    Ensure browser is connected and return page.
   353→
   354→    Convenience function for modules that need a browser.
   355→
   356→    Returns:
   357→        Active page instance
   358→    """
   359→    manager = get_browser_manager()
   360→    if not manager.is_running:
   361→        return await manager.start()
   362→    return manager.page
