     1→"""Rate limiter for preventing account bans."""

     3→import asyncio
     4→from datetime import datetime, timedelta
     5→from typing import Tuple, Optional
     6→
     7→from loguru import logger
     8→from sqlalchemy import func
     9→
    10→from src.models.database import ActionLog, get_db_manager, DatabaseManager

    15→class RateLimiter:
    16→    """Rate limiter for controlling action frequency.
    17→
    18→    Supports adaptive throttling: when captchas or errors are reported,
    19→    limits are temporarily tightened to reduce detection risk.
    20→    """
    21→
    22→    def __init__(self, config: Optional[Config] = None, db_manager: Optional[DatabaseManager] = None):
    23→        """Initialize rate limiter with configuration.
    24→
    25→        Args:
    26→            config: Config instance (uses global singleton if None)
    27→            db_manager: DatabaseManager instance (uses global singleton if None)
    28→        """
    29→        self.config = config or get_config()
    30→        self.db_manager = db_manager or get_db_manager()
    31→
    32→        # Adaptive throttle state
    33→        self._throttle_factor: float = 1.0  # 1.0 = normal, <1.0 = stricter
    34→        self._captcha_timestamps: list = []
    35→        self._error_timestamps: list = []
    36→        self._pruned: bool = False  # one-time prune flag
    37→
    38→    def report_captcha(self) -> None:
    39→        """Report a captcha occurrence to trigger adaptive throttling."""
    40→        now = datetime.utcnow()
    41→        self._captcha_timestamps.append(now)
    42→        # Keep only last 30 minutes of captcha events
    43→        cutoff = now - timedelta(minutes=30)
    44→        self._captcha_timestamps = [t for t in self._captcha_timestamps if t >= cutoff]
    45→        self._update_throttle_factor()
    46→
    47→    def report_error(self) -> None:
    48→        """Report a request error/failure to trigger adaptive throttling."""
    49→        now = datetime.utcnow()
    50→        self._error_timestamps.append(now)
    51→        cutoff = now - timedelta(minutes=30)
    52→        self._error_timestamps = [t for t in self._error_timestamps if t >= cutoff]
    53→        self._update_throttle_factor()
    54→
    55→    def _update_throttle_factor(self) -> None:
    56→        """Recalculate throttle factor based on recent captcha/error frequency.
    57→
    58→        Throttle factor ranges from 1.0 (normal) to 0.3 (heavily throttled).
    59→        - 1 captcha in 30min: reduce to 0.7x (30% fewer operations allowed)
    60→        - 2 captchas: reduce to 0.5x
    61→        - 3+ captchas: reduce to 0.3x
    62→        - Errors contribute at half the weight of captchas
    63→        """
    64→        captcha_count = len(self._captcha_timestamps)
    65→        error_count = len(self._error_timestamps)
    66→        combined = captcha_count + error_count * 0.5
    67→
    68→        if combined >= 3:
    69→            self._throttle_factor = 0.3
    70→        elif combined >= 2:
    71→            self._throttle_factor = 0.5
    72→        elif combined >= 1:
    73→            self._throttle_factor = 0.7
    74→        else:
    75→            self._throttle_factor = 1.0
    76→
    77→        if self._throttle_factor < 1.0:
    78→            logger.warning(
    79→                f"Adaptive throttle active: factor={self._throttle_factor:.1f} "
    80→                f"(captchas={captcha_count}, errors={error_count} in last 30min)"
    81→            )
    82→
    83→    def _refresh_throttle(self) -> None:
    84→        """Prune expired timestamps and recalculate throttle factor."""
    85→        cutoff = datetime.utcnow() - timedelta(minutes=30)
    86→        self._captcha_timestamps = [t for t in self._captcha_timestamps if t >= cutoff]
    87→        self._error_timestamps = [t for t in self._error_timestamps if t >= cutoff]
    88→        self._update_throttle_factor()
    89→
    90→    @property
    91→    def throttle_factor(self) -> float:
    92→        """Current adaptive throttle factor (1.0 = normal, <1.0 = throttled)."""
    93→        return self._throttle_factor
    94→
    95→    def _get_limits(self, action_type: str) -> Tuple[int, int, int]:
    96→        """
    97→        Get rate limits for an action type, adjusted by adaptive throttle.
    98→
    99→        Args:
   100→            action_type: Type of action ('search', 'view_profile', 'send_message')
   101→
   102→        Returns:
   103→            Tuple of (per_hour_limit, per_day_limit, min_interval_seconds)
   104→        """
   105→        action_map = {
   106→            'search': ('search_per_hour', 'search_per_day'),
   107→            'view_profile': ('profile_view_per_hour', 'profile_view_per_day'),
   108→            'send_message': ('message_per_hour', 'message_per_day'),
   109→        }
   110→
   111→        if action_type not in action_map:
   112→            raise ValueError(f"Unknown action type: {action_type}")
   113→
   114→        hour_key, day_key = action_map[action_type]
   115→
   116→        per_hour = self.config.get(f'rate_limits.{hour_key}', 20)
   117→        per_day = self.config.get(f'rate_limits.{day_key}', 100)
   118→        min_interval = self.config.get('rate_limits.min_interval_seconds', 30)
   119→
   120→        # Apply adaptive throttle: reduce limits and increase interval
   121→        per_hour = max(1, int(per_hour * self._throttle_factor))
   122→        per_day = max(1, int(per_day * self._throttle_factor))
   123→        if self._throttle_factor < 1.0:
   124→            min_interval = int(min_interval / self._throttle_factor)
   125→
   126→        return per_hour, per_day, min_interval
   127→
   128→    def _count_actions(self, action_type: str, since: datetime) -> int:
   129→        """
   130→        Count actions of a type since a given time.
   131→
   132→        Args:
   133→            action_type: Type of action
   134→            since: Count actions since this time
   135→
   136→        Returns:
   137→            Number of actions
   138→        """
   139→        with self.db_manager.session_scope() as session:
   140→            count = session.query(func.count(ActionLog.id)).filter(
   141→                ActionLog.action_type == action_type,
   142→                ActionLog.timestamp >= since
   143→            ).scalar()
   144→
   145→            return count or 0
   146→
   147→    def _get_last_action_time(self, action_type: Optional[str] = None) -> Optional[datetime]:
   148→        """
   149→        Get timestamp of last action.
   150→
   151→        Args:
   152→            action_type: Type of action, or None for any action
   153→
   154→        Returns:
   155→            Timestamp of last action, or None if no actions
   156→        """
   157→        with self.db_manager.session_scope() as session:
   158→            query = session.query(func.max(ActionLog.timestamp))
   159→
   160→            if action_type:
   161→                query = query.filter(ActionLog.action_type == action_type)
   162→
   163→            last_time = query.scalar()
   164→            return last_time
   165→
   166→    def check_rate_limit(self, action_type: str) -> None:
   167→        """
   168→        Check if action is allowed under rate limits.
   169→
   170→        Args:
   171→            action_type: Type of action to check
   172→
   173→        Raises:
   174→            RateLimitExceededError: If rate limit is exceeded
   175→        """
   176→        # Prune stale entries once per session to keep counts accurate
   177→        if not self._pruned:
   178→            self.prune_old_entries()
   179→            self._pruned = True
   180→
   181→        self._refresh_throttle()
   182→        per_hour, per_day, min_interval = self._get_limits(action_type)
   183→
   184→        now = datetime.utcnow()
   185→        one_hour_ago = now - timedelta(hours=1)
   186→        one_day_ago = now - timedelta(days=1)
   187→
   188→        # Check hourly limit
   189→        hourly_count = self._count_actions(action_type, one_hour_ago)
   190→        if hourly_count >= per_hour:
   191→            # Calculate wait time until oldest action expires
   192→            with self.db_manager.session_scope() as session:
   193→                oldest_in_window = session.query(ActionLog.timestamp).filter(
   194→                    ActionLog.action_type == action_type,
   195→                    ActionLog.timestamp >= one_hour_ago
   196→                ).order_by(ActionLog.timestamp.asc()).first()
   197→
   198→                if oldest_in_window:
   199→                    wait_until = oldest_in_window[0] + timedelta(hours=1)
   200→                    wait_seconds = int((wait_until - now).total_seconds())
   201→                else:
   202→                    wait_seconds = 3600
   203→
   204→            raise RateLimitExceededError(
   205→                action_type=action_type,
   206→                wait_time=wait_seconds,
   207→                message=f"Hourly rate limit exceeded for {action_type}. "
   208→                        f"Limit: {per_hour}/hour. Wait {wait_seconds // 60} minutes."
   209→            )
   210→
   211→        # Check daily limit
   212→        daily_count = self._count_actions(action_type, one_day_ago)
   213→        if daily_count >= per_day:
   214→            with self.db_manager.session_scope() as session:
   215→                oldest_in_window = session.query(ActionLog.timestamp).filter(
   216→                    ActionLog.action_type == action_type,
   217→                    ActionLog.timestamp >= one_day_ago
   218→                ).order_by(ActionLog.timestamp.asc()).first()
   219→
   220→                if oldest_in_window:
   221→                    wait_until = oldest_in_window[0] + timedelta(days=1)
   222→                    wait_seconds = int((wait_until - now).total_seconds())
   223→                else:
   224→                    wait_seconds = 86400
   225→
   226→            raise RateLimitExceededError(
   227→                action_type=action_type,
   228→                wait_time=wait_seconds,
   229→                message=f"Daily rate limit exceeded for {action_type}. "
   230→                        f"Limit: {per_day}/day. Wait {wait_seconds // 3600} hours."
   231→            )
   232→
   233→        # Check minimum interval
   234→        last_action = self._get_last_action_time()
   235→        if last_action:
   236→            elapsed = (now - last_action).total_seconds()
   237→            if elapsed < min_interval:
   238→                wait_seconds = int(min_interval - elapsed)
   239→                raise RateLimitExceededError(
   240→                    action_type='any',
   241→                    wait_time=wait_seconds,
   242→                    message=f"Minimum interval not met. Wait {wait_seconds} seconds."
   243→                )
   244→
   245→        logger.debug(f"Rate limit check passed for {action_type}")
   246→
   247→    def log_action(self, action_type: str, details: Optional[dict] = None) -> None:
   248→        """
   249→        Log an action to the database.
   250→
   251→        Args:
   252→            action_type: Type of action
   253→            details: Optional action details
   254→        """
   255→        with self.db_manager.session_scope() as session:
   256→            action_log = ActionLog(
   257→                action_type=action_type,
   258→                timestamp=datetime.utcnow(),
   259→                details=details or {}
   260→            )
   261→            session.add(action_log)
   262→
   263→        logger.debug(f"Logged action: {action_type}")
   264→
   265→    def prune_old_entries(self, retention_hours: int = 48) -> int:
   266→        """Remove ActionLog entries older than the retention window.
   267→
   268→        This prevents stale counts from accumulating across sessions and
   269→        keeps the ActionLog table from growing indefinitely.
   270→
   271→        Args:
   272→            retention_hours: Delete entries older than this many hours (default 48)
   273→
   274→        Returns:
   275→            Number of entries deleted
   276→        """
   277→        cutoff = datetime.utcnow() - timedelta(hours=retention_hours)
   278→        try:
   279→            with self.db_manager.session_scope() as session:
   280→                count = session.query(ActionLog).filter(
   281→                    ActionLog.timestamp < cutoff
   282→                ).delete()
   283→                logger.debug(f"Pruned {count} old ActionLog entries (older than {retention_hours}h)")
   284→                return count
   285→        except Exception as e:
   286→            logger.error(f"Failed to prune ActionLog entries: {e}")
   287→            return 0
   288→
   289→    def get_quota_status(self, action_type: str) -> dict:
   290→        """
   291→        Get current quota status for an action type.
   292→
   293→        Args:
   294→            action_type: Type of action
   295→
   296→        Returns:
   297→            Dictionary with quota information
   298→        """
   299→        per_hour, per_day, _ = self._get_limits(action_type)
   300→
   301→        now = datetime.utcnow()
   302→        one_hour_ago = now - timedelta(hours=1)
   303→        one_day_ago = now - timedelta(days=1)
   304→
   305→        hourly_used = self._count_actions(action_type, one_hour_ago)
   306→        daily_used = self._count_actions(action_type, one_day_ago)
   307→
   308→        return {
   309→            'action_type': action_type,
   310→            'hourly': {
   311→                'used': hourly_used,
   312→                'limit': per_hour,
   313→                'remaining': max(0, per_hour - hourly_used)
   314→            },
   315→            'daily': {
   316→                'used': daily_used,
   317→                'limit': per_day,
   318→                'remaining': max(0, per_day - daily_used)
   319→            }
   320→        }

   322→
   323→async def async_check_rate_limit(action_type: str) -> None:
   324→    """
   325→    Check rate limit, waiting through any violations until the action is allowed.
   326→
   327→    Blocks until the rate limit clears — never raises RateLimitExceededError.
   328→
   329→    Args:
   330→        action_type: Type of action to check
   331→    """
   332→    rate_limiter = get_rate_limiter()
   333→    while True:
   334→        try:
   335→            rate_limiter.check_rate_limit(action_type)
   336→            return
   337→        except RateLimitExceededError as e:
   338→            wait = e.wait_time + 1  # +1s buffer
   339→            logger.info(f"Rate limit hit: {e}. Waiting {wait}s before retrying...")
   340→            await asyncio.sleep(wait)
   341→
   342→
   343→# Global rate limiter instance
   344→_rate_limiter = None
   345→
   346→
   347→def get_rate_limiter() -> RateLimiter:
   348→    """Get global rate limiter instance."""
   349→    global _rate_limiter
   350→    if _rate_limiter is None:
   351→        _rate_limiter = RateLimiter()
   352→    return _rate_limiter
