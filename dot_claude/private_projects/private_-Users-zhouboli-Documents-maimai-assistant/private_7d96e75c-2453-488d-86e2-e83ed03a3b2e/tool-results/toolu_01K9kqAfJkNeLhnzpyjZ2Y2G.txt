    53→    53→def _is_non_rd_title(title: str) -> bool:
    54→    54→    """Check if a title indicates a non-R&D role.
    55→    55→
    56→    56→    Strict mode: if any non-R&D keyword appears, filter out regardless of
    57→    57→    whether R&D keywords are also present (e.g. "技术产品经理" is filtered).
    58→    58→    Returns False (keep) only if title is empty or contains no non-R&D keywords.
    59→    59→    """
    60→    60→    if not title:
    61→    61→        return False
    62→    62→
    63→    63→    # Strict: any non-R&D keyword present → filter out
    64→    64→    if any(kw in title for kw in NON_RD_KEYWORDS):
    65→    65→        return True
    66→    66→
    67→    67→    return False

    82→    82→async def search_candidates(
    83→    83→    keywords: str,
    84→    84→    company: Optional[tuple] = None,
    85→    85→    location: Optional[str] = None,
    86→    86→    pages: int = 1,
    87→    87→    start_page: int = 1,
    88→    88→    browser: Optional[BrowserManager] = None
    89→    89→) -> List[CandidateBasic]:
    90→    90→    """
    91→    91→    Search for candidates on Maimai.
    92→    92→
    93→    93→    Args:
    94→    94→        keywords: Search keywords (e.g., "高级前端工程师")
    95→    95→        company: Filter by company names, tuple of strings (optional)
    96→    96→        location: Filter by location (optional)
    97→    97→        pages: Number of pages to scrape (default 1)
    98→    98→        start_page: First page number to scrape (default 1)
    99→    99→        browser: Browser manager instance
   100→   100→
   101→   101→    Returns:
   102→   102→        List of CandidateBasic objects
   103→   103→
   104→   104→    Raises:
   105→   105→        RateLimitExceededError: If rate limit exceeded
   106→   106→        SelectorNotFoundError: If page elements not found
   107→   107→    """
   108→   108→    # Validate inputs
   109→   109→    validate_keywords(keywords)
   110→   110→
   111→   111→    if browser is None:
   112→   112→        browser = get_browser_manager()
   113→   113→
   114→   114→    config = get_config()
   115→   115→    rate_limiter = get_rate_limiter()
   116→   116→    db_manager = get_db_manager()
   117→   117→
   118→   118→    # Ensure logged in
   119→   119→    await ensure_logged_in(browser)
   120→   120→
   121→   121→    all_candidates: List[CandidateBasic] = []
   122→   122→    seen_ids: set = set()  # Deduplicate across infinite-scroll extractions
   123→   123→
   124→   124→    with Progress(
   125→   125→        SpinnerColumn(),
   126→   126→        TextColumn("[progress.description]{task.description}"),
   127→   127→        console=console
   128→   128→    ) as progress:
   129→   129→
   130→   130→        end_page = start_page + pages
   131→   131→        for page_num in range(start_page, end_page):
   132→   132→            task = progress.add_task(f"Searching page {page_num}/{end_page - 1}...", total=None)
   133→   133→
   134→   134→            try:
   135→   135→                # Check rate limit (waits through min-interval)
   136→   136→                await async_check_rate_limit('search')
   137→   137→
   138→   138→                page = browser.page
   139→   139→
   140→   140→                if page_num == start_page:
   141→   141→                    # First page: interactive search (maimai SPA ignores URL query params)
   142→   142→                    await retry_navigation(page, MAIMAI_SEARCH_URL, wait_until='domcontentloaded')
   143→   143→                    await asyncio.sleep(1)
   144→   144→
   145→   145→                    # Find search input and type keyword
   146→   146→                    search_input = (
   147→   147→                        await page.query_selector('input[placeholder="搜索"]')
   148→   148→                        or await page.query_selector('input[type="search"]')
   149→   149→                        or await page.query_selector('.search-input input, .header input[type="text"]')
   150→   150→                    )
   151→   151→                    if not search_input:
   152→   152→                        raise SelectorNotFoundError("Search input not found on search center page")
   153→   153→
   154→   154→                    await search_input.click()
   155→   155→                    await search_input.fill(keywords)
   156→   156→                    await asyncio.sleep(0.5)
   157→   157→                    await page.keyboard.press('Enter')
   158→   158→                    await asyncio.sleep(2)
   159→   159→
   160→   160→                    # Default tab is 实名动态(feed); switch to 人脉(contacts)
   161→   161→                    renmai_tab = await page.query_selector(':text-is("人脉")')
   162→   162→                    if renmai_tab:
   163→   163→                        await renmai_tab.click()
   164→   164→                        await asyncio.sleep(2)
   165→   165→                    else:
   166→   166→                        logger.warning("Could not find '人脉' tab on search results page")
   167→   167→                else:
   168→   168→                    # Subsequent pages: scroll to load more (infinite scroll)
   169→   169→                    prev_count = await page.evaluate(
   170→   170→                        'document.querySelectorAll(\'[id^="contactcard_"]\').length'
   171→   171→                    )
   172→   172→                    await human_scroll(page, 'down', distance=1500)
   173→   173→                    # Wait for new cards to load
   174→   174→                    try:
   175→   175→                        await page.wait_for_function(
   176→   176→                            f'''() => {{
   177→   177→                                const cards = document.querySelectorAll('[id^="contactcard_"]');
   178→   178→                                return cards.length > {prev_count};
   179→   179→                            }}''',
   180→   180→                            timeout=8000,
   181→   181→                        )
   182→   182→                    except TimeoutError:
   183→   183→                        logger.debug(f"No new cards after scroll (had {prev_count})")
   184→   184→                    except Exception as e:
   185→   185→                        logger.warning(f"Error waiting for new cards after scroll: {e}")
   186→   186→
   187→   187→                # Wait for contactcard elements
   188→   188→                try:
   189→   189→                    await page.wait_for_function(
   190→   190→                        '''() => {
   191→   191→                            const cards = document.querySelectorAll('[id^="contactcard_"]');
   192→   192→                            return cards.length >= 1;
   193→   193→                        }''',
   194→   194→                        timeout=8000,
   195→   195→                    )
   196→   196→                except TimeoutError:
   197→   197→                    logger.debug("Timed out waiting for contactcard elements, proceeding anyway")
   198→   198→                except Exception as e:
   199→   199→                    logger.warning(f"Error waiting for contactcard elements: {e}")
   200→   200→
   201→   201→                # Extract candidates from this page
   202→   202→                candidates = await _extract_search_results(page, config)
   203→   203→
   204→   204→                # Deduplicate: infinite scroll re-extracts all visible cards
   205→   205→                candidates = [c for c in candidates if c.maimai_id not in seen_ids]
   206→   206→                for c in candidates:
   207→   207→                    seen_ids.add(c.maimai_id)
   208→   208→
   209→   209→                # Anti-detection: human behavior AFTER extraction
   210→   210→                await check_captcha(page, report_to_rate_limiter=True)
   211→   211→                await human_scroll(page, 'down', distance=300)
   212→   212→                await async_random_delay('action')
   213→   213→
   214→   214→                if not candidates:
   215→   215→                    logger.info(f"No results found on page {page_num}")
   216→   216→                    await browser.screenshot(f"search_debug_page{page_num}")
   217→   217→                    progress.update(task, description=f"Page {page_num}: No results")
   218→   218→                    # Allow one empty page (could be transient), break on second consecutive
   219→   219→                    if page_num > start_page and not all_candidates:
   220→   220→                        # First page had no results either — genuinely empty
   221→   221→                        break
   222→   222→                    continue
   223→   223→
   224→   224→                # Filter out non-R&D roles
   225→   225→                before_rd_filter = len(candidates)
   226→   226→                candidates = [c for c in candidates if not _is_non_rd_title(c.title)]
   227→   227→                rd_filtered = before_rd_filter - len(candidates)
   228→   228→                if rd_filtered > 0:
   229→   229→                    logger.info(f"Filtered out {rd_filtered} non-R&D candidates")
   230→   230→
   231→   231→                # Apply filters if specified
   232→   232→                if company:
   233→   233→                    company_lower = [cn.lower() for cn in company]
   234→   234→                    candidates = [c for c in candidates if c.company and any(cn in c.company.lower() for cn in company_lower)]
   235→   235→
   236→   236→                if location:
   237→   237→                    candidates = [c for c in candidates if c.location and location.lower() in c.location.lower()]
   238→   238→
   239→   239→                all_candidates.extend(candidates)
   240→   240→                progress.update(task, description=f"Page {page_num}: Found {len(candidates)} candidates")
   241→   241→
   242→   242→                # Log the action
   243→   243→                rate_limiter.log_action('search', {
   244→   244→                    'keywords': keywords,
   245→   245→                    'page': page_num,
   246→   246→                    'results': len(candidates)
   247→   247→                })
   248→   248→
   249→   249→                # Save candidates to database
   250→   250→                try:
   251→   251→                    _save_candidates_to_db(db_manager, candidates, keywords)
   252→   252→                except Exception as e:
   253→   253→                    console.print(f"[yellow]Warning: Failed to save candidates to DB: {e}[/yellow]")
   254→   254→
   255→   255→                # Delay before next page
   256→   256→                if page_num < end_page - 1:
   257→   257→                    await async_random_delay('page_load')
   258→   258→
   259→   259→            except (BrowserError, SessionExpiredError):
   260→   260→                raise  # Critical errors must propagate
   261→   261→            except Exception as e:
   262→   262→                logger.error(f"Error on page {page_num}: {e}")
   263→   263→                progress.update(task, description=f"Page {page_num}: Error - {str(e)[:30]}")
   264→   264→                # Take screenshot for debugging
   265→   265→                await browser.screenshot(f"search_error_page{page_num}")
   266→   266→                continue
   267→   267→
   268→   268→    # Log search to history
   269→   269→    _log_search_history(db_manager, keywords, company, location, pages, len(all_candidates))
   270→   270→
   271→   271→    console.print(f"[green]Search complete. Found {len(all_candidates)} candidates.[/green]")
   272→   272→    return all_candidates

   275→   275→async def _extract_search_results(page: Page, config) -> List[CandidateBasic]:
   276→   276→    """
   277→   277→    Extract candidate information from search results page.
   278→   278→
   279→   279→    Args:
   280→   280→        page: Playwright page instance
   281→   281→        config: Config instance
   282→   282→
   283→   283→    Returns:
   284→   284→        List of CandidateBasic objects
   285→   285→    """
   286→   286→    candidates = []
   287→   287→
   288→   288→    # NOTE: The caller (search_candidates) already waits for contactcard_
   289→   289→    # elements before calling this function.  Maimai's SPA strips URL
   290→   290→    # params within ~5-8s and clears results, so we must NOT add long
   291→   291→    # waits here (no networkidle, no loading-text polling).
   292→   292→
   293→   293→    try:
   294→   294→        # Strategy: Extract candidates by finding avatar images and their associated text
   295→   295→        candidates_data = await page.evaluate('''() => {
   296→   296→            const candidates = [];
   297→   297→            const seenNames = new Set();
   298→   298→
   299→   299→            // Find all images that look like avatars
   300→   300→            const images = document.querySelectorAll('img');
   301→   301→
   302→   302→            for (const img of images) {
   303→   303→                // Skip large images (probably not avatars)
   304→   304→                const rect = img.getBoundingClientRect();
   305→   305→                if (rect.width > 100 || rect.height > 100) continue;
   306→   306→                if (rect.width < 20 || rect.height < 20) continue;
   307→   307→
   308→   308→                // Find parent container (the card)
   309→   309→                let card = img.parentElement;
   310→   310→                for (let i = 0; i < 8 && card; i++) {
   311→   311→                    const cardRect = card.getBoundingClientRect();
   312→   312→                    // Look for a reasonable card size
   313→   313→                    if (cardRect.width > 200 && cardRect.height > 50 && cardRect.height < 200) {
   314→   314→                        break;
   315→   315→                    }
   316→   316→                    card = card.parentElement;
   317→   317→                }
   318→   318→
   319→   319→                if (!card) continue;
   320→   320→
   321→   321→                // Get all text nodes from card, preserving order
   322→   322→                const textParts = [];
   323→   323→                const walker = document.createTreeWalker(card, NodeFilter.SHOW_TEXT, null, false);
   324→   324→                while (walker.nextNode()) {
   325→   325→                    const text = walker.currentNode.textContent.trim();
   326→   326→                    if (text && text.length > 0 && text.length < 100) {
   327→   327→                        textParts.push(text);
   328→   328→                    }
   329→   329→                }
   330→   330→
   331→   331→                // Parse candidate info from text parts
   332→   332→                // Pattern: Name, Activity time, Company+Position, Industry|Department
   333→   333→                let name = '';
   334→   334→                let company = '';
   335→   335→                let title = '';
   336→   336→                let industry = '';
   337→   337→
   338→   338→                // Compound job title patterns (longer patterns first for greedy match)
   339→   339→                const jobPatterns = [
   340→   340→                    // Compound titles with prefix modifiers
   341→   341→                    /(?:高级|资深|首席|初级|中级|senior|junior|staff|principal|lead)?\s*(?:前端|后端|全栈|web|Web|iOS|Android|Java|Python|Go|C\+\+|PHP|.NET|算法|数据|AI|机器学习|深度学习|NLP|大数据|云计算|DevOps|SRE|安全|测试|嵌入式|移动端|客户端|服务端|区块链|具身智能|自动驾驶|机器人)?\s*(?:工程师|开发|架构师|专家|技术经理|技术总监|技术负责人|Tech\s*Lead)/i,
   342→   342→                    // Management titles
   343→   343→                    /(?:技术|产品|运营|市场|销售|人力|财务|行政|项目|研发)?\s*(?:总监|经理|主管|负责人|VP|总裁|副总|合伙人|创始人|董事|CEO|CTO|COO|CFO|CMO|CHO|CIO)/i,
   344→   344→                    // Other roles
   345→   345→                    /(?:产品|UI|UX|交互|视觉)?\s*(?:设计师|分析师|顾问|运营|助理|实习生?)/i,
   346→   346→                ];
   347→   347→
   348→   348→                // Patterns for text that should be skipped
   349→   349→                const skipPatterns = [
   350→   350→                    /^[+＋]\s*好友$/,          // "+ 好友" button
   351→   351→                    /^\d+个共同好友$/,           // "1个共同好友" social label
   352→   352→                    /^共同好友/,                 // "共同好友..." variants
   353→   353→                    /^已关注$/,                  // "已关注" button
   354→   354→                    /^添加好友$/,                // "添加好友" button
   355→   355→                    /^关注$/,                    // "关注" button
   356→   356→                    /^好友$/,                    // "好友" label
   357→   357→                    /^[|｜·]$/,                  // separators
   358→   358→                ];
   359→   359→
   360→   360→                const isSkippable = (text) => {
   361→   361→                    if (skipPatterns.some(p => p.test(text))) return true;
   362→   362→                    // Activity time
   363→   363→                    if (/天前|小时前|分钟前|刚刚|活跃|在线/.test(text)) return true;
   364→   364→                    // Social relationship labels
   365→   365→                    if (/同行|前同事|校友|共同好友/.test(text)) return true;
   366→   366→                    return false;
   367→   367→                };
   368→   368→
   369→   369→                for (let i = 0; i < textParts.length; i++) {
   370→   370→                    const part = textParts[i];
   371→   371→
   372→   372→                    if (isSkippable(part)) continue;
   373→   373→
   374→   374→                    // First short text (2-15 chars, pure CJK/letters) is likely the name
   375→   375→                    if (!name && part.length >= 2 && part.length <= 15) {
   376→   376→                        if (/^[\u4e00-\u9fa5a-zA-Z·\s()（）]+$/.test(part)) {
   377→   377→                            name = part.replace(/[（(][^)）]*[）)]/g, '').trim();
   378→   378→                            continue;
   379→   379→                        }
   380→   380→                    }
   381→   381→
   382→   382→                    // After name, next substantial text is company+position
   383→   383→                    if (name && !company && part.length > 2) {
   384→   384→                        // Try to split company and position using regex patterns
   385→   385→                        let splitDone = false;
   386→   386→                        for (const pattern of jobPatterns) {
   387→   387→                            const match = part.match(pattern);
   388→   388→                            if (match && match.index > 0) {
   389→   389→                                company = part.substring(0, match.index).trim();
   390→   390→                                title = match[0].trim();
   391→   391→                                splitDone = true;
   392→   392→                                break;
   393→   393→                            }
   394→   394→                        }
   395→   395→                        if (!splitDone) {
   396→   396→                            company = part;
   397→   397→                        }
   398→   398→                        continue;
   399→   399→                    }
   400→   400→
   401→   401→                    // Next part is industry|department info
   402→   402→                    if (name && company && !industry && part.length > 2) {
   403→   403→                        industry = part;
   404→   404→                        // If title wasn't split from company, use industry as fallback context
   405→   405→                        if (!title) {
   406→   406→                            title = industry;
   407→   407→                        }
   408→   408→                        break;
   409→   409→                    }
   410→   410→                }
   411→   411→
   412→   412→                // Skip if no name found or premium placeholder
   413→   413→                if (!name) continue;
   414→   414→                if (name === '脉脉高价值人脉') continue;
   415→   415→
   416→   416→                // Extract user ID from DOM patterns on card and ancestor elements.
   417→   417→                // Two patterns exist:
   418→   418→                //   1. contactcard_{numeric_id} — on ancestors (colleagues layout)
   419→   419→                //   2. member_entry_{numeric_id} — on descendants (standard layout)
   420→   420→                let userId = '';
   421→   421→
   422→   422→                // Check ancestors for numeric contactcard_{id}
   423→   423→                let el = card;
   424→   424→                while (el) {
   425→   425→                    const elId = el.id || '';
   426→   426→                    const ccMatch = elId.match(/^contactcard_(\d+)$/);
   427→   427→                    if (ccMatch) {
   428→   428→                        userId = ccMatch[1];
   429→   429→                        break;
   430→   430→                    }
   431→   431→                    el = el.parentElement;
   432→   432→                }
   433→   433→
   434→   434→                // Check descendants for member_entry_{id} (most common pattern)
   435→   435→                if (!userId) {
   436→   436→                    const meEl = card.querySelector('[id^="member_entry_"]');
   437→   437→                    if (meEl) {
   438→   438→                        const meMatch = meEl.id.match(/^member_entry_(\d+)$/);
   439→   439→                        if (meMatch) {
   440→   440→                            userId = meMatch[1];
   441→   441→                        }
   442→   442→                    }
   443→   443→                }
   444→   444→
   445→   445→                // Also check ancestor descendants for contactcard with numeric ID
   446→   446→                if (!userId) {
   447→   447→                    const ccEl = card.querySelector('[id^="contactcard_"]');
   448→   448→                    if (ccEl) {
   449→   449→                        const ccMatch = ccEl.id.match(/^contactcard_(\d+)$/);
   450→   450→                        if (ccMatch) {
   451→   451→                            userId = ccMatch[1];
   452→   452→                        }
   453→   453→                    }
   454→   454→                }
   455→   455→

   692→   692→def _save_candidates_to_db(db_manager, candidates: List[CandidateBasic], source_search: str) -> None:
   693→   693→    """
   694→   694→    Save or update candidates in database.
   695→   695→
   696→   696→    Args:
   697→   697→        db_manager: Database manager instance
   698→   698→        candidates: List of candidates to save
   699→   699→        source_search: Search keywords that found these candidates
   700→   700→    """
   701→   701→    try:
   702→   702→        with db_manager.session_scope() as session:
   703→   703→            # Batch pre-fetch existing candidates to avoid N+1 queries
   704→   704→            maimai_ids = [c.maimai_id for c in candidates]

   706→   706→                row.maimai_id: row

   719→   719→                ).all():

   721→   721→
   722→   722→            for c in candidates:
   723→   723→                existing = existing_by_id.get(c.maimai_id)
   724→   724→                if not existing:
   725→   725→                    existing = existing_by_name.get((c.name, c.company))
   726→   726→
   727→   727→                if existing:
   728→   728→                    existing.name = c.name
   729→   729→                    existing.title = c.title
   730→   730→                    existing.company = c.company
   731→   731→                    existing.location = c.location
   732→   732→                    existing.profile_url = c.profile_url
   733→   733→                else:
   734→   734→                    candidate = Candidate(
   735→   735→                        maimai_id=c.maimai_id,
   736→   736→                        name=c.name,

   738→   738→                        company=c.company,
   739→   739→                        location=c.location,
   740→   740→                        profile_url=c.profile_url,

   777→   777→async def _click_avatar_for_profile_url(page: Page, name: str, company: str) -> Optional[str]:
   778→   778→    """Try to extract profile URL for a candidate from the DOM contactcard pattern.
   779→   779→
   780→   780→    Searches the current page for a contactcard element whose parent card
   781→   781→    contains the candidate's name (and optionally company), then constructs
   782→   782→    the profile URL from the extracted user ID.
   783→   783→
   784→   784→    Args:
   785→   785→        page: Playwright page on the search results
   786→   786→        name: Candidate name to match
   787→   787→        company: Candidate company to help disambiguate
   788→   788→
   789→   789→    Returns:
   790→   790→        Profile URL string, or None if not found
   791→   791→    """
   792→   792→    try:
   793→   793→        user_id = await page.evaluate('''(args) => {
   794→   794→            const { name, company } = args;
   795→   795→            // Check contactcard_{numeric_id} ancestors
   796→   796→            const ccElements = document.querySelectorAll('[id^="contactcard_"]');
   797→   797→            for (const el of ccElements) {
   798→   798→                const text = el.textContent || '';
   799→   799→                if (!text.includes(name)) continue;
   800→   800→                if (company && !text.includes(company)) continue;
   801→   801→                const match = el.id.match(/^contactcard_(\\d+)$/);
   802→   802→                if (match) return match[1];
   803→   803→            }
   804→   804→            // Check member_entry_{numeric_id} descendants
   805→   805→            const meElements = document.querySelectorAll('[id^="member_entry_"]');
   806→   806→            for (const el of meElements) {
   807→   807→                // Walk up to find the card containing this member_entry
   808→   808→                let card = el.parentElement;
   809→   809→                for (let i = 0; i < 8 && card; i++) {
   810→   810→                    const text = card.textContent || '';
   811→   811→                    if (text.includes(name) && (!company || text.includes(company))) {
   812→   812→                        const match = el.id.match(/^member_entry_(\\d+)$/);
   813→   813→                        if (match) return match[1];
   814→   814→                    }
   815→   815→                    card = card.parentElement;
   816→   816→                }
   817→   817→            }
   818→   818→            return null;
   819→   819→        }''', {'name': name, 'company': company})
   820→   820→
   821→   821→        if user_id:
   822→   822→            profile_url = f"https://maimai.cn/contact/detail/{user_id}?dstu={user_id}"
   823→   823→            logger.info(f"Resolved profile_url for '{name}' via contactcard: {profile_url}")
   824→   824→            return profile_url
   825→   825→
   826→   826→        logger.debug(f"No contactcard found for '{name}'")
   827→   827→        return None
   828→   828→
   829→   829→    except Exception as e:
   830→   830→        logger.debug(f"_click_avatar_for_profile_url error for '{name}': {e}")
   831→   831→        return None
