    73→def _extract_user_id(profile_url: str) -> Optional[str]:
    74→    """Extract dstu/user_id from a Maimai profile URL for direct chat navigation.
    75→
    76→    Only returns IDs that are known to work with the chat URL (u2= parameter).
    77→    Query-param IDs (dstu, u2, egid) are reliable for chat navigation.
    78→    Path-based IDs (/contact/detail/xxx) are NOT the same as chat user IDs
    79→    and should not be used for direct chat URL construction.
    80→    """
    81→    from urllib.parse import urlparse, parse_qs
    82→
    83→    parsed = urlparse(profile_url)
    84→    params = parse_qs(parsed.query)
    85→    # Only use query parameter IDs — these are confirmed to work with uh_chat_pc
    86→    for key in ("dstu", "u2", "egid"):
    87→        if key in params:
    88→            return params[key][0]
    89→    return None

    92→async def _navigate_to_chat(
    93→    page: Page,
    94→    profile_url: Optional[str],
    95→    config,
    96→    browser: BrowserManager,
    97→    maimai_id: str,
    98→    interactive: bool = False,
    99→    discovered_selectors=None,
   100→) -> bool:
   101→    """Navigate to the chat page for a candidate.
   102→
   103→    Tries in order:
   104→    1. Extract user_id from profile_url query params → direct chat URL
   105→    2. Navigate to profile_url, extract user_id from redirected URL → direct chat URL
   106→    3. Click chat button on profile page as fallback
   107→
   108→    If no profile_url is provided, looks up the candidate's profile_url from DB.
   109→
   110→    Args:
   111→        page: Playwright page instance
   112→        profile_url: Candidate's profile URL (may be None)
   113→        config: Config instance
   114→        browser: Browser manager instance
   115→        maimai_id: Candidate's Maimai ID
   116→        interactive: Enable interactive selector discovery
   117→        discovered_selectors: Shared DiscoveredSelectors instance
   118→
   119→    Returns:
   120→        True if successfully navigated to chat page
   121→    """
   122→    db_manager = get_db_manager()
   123→
   124→    # Resolve profile URL from DB if not provided
   125→    if not profile_url:
   126→        with db_manager.session_scope() as session:
   127→            db_candidate = session.query(Candidate).filter_by(maimai_id=maimai_id).first()
   128→            profile_url = db_candidate.profile_url if db_candidate else None
   129→
   130→        if not profile_url:
   131→            raise PermanentFailureError(f"No profile URL for {maimai_id}")
   132→
   133→    # Try extracting user_id directly from profile URL
   134→    user_id = _extract_user_id(profile_url)
   135→    logger.debug(f"_navigate_to_chat: profile_url={profile_url!r}, extracted user_id={user_id}")
   136→
   137→    if user_id:
   138→        chat_url = MAIMAI_CHAT_URL.format(user_id=user_id)
   139→        await retry_navigation(page, chat_url, wait_until="domcontentloaded")
   140→        await async_random_delay("page_load")
   141→        return True
   142→
   143→    # Navigate to profile page, then try to extract user_id from resulting URL
   144→    logger.debug(f"_navigate_to_chat: navigating to profile {profile_url}")
   145→    await retry_navigation(page, profile_url, wait_until="domcontentloaded")
   146→    await async_random_delay("page_load")
   147→
   148→    current_url = page.url
   149→    logger.debug(f"_navigate_to_chat: landed on {current_url}")
   150→    redirected_user_id = _extract_user_id(current_url)
   151→    if redirected_user_id:
   152→        logger.info(f"Extracted user_id={redirected_user_id} from page URL, redirecting to chat")
   153→        chat_url = MAIMAI_CHAT_URL.format(user_id=redirected_user_id)
   154→        await retry_navigation(page, chat_url, wait_until="domcontentloaded")
   155→        await async_random_delay("page_load")
   156→        return True
   157→
   158→    # Fallback: click chat button on profile page
   159→    chat_button_found = await _click_chat_button(page, config)
   160→    if chat_button_found:
   161→        await async_random_delay("action")
   162→        return True
   163→
   164→    # If interactive mode, trigger selector discovery
   165→    if interactive and discovered_selectors is not None and not discovered_selectors.is_populated:
   166→        from src.modules.selector_discovery import discover_chat_page
   167→
   168→        console.print(
   169→            "[yellow]Chat button not found. Entering interactive discovery mode...[/yellow]"
   170→        )
   171→        result = await discover_chat_page(page, maimai_id, browser)
   172→        if result.is_populated:
   173→            discovered_selectors.chat_button = result.chat_button
   174→            discovered_selectors.message_input = result.message_input
   175→            discovered_selectors.send_button = result.send_button
   176→            discovered_selectors.chat_url_pattern = result.chat_url_pattern
   177→            discovered_selectors.skip_chat_button = result.skip_chat_button
   178→            discovered_selectors.is_populated = True
   179→            return True
   180→
   181→    console.print("[red]Could not find chat button[/red]")
   182→    await browser.screenshot(f"message_error_{maimai_id}")
   183→    return False

   186→async def extract_contact_quota(page: Page) -> Optional[dict]:
   187→    """Extract contact quota from Quick Contact page's embedded share_data.
   188→
   189→    The Quick Contact page server-renders ``window.share_data`` which includes
   190→    ``mem_status.mem_info`` with remaining quotas for various member actions.
   191→
   192→    Args:
   193→        page: Playwright page on a Quick Contact (极速联系) page
   194→
   195→    Returns:
   196→        Dict with quota info, or None if data is unavailable::
   197→
   198→            {
   199→                'uh_left': int,        # 极速联系 remaining
   200→                'addfr_left': int,     # 加好友 remaining
   201→                'dc_left': int,        # 直聊 remaining
   202→                'expire_time': str,    # membership expiry
   203→                'cost': int,           # vouchers consumed per contact
   204→            }
   205→    """
   206→    try:
   207→        quota = await page.evaluate(
   208→            """() => {
   209→            if (window.share_data && window.share_data.mem_status &&
   210→                window.share_data.mem_status.mem_info) {
   211→                const info = window.share_data.mem_status.mem_info;
   212→                return {
   213→                    uh_left: info.uh_left,
   214→                    addfr_left: info.addfr_left,
   215→                    dc_left: info.dc_left,
   216→                    expire_time: info.expire_time || null,
   217→                    cost: (window.share_data.fast_contact_pay_num || {}).final_num || 1,
   218→                };
   219→            }
   220→            return null;
   221→        }"""
   222→        )
   223→        return quota
   224→    except Exception as e:
   225→        logger.debug(f"Failed to extract contact quota: {e}")
   226→        return None

   271→async def _handle_quick_contact(page: Page, content: str) -> Optional[bool]:
   272→    """Handle Maimai's "极速联系" (Quick Contact) page.
   273→
   274→    This page appears instead of regular chat for some candidates. It has a
   275→    category selector, a textarea, and a "提交" (Submit) button.
   276→
   277→    Submission works by manipulating the React component directly: setting
   278→    the message via setState and calling the debounced send method (with
   279→    flush) to trigger the POST to /bizmember/uh-chat/api/send_v2.
   280→
   281→    DOM-level textarea.fill() does NOT update React's controlled state, and
   282→    DOM click on the submit button does NOT reliably trigger the send API.
   283→
   284→    Args:
   285→        page: Playwright page instance
   286→        content: Message content to send
   287→
   288→    Returns:
   289→        True if message was sent via quick contact form,
   290→        False if quick contact page was detected but sending failed,
   291→        None if this is not a quick contact page (caller should continue normal flow)
   292→    """
   293→    try:
   294→        # Detect quick contact page via page title or page content
   295→        is_quick_contact = await page.evaluate(
   296→            """() => {
   297→            const title = document.title || '';
   298→            const bodyText = document.body ? document.body.innerText : '';
   299→            return title.includes('极速联系') || bodyText.includes('极速联系');
   300→        }"""
   301→        )
   302→
   303→        if not is_quick_contact:
   304→            return None
   305→
   306→        logger.info("Detected '极速联系' (Quick Contact) page")
   307→
   308→        # Validate message length (Quick Contact enforces 10-300 chars)
   309→        if len(content) < 10:
   310→            logger.error(
   311→                f"Quick contact: message too short ({len(content)} chars, "
   312→                "minimum 10). Quick Contact requires 10-300 characters."
   313→            )
   314→            return False
   315→        if len(content) > 300:
   316→            logger.error(
   317→                f"Quick contact: message too long ({len(content)} chars, "
   318→                "maximum 300). Quick Contact requires 10-300 characters."
   319→            )
   320→            return False
   321→
   322→        # Extract quota info before submitting (share_data is available on page load)
   323→        quota = await extract_contact_quota(page)
   324→
   325→        # Check if quota is exhausted
   326→        if quota and quota.get("uh_left", 0) <= 0:
   327→            raise QuotaExhaustedError(remaining=quota.get("uh_left", 0))
   328→
   329→        # Install XHR interceptor to capture the send_v2 response
   330→        await page.evaluate(
   331→            """() => {
   332→            window.__sendV2Result = null;
   333→            const origOpen = XMLHttpRequest.prototype.open;
   334→            const origSend = XMLHttpRequest.prototype.send;
   335→            XMLHttpRequest.prototype.open = function(m, u) {
   336→                this.__url = u; this.__method = m;
   337→                return origOpen.apply(this, arguments);
   338→            };
   339→            XMLHttpRequest.prototype.send = function(b) {
   340→                if (this.__url && this.__url.includes('send_v2')) {
   341→                    const xhr = this;
   342→                    xhr.addEventListener('load', function() {
   343→                        try {
   344→                            window.__sendV2Result = {
   345→                                status: xhr.status,
   346→                                response: xhr.responseText,
   347→                            };
   348→                        } catch(e) {
   349→                            window.__sendV2Result = {status: xhr.status, error: e.message};
   350→                        }
   351→                    });
   352→                    xhr.addEventListener('error', function() {
   353→                        window.__sendV2Result = {status: 0, error: 'network_error'};
   354→                    });
   355→                }
   356→                return origSend.apply(this, arguments);
   357→            };
   358→        }"""
   359→        )
   360→
   361→        # Submit via React component internals:
   362→        # 1. Find the class component that owns the form state
   363→        # 2. Set message + unlock push_type via setState
   364→        # 3. Call the debounced send() method and flush() to execute immediately
   365→        send_triggered = await page.evaluate(
   366→            """(msg) => {
   367→            const textarea = document.querySelector('textarea');
   368→            if (!textarea) return {success: false, error: 'no textarea'};
   369→            const keys = Object.keys(textarea);
   370→            const fiberKey = keys.find(
   371→                k => k.startsWith('__reactInternalInstance') || k.startsWith('__reactFiber')
   372→            );
   373→            if (!fiberKey) return {success: false, error: 'no React fiber'};
   374→
   375→            let fiber = textarea[fiberKey];
   376→            while (fiber) {
   377→                if (fiber.stateNode && fiber.stateNode.state
   378→                    && 'message' in fiber.stateNode.state
   379→                    && typeof fiber.stateNode.send === 'function') {
   380→                    const component = fiber.stateNode;
   381→                    // Set message and ensure push_type is not locked (true = sending lock)
   382→                    component.setState({message: msg, push_type: false}, () => {
   383→                        component.send();
   384→                        if (typeof component.send.flush === 'function') {
   385→                            component.send.flush();
   386→                        }
   387→                    });
   388→                    return {success: true};
   389→                }
   390→                fiber = fiber.return;
   391→            }
   392→            return {success: false, error: 'React component not found'};
   393→        }""",
   394→            content,
   395→        )
   396→
   397→        if not send_triggered.get("success"):
   398→            logger.warning(
   399→                f"Quick contact: failed to trigger send via React: "
   400→                f"{send_triggered.get('error', 'unknown')}"
   401→            )
   402→            return False
   403→
   404→        # Wait for the API response (send_v2 XHR)
   405→        send_result = None
   406→        for _ in range(10):
   407→            await asyncio.sleep(0.5)
   408→            send_result = await page.evaluate("() => window.__sendV2Result")
   409→            if send_result is not None:
   410→                break
   411→
   412→        if send_result is None:
   413→            logger.warning("Quick contact: send_v2 API call was not captured (timeout)")
   414→            return False
   415→
   416→        # Parse the API response
   417→        response_text = send_result.get("response", "")
   418→        try:
   419→            resp_data = json.loads(response_text) if response_text else {}
   420→        except (json.JSONDecodeError, TypeError):
   421→            resp_data = {}
   422→
   423→        ret = resp_data.get("ret")
   424→        error_msg = resp_data.get("error", "")
   425→
   426→        if ret == -1:
   427→            logger.warning(f"Quick contact: send failed — {error_msg}")
   428→            if error_msg:
   429→                console.print(f"  [red]Quick Contact 失败: {error_msg}[/red]")
   430→            return False
   431→
   432→        logger.info("Quick contact: message sent successfully")
   433→        await async_random_delay("action")
   434→
   435→        # Display remaining quota
   436→        if quota:
   437→            _display_quota(quota, just_sent=True)
   438→
   439→        return True
   440→
   441→    except PermanentFailureError:
   442→        raise  # Let quota exhaustion etc. propagate to abort the batch
   443→    except Exception as e:
   444→        logger.error(f"Quick contact handling failed: {e}\n{traceback.format_exc()}")
   445→        return False

   448→async def _detect_vip_paywall(page: Page) -> bool:
   449→    """Detect if a VIP/membership paywall is blocking the current page.
   450→
   451→    Non-VIP accounts may be redirected to a recharge/upgrade page when
   452→    trying to message certain candidates. This checks the page URL and
   453→    visible text for common paywall indicators.
   454→
   455→    Args:
   456→        page: Playwright page instance
   457→
   458→    Returns:
   459→        True if a VIP paywall is detected
   460→    """
   461→    try:
   462→        return await page.evaluate(
   463→            """() => {
   464→            const url = window.location.href;
   465→            const urlKeywords = ['vip', 'member', 'pay', 'recharge', 'upgrade', 'subscribe'];
   466→            if (urlKeywords.some(kw => url.toLowerCase().includes(kw))) {
   467→                return true;
   468→            }
   469→
   470→            const title = document.title || '';
   471→            const bodyText = document.body ? document.body.innerText.substring(0, 3000) : '';
   472→            const text = title + ' ' + bodyText;
   473→
   474→            const paywallPatterns = [
   475→                '开通会员', '升级会员', '会员专享', '会员权益',
   476→                '充值', '付费', '购买会员', '开通VIP', '升级VIP',
   477→                '立即开通', '立即升级', '立即购买',
   478→                '会员到期', '续费', '解锁', '次数已用完',
   479→                '今日沟通次数', '沟通人数已达上限',
   480→            ];
   481→
   482→            return paywallPatterns.some(p => text.includes(p));
   483→        }"""
   484→        )
   485→    except Exception as e:
   486→        logger.debug(f"VIP paywall detection failed: {e}")
   487→        return False

   490→async def send_message(
   491→    maimai_id: str,
   492→    template_name: str,
   493→    variables: Dict[str, str],
   494→    browser: Optional[BrowserManager] = None,
   495→    skip_confirmation: bool = False,
   496→    interactive: bool = False,
   497→    discovered_selectors: Optional["DiscoveredSelectors"] = None,
   498→    _skip_rate_check: bool = False,
   499→    profile_url: Optional[str] = None,
   500→) -> bool:
   501→    """
   502→    Send a message to a candidate.
   503→
   504→    Args:
   505→        maimai_id: Candidate's Maimai ID
   506→        template_name: Name of message template to use
   507→        variables: Variables to substitute in template
   508→        browser: Browser manager instance
   509→        skip_confirmation: Skip confirmation (only for testing)
   510→        interactive: Enable interactive selector discovery on failure
   511→        discovered_selectors: Shared DiscoveredSelectors instance for reuse
   512→
   513→    Returns:
   514→        True if message was sent successfully
   515→
   516→    Raises:
   517→        RateLimitExceededError: If rate limit exceeded
   518→        ValidationError: If template or variables invalid
   519→    """
   520→    # Validate and clean inputs
   521→    maimai_id = validate_maimai_id(maimai_id)
   522→
   523→    if browser is None:
   524→        browser = get_browser_manager()
   525→
   526→    config = get_config()
   527→    rate_limiter = get_rate_limiter()
   528→    db_manager = get_db_manager()
   529→
   530→    # Get template
   531→    try:
   532→        template = config.get_template(template_name)
   533→    except Exception as e:
   534→        raise ValidationError(f"Template not found: {template_name}")
   535→
   536→    # Validate template variables
   537→    validate_template_variables(template, variables)
   538→
   539→    # Render message content
   540→    content = render_template(template["content"], variables)
   541→
   542→    # Get candidate info for display
   543→    candidate_info = _get_candidate_info(db_manager, maimai_id)
   544→
   545→    # Preview and confirm
   546→    if not skip_confirmation:
   547→        action = await interactive_confirm(content, candidate_info, template_name)
   548→
   549→        if action == "skip":
   550→            console.print("[yellow]Message skipped[/yellow]")
   551→            _log_message(db_manager, maimai_id, template_name, content, "cancelled")
   552→            return False
   553→
   554→        elif action == "edit":
   555→            # Allow user to edit content
   556→            content = await edit_message(content)
   557→            if content is None:
   558→                console.print("[yellow]Message cancelled[/yellow]")
   559→                return False
   560→
   561→            # Re-confirm after edit
   562→            action = await interactive_confirm(content, candidate_info, template_name)
   563→            if action != "send":
   564→                console.print("[yellow]Message skipped[/yellow]")
   565→                return False
   566→
   567→        elif action != "send":
   568→            console.print("[yellow]Message cancelled[/yellow]")
   569→            return False
   570→
   571→    # Check rate limit (waits through min-interval) - skip when called from batch
   572→    if not _skip_rate_check:
   573→        await async_check_rate_limit("send_message")
   574→
   575→    # Ensure logged in
   576→    await ensure_logged_in(browser)
   577→
   578→    try:
   579→        page = browser.page
   580→
   581→        # If we have populated discovered selectors, use them directly
   582→        if discovered_selectors and discovered_selectors.is_populated:

   750→async def _send_message_batch_common(
   751→    candidates: List[dict],
   752→    template_name: str,
   753→    common_variables: Dict[str, str],
   754→    send_one: Callable[[dict, str], Awaitable[CandidateSendResult]],
   755→    batch_config: BatchConfig,
   756→    skip_messaged: bool = True,
   757→    skip_confirmation: bool = False,
   758→) -> Dict[str, Any]:
   759→    """Common orchestration for all three batch-send paths.
   760→
   761→    Handles: template loading, filtering already-messaged, preview panel,
   762→    confirmation, progress bar, rate limiting, template rendering, and
   763→    result collection.  The actual per-candidate send logic is delegated
   764→    to the ``send_one`` callback.
   765→
   766→    Args:
   767→        candidates: List of dicts with maimai_id, name, company, title
   768→        template_name: Name of message template
   769→        common_variables: Shared template variables
   770→        send_one: async callback(candidate_dict, rendered_content) → CandidateSendResult
   771→        batch_config: Display / behaviour config for this path
   772→        skip_messaged: Skip already-messaged candidates
   773→        skip_confirmation: Skip interactive confirmation prompt
   774→
   775→    Returns:
   776→        Dict with total, sent, failed, skipped, rate_limited, details
   777→    """
   778→    config = get_config()
   779→    db_manager = get_db_manager()
   780→
   781→    try:
   782→        template = config.get_template(template_name)
   783→    except Exception:
   784→        raise ValidationError(f"Template not found: {template_name}")
   785→
   786→    results = {
   787→        "total": len(candidates),
   788→        "sent": 0,
   789→        "failed": 0,
   790→        "skipped": 0,
   791→        "rate_limited": 0,
   792→        "quota_exhausted": 0,
   793→        "details": [],
   794→    }
   795→
   796→    # --- Filter already-messaged candidates ---
   797→    send_list = []
   798→    if skip_messaged:
   799→        all_ids = [c["maimai_id"] for c in candidates]
   800→        with db_manager.session_scope() as session:
   801→            rows = (
   802→                session.query(Message.candidate_maimai_id)
   803→                .filter(Message.candidate_maimai_id.in_(all_ids), Message.status == "sent")
   804→                .distinct()
   805→                .all()
   806→            )
   807→            messaged_ids = {row[0] for row in rows}
   808→
   809→        for c in candidates:
   810→            if c["maimai_id"] in messaged_ids:
   811→                results["skipped"] += 1
   812→                results["details"].append(
   813→                    {
   814→                        "maimai_id": c["maimai_id"],
   815→                        "name": c["name"],
   816→                        "status": "skipped",
   817→                        "reason": "already messaged",
   818→                    }
   819→                )
   820→            else:
   821→                send_list.append(c)
   822→    else:
   823→        send_list = list(candidates)
   824→
   825→    if not send_list:
   826→        if results["skipped"] > 0:
   827→            console.print(
   828→                f"[yellow]All {results['skipped']} candidates already messaged. Skipping send.[/yellow]"
   829→            )
   830→        else:
   831→            console.print("[yellow]No candidates to send messages to.[/yellow]")
   832→        return results
   833→
   834→    # --- Optional pre-validation (Path A) ---
   835→    if batch_config.pre_validate:
   836→        console.print("[dim]Validating template for all candidates...[/dim]")
   837→        for c in send_list:
   838→            test_vars = {
   839→                **common_variables,
   840→                "name": c.get("name", ""),
   841→                "company": c.get("company", ""),
   842→                "position": c.get("title", ""),
   843→            }
   844→            is_valid, missing = validate_template_rendering(template["content"], test_vars, c)
   845→            if not is_valid:
   846→                raise ValidationError(
   847→                    f"候选人 {c.get('name', 'Unknown')} 的模板变量缺失或为空:\n"
   848→                    f"  缺失: {', '.join(missing)}\n"
   849→                    f"  候选人信息: {c}\n"
   850→                    f"  请检查模板或补充变量"
   851→                )
   852→
   853→    # --- Build preview ---
   854→    first = send_list[0]
   855→    first_vars = {
   856→        **common_variables,
   857→        "name": first["name"],
   858→        "company": first["company"],
   859→        "position": first["title"],
   860→    }
   861→    # If pre-validated, skip redundant validation in render
   862→    first_content = render_template(
   863→        template["content"], first_vars, validate=not batch_config.pre_validate
   864→    )
   865→
   866→    summary_text = Text()
   867→    summary_text.append("Template: ", style="bold")
   868→    summary_text.append(f"{template_name}\n")
   869→    if batch_config.mode_label:
   870→        summary_text.append("Mode: ", style="bold")
   871→        summary_text.append(f"{batch_config.mode_label}\n")
   872→    summary_text.append("Candidates: ", style="bold")
   873→    summary_text.append(f"{len(candidates)} found, {results['skipped']} already messaged\n")
   874→    summary_text.append("Will send to: ", style="bold")
   875→    summary_text.append(f"{len(send_list)} candidates\n\n")
   876→    summary_text.append("Message Preview ", style="bold cyan")
   877→    summary_text.append("(first candidate):\n", style="dim")
   878→    summary_text.append("To: ", style="bold cyan")
   879→    recipient = first["name"]
   880→    if first.get("title") and first.get("company"):
   881→        recipient += f" ({first['title']} @ {first['company']})"
   882→    summary_text.append(f"{recipient}\n\n")
   883→    summary_text.append(first_content)
   884→
   885→    console.print(
   886→        Panel(
   887→            summary_text,
   888→            title=batch_config.panel_title,
   889→            border_style="green",
   890→            padding=(1, 2),
   891→        )
   892→    )
   893→
   894→    # --- Confirmation ---
   895→    if skip_confirmation:
   896→        console.print(f"[green]Auto-confirmed: sending to {len(send_list)} candidates[/green]")
   897→    else:
   898→        loop = asyncio.get_running_loop()
   899→        confirm = await loop.run_in_executor(
   900→            None,
   901→            lambda: questionary.confirm(
   902→                f"Confirm batch send to {len(send_list)} candidates?"
   903→            ).unsafe_ask(),
   904→        )
   905→        if not confirm:
   906→            console.print("[yellow]Batch send cancelled.[/yellow]")
   907→            return results
   908→
   909→    # --- Send loop ---
   910→    from rich.progress import (
   911→        Progress,
   912→        BarColumn,
   913→        TextColumn as ProgressTextColumn,
   914→        MofNCompleteColumn,
   915→    )
   916→
   917→    with Progress(
   918→        ProgressTextColumn("[progress.description]{task.description}"),
   919→        BarColumn(),
   920→        MofNCompleteColumn(),
   921→        console=console,
   922→    ) as progress:
   923→        task = progress.add_task(batch_config.progress_description, total=len(send_list))
   924→
   925→        for c in send_list:
   926→            # Rate limit check
   927→            try:
   928→                await async_check_rate_limit("send_message")
   929→            except RateLimitExceededError as e:
   930→                console.print(f"\n[yellow]Rate limit reached: {e}[/yellow]")
   931→                remaining = len(send_list) - (results["sent"] + results["failed"])
   932→                results["rate_limited"] = remaining
   933→                results["details"].append(
   934→                    {
   935→                        "maimai_id": c["maimai_id"],
   936→                        "name": c["name"],
   937→                        "status": "rate_limited",
   938→                        "reason": str(e),
   939→                    }
   940→                )
   941→                break
   942→
   943→            # Build per-candidate variables and render template
   944→            variables = {
   945→                **common_variables,
   946→                "name": c["name"],
   947→                "company": c["company"],
   948→                "position": c["title"],
   949→            }
   950→            try:
   951→                content = render_template(
   952→                    template["content"],
   953→                    variables,
   954→                    validate=not batch_config.pre_validate,
   955→                )
   956→            except ValidationError as e:
   957→                logger.error(f"Template rendering failed for {c['name']}: {e}")
   958→                results["failed"] += 1
   959→                results["details"].append(
   960→                    {
   961→                        "maimai_id": c["maimai_id"],
   962→                        "name": c["name"],
   963→                        "status": "failed",
   964→                        "reason": f"模板渲染失败: {e}",
   965→                    }
   966→                )
   967→                progress.advance(task)
   968→                continue
   969→
   970→            # Delegate to path-specific sender
   971→            result = await send_one(c, content)
   972→
   973→            # Record result
   974→            if result.status == "sent":
   975→                results["sent"] += 1
   976→            elif result.status in ("quota_exhausted", "rate_limited"):
   977→                if result.status == "rate_limited":
   978→                    remaining = len(send_list) - (results["sent"] + results["failed"])
   979→                    results["rate_limited"] = remaining
   980→                else:
   981→                    results["quota_exhausted"] += 1
   982→            else:
   983→                results["failed"] += 1
   984→
   985→            detail = {
   986→                "maimai_id": c["maimai_id"],
   987→                "name": c["name"],
   988→                "status": result.status,
   989→            }
   990→            if result.reason:
   991→                detail["reason"] = result.reason
   992→            if result.attempts > 1:
   993→                detail["attempts"] = result.attempts
   994→            if result.content:
   995→                detail["content"] = result.content
   996→            elif content:
   997→                detail["content"] = content
   998→            results["details"].append(detail)
   999→
  1000→            if result.abort_batch:
  1001→                remaining = len(send_list) - (results["sent"] + results["failed"])
  1002→                if remaining > 0:
  1003→                    console.print(f"[yellow]跳过剩余 {remaining} 位候选人[/yellow]")
  1004→                break
  1005→
  1006→            progress.advance(task)
  1007→
  1008→            await async_random_delay("message")
  1009→
  1010→    return results

  1018→def _make_search_center_sender(
  1019→    browser: BrowserManager,
  1020→    template_name: str,
  1021→    interactive: bool,
  1022→    discovered_selectors,
  1023→) -> Callable[[dict, str], Awaitable[CandidateSendResult]]:
  1024→    """Create a sender callback for search-center (Path A) with retry + timeout."""
  1025→
  1026→    max_retries = 2
  1027→    per_candidate_timeout = 90
  1028→
  1029→    async def _send(c: dict, content: str) -> CandidateSendResult:
  1030→        variables = {
  1031→            "name": c["name"],
  1032→            "company": c["company"],
  1033→            "position": c["title"],
  1034→        }
  1035→        last_error = None
  1036→        for attempt in range(1 + max_retries):
  1037→            try:
  1038→                success = await asyncio.wait_for(
  1039→                    send_message(
  1040→                        maimai_id=c["maimai_id"],
  1041→                        template_name=template_name,
  1042→                        variables=variables,
  1043→                        browser=browser,
  1044→                        skip_confirmation=True,
  1045→                        interactive=interactive,
  1046→                        discovered_selectors=discovered_selectors,
  1047→                        _skip_rate_check=True,
  1048→                        profile_url=c.get("profile_url"),
  1049→                    ),
  1050→                    timeout=per_candidate_timeout,
  1051→                )
  1052→                if success:
  1053→                    return CandidateSendResult(
  1054→                        success=True,
  1055→                        status="sent",
  1056→                        attempts=attempt + 1,
  1057→                        content=content,
  1058→                    )
  1059→                last_error = "send returned False"
  1060→
  1061→            except asyncio.TimeoutError:
  1062→                console.print(
  1063→                    f"[yellow]Timeout for {c['name']} ({c['maimai_id']}), skipping.[/yellow]"
  1064→                )
  1065→                return CandidateSendResult(
  1066→                    success=False,
  1067→                    status="failed",
  1068→                    reason=f"Timed out after {per_candidate_timeout}s",
  1069→                    attempts=attempt + 1,
  1070→                    content=content,
  1071→                )
  1072→            except QuotaExhaustedError as e:
  1073→                console.print(f"\n[red]联系券已用尽，终止批量发送[/red]")
  1074→                return CandidateSendResult(
  1075→                    success=False,
  1076→                    status="quota_exhausted",
  1077→                    reason=str(e),
  1078→                    attempts=attempt + 1,
  1079→                    content=content,
  1080→                    abort_batch=True,
  1081→                )
  1082→            except PermanentFailureError as e:
  1083→                console.print(f"[yellow]{e}. Skipping.[/yellow]")
  1084→                return CandidateSendResult(
  1085→                    success=False,
  1086→                    status="failed",
  1087→                    reason=str(e),
  1088→                    attempts=attempt + 1,
  1089→                    content=content,
  1090→                )
  1091→            except RateLimitExceededError as e:
  1092→                console.print(f"\n[yellow]Rate limit reached: {e}[/yellow]")
  1093→                return CandidateSendResult(
  1094→                    success=False,
  1095→                    status="rate_limited",
  1096→                    reason=str(e),
  1097→                    attempts=attempt + 1,
  1098→                    content=content,
  1099→                    abort_batch=True,
  1100→                )
  1101→            except Exception as e:
  1102→                last_error = str(e)
  1103→
  1104→            # Retry with exponential backoff
  1105→            if attempt < max_retries:
  1106→                from src.utils.delays import get_retry_delay
  1107→
  1108→                delay = get_retry_delay(attempt, base_delay=3.0, max_delay=30.0)
  1109→                logger.info(f"Retry {attempt + 1}/{max_retries} for {c['name']} in {delay:.1f}s")
  1110→                await asyncio.sleep(delay)
  1111→
  1112→        # All retries exhausted
  1113→        actual_attempts = max_retries + 1
  1114→        logger.error(
  1115→            f"Failed to send message to {c['name']} after {actual_attempts} attempt(s): {last_error}"
  1116→        )
  1117→        return CandidateSendResult(
  1118→            success=False,
  1119→            status="failed",
  1120→            reason=last_error or "unknown error",
  1121→            attempts=actual_attempts,
  1122→            content=content,
  1123→        )
  1124→
  1125→    return _send

  1302→    """
  1303→    Render template with variables.

  1615→async def _verify_message_sent(page: Page, content: str, timeout: int = 5000) -> bool:
  1616→    """
  1617→    Verify that a message was actually sent by checking the conversation area.
  1618→
  1619→    Polls up to `timeout` ms with short intervals to handle async rendering.
  1620→
  1621→    Checks for:
  1622→    1. The sent message content appearing in the conversation
  1623→    2. The message input being cleared (indicating successful send)
  1624→    3. A sent/delivered indicator element
  1625→
  1626→    Args:
  1627→        page: Playwright page instance
  1628→        content: The message content that was sent
  1629→        timeout: Max time to wait for verification in ms
  1630→
  1631→    Returns:
  1632→        True if message send was verified
  1633→    """
  1634→    import time
  1635→
  1636→    # Short snippet for matching (first 20 chars, avoid template artifacts)
  1637→    snippet = content[:20].strip()
  1638→
  1639→    poll_interval_ms = 500
  1640→    deadline = time.monotonic() + timeout / 1000.0
  1641→
  1642→    while True:
  1643→        try:
  1644→            # Strategy 1: Check if message content appears in conversation area
  1645→            verified = await page.evaluate(
  1646→                """(snippet) => {
  1647→                const chatSelectors = [
  1648→                    '[class*="message-list"]',
  1649→                    '[class*="chat-content"]',
  1650→                    '[class*="conversation"]',
  1651→                    '[class*="msg-list"]',
  1652→                    '[class*="im-message"]',
  1653→                    '[class*="chat-msg"]',
  1654→                ];
  1655→
  1656→                for (const sel of chatSelectors) {
  1657→                    const container = document.querySelector(sel);
  1658→                    if (container && container.textContent.includes(snippet)) {
  1659→                        return true;
  1660→                    }
  1661→                }
  1662→
  1663→                // Fallback: scan leaf-ish elements in the lower portion of the page
  1664→                const viewportHeight = window.innerHeight;
  1665→                const elements = document.querySelectorAll('div, p, span');
  1666→                for (const el of elements) {
  1667→                    const rect = el.getBoundingClientRect();
  1668→                    if (rect.top > viewportHeight * 0.3 && el.textContent.includes(snippet)) {
  1669→                        if (el.textContent.length < snippet.length + 200) {
  1670→                            return true;
  1671→                        }
  1672→                    }
  1673→                }
  1674→
  1675→                return false;
  1676→            }""",
  1677→                snippet,
  1678→            )
  1679→
  1680→            if verified:
  1681→                logger.debug("Message send verified: content found in conversation")
  1682→                return True
  1683→
  1684→            # Strategy 2: Check if the chat input was cleared (indicates successful send)
  1685→            # Only check textarea/contenteditable inside chat containers to avoid
  1686→            # false positives from search boxes or other form inputs.
  1687→            input_cleared = await page.evaluate(
  1688→                """() => {
  1689→                const chatContainers = document.querySelectorAll(
  1690→                    '[class*="chat"], [class*="im-"], [class*="message"], [class*="editor"]'
  1691→                );
  1692→                for (const container of chatContainers) {
  1693→                    const inputs = container.querySelectorAll(
  1694→                        'textarea, [contenteditable="true"]'
  1695→                    );
  1696→                    for (const input of inputs) {
  1697→                        const rect = input.getBoundingClientRect();
  1698→                        if (rect.width > 0 && rect.height > 0) {
  1699→                            const value = input.value || input.textContent || '';
  1700→                            if (value.trim() === '') {
  1701→                                return true;
  1702→                            }
  1703→                        }
  1704→                    }
  1705→                }
  1706→                return false;
  1707→            }"""
  1708→            )
  1709→
  1710→            if input_cleared:
  1711→                logger.debug("Message send verified: input field was cleared")
  1712→                return True
  1713→
  1714→            # Strategy 3: Check for sent/delivered indicators
  1715→            sent_indicators = [
  1716→                '[class*="sent"]',
  1717→                '[class*="delivered"]',
  1718→                '[class*="success"]',
  1719→                '[class*="msg-status"]',
  1720→            ]
  1721→            for selector in sent_indicators:
  1722→                try:
  1723→                    indicator = await page.query_selector(selector)
  1724→                    if indicator and await indicator.is_visible():
  1725→                        logger.debug(f"Message send verified: indicator found ({selector})")
  1726→                        return True
  1727→                except Exception:
  1728→                    continue
  1729→
  1730→        except Exception as e:
  1731→            logger.warning(f"Message verification poll failed: {e}")
  1732→
  1733→        # Check if we've exceeded the deadline

  1743→async def _check_existing_conversation(page: Page, maimai_id: str) -> bool:
  1744→    """
  1745→    Check if there's already an existing conversation with this candidate.
  1746→
  1747→    Checks for actual user message bubbles (not system prompts/tips) in the
  1748→    conversation area.
  1749→
  1750→    Args:
  1751→        page: Playwright page instance
  1752→        maimai_id: Candidate's Maimai ID
  1753→
  1754→    Returns:
  1755→        True if an existing conversation was detected
  1756→    """
  1757→    try:
  1758→        has_conversation = await page.evaluate(
  1759→            """() => {
  1760→            // Look for conversation message containers
  1761→            const chatSelectors = [
  1762→                '[class*="message-list"]',
  1763→                '[class*="chat-content"]',
  1764→                '[class*="conversation"]',
  1765→                '[class*="msg-list"]',
  1766→                '[class*="im-message"]',
  1767→                '[class*="chat-msg"]',
  1768→                '[class*="chat-record"]',
  1769→            ];
  1770→
  1771→            // System/tip patterns that should NOT count as existing conversation
  1772→            const systemPatterns = [
  1773→                '还不是好友', '添加好友', '发送好友申请', '对方未回复',
  1774→                '以上是打招呼内容', '暂无消息', '开始聊天', '你们已经是好友',
  1775→                '系统消息', '温馨提示', '安全提示', '注意事项',
  1776→                '开启招呼语', '求职意向', '目前情况', '竞业点',
  1777→                '欢迎联系', '不感兴趣',
  1778→            ];
  1779→
  1780→            const isSystemText = (text) => {
  1781→                const t = text.trim();
  1782→                if (t.length < 5) return true;  // Too short to be a real message
  1783→                return systemPatterns.some(p => t.includes(p));
  1784→            };
  1785→
  1786→            for (const sel of chatSelectors) {
  1787→                const container = document.querySelector(sel);
  1788→                if (!container) continue;
  1789→
  1790→                // Strategy 1: Look for message bubble elements
  1791→                const bubbleSelectors = [
  1792→                    '[class*="msg-item"]', '[class*="message-item"]',
  1793→                    '[class*="bubble"]', '[class*="chat-item"]',
  1794→                    '[class*="msg-bubble"]', '[class*="chat-bubble"]',
  1795→                ];
  1796→                for (const bSel of bubbleSelectors) {
  1797→                    const bubbles = container.querySelectorAll(bSel);
  1798→                    // Filter out system/tip bubbles
  1799→                    let realMessageCount = 0;
  1800→                    for (const bubble of bubbles) {
  1801→                        const text = bubble.textContent.trim();
  1802→                        if (text.length > 5 && !isSystemText(text)) {
  1803→                            realMessageCount++;
  1804→                        }
  1805→                    }
  1806→                    if (realMessageCount > 0) {
  1807→                        return true;
  1808→                    }
  1809→                }
  1810→
  1811→                // Strategy 2: If no bubble selectors matched, check direct children
  1812→                // that look like message blocks (not system tips)
  1813→                const children = container.children;
  1814→                let realChildCount = 0;
  1815→                for (const child of children) {
  1816→                    const text = child.textContent.trim();
  1817→                    // Must have meaningful content and not be a system message
  1818→                    if (text.length > 10 && !isSystemText(text)) {
  1819→                        // Check it's not a timestamp-only or status-only element
  1820→                        if (child.querySelector('img[class*="avatar"]') || text.length > 20) {
  1821→                            realChildCount++;
  1822→                        }
  1823→                    }
  1824→                }
  1825→                if (realChildCount > 0) {
  1826→                    return true;
  1827→                }
  1828→            }
  1829→            return false;
  1830→        }"""
  1831→        )
  1832→
  1833→        if has_conversation:
  1834→            logger.info(f"Existing conversation detected for {maimai_id}")
  1835→        return has_conversation
  1836→
  1837→    except Exception as e:
  1838→        # Conservative: assume conversation exists to avoid duplicate messages.
  1839→        # The caller logs this as "cancelled" in the audit DB.
  1840→        logger.warning(
  1841→            f"Conversation check failed for {maimai_id}: {e} — skipping to avoid potential duplicate"
  1842→        )
  1843→        return True
