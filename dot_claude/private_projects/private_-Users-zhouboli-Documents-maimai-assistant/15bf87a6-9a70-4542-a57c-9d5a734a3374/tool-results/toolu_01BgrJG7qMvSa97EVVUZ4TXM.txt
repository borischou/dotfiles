    76→def _extract_user_id(profile_url: str) -> Optional[str]:
    77→    """Extract dstu/user_id from a Maimai profile URL for direct chat navigation.
    78→
    79→    Only returns IDs that are known to work with the chat URL (u2= parameter).
    80→    Query-param IDs (dstu, u2, egid) are reliable for chat navigation.
    81→    Path-based IDs (/contact/detail/xxx) are NOT the same as chat user IDs
    82→    and should not be used for direct chat URL construction.
    83→    """
    84→    from urllib.parse import urlparse, parse_qs
    85→
    86→    parsed = urlparse(profile_url)
    87→    params = parse_qs(parsed.query)
    88→    # Only use query parameter IDs — these are confirmed to work with uh_chat_pc
    89→    for key in ("dstu", "u2", "egid"):
    90→        if key in params:
    91→            return params[key][0]
    92→    return None

    95→async def _navigate_to_chat(
    96→    page: Page,
    97→    profile_url: Optional[str],
    98→    config,
    99→    browser: BrowserManager,
   100→    maimai_id: str,
   101→    interactive: bool = False,
   102→    discovered_selectors=None,
   103→) -> bool:
   104→    """Navigate to the chat page for a candidate.
   105→
   106→    Tries in order:
   107→    1. Extract user_id from profile_url query params → direct chat URL
   108→    2. Navigate to profile_url, extract user_id from redirected URL → direct chat URL
   109→    3. Click chat button on profile page as fallback
   110→
   111→    If no profile_url is provided, looks up the candidate's profile_url from DB.
   112→
   113→    Args:
   114→        page: Playwright page instance
   115→        profile_url: Candidate's profile URL (may be None)
   116→        config: Config instance
   117→        browser: Browser manager instance
   118→        maimai_id: Candidate's Maimai ID
   119→        interactive: Enable interactive selector discovery
   120→        discovered_selectors: Shared DiscoveredSelectors instance
   121→
   122→    Returns:
   123→        True if successfully navigated to chat page
   124→    """
   125→    db_manager = get_db_manager()
   126→
   127→    # Resolve profile URL from DB if not provided
   128→    if not profile_url:
   129→        with db_manager.session_scope() as session:
   130→            db_candidate = session.query(Candidate).filter_by(maimai_id=maimai_id).first()
   131→            profile_url = db_candidate.profile_url if db_candidate else None
   132→
   133→        if not profile_url:
   134→            raise PermanentFailureError(f"No profile URL for {maimai_id}")
   135→
   136→    # Try extracting user_id directly from profile URL
   137→    user_id = _extract_user_id(profile_url)
   138→    logger.debug(f"_navigate_to_chat: profile_url={profile_url!r}, extracted user_id={user_id}")
   139→
   140→    if user_id:
   141→        chat_url = MAIMAI_CHAT_URL.format(user_id=user_id)
   142→        await retry_navigation(page, chat_url, wait_until="domcontentloaded")
   143→        await async_random_delay("page_load")
   144→        return True
   145→
   146→    # Navigate to profile page, then try to extract user_id from resulting URL
   147→    logger.debug(f"_navigate_to_chat: navigating to profile {profile_url}")
   148→    await retry_navigation(page, profile_url, wait_until="domcontentloaded")
   149→    await async_random_delay("page_load")
   150→
   151→    current_url = page.url
   152→    logger.debug(f"_navigate_to_chat: landed on {current_url}")
   153→    redirected_user_id = _extract_user_id(current_url)
   154→    if redirected_user_id:
   155→        logger.info(f"Extracted user_id={redirected_user_id} from page URL, redirecting to chat")
   156→        chat_url = MAIMAI_CHAT_URL.format(user_id=redirected_user_id)
   157→        await retry_navigation(page, chat_url, wait_until="domcontentloaded")
   158→        await async_random_delay("page_load")
   159→        return True
   160→
   161→    # Fallback: click chat button on profile page
   162→    chat_button_found = await _click_chat_button(page, config)
   163→    if chat_button_found:
   164→        await async_random_delay("action")
   165→        return True
   166→
   167→    # If interactive mode, trigger selector discovery
   168→    if interactive and discovered_selectors is not None and not discovered_selectors.is_populated:
   169→        from src.modules.selector_discovery import discover_chat_page
   170→
   171→        console.print(
   172→            "[yellow]Chat button not found. Entering interactive discovery mode...[/yellow]"
   173→        )
   174→        result = await discover_chat_page(page, maimai_id, browser)
   175→        if result.is_populated:
   176→            discovered_selectors.chat_button = result.chat_button
   177→            discovered_selectors.message_input = result.message_input
   178→            discovered_selectors.send_button = result.send_button
   179→            discovered_selectors.chat_url_pattern = result.chat_url_pattern
   180→            discovered_selectors.skip_chat_button = result.skip_chat_button
   181→            discovered_selectors.is_populated = True
   182→            return True
   183→
   184→    console.print("[red]Could not find chat button[/red]")
   185→    await browser.screenshot(f"message_error_{maimai_id}")
   186→    return False

   189→async def extract_contact_quota(page: Page) -> Optional[dict]:
   190→    """Extract contact quota from Quick Contact page's embedded share_data.
   191→
   192→    The Quick Contact page server-renders ``window.share_data`` which includes
   193→    ``mem_status.mem_info`` with remaining quotas for various member actions.
   194→
   195→    Args:
   196→        page: Playwright page on a Quick Contact (极速联系) page
   197→
   198→    Returns:
   199→        Dict with quota info, or None if data is unavailable::
   200→
   201→            {
   202→                'uh_left': int,        # 极速联系 remaining
   203→                'addfr_left': int,     # 加好友 remaining
   204→                'dc_left': int,        # 直聊 remaining
   205→                'expire_time': str,    # membership expiry
   206→                'cost': int,           # vouchers consumed per contact
   207→            }
   208→    """
   209→    try:
   210→        quota = await page.evaluate(
   211→            """() => {
   212→            if (window.share_data && window.share_data.mem_status &&
   213→                window.share_data.mem_status.mem_info) {
   214→                const info = window.share_data.mem_status.mem_info;
   215→                return {
   216→                    uh_left: info.uh_left,
   217→                    addfr_left: info.addfr_left,
   218→                    dc_left: info.dc_left,
   219→                    expire_time: info.expire_time || null,
   220→                    cost: (window.share_data.fast_contact_pay_num || {}).final_num || 1,
   221→                };
   222→            }
   223→            return null;
   224→        }"""
   225→        )
   226→        return quota
   227→    except Exception as e:
   228→        logger.debug(f"Failed to extract contact quota: {e}")
   229→        return None

   274→async def _handle_quick_contact(page: Page, content: str) -> Optional[bool]:
   275→    """Handle Maimai's "极速联系" (Quick Contact) page.
   276→
   277→    This page appears instead of regular chat for some candidates. It has a
   278→    category selector, a textarea, and a "提交" (Submit) button.
   279→
   280→    Submission works by manipulating the React component directly: setting
   281→    the message via setState and calling the debounced send method (with
   282→    flush) to trigger the POST to /bizmember/uh-chat/api/send_v2.
   283→
   284→    DOM-level textarea.fill() does NOT update React's controlled state, and
   285→    DOM click on the submit button does NOT reliably trigger the send API.
   286→
   287→    Args:
   288→        page: Playwright page instance
   289→        content: Message content to send
   290→
   291→    Returns:
   292→        True if message was sent via quick contact form,
   293→        False if quick contact page was detected but sending failed,
   294→        None if this is not a quick contact page (caller should continue normal flow)
   295→    """
   296→    try:
   297→        # Detect quick contact page via page title or page content
   298→        is_quick_contact = await page.evaluate(
   299→            """() => {
   300→            const title = document.title || '';
   301→            const bodyText = document.body ? document.body.innerText : '';
   302→            return title.includes('极速联系') || bodyText.includes('极速联系');
   303→        }"""
   304→        )
   305→
   306→        if not is_quick_contact:
   307→            return None
   308→
   309→        logger.info("Detected '极速联系' (Quick Contact) page")
   310→
   311→        # Validate message length (Quick Contact enforces 10-300 chars)
   312→        if len(content) < 10:
   313→            logger.error(
   314→                f"Quick contact: message too short ({len(content)} chars, "
   315→                "minimum 10). Quick Contact requires 10-300 characters."
   316→            )
   317→            return False
   318→        if len(content) > 300:
   319→            logger.error(
   320→                f"Quick contact: message too long ({len(content)} chars, "
   321→                "maximum 300). Quick Contact requires 10-300 characters."
   322→            )
   323→            return False
   324→
   325→        # Extract quota info before submitting (share_data is available on page load)
   326→        quota = await extract_contact_quota(page)
   327→
   328→        # Check if quota is exhausted
   329→        if quota and quota.get("uh_left", 0) <= 0:
   330→            raise QuotaExhaustedError(remaining=quota.get("uh_left", 0))
   331→
   332→        # Install XHR interceptor to capture the send_v2 response
   333→        await page.evaluate(
   334→            """() => {
   335→            window.__sendV2Result = null;
   336→            const origOpen = XMLHttpRequest.prototype.open;
   337→            const origSend = XMLHttpRequest.prototype.send;
   338→            XMLHttpRequest.prototype.open = function(m, u) {
   339→                this.__url = u; this.__method = m;
   340→                return origOpen.apply(this, arguments);
   341→            };
   342→            XMLHttpRequest.prototype.send = function(b) {
   343→                if (this.__url && this.__url.includes('send_v2')) {
   344→                    const xhr = this;
   345→                    xhr.addEventListener('load', function() {
   346→                        try {
   347→                            window.__sendV2Result = {
   348→                                status: xhr.status,
   349→                                response: xhr.responseText,
   350→                            };
   351→                        } catch(e) {
   352→                            window.__sendV2Result = {status: xhr.status, error: e.message};
   353→                        }
   354→                    });
   355→                    xhr.addEventListener('error', function() {
   356→                        window.__sendV2Result = {status: 0, error: 'network_error'};
   357→                    });
   358→                }
   359→                return origSend.apply(this, arguments);
   360→            };
   361→        }"""
   362→        )
   363→
   364→        # Submit via React component internals:
   365→        # 1. Find the class component that owns the form state
   366→        # 2. Set message + unlock push_type via setState
   367→        # 3. Call the debounced send() method and flush() to execute immediately
   368→        send_triggered = await page.evaluate(
   369→            """(msg) => {
   370→            const textarea = document.querySelector('textarea');
   371→            if (!textarea) return {success: false, error: 'no textarea'};
   372→            const keys = Object.keys(textarea);
   373→            const fiberKey = keys.find(
   374→                k => k.startsWith('__reactInternalInstance') || k.startsWith('__reactFiber')
   375→            );
   376→            if (!fiberKey) return {success: false, error: 'no React fiber'};
   377→
   378→            let fiber = textarea[fiberKey];
   379→            while (fiber) {
   380→                if (fiber.stateNode && fiber.stateNode.state
   381→                    && 'message' in fiber.stateNode.state
   382→                    && typeof fiber.stateNode.send === 'function') {
   383→                    const component = fiber.stateNode;
   384→                    // Set message and ensure push_type is not locked (true = sending lock)
   385→                    component.setState({message: msg, push_type: false}, () => {
   386→                        component.send();
   387→                        if (typeof component.send.flush === 'function') {
   388→                            component.send.flush();
   389→                        }
   390→                    });
   391→                    return {success: true};
   392→                }
   393→                fiber = fiber.return;
   394→            }
   395→            return {success: false, error: 'React component not found'};
   396→        }""",
   397→            content,
   398→        )
   399→
   400→        if not send_triggered.get("success"):
   401→            logger.warning(
   402→                f"Quick contact: failed to trigger send via React: "
   403→                f"{send_triggered.get('error', 'unknown')}"
   404→            )
   405→            return False
   406→
   407→        # Wait for the API response (send_v2 XHR)
   408→        send_result = None
   409→        for _ in range(10):
   410→            await asyncio.sleep(0.5)
   411→            send_result = await page.evaluate("() => window.__sendV2Result")
   412→            if send_result is not None:
   413→                break
   414→
   415→        if send_result is None:
   416→            logger.warning("Quick contact: send_v2 API call was not captured (timeout)")
   417→            return False
   418→
   419→        # Parse the API response
   420→        response_text = send_result.get("response", "")
   421→        try:
   422→            resp_data = json.loads(response_text) if response_text else {}
   423→        except (json.JSONDecodeError, TypeError):
   424→            resp_data = {}
   425→
   426→        ret = resp_data.get("ret")
   427→        error_msg = resp_data.get("error", "")
   428→
   429→        if ret == -1:
   430→            logger.warning(f"Quick contact: send failed — {error_msg}")
   431→            if error_msg:
   432→                console.print(f"  [red]Quick Contact 失败: {error_msg}[/red]")
   433→            return False
   434→
   435→        logger.info("Quick contact: message sent successfully")
   436→        await async_random_delay("action")
   437→
   438→        # Display remaining quota
   439→        if quota:
   440→            _display_quota(quota, just_sent=True)
   441→
   442→        return True
   443→
   444→    except PermanentFailureError:
   445→        raise  # Let quota exhaustion etc. propagate to abort the batch
   446→    except Exception as e:
   447→        logger.error(f"Quick contact handling failed: {e}\n{traceback.format_exc()}")
   448→        return False

   451→async def _detect_vip_paywall(page: Page) -> bool:
   452→    """Detect if a VIP/membership paywall is blocking the current page.
   453→
   454→    Non-VIP accounts may be redirected to a recharge/upgrade page when
   455→    trying to message certain candidates. This checks the page URL and
   456→    visible text for common paywall indicators.
   457→
   458→    Args:
   459→        page: Playwright page instance
   460→
   461→    Returns:
   462→        True if a VIP paywall is detected
   463→    """
   464→    try:
   465→        return await page.evaluate(
   466→            """() => {
   467→            const url = window.location.href;
   468→            const urlKeywords = ['vip', 'member', 'pay', 'recharge', 'upgrade', 'subscribe'];
   469→            if (urlKeywords.some(kw => url.toLowerCase().includes(kw))) {
   470→                return true;
   471→            }
   472→
   473→            const title = document.title || '';
   474→            const bodyText = document.body ? document.body.innerText.substring(0, 3000) : '';
   475→            const text = title + ' ' + bodyText;
   476→
   477→            const paywallPatterns = [
   478→                '开通会员', '升级会员', '会员专享', '会员权益',
   479→                '充值', '付费', '购买会员', '开通VIP', '升级VIP',
   480→                '立即开通', '立即升级', '立即购买',
   481→                '会员到期', '续费', '解锁', '次数已用完',
   482→                '今日沟通次数', '沟通人数已达上限',
   483→            ];
   484→
   485→            return paywallPatterns.some(p => text.includes(p));
   486→        }"""
   487→        )
   488→    except Exception as e:
   489→        logger.debug(f"VIP paywall detection failed: {e}")
   490→        return False

   493→async def send_message(
   494→    maimai_id: str,
   495→    template_name: str,
   496→    variables: Dict[str, str],
   497→    browser: Optional[BrowserManager] = None,
   498→    skip_confirmation: bool = False,
   499→    interactive: bool = False,
   500→    discovered_selectors: Optional["DiscoveredSelectors"] = None,
   501→    _skip_rate_check: bool = False,
   502→    profile_url: Optional[str] = None,
   503→) -> bool:
   504→    """
   505→    Send a message to a candidate.
   506→
   507→    Args:
   508→        maimai_id: Candidate's Maimai ID
   509→        template_name: Name of message template to use
   510→        variables: Variables to substitute in template
   511→        browser: Browser manager instance
   512→        skip_confirmation: Skip confirmation (only for testing)
   513→        interactive: Enable interactive selector discovery on failure
   514→        discovered_selectors: Shared DiscoveredSelectors instance for reuse
   515→
   516→    Returns:
   517→        True if message was sent successfully
   518→
   519→    Raises:
   520→        RateLimitExceededError: If rate limit exceeded
   521→        ValidationError: If template or variables invalid
   522→    """
   523→    # Validate and clean inputs
   524→    maimai_id = validate_maimai_id(maimai_id)
   525→
   526→    if browser is None:
   527→        browser = get_browser_manager()
   528→
   529→    config = get_config()
   530→    rate_limiter = get_rate_limiter()
   531→    db_manager = get_db_manager()
   532→
   533→    # Get template
   534→    try:
   535→        template = config.get_template(template_name)
   536→    except Exception as e:
   537→        raise ValidationError(f"Template not found: {template_name}")
   538→
   539→    # Validate template variables
   540→    validate_template_variables(template, variables)
   541→
   542→    # Render message content
   543→    content = render_template(template["content"], variables)
   544→
   545→    # Get candidate info for display
   546→    candidate_info = _get_candidate_info(db_manager, maimai_id)
   547→
   548→    # Preview and confirm
   549→    if not skip_confirmation:
   550→        action = await interactive_confirm(content, candidate_info, template_name)
   551→
   552→        if action == "skip":
   553→            console.print("[yellow]Message skipped[/yellow]")
   554→            _log_message(db_manager, maimai_id, template_name, content, "cancelled")
   555→            return False
   556→
   557→        elif action == "edit":
   558→            # Allow user to edit content
   559→            content = await edit_message(content)
   560→            if content is None:
   561→                console.print("[yellow]Message cancelled[/yellow]")
   562→                return False
   563→
   564→            # Re-confirm after edit
   565→            action = await interactive_confirm(content, candidate_info, template_name)
   566→            if action != "send":
   567→                console.print("[yellow]Message skipped[/yellow]")
   568→                return False
   569→
   570→        elif action != "send":
   571→            console.print("[yellow]Message cancelled[/yellow]")
   572→            return False
   573→
   574→    # Check rate limit (waits through min-interval) - skip when called from batch
   575→    if not _skip_rate_check:
   576→        await async_check_rate_limit("send_message")
   577→
   578→    # Ensure logged in
   579→    await ensure_logged_in(browser)
   580→
   581→    try:
   582→        page = browser.page
   583→

   753→async def _send_message_batch_common(
   754→    candidates: List[dict],
   755→    template_name: str,
   756→    common_variables: Dict[str, str],
   757→    send_one: Callable[[dict, str], Awaitable[CandidateSendResult]],
   758→    batch_config: BatchConfig,
   759→    skip_messaged: bool = True,
   760→    skip_confirmation: bool = False,
   761→) -> Dict[str, Any]:
   762→    """Common orchestration for all three batch-send paths.
   763→
   764→    Handles: template loading, filtering already-messaged, preview panel,
   765→    confirmation, progress bar, rate limiting, template rendering, and
   766→    result collection.  The actual per-candidate send logic is delegated
   767→    to the ``send_one`` callback.
   768→
   769→    Args:
   770→        candidates: List of dicts with maimai_id, name, company, title
   771→        template_name: Name of message template
   772→        common_variables: Shared template variables
   773→        send_one: async callback(candidate_dict, rendered_content) → CandidateSendResult
   774→        batch_config: Display / behaviour config for this path
   775→        skip_messaged: Skip already-messaged candidates
   776→        skip_confirmation: Skip interactive confirmation prompt
   777→
   778→    Returns:
   779→        Dict with total, sent, failed, skipped, rate_limited, details
   780→    """
   781→    config = get_config()
   782→    db_manager = get_db_manager()
   783→
   784→    try:
   785→        template = config.get_template(template_name)
   786→    except Exception:
   787→        raise ValidationError(f"Template not found: {template_name}")
   788→
   789→    results = {
   790→        "total": len(candidates),
   791→        "sent": 0,
   792→        "failed": 0,
   793→        "skipped": 0,
   794→        "rate_limited": 0,
   795→        "details": [],
   796→    }
   797→
   798→    # --- Filter already-messaged candidates ---
   799→    send_list = []
   800→    if skip_messaged:
   801→        all_ids = [c["maimai_id"] for c in candidates]
   802→        with db_manager.session_scope() as session:
   803→            rows = (
   804→                session.query(Message.candidate_maimai_id)
   805→                .filter(Message.candidate_maimai_id.in_(all_ids), Message.status == "sent")
   806→                .distinct()
   807→                .all()
   808→            )
   809→            messaged_ids = {row[0] for row in rows}
   810→
   811→        for c in candidates:
   812→            if c["maimai_id"] in messaged_ids:
   813→                results["skipped"] += 1
   814→                results["details"].append(
   815→                    {
   816→                        "maimai_id": c["maimai_id"],
   817→                        "name": c["name"],
   818→                        "status": "skipped",
   819→                        "reason": "already messaged",
   820→                    }
   821→                )
   822→            else:
   823→                send_list.append(c)
   824→    else:
   825→        send_list = list(candidates)
   826→
   827→    if not send_list:
   828→        if results["skipped"] > 0:
   829→            console.print(
   830→                f"[yellow]All {results['skipped']} candidates already messaged. Skipping send.[/yellow]"
   831→            )
   832→        else:
   833→            console.print("[yellow]No candidates to send messages to.[/yellow]")
   834→        return results
   835→
   836→    # --- Optional pre-validation (Path A) ---
   837→    if batch_config.pre_validate:
   838→        console.print("[dim]Validating template for all candidates...[/dim]")
   839→        for c in send_list:
   840→            test_vars = {
   841→                **common_variables,
   842→                "name": c.get("name", ""),
   843→                "company": c.get("company", ""),
   844→                "position": c.get("title", ""),
   845→            }
   846→            is_valid, missing = validate_template_rendering(template["content"], test_vars, c)
   847→            if not is_valid:
   848→                raise ValidationError(
   849→                    f"候选人 {c.get('name', 'Unknown')} 的模板变量缺失或为空:\n"
   850→                    f"  缺失: {', '.join(missing)}\n"
   851→                    f"  候选人信息: {c}\n"
   852→                    f"  请检查模板或补充变量"
   853→                )
   854→
   855→    # --- Build preview ---
   856→    first = send_list[0]
   857→    first_vars = {
   858→        **common_variables,
   859→        "name": first["name"],
   860→        "company": first["company"],
   861→        "position": first["title"],
   862→    }
   863→    # If pre-validated, skip redundant validation in render
   864→    first_content = render_template(
   865→        template["content"], first_vars, validate=not batch_config.pre_validate
   866→    )
   867→
   868→    summary_text = Text()
   869→    summary_text.append("Template: ", style="bold")
   870→    summary_text.append(f"{template_name}\n")
   871→    if batch_config.mode_label:
   872→        summary_text.append("Mode: ", style="bold")
   873→        summary_text.append(f"{batch_config.mode_label}\n")
   874→    summary_text.append("Candidates: ", style="bold")
   875→    summary_text.append(f"{len(candidates)} found, {results['skipped']} already messaged\n")
   876→    summary_text.append("Will send to: ", style="bold")
   877→    summary_text.append(f"{len(send_list)} candidates\n\n")
   878→    summary_text.append("Message Preview ", style="bold cyan")
   879→    summary_text.append("(first candidate):\n", style="dim")
   880→    summary_text.append("To: ", style="bold cyan")
   881→    recipient = first["name"]
   882→    if first.get("title") and first.get("company"):
   883→        recipient += f" ({first['title']} @ {first['company']})"
   884→    summary_text.append(f"{recipient}\n\n")
   885→    summary_text.append(first_content)
   886→
   887→    console.print(
   888→        Panel(
   889→            summary_text,
   890→            title=batch_config.panel_title,
   891→            border_style="green",
   892→            padding=(1, 2),
   893→        )
   894→    )
   895→
   896→    # --- Confirmation ---
   897→    if skip_confirmation:
   898→        console.print(f"[green]Auto-confirmed: sending to {len(send_list)} candidates[/green]")
   899→    else:
   900→        loop = asyncio.get_running_loop()
   901→        confirm = await loop.run_in_executor(
   902→            None,
   903→            lambda: questionary.confirm(
   904→                f"Confirm batch send to {len(send_list)} candidates?"
   905→            ).unsafe_ask(),
   906→        )
   907→        if not confirm:
   908→            console.print("[yellow]Batch send cancelled.[/yellow]")
   909→            return results
   910→
   911→    # --- Send loop ---
   912→    from rich.progress import (
   913→        Progress,
   914→        BarColumn,
   915→        TextColumn as ProgressTextColumn,
   916→        MofNCompleteColumn,
   917→    )
   918→
   919→    with Progress(
   920→        ProgressTextColumn("[progress.description]{task.description}"),
   921→        BarColumn(),
   922→        MofNCompleteColumn(),
   923→        console=console,
   924→    ) as progress:
   925→        task = progress.add_task(batch_config.progress_description, total=len(send_list))
   926→
   927→        for c in send_list:
   928→            # Rate limit check
   929→            try:
   930→                await async_check_rate_limit("send_message")
   931→            except RateLimitExceededError as e:
   932→                console.print(f"\n[yellow]Rate limit reached: {e}[/yellow]")
   933→                remaining = len(send_list) - (results["sent"] + results["failed"])
   934→                results["rate_limited"] = remaining
   935→                results["details"].append(
   936→                    {
   937→                        "maimai_id": c["maimai_id"],
   938→                        "name": c["name"],
   939→                        "status": "rate_limited",
   940→                        "reason": str(e),
   941→                    }
   942→                )
   943→                break
   944→
   945→            # Build per-candidate variables and render template
   946→            variables = {
   947→                **common_variables,
   948→                "name": c["name"],
   949→                "company": c["company"],
   950→                "position": c["title"],
   951→            }
   952→            try:
   953→                content = render_template(
   954→                    template["content"],
   955→                    variables,
   956→                    validate=not batch_config.pre_validate,
   957→                )
   958→            except ValidationError as e:
   959→                logger.error(f"Template rendering failed for {c['name']}: {e}")
   960→                results["failed"] += 1
   961→                results["details"].append(
   962→                    {
   963→                        "maimai_id": c["maimai_id"],
   964→                        "name": c["name"],
   965→                        "status": "failed",
   966→                        "reason": f"模板渲染失败: {e}",
   967→                    }
   968→                )
   969→                progress.advance(task)
   970→                continue
   971→
   972→            # Delegate to path-specific sender
   973→            result = await send_one(c, content)
   974→
   975→            # Record result
   976→            if result.status == "sent":
   977→                results["sent"] += 1
   978→            elif result.status in ("quota_exhausted", "rate_limited"):
   979→                if result.status == "rate_limited":
   980→                    remaining = len(send_list) - (results["sent"] + results["failed"])
   981→                    results["rate_limited"] = remaining
   982→                else:
   983→                    results["failed"] += 1
   984→            else:
   985→                results["failed"] += 1
   986→
   987→            detail = {
   988→                "maimai_id": c["maimai_id"],
   989→                "name": c["name"],
   990→                "status": result.status,
   991→            }
   992→            if result.reason:
   993→                detail["reason"] = result.reason
   994→            if result.attempts > 1:
   995→                detail["attempts"] = result.attempts
   996→            if result.content:
   997→                detail["content"] = result.content
   998→            elif content:
   999→                detail["content"] = content
  1000→            results["details"].append(detail)
  1001→
  1002→            if result.abort_batch:
  1003→                break
  1004→
  1005→            progress.advance(task)
  1006→
  1007→            # Emit progress for web service (Path A only)
  1008→            if batch_config.emit_web_progress:
  1009→                emit_progress(
  1010→                    "send_progress",
  1011→                    sent=results["sent"],
  1012→                    failed=results["failed"],
  1013→                    total=len(send_list),
  1014→                    current=c["name"],
  1015→                )
  1016→
  1017→            await async_random_delay("message")
  1018→
  1019→    return results

  1027→def _make_search_center_sender(
  1028→    browser: BrowserManager,
  1029→    template_name: str,
  1030→    interactive: bool,
  1031→    discovered_selectors,
  1032→) -> Callable[[dict, str], Awaitable[CandidateSendResult]]:
  1033→    """Create a sender callback for search-center (Path A) with retry + timeout."""
  1034→
  1035→    max_retries = 2
  1036→    per_candidate_timeout = 90
  1037→
  1038→    async def _send(c: dict, content: str) -> CandidateSendResult:
  1039→        variables = {
  1040→            "name": c["name"],
  1041→            "company": c["company"],
  1042→            "position": c["title"],
  1043→        }
  1044→        last_error = None
  1045→        for attempt in range(1 + max_retries):
  1046→            try:
  1047→                success = await asyncio.wait_for(
  1048→                    send_message(
  1049→                        maimai_id=c["maimai_id"],
  1050→                        template_name=template_name,
  1051→                        variables=variables,
  1052→                        browser=browser,
  1053→                        skip_confirmation=True,
  1054→                        interactive=interactive,
  1055→                        discovered_selectors=discovered_selectors,
  1056→                        _skip_rate_check=True,
  1057→                        profile_url=c.get("profile_url"),
  1058→                    ),
  1059→                    timeout=per_candidate_timeout,
  1060→                )
  1061→                if success:
  1062→                    return CandidateSendResult(
  1063→                        success=True,
  1064→                        status="sent",
  1065→                        attempts=attempt + 1,
  1066→                        content=content,
  1067→                    )
  1068→                last_error = "send returned False"
  1069→
  1070→            except asyncio.TimeoutError:
  1071→                console.print(
  1072→                    f"[yellow]Timeout for {c['name']} ({c['maimai_id']}), skipping.[/yellow]"
  1073→                )
  1074→                return CandidateSendResult(
  1075→                    success=False,
  1076→                    status="failed",
  1077→                    reason=f"Timed out after {per_candidate_timeout}s",
  1078→                    attempts=attempt + 1,
  1079→                    content=content,
  1080→                )
  1081→            except QuotaExhaustedError as e:
  1082→                console.print(f"\n[red]联系券已用尽，终止批量发送[/red]")
  1083→                return CandidateSendResult(
  1084→                    success=False,
  1085→                    status="quota_exhausted",
  1086→                    reason=str(e),
  1087→                    attempts=attempt + 1,
  1088→                    content=content,
  1089→                    abort_batch=True,
  1090→                )
  1091→            except PermanentFailureError as e:
  1092→                console.print(f"[yellow]{e}. Skipping.[/yellow]")
  1093→                return CandidateSendResult(
  1094→                    success=False,
  1095→                    status="failed",
  1096→                    reason=str(e),
  1097→                    attempts=attempt + 1,
  1098→                    content=content,
  1099→                )
  1100→            except RateLimitExceededError as e:
  1101→                console.print(f"\n[yellow]Rate limit reached: {e}[/yellow]")
  1102→                return CandidateSendResult(
  1103→                    success=False,
  1104→                    status="rate_limited",
  1105→                    reason=str(e),
  1106→                    attempts=attempt + 1,
  1107→                    content=content,
  1108→                    abort_batch=True,
  1109→                )
  1110→            except Exception as e:
  1111→                last_error = str(e)
  1112→
  1113→            # Retry with exponential backoff
  1114→            if attempt < max_retries:
  1115→                from src.utils.delays import get_retry_delay
  1116→
  1117→                delay = get_retry_delay(attempt, base_delay=3.0, max_delay=30.0)
  1118→                logger.info(f"Retry {attempt + 1}/{max_retries} for {c['name']} in {delay:.1f}s")
  1119→                await asyncio.sleep(delay)
  1120→
  1121→        # All retries exhausted
  1122→        actual_attempts = max_retries + 1
  1123→        logger.error(
  1124→            f"Failed to send message to {c['name']} after {actual_attempts} attempt(s): {last_error}"
  1125→        )
  1126→        return CandidateSendResult(
  1127→            success=False,
  1128→            status="failed",
  1129→            reason=last_error or "unknown error",
  1130→            attempts=actual_attempts,
  1131→            content=content,
  1132→        )
  1133→
  1134→    return _send

  1137→def _make_enterprise_sender(

  1312→    """
  1313→    Render template with variables.

  1625→async def _verify_message_sent(page: Page, content: str, timeout: int = 5000) -> bool:
  1626→    """
  1627→    Verify that a message was actually sent by checking the conversation area.
  1628→
  1629→    Polls up to `timeout` ms with short intervals to handle async rendering.
  1630→
  1631→    Checks for:
  1632→    1. The sent message content appearing in the conversation
  1633→    2. The message input being cleared (indicating successful send)
  1634→    3. A sent/delivered indicator element
  1635→
  1636→    Args:
  1637→        page: Playwright page instance
  1638→        content: The message content that was sent
  1639→        timeout: Max time to wait for verification in ms
  1640→
  1641→    Returns:
  1642→        True if message send was verified
  1643→    """
  1644→    import time
  1645→
  1646→    # Short snippet for matching (first 20 chars, avoid template artifacts)
  1647→    snippet = content[:20].strip()
  1648→
  1649→    poll_interval_ms = 500
  1650→    deadline = time.monotonic() + timeout / 1000.0
  1651→
  1652→    while True:
  1653→        try:
  1654→            # Strategy 1: Check if message content appears in conversation area
  1655→            verified = await page.evaluate(
  1656→                """(snippet) => {
  1657→                const chatSelectors = [
  1658→                    '[class*="message-list"]',
  1659→                    '[class*="chat-content"]',
  1660→                    '[class*="conversation"]',
  1661→                    '[class*="msg-list"]',
  1662→                    '[class*="im-message"]',
  1663→                    '[class*="chat-msg"]',
  1664→                ];
  1665→
  1666→                for (const sel of chatSelectors) {
  1667→                    const container = document.querySelector(sel);
  1668→                    if (container && container.textContent.includes(snippet)) {
  1669→                        return true;
  1670→                    }
  1671→                }
  1672→
  1673→                // Fallback: scan leaf-ish elements in the lower portion of the page
  1674→                const viewportHeight = window.innerHeight;
  1675→                const elements = document.querySelectorAll('div, p, span');
  1676→                for (const el of elements) {
  1677→                    const rect = el.getBoundingClientRect();
  1678→                    if (rect.top > viewportHeight * 0.3 && el.textContent.includes(snippet)) {
  1679→                        if (el.textContent.length < snippet.length + 200) {
  1680→                            return true;
  1681→                        }
  1682→                    }
  1683→                }
  1684→
  1685→                return false;
  1686→            }""",
  1687→                snippet,
  1688→            )
  1689→
  1690→            if verified:
  1691→                logger.debug("Message send verified: content found in conversation")
  1692→                return True
  1693→
  1694→            # Strategy 2: Check if input was cleared (indicates successful send)
  1695→            input_cleared = await page.evaluate(
  1696→                """() => {
  1697→                const inputs = document.querySelectorAll(
  1698→                    'textarea, [contenteditable="true"], input[type="text"]'
  1699→                );
  1700→                for (const input of inputs) {
  1701→                    const rect = input.getBoundingClientRect();
  1702→                    if (rect.width > 0 && rect.height > 0) {
  1703→                        const value = input.value || input.textContent || '';
  1704→                        if (value.trim() === '') {
  1705→                            return true;
  1706→                        }
  1707→                    }
  1708→                }
  1709→                return false;
  1710→            }"""
  1711→            )
  1712→
  1713→            if input_cleared:
  1714→                logger.debug("Message send verified: input field was cleared")
  1715→                return True
  1716→
  1717→            # Strategy 3: Check for sent/delivered indicators
  1718→            sent_indicators = [
  1719→                '[class*="sent"]',
  1720→                '[class*="delivered"]',
  1721→                '[class*="success"]',
  1722→                '[class*="msg-status"]',
  1723→            ]
  1724→            for selector in sent_indicators:
  1725→                try:
  1726→                    indicator = await page.query_selector(selector)
  1727→                    if indicator and await indicator.is_visible():
  1728→                        logger.debug(f"Message send verified: indicator found ({selector})")
  1729→                        return True
  1730→                except Exception:
  1731→                    continue
  1732→
  1733→        except Exception as e:
  1734→            logger.warning(f"Message verification poll failed: {e}")
  1735→
  1736→        # Check if we've exceeded the deadline
  1737→        if time.monotonic() >= deadline:
  1738→            break
  1739→
  1740→        await asyncio.sleep(poll_interval_ms / 1000.0)
  1741→

  1746→async def _check_existing_conversation(page: Page, maimai_id: str) -> bool:
  1747→    """
  1748→    Check if there's already an existing conversation with this candidate.
  1749→
  1750→    Checks for actual user message bubbles (not system prompts/tips) in the
  1751→    conversation area.
  1752→
  1753→    Args:
  1754→        page: Playwright page instance
  1755→        maimai_id: Candidate's Maimai ID
  1756→
  1757→    Returns:
  1758→        True if an existing conversation was detected
  1759→    """
  1760→    try:
  1761→        has_conversation = await page.evaluate(
  1762→            """() => {
  1763→            // Look for conversation message containers
  1764→            const chatSelectors = [
  1765→                '[class*="message-list"]',
  1766→                '[class*="chat-content"]',
  1767→                '[class*="conversation"]',
  1768→                '[class*="msg-list"]',
  1769→                '[class*="im-message"]',
  1770→                '[class*="chat-msg"]',
  1771→                '[class*="chat-record"]',
  1772→            ];
  1773→
  1774→            // System/tip patterns that should NOT count as existing conversation
  1775→            const systemPatterns = [
  1776→                '还不是好友', '添加好友', '发送好友申请', '对方未回复',
  1777→                '以上是打招呼内容', '暂无消息', '开始聊天', '你们已经是好友',
  1778→                '系统消息', '温馨提示', '安全提示', '注意事项',
  1779→                '开启招呼语', '求职意向', '目前情况', '竞业点',
  1780→                '欢迎联系', '不感兴趣',
  1781→            ];
  1782→
  1783→            const isSystemText = (text) => {
  1784→                const t = text.trim();
  1785→                if (t.length < 5) return true;  // Too short to be a real message
  1786→                return systemPatterns.some(p => t.includes(p));
  1787→            };
  1788→
  1789→            for (const sel of chatSelectors) {
  1790→                const container = document.querySelector(sel);
  1791→                if (!container) continue;
  1792→
  1793→                // Strategy 1: Look for message bubble elements
  1794→                const bubbleSelectors = [
  1795→                    '[class*="msg-item"]', '[class*="message-item"]',
  1796→                    '[class*="bubble"]', '[class*="chat-item"]',
  1797→                    '[class*="msg-bubble"]', '[class*="chat-bubble"]',
  1798→                ];
  1799→                for (const bSel of bubbleSelectors) {
  1800→                    const bubbles = container.querySelectorAll(bSel);
  1801→                    // Filter out system/tip bubbles
  1802→                    let realMessageCount = 0;
  1803→                    for (const bubble of bubbles) {
  1804→                        const text = bubble.textContent.trim();
  1805→                        if (text.length > 5 && !isSystemText(text)) {
  1806→                            realMessageCount++;
  1807→                        }
  1808→                    }
  1809→                    if (realMessageCount > 0) {
  1810→                        return true;
  1811→                    }
  1812→                }
  1813→
  1814→                // Strategy 2: If no bubble selectors matched, check direct children
  1815→                // that look like message blocks (not system tips)
  1816→                const children = container.children;
  1817→                let realChildCount = 0;
  1818→                for (const child of children) {
  1819→                    const text = child.textContent.trim();
  1820→                    // Must have meaningful content and not be a system message
  1821→                    if (text.length > 10 && !isSystemText(text)) {
  1822→                        // Check it's not a timestamp-only or status-only element
  1823→                        if (child.querySelector('img[class*="avatar"]') || text.length > 20) {
  1824→                            realChildCount++;
  1825→                        }
  1826→                    }
  1827→                }
  1828→                if (realChildCount > 0) {
  1829→                    return true;
  1830→                }
  1831→            }
  1832→            return false;
  1833→        }"""
  1834→        )
  1835→
  1836→        if has_conversation:
  1837→            logger.info(f"Existing conversation detected for {maimai_id}")
  1838→        return has_conversation
  1839→
  1840→    except Exception as e:
  1841→        # On error, assume conversation exists (safe side: avoid duplicate messages)
  1842→        logger.warning(
  1843→            f"Conversation check failed for {maimai_id}: {e} — assuming existing conversation"
  1844→        )
  1845→        return True
