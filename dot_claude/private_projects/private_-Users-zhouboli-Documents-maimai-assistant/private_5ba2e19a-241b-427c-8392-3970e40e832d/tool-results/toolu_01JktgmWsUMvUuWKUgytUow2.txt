    45→def _extract_user_id(profile_url: str) -> Optional[str]:
    46→    """Extract dstu/user_id from a Maimai profile URL for direct chat navigation.
    47→
    48→    Only returns IDs that are known to work with the chat URL (u2= parameter).
    49→    Query-param IDs (dstu, u2, egid) are reliable for chat navigation.
    50→    Path-based IDs (/contact/detail/xxx) are NOT the same as chat user IDs
    51→    and should not be used for direct chat URL construction.
    52→    """
    53→    from urllib.parse import urlparse, parse_qs
    54→    parsed = urlparse(profile_url)
    55→    params = parse_qs(parsed.query)
    56→    # Only use query parameter IDs — these are confirmed to work with uh_chat_pc
    57→    for key in ('dstu', 'u2', 'egid'):
    58→        if key in params:
    59→            return params[key][0]
    60→    return None

    63→async def _navigate_to_chat(
    64→    page: Page,
    65→    profile_url: Optional[str],
    66→    config,
    67→    browser: BrowserManager,
    68→    maimai_id: str,
    69→    interactive: bool = False,
    70→    discovered_selectors=None,
    71→) -> bool:
    72→    """Navigate to the chat page for a candidate.
    73→
    74→    Tries in order:
    75→    1. Extract user_id from profile_url query params → direct chat URL
    76→    2. Navigate to profile_url, extract user_id from redirected URL → direct chat URL
    77→    3. Click chat button on profile page as fallback
    78→
    79→    If no profile_url is provided, looks up the candidate's profile_url from DB.
    80→
    81→    Args:
    82→        page: Playwright page instance
    83→        profile_url: Candidate's profile URL (may be None)
    84→        config: Config instance
    85→        browser: Browser manager instance
    86→        maimai_id: Candidate's Maimai ID
    87→        interactive: Enable interactive selector discovery
    88→        discovered_selectors: Shared DiscoveredSelectors instance
    89→
    90→    Returns:
    91→        True if successfully navigated to chat page
    92→    """
    93→    db_manager = get_db_manager()
    94→
    95→    # Resolve profile URL from DB if not provided
    96→    if not profile_url:
    97→        with db_manager.session_scope() as session:
    98→            db_candidate = session.query(Candidate).filter_by(maimai_id=maimai_id).first()
    99→            profile_url = db_candidate.profile_url if db_candidate else None
   100→
   101→        if not profile_url:
   102→            raise PermanentFailureError(f"No profile URL for {maimai_id}")
   103→
   104→    # Try extracting user_id directly from profile URL
   105→    user_id = _extract_user_id(profile_url)
   106→    logger.debug(f"_navigate_to_chat: profile_url={profile_url!r}, extracted user_id={user_id}")
   107→
   108→    if user_id:
   109→        chat_url = MAIMAI_CHAT_URL.format(user_id=user_id)
   110→        await retry_navigation(page, chat_url, wait_until='domcontentloaded')
   111→        await async_random_delay('page_load')
   112→        return True
   113→
   114→    # Navigate to profile page, then try to extract user_id from resulting URL
   115→    logger.debug(f"_navigate_to_chat: navigating to profile {profile_url}")
   116→    await retry_navigation(page, profile_url, wait_until='domcontentloaded')
   117→    await async_random_delay('page_load')
   118→
   119→    current_url = page.url
   120→    logger.debug(f"_navigate_to_chat: landed on {current_url}")
   121→    redirected_user_id = _extract_user_id(current_url)
   122→    if redirected_user_id:
   123→        logger.info(f"Extracted user_id={redirected_user_id} from page URL, redirecting to chat")
   124→        chat_url = MAIMAI_CHAT_URL.format(user_id=redirected_user_id)
   125→        await retry_navigation(page, chat_url, wait_until='domcontentloaded')
   126→        await async_random_delay('page_load')
   127→        return True
   128→
   129→    # Fallback: click chat button on profile page
   130→    chat_button_found = await _click_chat_button(page, config)
   131→    if chat_button_found:
   132→        await async_random_delay('action')
   133→        return True
   134→
   135→    # If interactive mode, trigger selector discovery
   136→    if interactive and discovered_selectors is not None and not discovered_selectors.is_populated:
   137→        from src.modules.selector_discovery import discover_chat_page
   138→        console.print("[yellow]Chat button not found. Entering interactive discovery mode...[/yellow]")
   139→        result = await discover_chat_page(page, maimai_id, browser)
   140→        if result.is_populated:
   141→            discovered_selectors.chat_button = result.chat_button
   142→            discovered_selectors.message_input = result.message_input
   143→            discovered_selectors.send_button = result.send_button
   144→            discovered_selectors.chat_url_pattern = result.chat_url_pattern
   145→            discovered_selectors.skip_chat_button = result.skip_chat_button
   146→            discovered_selectors.is_populated = True
   147→            return True
   148→
   149→    console.print("[red]Could not find chat button[/red]")
   150→    await browser.screenshot(f"message_error_{maimai_id}")
   151→    return False

   154→async def extract_contact_quota(page: Page) -> Optional[dict]:
   155→    """Extract contact quota from Quick Contact page's embedded share_data.
   156→
   157→    The Quick Contact page server-renders ``window.share_data`` which includes
   158→    ``mem_status.mem_info`` with remaining quotas for various member actions.
   159→
   160→    Args:
   161→        page: Playwright page on a Quick Contact (极速联系) page
   162→
   163→    Returns:
   164→        Dict with quota info, or None if data is unavailable::
   165→
   166→            {
   167→                'uh_left': int,        # 极速联系 remaining
   168→                'addfr_left': int,     # 加好友 remaining
   169→                'dc_left': int,        # 直聊 remaining
   170→                'expire_time': str,    # membership expiry
   171→                'cost': int,           # vouchers consumed per contact
   172→            }
   173→    """
   174→    try:
   175→        quota = await page.evaluate('''() => {
   176→            if (window.share_data && window.share_data.mem_status &&
   177→                window.share_data.mem_status.mem_info) {
   178→                const info = window.share_data.mem_status.mem_info;
   179→                return {
   180→                    uh_left: info.uh_left,
   181→                    addfr_left: info.addfr_left,
   182→                    dc_left: info.dc_left,
   183→                    expire_time: info.expire_time || null,
   184→                    cost: (window.share_data.fast_contact_pay_num || {}).final_num || 1,
   185→                };
   186→            }
   187→            return null;
   188→        }''')
   189→        return quota
   190→    except Exception as e:
   191→        logger.debug(f"Failed to extract contact quota: {e}")
   192→        return None

   203→        browser: BrowserManager instance (uses global singleton if None)

   219→        logger.debug(f"Failed to fetch contact quota: {e}")

   237→async def _handle_quick_contact(page: Page, content: str) -> Optional[bool]:
   238→    """Handle Maimai's "极速联系" (Quick Contact) page.
   239→
   240→    This page appears instead of regular chat for some candidates. It has a
   241→    category selector, a textarea, and a "提交" (Submit) button.
   242→
   243→    Submission works by manipulating the React component directly: setting
   244→    the message via setState and calling the debounced send method (with
   245→    flush) to trigger the POST to /bizmember/uh-chat/api/send_v2.
   246→
   247→    DOM-level textarea.fill() does NOT update React's controlled state, and
   248→    DOM click on the submit button does NOT reliably trigger the send API.
   249→
   250→    Args:
   251→        page: Playwright page instance
   252→        content: Message content to send
   253→
   254→    Returns:
   255→        True if message was sent via quick contact form,
   256→        False if quick contact page was detected but sending failed,
   257→        None if this is not a quick contact page (caller should continue normal flow)
   258→    """
   259→    try:
   260→        # Detect quick contact page via page title or page content
   261→        is_quick_contact = await page.evaluate('''() => {
   262→            const title = document.title || '';
   263→            const bodyText = document.body ? document.body.innerText : '';
   264→            return title.includes('极速联系') || bodyText.includes('极速联系');
   265→        }''')
   266→
   267→        if not is_quick_contact:
   268→            return None
   269→
   270→        logger.info("Detected '极速联系' (Quick Contact) page")
   271→
   272→        # Validate message length (Quick Contact enforces 10-300 chars)
   273→        if len(content) < 10:
   274→            logger.error(
   275→                f"Quick contact: message too short ({len(content)} chars, "
   276→                "minimum 10). Quick Contact requires 10-300 characters."
   277→            )
   278→            return False
   279→        if len(content) > 300:
   280→            logger.error(
   281→                f"Quick contact: message too long ({len(content)} chars, "
   282→                "maximum 300). Quick Contact requires 10-300 characters."
   283→            )
   284→            return False
   285→
   286→        # Extract quota info before submitting (share_data is available on page load)
   287→        quota = await extract_contact_quota(page)
   288→
   289→        # Check if quota is exhausted
   290→        if quota and quota.get('uh_left', 0) <= 0:
   291→            raise QuotaExhaustedError(remaining=quota.get('uh_left', 0))
   292→
   293→        # Install XHR interceptor to capture the send_v2 response
   294→        await page.evaluate('''() => {
   295→            window.__sendV2Result = null;
   296→            const origOpen = XMLHttpRequest.prototype.open;
   297→            const origSend = XMLHttpRequest.prototype.send;
   298→            XMLHttpRequest.prototype.open = function(m, u) {
   299→                this.__url = u; this.__method = m;
   300→                return origOpen.apply(this, arguments);
   301→            };
   302→            XMLHttpRequest.prototype.send = function(b) {
   303→                if (this.__url && this.__url.includes('send_v2')) {
   304→                    const xhr = this;
   305→                    xhr.addEventListener('load', function() {
   306→                        try {
   307→                            window.__sendV2Result = {
   308→                                status: xhr.status,
   309→                                response: xhr.responseText,
   310→                            };
   311→                        } catch(e) {
   312→                            window.__sendV2Result = {status: xhr.status, error: e.message};
   313→                        }
   314→                    });
   315→                    xhr.addEventListener('error', function() {
   316→                        window.__sendV2Result = {status: 0, error: 'network_error'};
   317→                    });
   318→                }
   319→                return origSend.apply(this, arguments);
   320→            };
   321→        }''')
   322→
   323→        # Submit via React component internals:
   324→        # 1. Find the class component that owns the form state
   325→        # 2. Set message + unlock push_type via setState
   326→        # 3. Call the debounced send() method and flush() to execute immediately
   327→        send_triggered = await page.evaluate('''(msg) => {
   328→            const textarea = document.querySelector('textarea');
   329→            if (!textarea) return {success: false, error: 'no textarea'};
   330→            const keys = Object.keys(textarea);
   331→            const fiberKey = keys.find(
   332→                k => k.startsWith('__reactInternalInstance') || k.startsWith('__reactFiber')
   333→            );
   334→            if (!fiberKey) return {success: false, error: 'no React fiber'};
   335→
   336→            let fiber = textarea[fiberKey];
   337→            while (fiber) {
   338→                if (fiber.stateNode && fiber.stateNode.state
   339→                    && 'message' in fiber.stateNode.state
   340→                    && typeof fiber.stateNode.send === 'function') {
   341→                    const component = fiber.stateNode;
   342→                    // Set message and ensure push_type is not locked (true = sending lock)
   343→                    component.setState({message: msg, push_type: false}, () => {
   344→                        component.send();
   345→                        if (typeof component.send.flush === 'function') {
   346→                            component.send.flush();
   347→                        }
   348→                    });
   349→                    return {success: true};
   350→                }
   351→                fiber = fiber.return;
   352→            }
   353→            return {success: false, error: 'React component not found'};
   354→        }''', content)
   355→
   356→        if not send_triggered.get('success'):
   357→            logger.warning(
   358→                f"Quick contact: failed to trigger send via React: "
   359→                f"{send_triggered.get('error', 'unknown')}"
   360→            )
   361→            return False
   362→
   363→        # Wait for the API response (send_v2 XHR)
   364→        send_result = None
   365→        for _ in range(10):
   366→            await asyncio.sleep(0.5)
   367→            send_result = await page.evaluate('() => window.__sendV2Result')
   368→            if send_result is not None:
   369→                break
   370→
   371→        if send_result is None:
   372→            logger.warning("Quick contact: send_v2 API call was not captured (timeout)")
   373→            return False
   374→
   375→        # Parse the API response
   376→        response_text = send_result.get('response', '')
   377→        try:
   378→            resp_data = json.loads(response_text) if response_text else {}
   379→        except (json.JSONDecodeError, TypeError):
   380→            resp_data = {}
   381→
   382→        ret = resp_data.get('ret')
   383→        error_msg = resp_data.get('error', '')
   384→
   385→        if ret == -1:
   386→            logger.warning(f"Quick contact: send failed — {error_msg}")
   387→            if error_msg:
   388→                console.print(f"  [red]Quick Contact 失败: {error_msg}[/red]")
   389→            return False
   390→
   391→        logger.info("Quick contact: message sent successfully")
   392→        await async_random_delay('action')
   393→
   394→        # Display remaining quota
   395→        if quota:
   396→            _display_quota(quota, just_sent=True)
   397→
   398→        return True
   399→
   400→    except PermanentFailureError:
   401→        raise  # Let quota exhaustion etc. propagate to abort the batch
   402→    except Exception as e:
   403→        logger.error(f"Quick contact handling failed: {e}\n{traceback.format_exc()}")
   404→        return False

   407→async def _detect_vip_paywall(page: Page) -> bool:
   408→    """Detect if a VIP/membership paywall is blocking the current page.
   409→
   410→    Non-VIP accounts may be redirected to a recharge/upgrade page when
   411→    trying to message certain candidates. This checks the page URL and
   412→    visible text for common paywall indicators.
   413→
   414→    Args:
   415→        page: Playwright page instance
   416→
   417→    Returns:
   418→        True if a VIP paywall is detected
   419→    """
   420→    try:
   421→        return await page.evaluate('''() => {
   422→            const url = window.location.href;
   423→            const urlKeywords = ['vip', 'member', 'pay', 'recharge', 'upgrade', 'subscribe'];
   424→            if (urlKeywords.some(kw => url.toLowerCase().includes(kw))) {
   425→                return true;
   426→            }
   427→
   428→            const title = document.title || '';
   429→            const bodyText = document.body ? document.body.innerText.substring(0, 3000) : '';
   430→            const text = title + ' ' + bodyText;
   431→
   432→            const paywallPatterns = [
   433→                '开通会员', '升级会员', '会员专享', '会员权益',
   434→                '充值', '付费', '购买会员', '开通VIP', '升级VIP',
   435→                '立即开通', '立即升级', '立即购买',
   436→                '会员到期', '续费', '解锁', '次数已用完',
   437→                '今日沟通次数', '沟通人数已达上限',
   438→            ];
   439→
   440→            return paywallPatterns.some(p => text.includes(p));
   441→        }''')
   442→    except Exception as e:
   443→        logger.debug(f"VIP paywall detection failed: {e}")
   444→        return False

   447→async def send_message(
   448→    maimai_id: str,
   449→    template_name: str,
   450→    variables: Dict[str, str],
   451→    browser: Optional[BrowserManager] = None,
   452→    skip_confirmation: bool = False,
   453→    interactive: bool = False,
   454→    discovered_selectors: Optional['DiscoveredSelectors'] = None,
   455→    _skip_rate_check: bool = False,
   456→    profile_url: Optional[str] = None,
   457→) -> bool:
   458→    """
   459→    Send a message to a candidate.
   460→
   461→    Args:
   462→        maimai_id: Candidate's Maimai ID
   463→        template_name: Name of message template to use
   464→        variables: Variables to substitute in template
   465→        browser: Browser manager instance
   466→        skip_confirmation: Skip confirmation (only for testing)
   467→        interactive: Enable interactive selector discovery on failure
   468→        discovered_selectors: Shared DiscoveredSelectors instance for reuse
   469→
   470→    Returns:
   471→        True if message was sent successfully
   472→
   473→    Raises:
   474→        RateLimitExceededError: If rate limit exceeded
   475→        ValidationError: If template or variables invalid
   476→    """
   477→    # Validate and clean inputs
   478→    maimai_id = validate_maimai_id(maimai_id)
   479→
   480→    if browser is None:
   481→        browser = get_browser_manager()
   482→
   483→    config = get_config()
   484→    rate_limiter = get_rate_limiter()
   485→    db_manager = get_db_manager()
   486→
   487→    # Get template
   488→    try:
   489→        template = config.get_template(template_name)
   490→    except Exception as e:
   491→        raise ValidationError(f"Template not found: {template_name}")
   492→
   493→    # Validate template variables
   494→    validate_template_variables(template, variables)
   495→
   496→    # Render message content
   497→    content = render_template(template['content'], variables)
   498→
   499→    # Get candidate info for display
   500→    candidate_info = _get_candidate_info(db_manager, maimai_id)
   501→
   502→    # Preview and confirm
   503→    if not skip_confirmation:
   504→        action = await interactive_confirm(content, candidate_info, template_name)
   505→
   506→        if action == 'skip':
   507→            console.print("[yellow]Message skipped[/yellow]")
   508→            _log_message(db_manager, maimai_id, template_name, content, 'cancelled')
   509→            return False
   510→
   511→        elif action == 'edit':
   512→            # Allow user to edit content
   513→            content = await edit_message(content)
   514→            if content is None:
   515→                console.print("[yellow]Message cancelled[/yellow]")
   516→                return False
   517→
   518→            # Re-confirm after edit
   519→            action = await interactive_confirm(content, candidate_info, template_name)
   520→            if action != 'send':
   521→                console.print("[yellow]Message skipped[/yellow]")
   522→                return False
   523→
   524→        elif action != 'send':
   525→            console.print("[yellow]Message cancelled[/yellow]")
   526→            return False
   527→
   528→    # Check rate limit (waits through min-interval) - skip when called from batch
   529→    if not _skip_rate_check:
   530→        await async_check_rate_limit('send_message')
   531→
   532→    # Ensure logged in
   533→    await ensure_logged_in(browser)
   534→
   535→    try:
   536→        page = browser.page
   537→
   538→        # If we have populated discovered selectors, use them directly
   539→        if discovered_selectors and discovered_selectors.is_populated:
   540→            from src.modules.selector_discovery import apply_discovered_selectors
   541→            success = await apply_discovered_selectors(
   542→                page, maimai_id, discovered_selectors, content
   543→            )
   544→            if success:
   545→                rate_limiter.log_action('send_message', {
   546→                    'maimai_id': maimai_id,
   547→                    'template': template_name
   548→                })
   549→                _log_message(db_manager, maimai_id, template_name, content, 'sent')
   550→                console.print("[green]Message sent successfully![/green]")
   551→                return True
   552→            else:
   553→                console.print("[red]Failed to send with discovered selectors[/red]")
   554→                await browser.screenshot(f"message_error_{maimai_id}")
   555→                _log_message(db_manager, maimai_id, template_name, content, 'failed')
   556→                return False
   557→
   558→        nav_ok = await _navigate_to_chat(
   559→            page, profile_url, config, browser, maimai_id,
   560→            interactive, discovered_selectors,
   561→        )
   562→        if not nav_ok:
   563→            _log_message(db_manager, maimai_id, template_name, content, 'failed')
   564→            return False
   565→
   566→        # Check for "等待回应" (Awaiting Response) — already messaged this candidate
   567→        try:
   568→            awaiting = await page.evaluate('''() => {
   569→                const text = document.body ? document.body.innerText : '';
   570→                return text.includes('等待回应') || text.includes('等待对方回复');
   571→            }''')
   572→            if awaiting:
   573→                logger.info(f"Candidate {maimai_id} already messaged (awaiting response), skipping")
   574→                console.print(f"[yellow]Already messaged {maimai_id} (等待回应), skipping[/yellow]")
   575→                _log_message(db_manager, maimai_id, template_name, content, 'cancelled')
   576→                return False
   577→        except Exception as e:
   578→            logger.debug(f"Awaiting-response check failed for {maimai_id}: {e}")
   579→
   580→        # Check for "极速联系" (Quick Contact) page
   581→        qc_result = await _handle_quick_contact(page, content)
   582→        if qc_result is True:
   583→            rate_limiter.log_action('send_message', {
   584→                'maimai_id': maimai_id,
   585→                'template': template_name,
   586→                'mode': 'quick_contact',
   587→            })
   588→            _log_message(db_manager, maimai_id, template_name, content, 'sent')
   589→            console.print("[green]Message sent via Quick Contact![/green]")
   590→            return True
   591→        elif qc_result is False:
   592→            console.print("[red]Quick Contact page detected but send failed[/red]")
   593→            await browser.screenshot(f"message_error_qc_{maimai_id}")
   594→            _log_message(db_manager, maimai_id, template_name, content, 'failed')
   595→            return False
   596→        # qc_result is None → not a quick contact page, continue normal flow
   597→
   598→        # If interactive discovery succeeded, apply discovered selectors
   599→        if discovered_selectors and discovered_selectors.is_populated:
   600→            from src.modules.selector_discovery import apply_discovered_selectors
   601→            success = await apply_discovered_selectors(
   602→                page, maimai_id, discovered_selectors, content
   603→            )
   604→            if success:
   605→                rate_limiter.log_action('send_message', {
   606→                    'maimai_id': maimai_id,
   607→                    'template': template_name
   608→                })
   609→                _log_message(db_manager, maimai_id, template_name, content, 'sent')
   610→                console.print("[green]Message sent successfully![/green]")
   611→                return True
   612→
   613→        # Check for existing conversation to avoid duplicate greetings
   614→        has_existing = await _check_existing_conversation(page, maimai_id)
   615→        if has_existing:
   616→            console.print(f"[yellow]Existing conversation detected with {maimai_id}. Skipping to avoid duplicate.[/yellow]")
   617→            _log_message(db_manager, maimai_id, template_name, content, 'cancelled')
   618→            return False
   619→
   620→        # Find message input and type message
   621→        input_found = await _type_message(page, config, content)
   622→        if not input_found:
   623→            console.print("[red]Could not find message input[/red]")
   624→            await browser.screenshot(f"message_error_{maimai_id}")
   625→            _log_message(db_manager, maimai_id, template_name, content, 'failed')
   626→            return False
   627→
   628→        await async_random_delay('action')
   629→
   630→        # Click send button
   631→        sent = await _click_send_button(page, config)
   632→        if not sent:
   633→            console.print("[red]Could not send message[/red]")
   634→            await browser.screenshot(f"message_error_{maimai_id}")
   635→            _log_message(db_manager, maimai_id, template_name, content, 'failed')
   636→            return False
   637→
   638→        # Check if clicking send triggered a VIP paywall popup
   639→        await asyncio.sleep(1)
   640→        if await _detect_vip_paywall(page):
   641→            console.print(f"[yellow]VIP paywall popup appeared after send for {maimai_id}, skipping[/yellow]")
   642→            _log_message(db_manager, maimai_id, template_name, content, 'cancelled')
   643→            raise PermanentFailureError(f"VIP paywall detected for {maimai_id}")
   644→
   645→        # Verify message was actually sent
   646→        verified = await _verify_message_sent(page, content)
   647→        if not verified:
   648→            logger.warning(f"Message to {maimai_id}: send button clicked but could not verify delivery")
   649→            console.print("[yellow]Warning: Message send could not be verified. Marking as unverified.[/yellow]")
   650→            await browser.screenshot(f"message_unverified_{maimai_id}")
   651→            # Still log as sent but with unverified note in details
   652→            rate_limiter.log_action('send_message', {
   653→                'maimai_id': maimai_id,
   654→                'template': template_name,
   655→                'verified': False,
   656→            })
   657→            _log_message(db_manager, maimai_id, template_name, content, 'sent')
   658→            console.print("[yellow]Message marked as sent (unverified)[/yellow]")
   659→            return True
   660→
   661→        # Log action
   662→        rate_limiter.log_action('send_message', {
   663→            'maimai_id': maimai_id,
   664→            'template': template_name,
   665→            'verified': True,
   666→        })
   667→
   668→        # Log message to database
   669→        _log_message(db_manager, maimai_id, template_name, content, 'sent')
   670→
   671→        console.print("[green]Message sent and verified![/green]")
   672→        return True
   673→
   674→    except PermanentFailureError:
   675→        raise
   676→    except Exception as e:
   677→        logger.error(f"Error sending message: {e}")
   678→        await browser.screenshot(f"message_error_{maimai_id}")
   679→        _log_message(db_manager, maimai_id, template_name, content, 'failed')
   680→        raise

   683→async def send_message_batch(
   684→    candidates: List[dict],
   685→    template_name: str,
   686→    common_variables: Dict[str, str],
   687→    browser: BrowserManager,
   688→    skip_messaged: bool = True,
   689→    interactive: bool = False,
   690→    skip_confirmation: bool = False,
   691→) -> Dict[str, Any]:
   692→    """
   693→    Batch send messages to a list of candidates.
   694→
   695→    Flow:
   696→    1. Filter already-messaged candidates
   697→    2. Auto-fill name/company/position variables per candidate
   698→    3. Render message template
   699→    4. Show candidate list + first message preview
   700→    5. Single confirmation
   701→    6. Send one by one, checking rate limit each time
   702→    7. Skip failures and record
   703→    8. Return result stats
   704→
   705→    Args:
   706→        candidates: List of dicts with maimai_id, name, company, title
   707→        template_name: Name of message template
   708→        common_variables: Shared variables (recruiter_name, target_position, etc.)
   709→        browser: Browser manager instance
   710→        skip_messaged: Skip candidates already messaged (default True)
   711→        interactive: Enable interactive selector discovery on first failure
   712→
   713→    Returns:
   714→        Dict with total, sent, failed, skipped, rate_limited, details
   715→    """
   716→    config = get_config()
   717→    rate_limiter = get_rate_limiter()
   718→    db_manager = get_db_manager()
   719→
   720→    # Get template
   721→    try:
   722→        template = config.get_template(template_name)
   723→    except Exception:
   724→        raise ValidationError(f"Template not found: {template_name}")
   725→
   726→    # Create shared discovered selectors for interactive mode
   727→    if interactive:
   728→        from src.modules.selector_discovery import DiscoveredSelectors
   729→        discovered_selectors = DiscoveredSelectors()
   730→    else:
   731→        discovered_selectors = None
   732→
   733→    results = {
   734→        'total': len(candidates),
   735→        'sent': 0,
   736→        'failed': 0,
   737→        'skipped': 0,
   738→        'rate_limited': 0,
   739→        'details': [],
   740→    }
   741→
   742→    # Filter already-messaged candidates
   743→    send_list = []
   744→    if skip_messaged:
   745→        # Batch pre-fetch all sent message IDs to avoid N+1 queries
   746→        all_ids = [c['maimai_id'] for c in candidates]
   747→        messaged_ids = set()
   748→        with db_manager.session_scope() as session:
   749→            rows = session.query(Message.candidate_maimai_id).filter(
   750→                Message.candidate_maimai_id.in_(all_ids),
   751→                Message.status == 'sent'
   752→            ).distinct().all()
   753→            messaged_ids = {row[0] for row in rows}
   754→
   755→        for c in candidates:
   756→            if c['maimai_id'] in messaged_ids:
   757→                results['skipped'] += 1
   758→                results['details'].append({
   759→                    'maimai_id': c['maimai_id'],
   760→                    'name': c['name'],
   761→                    'status': 'skipped',
   762→                    'reason': 'already messaged',
   763→                })
   764→            else:
   765→                send_list.append(c)
   766→    else:
   767→        send_list = list(candidates)
   768→
   769→    if not send_list:
   770→        console.print("[yellow]No candidates to send messages to.[/yellow]")
   771→        return results
   772→
   773→    # Pre-validate all candidates can render template (防止发送到一半才发现变量缺失)
   774→    console.print("[dim]Validating template for all candidates...[/dim]")
   775→    for c in send_list:
   776→        test_vars = {
   777→            **common_variables,
   778→            'name': c.get('name', ''),
   779→            'company': c.get('company', ''),
   780→            'position': c.get('title', ''),
   781→        }
   782→        is_valid, missing = validate_template_rendering(template['content'], test_vars, c)
   783→        if not is_valid:
   784→            raise ValidationError(
   785→                f"候选人 {c.get('name', 'Unknown')} 的模板变量缺失或为空:\n"
   786→                f"  缺失: {', '.join(missing)}\n"
   787→                f"  候选人信息: {c}\n"
   788→                f"  请检查模板或补充变量"
   789→            )
   790→
   791→    # Build preview for first candidate
   792→    first = send_list[0]
   793→    first_vars = {**common_variables, 'name': first['name'], 'company': first['company'], 'position': first['title']}
   794→    first_content = render_template(template['content'], first_vars, validate=False)  # Already validated above
   795→
   796→    # Display summary panel
   797→    summary_text = Text()
   798→    summary_text.append("Template: ", style="bold")
   799→    summary_text.append(f"{template_name}\n")
   800→    summary_text.append("Candidates: ", style="bold")
   801→    summary_text.append(f"{len(candidates)} found, {results['skipped']} already messaged\n")
   802→    summary_text.append("Will send to: ", style="bold")
   803→    summary_text.append(f"{len(send_list)} candidates\n\n")
   804→    summary_text.append("Message Preview ", style="bold cyan")
   805→    summary_text.append(f"(first candidate):\n", style="dim")
   806→    summary_text.append("To: ", style="bold cyan")
   807→    recipient = first['name']
   808→    if first.get('title') and first.get('company'):
   809→        recipient += f" ({first['title']} @ {first['company']})"
   810→    summary_text.append(f"{recipient}\n\n")
   811→    summary_text.append(first_content)
   812→
   813→    console.print(Panel(
   814→        summary_text,
   815→        title="Pipeline Summary",
   816→        border_style="green",
   817→        padding=(1, 2)
   818→    ))
   819→
   820→    # Single confirmation (skip if --yes flag or non-interactive environment)
   821→    if skip_confirmation:
   822→        console.print(f"[green]Auto-confirmed: sending to {len(send_list)} candidates[/green]")
   823→    else:
   824→        loop = asyncio.get_running_loop()
   825→        confirm = await loop.run_in_executor(
   826→            None,
   827→            lambda: questionary.confirm(
   828→                f"Confirm batch send to {len(send_list)} candidates?"
   829→            ).unsafe_ask()
   830→        )
   831→
   832→        if not confirm:
   833→            console.print("[yellow]Batch send cancelled.[/yellow]")
   834→            return results
   835→
   836→    # Send messages one by one
   837→    from rich.progress import Progress, BarColumn, TextColumn as ProgressTextColumn, MofNCompleteColumn
   838→
   839→    with Progress(
   840→        ProgressTextColumn("[progress.description]{task.description}"),
   841→        BarColumn(),
   842→        MofNCompleteColumn(),
   843→        console=console
   844→    ) as progress:
   845→        task = progress.add_task("Sending messages...", total=len(send_list))
   846→
   847→        max_retries = 2
   848→        rate_limit_hit = False
   849→
   850→        for c in send_list:
   851→            if rate_limit_hit:
   852→                break
   853→
   854→            # Check rate limit (waits through min-interval)
   855→            try:
   856→                await async_check_rate_limit('send_message')
   857→            except RateLimitExceededError as e:
   858→                console.print(f"\n[yellow]Rate limit reached: {e}[/yellow]")
   859→                remaining = len(send_list) - (results['sent'] + results['failed'])
   860→                results['rate_limited'] = remaining
   861→                results['details'].append({
   862→                    'maimai_id': c['maimai_id'],
   863→                    'name': c['name'],
   864→                    'status': 'rate_limited',
   865→                    'reason': str(e),
   866→                })
   867→                break
   868→
   869→            # Build per-candidate variables
   870→            variables = {
   871→                **common_variables,
   872→                'name': c['name'],
   873→                'company': c['company'],
   874→                'position': c['title'],
   875→            }
   876→            try:
   877→                content = render_template(template['content'], variables, validate=False)  # Already validated in batch
   878→            except ValidationError as e:
   879→                # Should not happen if pre-validation passed, but handle gracefully
   880→                logger.error(f"Template rendering failed for {c['name']}: {e}")
   881→                results['failed'] += 1
   882→                results['details'].append({
   883→                    'maimai_id': c['maimai_id'],
   884→                    'name': c['name'],
   885→                    'status': 'failed',
   886→                    'reason': f"模板渲染失败: {e}",
   887→                })
   888→                continue
   889→
   890→            # Retry loop for transient failures
   891→            sent = False
   892→            last_error = None
   893→            per_candidate_timeout = 90  # seconds per candidate
   894→            for attempt in range(1 + max_retries):
   895→                try:
   896→                    success = await asyncio.wait_for(
   897→                        send_message(
   898→                            maimai_id=c['maimai_id'],
   899→                            template_name=template_name,
   900→                            variables=variables,
   901→                            browser=browser,
   902→                            skip_confirmation=True,
   903→                            interactive=interactive,
   904→                            discovered_selectors=discovered_selectors,
   905→                            _skip_rate_check=True,
   906→                            profile_url=c.get('profile_url'),
   907→                        ),
   908→                        timeout=per_candidate_timeout,
   909→                    )
   910→
   911→                    if success:
   912→                        results['sent'] += 1
   913→                        results['details'].append({
   914→                            'maimai_id': c['maimai_id'],
   915→                            'name': c['name'],
   916→                            'status': 'sent',
   917→                            'attempts': attempt + 1,
   918→                            'content': content,
   919→                        })
   920→                        sent = True
   921→                        break
   922→                    else:
   923→                        last_error = "send returned False"
   924→
   925→                except asyncio.TimeoutError:
   926→                    last_error = f"Timed out after {per_candidate_timeout}s"
   927→                    console.print(f"[yellow]Timeout for {c['name']} ({c['maimai_id']}), skipping.[/yellow]")
   928→                    break
   929→                except QuotaExhaustedError as e:
   930→                    # Quota is a global resource — abort the entire batch
   931→                    remaining = len(send_list) - (results['sent'] + results['failed'])
   932→                    results['failed'] += 1
   933→                    results['details'].append({
   934→                        'maimai_id': c['maimai_id'],
   935→                        'name': c['name'],
   936→                        'status': 'quota_exhausted',
   937→                        'reason': str(e),
   938→                    })
   939→                    console.print(f"\n[red]联系券已用尽，终止批量发送 (剩余 {remaining - 1} 人未发送)[/red]")
   940→                    rate_limit_hit = True  # reuse flag to skip remaining candidates
   941→                    break
   942→                except PermanentFailureError as e:
   943→                    last_error = str(e)
   944→                    console.print(f"[yellow]{e}. Skipping.[/yellow]")
   945→                    break
   946→                except RateLimitExceededError as e:
   947→                    remaining = len(send_list) - (results['sent'] + results['failed'])
   948→                    results['rate_limited'] = remaining
   949→                    results['details'].append({
   950→                        'maimai_id': c['maimai_id'],
   951→                        'name': c['name'],
   952→                        'status': 'rate_limited',
   953→                        'reason': str(e),
   954→                    })
   955→                    console.print(f"\n[yellow]Rate limit reached: {e}[/yellow]")
   956→                    rate_limit_hit = True
   957→                    break
   958→                except Exception as e:
   959→                    last_error = str(e)
   960→
   961→                # Retry with exponential backoff if not last attempt
   962→                if attempt < max_retries and not rate_limit_hit:
   963→                    from src.utils.delays import get_retry_delay
   964→                    delay = get_retry_delay(attempt, base_delay=3.0, max_delay=30.0)
   965→                    logger.info(f"Retry {attempt + 1}/{max_retries} for {c['name']} in {delay:.1f}s")
   966→                    await asyncio.sleep(delay)
   967→
   968→            if not sent and not rate_limit_hit:
   969→                actual_attempts = attempt + 1
   970→                results['failed'] += 1
   971→                results['details'].append({
   972→                    'maimai_id': c['maimai_id'],
   973→                    'name': c['name'],
   974→                    'status': 'failed',
   975→                    'reason': last_error,
   976→                    'attempts': actual_attempts,
   977→                    'content': content,
   978→                })
   979→                logger.error(f"Failed to send message to {c['name']} after {actual_attempts} attempt(s): {last_error}")
   980→
   981→            progress.advance(task)
   982→
   983→            # Emit progress for web service
   984→            emit_progress("send_progress",
   985→                          sent=results['sent'], failed=results['failed'],
   986→                          total=len(send_list), current=c['name'])
   987→
   988→            # Human-like delay between messages
   989→            await async_random_delay('message')
   990→
   991→    return results

   994→def render_template(template_content: str, variables: Dict[str, str], validate: bool = True) -> str:
   995→    """
   996→    Render template with variables.
   997→
   998→    Maimai chat is plain text, so values are substituted as-is without
   999→    HTML escaping (which would produce visible &amp; entities in chat).
  1000→
  1001→    Args:
  1002→        template_content: Template string with {variable} placeholders
  1003→        variables: Variable values to substitute
  1004→        validate: Whether to validate template before rendering (default True)
  1005→
  1006→    Returns:
  1007→        Rendered message content
  1008→
  1009→    Raises:
  1010→        ValidationError: If template validation fails and validate=True
  1011→    """
  1012→    if validate:
  1013→        # Validate template can be fully rendered
  1014→        is_valid, missing = validate_template_rendering(template_content, variables, {})
  1015→        if not is_valid:
  1016→            raise ValidationError(
  1017→                f"模板变量缺失或为空: {', '.join(missing)}\n"
  1018→                f"请检查模板或提供完整变量"
  1019→            )
  1020→
  1021→    content = template_content
  1022→    for key, value in variables.items():
  1023→        content = content.replace(f"{{{key}}}", str(value))
  1024→    return content

  1092→async def interactive_confirm(
  1093→    content: str,
  1094→    candidate_info: Optional[Dict] = None,
  1095→    template_name: Optional[str] = None
  1096→) -> str:
  1097→    """
  1098→    Show interactive confirmation dialog.
  1099→
  1100→    Args:
  1101→        content: Message content to confirm
  1102→        candidate_info: Optional candidate info
  1103→        template_name: Optional template name
  1104→
  1105→    Returns:
  1106→        Action string: 'send', 'edit', or 'skip'
  1107→    """
  1108→    # Display preview
  1109→    display_message_preview(content, candidate_info, template_name)
  1110→
  1111→    console.print()
  1112→
  1113→    # Interactive prompt
  1114→    choices = [
  1115→        questionary.Choice("Send message", value="send"),
  1116→        questionary.Choice("Edit content", value="edit"),
  1117→        questionary.Choice("Skip (don't send)", value="skip"),

  1320→async def _verify_message_sent(page: Page, content: str, timeout: int = 5000) -> bool:
  1321→    """
  1322→    Verify that a message was actually sent by checking the conversation area.
  1323→
  1324→    Polls up to `timeout` ms with short intervals to handle async rendering.
  1325→
  1326→    Checks for:
  1327→    1. The sent message content appearing in the conversation
  1328→    2. The message input being cleared (indicating successful send)
  1329→    3. A sent/delivered indicator element
  1330→
  1331→    Args:
  1332→        page: Playwright page instance
  1333→        content: The message content that was sent
  1334→        timeout: Max time to wait for verification in ms
  1335→
  1336→    Returns:
  1337→        True if message send was verified
  1338→    """
  1339→    import time
  1340→
  1341→    # Short snippet for matching (first 20 chars, avoid template artifacts)
  1342→    snippet = content[:20].strip()
  1343→
  1344→    poll_interval_ms = 500
  1345→    deadline = time.monotonic() + timeout / 1000.0
  1346→
  1347→    while True:
  1348→        try:
  1349→            # Strategy 1: Check if message content appears in conversation area
  1350→            verified = await page.evaluate('''(snippet) => {
  1351→                const chatSelectors = [
  1352→                    '[class*="message-list"]',
  1353→                    '[class*="chat-content"]',
  1354→                    '[class*="conversation"]',
  1355→                    '[class*="msg-list"]',
  1356→                    '[class*="im-message"]',
  1357→                    '[class*="chat-msg"]',
  1358→                ];
  1359→
  1360→                for (const sel of chatSelectors) {
  1361→                    const container = document.querySelector(sel);
  1362→                    if (container && container.textContent.includes(snippet)) {
  1363→                        return true;
  1364→                    }
  1365→                }
  1366→
  1367→                // Fallback: scan leaf-ish elements in the lower portion of the page
  1368→                const viewportHeight = window.innerHeight;
  1369→                const elements = document.querySelectorAll('div, p, span');
  1370→                for (const el of elements) {
  1371→                    const rect = el.getBoundingClientRect();
  1372→                    if (rect.top > viewportHeight * 0.3 && el.textContent.includes(snippet)) {
  1373→                        if (el.textContent.length < snippet.length + 200) {
  1374→                            return true;
  1375→                        }
  1376→                    }
  1377→                }
  1378→
  1379→                return false;
  1380→            }''', snippet)
  1381→
  1382→            if verified:
  1383→                logger.debug("Message send verified: content found in conversation")
  1384→                return True
  1385→
  1386→            # Strategy 2: Check if input was cleared (indicates successful send)
  1387→            input_cleared = await page.evaluate('''() => {
  1388→                const inputs = document.querySelectorAll(
  1389→                    'textarea, [contenteditable="true"], input[type="text"]'
  1390→                );
  1391→                for (const input of inputs) {
  1392→                    const rect = input.getBoundingClientRect();
  1393→                    if (rect.width > 0 && rect.height > 0) {
  1394→                        const value = input.value || input.textContent || '';
  1395→                        if (value.trim() === '') {
  1396→                            return true;
  1397→                        }
  1398→                    }
  1399→                }
  1400→                return false;
  1401→            }''')
  1402→
  1403→            if input_cleared:
  1404→                logger.debug("Message send verified: input field was cleared")
  1405→                return True
  1406→
  1407→            # Strategy 3: Check for sent/delivered indicators
  1408→            sent_indicators = [
  1409→                '[class*="sent"]',
  1410→                '[class*="delivered"]',
  1411→                '[class*="success"]',
  1412→                '[class*="msg-status"]',
  1413→            ]
  1414→            for selector in sent_indicators:
  1415→                try:
  1416→                    indicator = await page.query_selector(selector)
  1417→                    if indicator and await indicator.is_visible():
  1418→                        logger.debug(f"Message send verified: indicator found ({selector})")
  1419→                        return True
  1420→                except Exception:
  1421→                    continue
  1422→
  1423→        except Exception as e:
  1424→            logger.warning(f"Message verification poll failed: {e}")
  1425→
  1426→        # Check if we've exceeded the deadline
  1427→        if time.monotonic() >= deadline:
  1428→            break
  1429→
  1430→        await asyncio.sleep(poll_interval_ms / 1000.0)
  1431→
  1432→    logger.warning("Could not verify message was sent after polling")
  1433→    return False
  1434→

  1436→async def _check_existing_conversation(page: Page, maimai_id: str) -> bool:
  1437→    """
  1438→    Check if there's already an existing conversation with this candidate.
  1439→
  1440→    Checks for actual user message bubbles (not system prompts/tips) in the
  1441→    conversation area.
  1442→
  1443→    Args:
  1444→        page: Playwright page instance
  1445→        maimai_id: Candidate's Maimai ID
  1446→
  1447→    Returns:
  1448→        True if an existing conversation was detected
  1449→    """
  1450→    try:
  1451→        has_conversation = await page.evaluate('''() => {
  1452→            // Look for conversation message containers
  1453→            const chatSelectors = [
  1454→                '[class*="message-list"]',
  1455→                '[class*="chat-content"]',
  1456→                '[class*="conversation"]',
  1457→                '[class*="msg-list"]',
  1458→                '[class*="im-message"]',
  1459→                '[class*="chat-msg"]',
  1460→                '[class*="chat-record"]',
  1461→            ];
  1462→
  1463→            // System/tip patterns that should NOT count as existing conversation
  1464→            const systemPatterns = [
  1465→                '还不是好友', '添加好友', '发送好友申请', '对方未回复',
  1466→                '以上是打招呼内容', '暂无消息', '开始聊天', '你们已经是好友',
  1467→                '系统消息', '温馨提示', '安全提示', '注意事项',
  1468→                '开启招呼语', '求职意向', '目前情况', '竞业点',
  1469→                '欢迎联系', '不感兴趣',
  1470→            ];
  1471→
  1472→            const isSystemText = (text) => {
  1473→                const t = text.trim();
  1474→                if (t.length < 5) return true;  // Too short to be a real message
  1475→                return systemPatterns.some(p => t.includes(p));
  1476→            };
  1477→
  1478→            for (const sel of chatSelectors) {
  1479→                const container = document.querySelector(sel);
  1480→                if (!container) continue;
  1481→
  1482→                // Strategy 1: Look for message bubble elements
  1483→                const bubbleSelectors = [
  1484→                    '[class*="msg-item"]', '[class*="message-item"]',
  1485→                    '[class*="bubble"]', '[class*="chat-item"]',
  1486→                    '[class*="msg-bubble"]', '[class*="chat-bubble"]',
  1487→                ];
  1488→                for (const bSel of bubbleSelectors) {
  1489→                    const bubbles = container.querySelectorAll(bSel);
  1490→                    // Filter out system/tip bubbles
  1491→                    let realMessageCount = 0;
  1492→                    for (const bubble of bubbles) {
  1493→                        const text = bubble.textContent.trim();
  1494→                        if (text.length > 5 && !isSystemText(text)) {
  1495→                            realMessageCount++;
  1496→                        }
  1497→                    }
  1498→                    if (realMessageCount > 0) {
  1499→                        return true;
  1500→                    }
  1501→                }
  1502→
  1503→                // Strategy 2: If no bubble selectors matched, check direct children
  1504→                // that look like message blocks (not system tips)
  1505→                const children = container.children;
  1506→                let realChildCount = 0;
  1507→                for (const child of children) {
  1508→                    const text = child.textContent.trim();
  1509→                    // Must have meaningful content and not be a system message
  1510→                    if (text.length > 10 && !isSystemText(text)) {
  1511→                        // Check it's not a timestamp-only or status-only element
  1512→                        if (child.querySelector('img[class*="avatar"]') || text.length > 20) {
  1513→                            realChildCount++;
  1514→                        }
  1515→                    }
  1516→                }
  1517→                if (realChildCount > 0) {
  1518→                    return true;
  1519→                }
  1520→            }
  1521→            return false;
  1522→        }''')
  1523→
  1524→        if has_conversation:
  1525→            logger.info(f"Existing conversation detected for {maimai_id}")
  1526→        return has_conversation
  1527→
  1528→    except Exception as e:
  1529→        # On error, assume conversation exists (safe side: avoid duplicate messages)
  1530→        logger.warning(f"Conversation check failed for {maimai_id}: {e} — assuming existing conversation")
  1531→        return True

  1900→async def _find_im_chat_frame(im_page: Page):
  1901→    """Find the chat iframe inside an IM tab page.

  1915→async def _exchange_in_im_tab(
  1916→    browser: 'BrowserManager',
  1917→    config,
  1918→    exchange_phone: bool,
  1919→    exchange_wechat: bool,

  1921→    """Switch to the IM tab and click exchange phone/WeChat buttons.
  1922→
  1923→    The exchange buttons live inside a chat iframe within the IM tab.
  1924→    Each button is a ``div.tool.normal`` with a ``.tool-text`` child
  1925→    whose text is "交换电话" or "交换微信".  After clicking, the button
  1926→    state changes to ``div.tool.disable`` ("申请中" / "交换中").
  1927→
  1928→    Args:
  1929→        browser: BrowserManager with open IM tab
  1930→        config: Config instance
  1931→        exchange_phone: Click "交换电话"
  1932→        exchange_wechat: Click "交换微信"
  1933→    """
  1934→    # Find the IM tab
  1935→    im_page = None
  1936→    for p in browser.context.pages:
  1937→        if "im" in p.url and "direct_modal" in p.url:
  1938→            im_page = p
  1939→            break
  1940→
  1941→    if not im_page:
  1942→        logger.warning("Talent bank exchange: IM tab not found")
  1943→        return
  1944→
  1945→    try:
  1946→        # Wait for iframe to load
  1947→        await im_page.wait_for_load_state("domcontentloaded", timeout=8000)
  1948→        await im_page.wait_for_timeout(2000)
  1949→
  1950→        chat_frame = await _find_im_chat_frame(im_page)
  1951→        if not chat_frame:
  1952→            logger.warning("Talent bank exchange: chat iframe not found")
  1953→            return
  1954→
  1955→        if exchange_phone:
  1956→            try:
  1957→                phone_btn = chat_frame.locator("div.tool.normal .tool-text", has_text="交换电话").first
  1958→                if await phone_btn.count() > 0:
  1959→                    await phone_btn.click()
  1960→                    logger.info("Talent bank: clicked 交换电话")
  1961→                    await im_page.wait_for_timeout(1000)
  1962→                else:
  1963→                    logger.debug("Talent bank: 交换电话 button not available (already exchanged or disabled)")
  1964→            except Exception as e:
  1965→                logger.debug(f"Talent bank: exchange phone failed: {e}")
  1966→
  1967→        if exchange_wechat:
  1968→            try:
  1969→                wechat_btn = chat_frame.locator("div.tool.normal .tool-text", has_text="交换微信").first
  1970→                if await wechat_btn.count() > 0:
  1971→                    await wechat_btn.click()
  1972→                    logger.info("Talent bank: clicked 交换微信")
  1973→                    await im_page.wait_for_timeout(1000)
  1974→                else:
  1975→                    logger.debug("Talent bank: 交换微信 button not available (already exchanged or disabled)")
  1976→            except Exception as e:
  1977→                logger.debug(f"Talent bank: exchange wechat failed: {e}")
  1978→
  1979→    except Exception as e:
  1980→        logger.error(f"Talent bank exchange error: {e}")
