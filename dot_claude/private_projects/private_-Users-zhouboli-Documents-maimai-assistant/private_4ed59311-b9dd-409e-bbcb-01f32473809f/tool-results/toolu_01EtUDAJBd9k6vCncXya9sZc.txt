     1→"""Browser manager for Playwright automation.
     2→
     3→Provides a singleton browser manager that connects to an existing Chrome
     4→instance via CDP (Chrome DevTools Protocol). Supports multi-tab management
     5→for workflows that open new tabs (e.g., enterprise recruitment).
     6→"""
     7→
     8→import asyncio
     9→from pathlib import Path
    10→from typing import List, Optional
    11→from datetime import datetime
    12→
    13→from patchright.async_api import async_playwright, Browser, BrowserContext, Page, Playwright
    14→from loguru import logger
    15→
    16→from src.utils.config import get_config, Config
    17→from src.core.session import SessionManager, get_session_manager
    18→from src.exceptions.errors import BrowserError
    19→
    20→
    21→class BrowserManager:
    22→    """
    23→    Singleton browser manager for Maimai automation.
    24→
    25→    Connects to a user-managed Chrome instance via CDP endpoint.
    26→    The user starts Chrome with --remote-debugging-port=9222 and logs in
    27→    manually. This manager connects/disconnects without closing Chrome.
    28→    """
    29→
    30→    _instance: Optional['BrowserManager'] = None
    31→    _lock: asyncio.Lock = None  # initialized once at class level
    32→
    33→    def __new__(cls, **kwargs):
    34→        if cls._instance is None:
    35→            cls._instance = super().__new__(cls)
    36→            cls._instance._initialized = False
    37→        return cls._instance
    38→
    39→    def __init__(
    40→        self,
    41→        config: Optional[Config] = None,
    42→        session_manager: Optional[SessionManager] = None,
    43→    ):
    44→        if self._initialized:
    45→            return
    46→
    47→        self._playwright: Optional[Playwright] = None
    48→        self._browser: Optional[Browser] = None
    49→        self._context: Optional[BrowserContext] = None
    50→        self._page: Optional[Page] = None
    51→        self._primary_page: Optional[Page] = None  # Original tab to return to
    52→        self._opened_tabs: List[Page] = []  # Tabs opened during our session
    53→        self._session_manager: SessionManager = session_manager or get_session_manager()
    54→        self._config: Config = config or get_config()
    55→        self._cdp_endpoint: Optional[str] = None
    56→        self._owned_page: bool = False  # True if we created the page (vs reusing existing)
    57→        self._initialized = True
    58→
    59→    @property
    60→    def is_running(self) -> bool:
    61→        """Check if browser is currently connected."""
    62→        return self._browser is not None and self._browser.is_connected()
    63→
    64→    @property
    65→    def page(self) -> Optional[Page]:
    66→        """Get current page instance."""
    67→        return self._page
    68→
    69→    @property
    70→    def context(self) -> Optional[BrowserContext]:
    71→        """Get current browser context."""
    72→        return self._context
    73→
    74→    @classmethod
    75→    def _ensure_lock(cls):
    76→        """Lazily create the asyncio.Lock inside a running event loop.
    77→
    78→        Uses a double-check pattern: if two coroutines race here, only
    79→        the first one creates the lock (asyncio is single-threaded within
    80→        one event loop, so the second check after creation is safe).
    81→        """
    82→        if cls._lock is None:
    83→            cls._lock = asyncio.Lock()
    84→
    85→    def set_cdp_endpoint(self, endpoint: str) -> None:
    86→        """Override CDP endpoint (e.g. from CLI --cdp flag)."""
    87→        self._cdp_endpoint = endpoint
    88→
    89→    async def start(self) -> Page:
    90→        """
    91→        Connect to Chrome via CDP and return a page.
    92→
    93→        Returns:
    94→            Configured Playwright page instance
    95→
    96→        Raises:
    97→            BrowserError: If connection fails
    98→        """
    99→        self._ensure_lock()
   100→        async with self._lock:
   101→            if self.is_running:
   102→                logger.debug("Browser already connected, returning existing page")
   103→                return self._page
   104→
   105→            try:
   106→                # Determine CDP endpoint
   107→                endpoint = self._cdp_endpoint or self._config.get(
   108→                    'browser.cdp_endpoint', 'http://localhost:9222'
   109→                )
   110→                timeout = self._config.get('browser.timeout', 30000)
   111→
   112→                logger.info(f"Connecting to Chrome via CDP: {endpoint}")
   113→
   114→                # Start Playwright
   115→                self._playwright = await async_playwright().start()
   116→
   117→                # Connect to existing Chrome via CDP
   118→                self._browser = await self._playwright.chromium.connect_over_cdp(endpoint)
   119→
   120→                # Get the default context (the user's browser context with cookies)
   121→                contexts = self._browser.contexts
   122→                if contexts:
   123→                    self._context = contexts[0]
   124→                else:
   125→                    # Fallback: create a new context (shouldn't happen with CDP)
   126→                    logger.warning("No existing context found, creating new one")
   127→                    self._context = await self._browser.new_context()
   128→
   129→                # Set default timeout
   130→                self._context.set_default_timeout(timeout)
   131→
   132→                # Reuse existing page or create one
   133→                pages = self._context.pages
   134→                if pages:
   135→                    self._page = pages[0]
   136→                    self._owned_page = False
   137→                    logger.info(f"Reusing existing tab: {self._page.url}")
   138→                else:
   139→                    self._page = await self._context.new_page()
   140→                    self._owned_page = True
   141→                    logger.info("Created new tab")
   142→
   143→                # Remember the primary tab
   144→                self._primary_page = self._page
   145→                self._opened_tabs = []
   146→
   147→                logger.info("Connected to Chrome via CDP successfully")
   148→                return self._page
   149→
   150→            except Exception as e:
   151→                logger.error(f"Failed to connect to Chrome via CDP: {e}")
   152→                await self._close_internal()
   153→                raise BrowserError(
   154→                    f"Failed to connect to Chrome via CDP at {endpoint}. "
   155→                    f"Make sure Chrome is running with --remote-debugging-port=9222. "
   156→                    f"Error: {e}"
   157→                )
   158→
   159→    async def close(self) -> None:
   160→        """Disconnect from Chrome via CDP. Does NOT close the user's Chrome."""
   161→        self._ensure_lock()
   162→        async with self._lock:
   163→            await self._close_internal()
   164→
   165→    async def _close_internal(self) -> None:
   166→        """Actual cleanup logic — callable without acquiring _lock.
   167→
   168→        Must only be called while the lock is already held, or from
   169→        start()'s except block (which already holds the lock).
   170→        """
   171→        try:
   172→            # Close tabs we opened during the session
   173→            for tab in self._opened_tabs:
   174→                try:
   175→                    if not tab.is_closed():
   176→                        await tab.close()
   177→                except Exception:
   178→                    pass
   179→            self._opened_tabs = []
   180→
   181→            # Only close pages we created, not the user's existing tabs
   182→            if self._page and self._owned_page:
   183→                try:
   184→                    await self._page.close()
   185→                except Exception:
   186→                    pass
   187→            self._page = None
   188→            self._primary_page = None
   189→            self._owned_page = False
   190→
   191→            # Don't close the context — it belongs to the user's Chrome
   192→            self._context = None
   193→
   194→            # browser.close() in CDP mode just disconnects, doesn't kill Chrome
   195→            if self._browser:
   196→                try:
   197→                    await self._browser.close()
   198→                except Exception as e:
   199→                    logger.error(f"Error closing browser connection: {e}")
   200→                finally:
   201→                    self._browser = None
   202→
   203→            if self._playwright:
   204→                try:
   205→                    await self._playwright.stop()
   206→                except Exception as e:
   207→                    logger.error(f"Error stopping Playwright: {e}")
   208→                finally:
   209→                    self._playwright = None
   210→
   211→            logger.info("Disconnected from Chrome (browser stays open)")
   212→
   213→        except Exception as e:
   214→            logger.error(f"Error disconnecting from Chrome: {e}")
   215→
   216→    async def wait_for_new_tab(self, timeout: int = 10000) -> Page:
   217→        """Wait for a click-triggered new tab, apply stealth, and switch to it.
   218→
   219→        Call this right after performing an action that opens a new tab
   220→        (e.g., clicking a link with target=_blank).
   221→
   222→        Args:
   223→            timeout: Max wait time in milliseconds
   224→
   225→        Returns:
   226→            The new Page instance (also set as the active page)
   227→
   228→        Raises:
   229→            BrowserError: If no new tab appears within timeout
   230→        """
   231→        if not self._context:
   232→            raise BrowserError("No active context for wait_for_new_tab")
   233→
   234→        try:
   235→            new_page = await self._context.wait_for_event(
   236→                "page", timeout=timeout
   237→            )
   238→            await new_page.wait_for_load_state("domcontentloaded")
   239→            self._opened_tabs.append(new_page)
   240→            self._page = new_page
   241→            logger.info(f"New tab captured: {new_page.url}")
   242→            return new_page
   243→        except Exception as e:
   244→            raise BrowserError(f"No new tab opened within {timeout}ms: {e}")
   245→
   246→    def switch_to_primary_tab(self) -> Page:
   247→        """Switch back to the original (primary) tab.
   248→
   249→        Returns:
   250→            The primary Page instance
   251→        """
   252→        if not self._primary_page or self._primary_page.is_closed():
   253→            raise BrowserError("Primary tab is not available")
   254→
   255→        self._page = self._primary_page
   256→        logger.debug(f"Switched to primary tab: {self._page.url}")
   257→        return self._page
   258→
   259→    async def close_current_tab(self) -> Page:
   260→        """Close the current active tab and switch to the previous one.
   261→
   262→        Will not close the primary tab. After closing, switches to the
   263→        most recent remaining tab or falls back to the primary tab.
   264→
   265→        Returns:
   266→            The page that became active after closing
   267→
   268→        Raises:
   269→            BrowserError: If trying to close the primary tab
   270→        """
   271→        if not self._page:
   272→            raise BrowserError("No active page to close")
   273→
   274→        if self._page == self._primary_page:
   275→            raise BrowserError("Cannot close the primary tab")
   276→
   277→        closing = self._page
   278→        try:
   279→            # Remove from opened_tabs tracking
   280→            if closing in self._opened_tabs:
   281→                self._opened_tabs.remove(closing)
   282→            await closing.close()
   283→        except Exception as e:
   284→            logger.warning(f"Error closing tab: {e}")
   285→
   286→        # Switch to previous tab
   287→        if self._context and self._context.pages:
   288→            self._page = self._context.pages[-1]
   289→        else:
   290→            self._page = self._primary_page
   291→
   292→        logger.debug(f"Closed tab, switched to: {self._page.url if self._page else 'none'}")
   293→        return self._page
   294→
   295→    async def save_session(self) -> None:
   296→        """Save current session to disk (backup of cookies)."""
   297→        if self._context:
   298→            await self._session_manager.save(self._context)
   299→        else:
   300→            logger.warning("No active context to save session from")
   301→
   302→    async def screenshot(self, name: str = None) -> str:
   303→        """
   304→        Take a screenshot of current page.
   305→
   306→        Args:
   307→            name: Screenshot name (timestamp used if None)
   308→
   309→        Returns:
   310→            Path to saved screenshot
   311→        """
   312→        if not self._page:
   313→            raise BrowserError("No active page for screenshot")
   314→
   315→        screenshot_dir = Path(self._config.get('data.screenshot_dir', 'data/screenshots'))
   316→        screenshot_dir.mkdir(parents=True, exist_ok=True)
   317→
   318→        if name is None:
   319→            name = datetime.now().strftime('%Y%m%d_%H%M%S')
   320→
   321→        filepath = screenshot_dir / f"{name}.png"
   322→
   323→        await self._page.screenshot(path=str(filepath), full_page=True)
   324→        logger.info(f"Screenshot saved: {filepath}")
   325→
   326→        return str(filepath)
   327→
   328→
   329→
   330→# Global browser manager instance
   331→_browser_manager: Optional[BrowserManager] = None
   332→
   333→
   334→def get_browser_manager() -> BrowserManager:
   335→    """Get global browser manager instance."""
   336→    global _browser_manager
   337→    if _browser_manager is None:
   338→        _browser_manager = BrowserManager()
   339→    return _browser_manager
