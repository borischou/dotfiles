    53→def _is_non_rd_title(title: str) -> bool:
    54→    """Check if a title indicates a non-R&D role.
    55→
    56→    Strict mode: if any non-R&D keyword appears, filter out regardless of
    57→    whether R&D keywords are also present (e.g. "技术产品经理" is filtered).
    58→    Returns False (keep) only if title is empty or contains no non-R&D keywords.
    59→    """
    60→    if not title:
    61→        return False
    62→
    63→    # Strict: any non-R&D keyword present → filter out
    64→    if any(kw in title for kw in NON_RD_KEYWORDS):
    65→        return True
    66→
    67→    return False

    81→async def search_candidates(
    82→    keywords: str,
    83→    company: Optional[tuple] = None,
    84→    location: Optional[str] = None,
    85→    pages: int = 1,
    86→    start_page: int = 1,
    87→    browser: Optional[BrowserManager] = None
    88→) -> List[CandidateBasic]:
    89→    """
    90→    Search for candidates on Maimai.
    91→
    92→    Args:
    93→        keywords: Search keywords (e.g., "高级前端工程师")
    94→        company: Filter by company names, tuple of strings (optional)
    95→        location: Filter by location (optional)
    96→        pages: Number of pages to scrape (default 1)
    97→        start_page: First page number to scrape (default 1)
    98→        browser: Browser manager instance
    99→
   100→    Returns:
   101→        List of CandidateBasic objects
   102→
   103→    Raises:
   104→        RateLimitExceededError: If rate limit exceeded
   105→        SelectorNotFoundError: If page elements not found
   106→    """
   107→    # Validate inputs
   108→    validate_keywords(keywords)
   109→
   110→    if browser is None:
   111→        browser = get_browser_manager()
   112→
   113→    config = get_config()
   114→    rate_limiter = get_rate_limiter()
   115→    db_manager = get_db_manager()
   116→
   117→    # Ensure logged in
   118→    await ensure_logged_in(browser)
   119→
   120→    all_candidates: List[CandidateBasic] = []
   121→
   122→    with Progress(
   123→        SpinnerColumn(),
   124→        TextColumn("[progress.description]{task.description}"),
   125→        console=console
   126→    ) as progress:
   127→
   128→        end_page = start_page + pages
   129→        for page_num in range(start_page, end_page):
   130→            task = progress.add_task(f"Searching page {page_num}/{end_page - 1}...", total=None)
   131→
   132→            try:
   133→                # Check rate limit (waits through min-interval)
   134→                await async_check_rate_limit('search')
   135→
   136→                # Build search URL
   137→                params = {
   138→                    'type': 'contact',
   139→                    'query': keywords,
   140→                    'highlight': 'true'
   141→                }
   142→                if page_num > 1:
   143→                    params['page'] = str(page_num)
   144→
   145→                search_url = f"{MAIMAI_SEARCH_URL}?{urlencode(params)}"
   146→
   147→                page = browser.page
   148→
   149→                # Navigate to search page
   150→                await retry_navigation(page, search_url, wait_until='domcontentloaded')
   151→
   152→                # Wait for page load
   153→                await async_random_delay('page_load')
   154→
   155→                # Check for captcha
   156→                await check_captcha(page, report_to_rate_limiter=True)
   157→
   158→                # Click "人脉" tab to show contact results
   159→                try:
   160→                    renmai_tab = await page.query_selector('text="人脉"')
   161→                    if renmai_tab:
   162→                        await renmai_tab.click()
   163→                        logger.info("Clicked '人脉' tab")
   164→                        await async_random_delay('action')
   165→                        # Wait for content to load
   166→                        await page.wait_for_timeout(2000)
   167→                except Exception as e:
   168→                    logger.debug(f"Could not click 人脉 tab: {e}")
   169→
   170→                # Simulate human behavior
   171→                await human_scroll(page, 'down', distance=300)
   172→                await async_random_delay('action')
   173→
   174→                # Extract candidates from this page
   175→                candidates = await _extract_search_results(page, config)
   176→
   177→                if not candidates:
   178→                    logger.info(f"No results found on page {page_num}")
   179→                    await browser.screenshot(f"search_debug_page{page_num}")
   180→                    progress.update(task, description=f"Page {page_num}: No results")
   181→                    # Allow one empty page (could be transient), break on second consecutive
   182→                    if page_num > start_page and not all_candidates:
   183→                        # First page had no results either — genuinely empty
   184→                        break
   185→                    continue
   186→
   187→                # Filter out non-R&D roles
   188→                before_rd_filter = len(candidates)
   189→                candidates = [c for c in candidates if not _is_non_rd_title(c.title)]
   190→                rd_filtered = before_rd_filter - len(candidates)
   191→                if rd_filtered > 0:
   192→                    logger.info(f"Filtered out {rd_filtered} non-R&D candidates")
   193→
   194→                # Apply filters if specified
   195→                if company:
   196→                    company_lower = [cn.lower() for cn in company]
   197→                    candidates = [c for c in candidates if any(cn in c.company.lower() for cn in company_lower)]
   198→
   199→                if location:
   200→                    candidates = [c for c in candidates if c.location and location.lower() in c.location.lower()]
   201→
   202→                all_candidates.extend(candidates)
   203→                progress.update(task, description=f"Page {page_num}: Found {len(candidates)} candidates")
   204→
   205→                # Log the action
   206→                rate_limiter.log_action('search', {
   207→                    'keywords': keywords,
   208→                    'page': page_num,
   209→                    'results': len(candidates)
   210→                })
   211→
   212→                # Save candidates to database
   213→                _save_candidates_to_db(db_manager, candidates, keywords)
   214→
   215→                # Delay before next page
   216→                if page_num < end_page - 1:
   217→                    await async_random_delay('page_load')
   218→
   219→            except (BrowserError, SessionExpiredError):
   220→                raise  # Critical errors must propagate
   221→            except Exception as e:
   222→                logger.error(f"Error on page {page_num}: {e}")
   223→                progress.update(task, description=f"Page {page_num}: Error - {str(e)[:30]}")
   224→                # Take screenshot for debugging
   225→                await browser.screenshot(f"search_error_page{page_num}")
   226→                continue
   227→
   228→    # Log search to history
   229→    _log_search_history(db_manager, keywords, company, location, pages, len(all_candidates))
   230→
   231→    console.print(f"[green]Search complete. Found {len(all_candidates)} candidates.[/green]")
   232→    return all_candidates

   235→async def _extract_search_results(page: Page, config) -> List[CandidateBasic]:
   236→    """
   237→    Extract candidate information from search results page.
   238→
   239→    Args:
   240→        page: Playwright page instance
   241→        config: Config instance
   242→
   243→    Returns:
   244→        List of CandidateBasic objects
   245→    """
   246→    candidates = []
   247→
   248→    try:
   249→        # Wait for page to be fully loaded
   250→        await page.wait_for_load_state('networkidle', timeout=10000)
   251→    except Exception:
   252→        pass  # Continue even if networkidle times out
   253→
   254→    # Wait for actual result cards to render (avoid "搜索数据中..." race condition)
   255→    try:
   256→        await page.wait_for_function(
   257→            '''() => {
   258→                const body = document.body ? document.body.innerText : '';
   259→                if (body.includes('搜索数据中') || body.includes('加载中')) return false;
   260→                // Check for avatar images indicating result cards are rendered
   261→                const imgs = document.querySelectorAll('img');
   262→                let avatarCount = 0;
   263→                for (const img of imgs) {
   264→                    const rect = img.getBoundingClientRect();
   265→                    if (rect.width >= 20 && rect.width <= 100 && rect.height >= 20 && rect.height <= 100) {
   266→                        avatarCount++;
   267→                    }
   268→                }
   269→                return avatarCount >= 1;
   270→            }''',
   271→            timeout=8000,
   272→        )
   273→    except Exception:
   274→        logger.debug("Timed out waiting for result cards to render, proceeding with extraction")
   275→
   276→    try:
   277→        # Strategy: Extract candidates by finding avatar images and their associated text
   278→        candidates_data = await page.evaluate('''() => {
   279→            const candidates = [];
   280→            const seenNames = new Set();
   281→
   282→            // Find all images that look like avatars
   283→            const images = document.querySelectorAll('img');
   284→
   285→            for (const img of images) {
   286→                // Skip large images (probably not avatars)
   287→                const rect = img.getBoundingClientRect();
   288→                if (rect.width > 100 || rect.height > 100) continue;
   289→                if (rect.width < 20 || rect.height < 20) continue;
   290→
   291→                // Find parent container (the card)
   292→                let card = img.parentElement;
   293→                for (let i = 0; i < 8 && card; i++) {
   294→                    const cardRect = card.getBoundingClientRect();
   295→                    // Look for a reasonable card size
   296→                    if (cardRect.width > 200 && cardRect.height > 50 && cardRect.height < 200) {
   297→                        break;
   298→                    }
   299→                    card = card.parentElement;
   300→                }
   301→
   302→                if (!card) continue;
   303→
   304→                // Get all text nodes from card, preserving order
   305→                const textParts = [];
   306→                const walker = document.createTreeWalker(card, NodeFilter.SHOW_TEXT, null, false);
   307→                while (walker.nextNode()) {
   308→                    const text = walker.currentNode.textContent.trim();
   309→                    if (text && text.length > 0 && text.length < 100) {
   310→                        textParts.push(text);
   311→                    }
   312→                }
   313→
   314→                // Parse candidate info from text parts
   315→                // Pattern: Name, Activity time, Company+Position, Industry|Department
   316→                let name = '';
   317→                let company = '';
   318→                let title = '';
   319→                let industry = '';
   320→
   321→                // Compound job title patterns (longer patterns first for greedy match)
   322→                const jobPatterns = [
   323→                    // Compound titles with prefix modifiers
   324→                    /(?:高级|资深|首席|初级|中级|senior|junior|staff|principal|lead)?\s*(?:前端|后端|全栈|web|Web|iOS|Android|Java|Python|Go|C\+\+|PHP|.NET|算法|数据|AI|机器学习|深度学习|NLP|大数据|云计算|DevOps|SRE|安全|测试|嵌入式|移动端|客户端|服务端|区块链|具身智能|自动驾驶|机器人)?\s*(?:工程师|开发|架构师|专家|技术经理|技术总监|技术负责人|Tech\s*Lead)/i,
   325→                    // Management titles
   326→                    /(?:技术|产品|运营|市场|销售|人力|财务|行政|项目|研发)?\s*(?:总监|经理|主管|负责人|VP|总裁|副总|合伙人|创始人|董事|CEO|CTO|COO|CFO|CMO|CHO|CIO)/i,
   327→                    // Other roles
   328→                    /(?:产品|UI|UX|交互|视觉)?\s*(?:设计师|分析师|顾问|运营|助理|实习生?)/i,
   329→                ];
   330→
   331→                // Patterns for text that should be skipped
   332→                const skipPatterns = [
   333→                    /^[+＋]\s*好友$/,          // "+ 好友" button
   334→                    /^\d+个共同好友$/,           // "1个共同好友" social label
   335→                    /^共同好友/,                 // "共同好友..." variants
   336→                    /^已关注$/,                  // "已关注" button
   337→                    /^添加好友$/,                // "添加好友" button
   338→                    /^关注$/,                    // "关注" button
   339→                    /^好友$/,                    // "好友" label
   340→                    /^[|｜·]$/,                  // separators
   341→                ];
   342→
   343→                const isSkippable = (text) => {
   344→                    if (skipPatterns.some(p => p.test(text))) return true;
   345→                    // Activity time
   346→                    if (/天前|小时前|分钟前|刚刚|活跃|在线/.test(text)) return true;
   347→                    return false;
   348→                };
   349→
   350→                for (let i = 0; i < textParts.length; i++) {
   351→                    const part = textParts[i];
   352→
   353→                    if (isSkippable(part)) continue;
   354→
   355→                    // First short text (2-15 chars, pure CJK/letters) is likely the name
   356→                    if (!name && part.length >= 2 && part.length <= 15) {
   357→                        if (/^[\u4e00-\u9fa5a-zA-Z·\s]+$/.test(part)) {
   358→                            name = part;
   359→                            continue;
   360→                        }
   361→                    }
   362→
   363→                    // After name, next substantial text is company+position
   364→                    if (name && !company && part.length > 2) {
   365→                        // Try to split company and position using regex patterns
   366→                        let splitDone = false;
   367→                        for (const pattern of jobPatterns) {
   368→                            const match = part.match(pattern);
   369→                            if (match && match.index > 0) {
   370→                                company = part.substring(0, match.index).trim();
   371→                                title = match[0].trim();
   372→                                splitDone = true;
   373→                                break;
   374→                            }
   375→                        }
   376→                        if (!splitDone) {
   377→                            company = part;
   378→                        }
   379→                        continue;
   380→                    }
   381→
   382→                    // Next part is industry|department info
   383→                    if (name && company && !industry && part.length > 2) {
   384→                        industry = part;
   385→                        // If title wasn't split from company, use industry as fallback context
   386→                        if (!title) {
   387→                            title = industry;
   388→                        }
   389→                        break;
   390→                    }
   391→                }
   392→
   393→                // Skip if no name found or premium placeholder
   394→                if (!name) continue;
   395→                if (name === '脉脉高价值人脉') continue;
   396→
   397→                // Extract user ID from DOM patterns on card and ancestor elements.
   398→                // Two patterns exist:
   399→                //   1. contactcard_{numeric_id} — on ancestors (colleagues layout)
   400→                //   2. member_entry_{numeric_id} — on descendants (standard layout)
   401→                let userId = '';
   402→
   403→                // Check ancestors for numeric contactcard_{id}
   404→                let el = card;
   405→                while (el) {
   406→                    const elId = el.id || '';
   407→                    const ccMatch = elId.match(/^contactcard_(\d+)$/);
   408→                    if (ccMatch) {
   409→                        userId = ccMatch[1];
   410→                        break;
   411→                    }
   412→                    el = el.parentElement;
   413→                }
   414→
   415→                // Check descendants for member_entry_{id} (most common pattern)
   416→                if (!userId) {
   417→                    const meEl = card.querySelector('[id^="member_entry_"]');
   418→                    if (meEl) {
   419→                        const meMatch = meEl.id.match(/^member_entry_(\d+)$/);
   420→                        if (meMatch) {
   421→                            userId = meMatch[1];
   422→                        }
   423→                    }
   424→                }
   425→
   426→                // Also check ancestor descendants for contactcard with numeric ID
   427→                if (!userId) {
   428→                    const ccEl = card.querySelector('[id^="contactcard_"]');
   429→                    if (ccEl) {
   430→                        const ccMatch = ccEl.id.match(/^contactcard_(\d+)$/);
   431→                        if (ccMatch) {
   432→                            userId = ccMatch[1];
   433→                        }
   434→                    }
   435→                }
   436→
   437→                // Use real user_id as maimai_id when available, fall back to hash
   438→                let maimaiId;
   439→                if (userId) {
   440→                    maimaiId = 'mm_' + userId;
   441→                } else {
   442→                    const idSource = (name || '') + '|' + (company || '') + '|' + (title || '');
   443→                    let hash = 0;
   444→                    for (let j = 0; j < idSource.length; j++) {
   445→                        const ch = idSource.charCodeAt(j);
   446→                        hash = ((hash << 5) - hash) + ch;
   447→                        hash = hash & hash; // Convert to 32bit integer
   448→                    }
   449→                    maimaiId = 'mm_' + Math.abs(hash).toString(16).padStart(8, '0');
   450→                }
   451→
   452→                // Skip duplicates based on the computed ID
   453→                if (seenNames.has(maimaiId)) continue;
   454→                seenNames.add(maimaiId);
   455→
   456→                // Build profile URL from user ID when available
   457→                let profileUrl = '';
   458→                if (userId) {
   459→                    profileUrl = 'https://maimai.cn/contact/detail/' + userId + '?dstu=' + userId;
   460→                }
   461→
   462→                // Fallback: extract profile URL from <a> tags inside the card
   463→                if (!profileUrl) {
   464→                    let fallbackUrl = '';
   465→
   466→                    // First: check <a> ancestors between img and card
   467→                    let ancestor = img.parentElement;
   468→                    while (ancestor && ancestor !== card) {
   469→                        if (ancestor.tagName === 'A' && ancestor.href) {
   470→                            const href = ancestor.getAttribute('href') || '';
   471→                            if (href.includes('/profile/detail') || href.includes('/contact/detail/') || href.includes('/u/') || href.includes('egid=') || href.includes('dstu=')) {
   472→                                profileUrl = href.startsWith('http') ? href : 'https://maimai.cn' + href;
   473→                                break;
   474→                            }
   475→                        }
   476→                        ancestor = ancestor.parentElement;
   477→                    }
   478→
   479→                    // Second: search all <a> descendants inside the card
   480→                    if (!profileUrl) {
   481→                        const links = card.querySelectorAll('a[href]');
   482→                        for (const link of links) {
   483→                            const href = link.getAttribute('href') || '';
   484→                            if (href.includes('/profile/detail') || href.includes('/contact/detail/') || href.includes('/u/') || href.includes('egid=') || href.includes('dstu=')) {
   485→                                profileUrl = href.startsWith('http') ? href : 'https://maimai.cn' + href;
   486→                                break;
   487→                            }
   488→                            if (!fallbackUrl && href && !href.startsWith('javascript:') && (href.startsWith('/') || href.includes('maimai.cn'))) {
   489→                                fallbackUrl = href.startsWith('http') ? href : 'https://maimai.cn' + href;
   490→                            }
   491→                        }
   492→                    }
   493→                    if (!profileUrl && fallbackUrl) {
   494→                        profileUrl = fallbackUrl;
   495→                    }
   496→                }
   497→
   498→                // Extract location: look for remaining short text parts after name/company/title
   499→                // Common patterns: city names like "北京", "上海", "深圳" etc.
   500→                let location = '';
   501→                const locationPatterns = /^(北京|上海|广州|深圳|杭州|成都|南京|武汉|西安|苏州|长沙|天津|重庆|郑州|东莞|青岛|大连|宁波|厦门|合肥|佛山|福州|昆明|济南|珠海|无锡|常州|海外|远程|香港|台北|新加坡|美国|日本|韩国|英国|全国|其他)/;
   502→                for (let i = 0; i < textParts.length; i++) {
   503→                    const part = textParts[i];
   504→                    if (part === name || part === company || part === title || part === industry) continue;
   505→                    if (isSkippable(part)) continue;
   506→                    if (locationPatterns.test(part) && part.length <= 10) {
   507→                        location = part;
   508→                        break;
   509→                    }
   510→                }
   511→
   512→                candidates.push({
   513→                    name: name,
   514→                    title: title || '',
   515→                    company: company || '',
   516→                    maimai_id: maimaiId,
   517→                    profile_url: profileUrl,
   518→                    location: location,
   519→                });
   520→            }
   521→
   522→            return candidates;
   523→        }''')
   524→
   525→        logger.info(f"JavaScript extraction found {len(candidates_data)} candidates")
   526→
   527→        for data in candidates_data:
   528→            if data.get('name'):
   529→                candidates.append(CandidateBasic(
   530→                    maimai_id=data.get('maimai_id', ''),
   531→                    name=data.get('name', ''),
   532→                    title=data.get('title', ''),
   533→                    company=data.get('company', ''),
   534→                    location=data.get('location', ''),
   535→                    profile_url=data.get('profile_url', '')
   536→                ))
   537→
   538→        # For candidates missing profile_url, try extracting from contactcard DOM
   539→        missing_url = [c for c in candidates if not c.profile_url]
   540→        if missing_url:
   541→            logger.info(f"{len(missing_url)} candidates missing profile_url, searching contactcard DOM")
   542→            for c in missing_url:
   543→                try:
   544→                    profile_url = await _click_avatar_for_profile_url(page, c.name, c.company)
   545→                    if profile_url:
   546→                        c.profile_url = profile_url
   547→                except Exception as e:
   548→                    logger.debug(f"Contactcard lookup failed for '{c.name}': {e}")
   549→
   550→        # Fallback: Strategy 1 - Find all profile links directly
   551→        if not candidates:
   552→            profile_links = await page.query_selector_all('a[href*="/contact/"], a[href*="/u/"], a[href*="egid="], a[href*="profile"]')
   553→        else:
   554→            profile_links = []
   555→
   556→        if profile_links and not candidates:
   557→            logger.debug(f"Found {len(profile_links)} profile links")
   558→
   559→            seen_ids = set()
   560→            for link in profile_links:
   561→                try:
   562→                    href = await link.get_attribute('href')
   563→                    if not href:
   564→                        continue
   565→
   566→                    # Extract maimai_id from URL
   567→                    maimai_id = None
   568→                    if '/contact/detail/' in href:
   569→                        maimai_id = href.split('/contact/detail/')[-1].split('?')[0].split('/')[0]
   570→                    elif '/u/' in href:
   571→                        maimai_id = href.split('/u/')[-1].split('?')[0].split('/')[0]
   572→
   573→                    if not maimai_id or maimai_id in seen_ids:
   574→                        continue
   575→                    seen_ids.add(maimai_id)
   576→
   577→                    # Get name from link text
   578→                    name = await link.inner_text()
   579→                    name = name.strip()
   580→                    if not name or len(name) > 50:  # Skip if no name or too long (probably not a name)
   581→                        continue
   582→
   583→                    # Try to get parent container for more info
   584→                    parent = await link.evaluate_handle('el => el.closest("div, li, article") || el.parentElement.parentElement')
   585→
   586→                    title = ""
   587→                    company = ""
   588→
   589→                    if parent:
   590→                        # Get all text content and try to parse
   591→                        try:
   592→                            full_text = await parent.inner_text()
   593→                            lines = [l.strip() for l in full_text.split('\n') if l.strip()]
   594→
   595→                            # Usually: name, then title/company info
   596→                            for line in lines:
   597→                                if line == name:
   598→                                    continue
   599→                                if '好友' in line or '关注' in line:
   600→                                    continue
   601→                                if not title and len(line) < 50:
   602→                                    title = line
   603→                                elif not company and len(line) < 50:
   604→                                    company = line
   605→                                    break
   606→                        except Exception:
   607→                            pass
   608→
   609→                    profile_url = href if href.startswith('http') else f"https://maimai.cn{href}"
   610→
   611→                    candidates.append(CandidateBasic(
   612→                        maimai_id=maimai_id,
   613→                        name=name,
   614→                        title=title,
   615→                        company=company,
   616→                        profile_url=profile_url
   617→                    ))
   618→
   619→                except Exception as e:
   620→                    logger.debug(f"Failed to extract from link: {e}")
   621→                    continue
   622→
   623→        # Strategy 2: If no links found, try button-based detection
   624→        if not candidates:
   625→            logger.debug("Trying button-based detection")
   626→            buttons = await page.query_selector_all('button:has-text("好友"), button:has-text("关注")')
   627→
   628→            for button in buttons:
   629→                try:
   630→                    parent = await button.evaluate_handle('el => el.closest("div, li, article")')
   631→                    if not parent:
   632→                        continue
   633→
   634→                    # Find link in parent
   635→                    link = await parent.query_selector('a[href*="/contact"], a[href*="/u/"]')
   636→                    if link:
   637→                        href = await link.get_attribute('href')
   638→                        name = await link.inner_text()
   639→
   640→                        maimai_id = None
   641→                        if '/contact/detail/' in href:
   642→                            maimai_id = href.split('/contact/detail/')[-1].split('?')[0]
   643→                        elif '/u/' in href:
   644→                            maimai_id = href.split('/u/')[-1].split('?')[0]
   645→
   646→                        if maimai_id and name:
   647→                            candidates.append(CandidateBasic(
   648→                                maimai_id=maimai_id,
   649→                                name=name.strip(),
   650→                                title="",
   651→                                company="",
   652→                                profile_url=href
   653→                            ))
   654→                except Exception:
   655→                    continue
   656→
   657→        # Filter out premium placeholder profiles
   658→        before_filter = len(candidates)
   659→        candidates = [c for c in candidates if c.name not in PREMIUM_PLACEHOLDER_NAMES]
   660→        filtered_count = before_filter - len(candidates)
   661→        if filtered_count > 0:
   662→            logger.info(f"Filtered out {filtered_count} premium placeholder profiles")
   663→
   664→        logger.info(f"Extracted {len(candidates)} candidates from page")
   665→
   666→    except Exception as e:
   667→        logger.error(f"Error extracting search results: {e}")
   668→
   669→    return candidates

   672→def _save_candidates_to_db(db_manager, candidates: List[CandidateBasic], source_search: str) -> None:
   673→    """
   674→    Save or update candidates in database.
   675→
   676→    Args:
   677→        db_manager: Database manager instance
   678→        candidates: List of candidates to save
   679→        source_search: Search keywords that found these candidates
   680→    """
   681→    try:
   682→        with db_manager.session_scope() as session:
   683→            # Batch pre-fetch existing candidates to avoid N+1 queries
   684→            maimai_ids = [c.maimai_id for c in candidates]
   685→            existing_by_id = {
   686→                row.maimai_id: row
   687→                for row in session.query(Candidate).filter(
   688→                    Candidate.maimai_id.in_(maimai_ids)
   689→                ).all()
   690→            }
   691→
   692→            # Also batch-fetch by name+company for dedup
   693→            name_company_pairs = [(c.name, c.company) for c in candidates]
   694→            from sqlalchemy import tuple_
   695→            existing_by_name = {}
   696→            if name_company_pairs:
   697→                for row in session.query(Candidate).filter(
   698→                    tuple_(Candidate.name, Candidate.company).in_(name_company_pairs)
   699→                ).all():
   700→                    existing_by_name[(row.name, row.company)] = row
   701→
   702→            for c in candidates:
   703→                existing = existing_by_id.get(c.maimai_id)
   704→                if not existing:
   705→                    existing = existing_by_name.get((c.name, c.company))
   706→
   707→                if existing:
   708→                    existing.name = c.name
   709→                    existing.title = c.title
   710→                    existing.company = c.company
   711→                    existing.location = c.location
   712→                    existing.profile_url = c.profile_url
   713→                else:
   714→                    candidate = Candidate(
   715→                        maimai_id=c.maimai_id,
   716→                        name=c.name,
   717→                        title=c.title,
   718→                        company=c.company,
   719→                        location=c.location,
   720→                        profile_url=c.profile_url,
   721→                        source_search=source_search
   722→                    )
   723→                    session.add(candidate)
   724→
   725→            logger.debug(f"Saved {len(candidates)} candidates to database")
   726→
   727→    except Exception as e:
   728→        logger.error(f"Failed to save candidates: {e}")

   756→async def _click_avatar_for_profile_url(page: Page, name: str, company: str) -> Optional[str]:
   757→    """Try to extract profile URL for a candidate from the DOM contactcard pattern.
   758→
   759→    Searches the current page for a contactcard element whose parent card
   760→    contains the candidate's name (and optionally company), then constructs
   761→    the profile URL from the extracted user ID.
   762→
   763→    Args:
   764→        page: Playwright page on the search results
   765→        name: Candidate name to match
   766→        company: Candidate company to help disambiguate
   767→
   768→    Returns:
   769→        Profile URL string, or None if not found
   770→    """
   771→    try:
   772→        user_id = await page.evaluate('''(args) => {
   773→            const { name, company } = args;
   774→            // Check contactcard_{numeric_id} ancestors
   775→            const ccElements = document.querySelectorAll('[id^="contactcard_"]');
   776→            for (const el of ccElements) {
   777→                const text = el.textContent || '';
   778→                if (!text.includes(name)) continue;
   779→                if (company && !text.includes(company)) continue;
   780→                const match = el.id.match(/^contactcard_(\\d+)$/);
   781→                if (match) return match[1];
   782→            }
   783→            // Check member_entry_{numeric_id} descendants
   784→            const meElements = document.querySelectorAll('[id^="member_entry_"]');
   785→            for (const el of meElements) {
   786→                // Walk up to find the card containing this member_entry
   787→                let card = el.parentElement;
   788→                for (let i = 0; i < 8 && card; i++) {
   789→                    const text = card.textContent || '';
   790→                    if (text.includes(name) && (!company || text.includes(company))) {
   791→                        const match = el.id.match(/^member_entry_(\\d+)$/);
   792→                        if (match) return match[1];
   793→                    }
   794→                    card = card.parentElement;
   795→                }
   796→            }
   797→            return null;
   798→        }''', {'name': name, 'company': company})
   799→
   800→        if user_id:
   801→            profile_url = f"https://maimai.cn/contact/detail/{user_id}?dstu={user_id}"
   802→            logger.info(f"Resolved profile_url for '{name}' via contactcard: {profile_url}")
   803→            return profile_url
   804→
   805→        logger.debug(f"No contactcard found for '{name}'")
   806→        return None
   807→
   808→    except Exception as e:
   809→        logger.debug(f"_click_avatar_for_profile_url error for '{name}': {e}")
   810→        return None

  1006→async def _extract_enterprise_results(page: Page) -> List[CandidateBasic]:
  1007→    """Extract candidate information from enterprise recruitment results page.
  1008→
  1009→    The enterprise UI has a different card layout from search_center.
  1010→    Uses JS-based extraction tailored to the enterprise talent cards.
  1011→
  1012→    Args:
  1013→        page: Playwright page instance on the enterprise results page
  1014→
  1015→    Returns:
  1016→        List of CandidateBasic objects
  1017→    """
  1018→    candidates = []
  1019→
  1020→    try:
  1021→        await page.wait_for_load_state("networkidle", timeout=10000)
  1022→    except Exception:
  1023→        pass
  1024→
  1025→    try:
  1026→        candidates_data = await page.evaluate('''() => {
  1027→            const candidates = [];
  1028→            const seenIds = new Set();
  1029→
  1030→            // Enterprise talent cards typically have a structured layout
  1031→            // Try multiple selectors for the card containers
  1032→            const cardSelectors = [
  1033→                '[class*="talent-card"]',
  1034→                '[class*="candidate-card"]',
  1035→                '[class*="recommend-card"]',
  1036→                '[class*="talent-item"]',
  1037→                '[class*="card-wrap"]',
  1038→            ];
  1039→
  1040→            let cards = [];
  1041→            for (const sel of cardSelectors) {
  1042→                cards = document.querySelectorAll(sel);
  1043→                if (cards.length > 0) break;
  1044→            }
  1045→
  1046→            // Fallback: find cards by avatar image pattern (same as search_center)
  1047→            if (cards.length === 0) {
  1048→                const images = document.querySelectorAll('img');
  1049→                const foundCards = [];
  1050→
  1051→                for (const img of images) {
  1052→                    const rect = img.getBoundingClientRect();
  1053→                    if (rect.width > 100 || rect.height > 100) continue;
  1054→                    if (rect.width < 30 || rect.height < 30) continue;
  1055→
  1056→                    let card = img.parentElement;
  1057→                    for (let i = 0; i < 8 && card; i++) {
  1058→                        const cardRect = card.getBoundingClientRect();
  1059→                        if (cardRect.width > 200 && cardRect.height > 80) {
  1060→                            break;
  1061→                        }
  1062→                        card = card.parentElement;
  1063→                    }
  1064→                    if (card) foundCards.push(card);
  1065→                }
  1066→                cards = foundCards;
  1067→            }
  1068→
  1069→            for (const card of cards) {
  1070→                const textParts = [];
  1071→                const walker = document.createTreeWalker(card, NodeFilter.SHOW_TEXT, null, false);
  1072→                while (walker.nextNode()) {
  1073→                    const text = walker.currentNode.textContent.trim();
  1074→                    if (text && text.length > 0 && text.length < 100) {
  1075→                        textParts.push(text);
  1076→                    }
  1077→                }
  1078→
  1079→                let name = '';
  1080→                let company = '';
  1081→                let title = '';
  1082→
  1083→                const skipPatterns = [
  1084→                    /^立即沟通$/,
  1085→                    /^不合适$/,
  1086→                    /^感兴趣$/,
  1087→                    /^不感兴趣$/,
  1088→                    /^(\\d+\\s*)?年/,
  1089→                    /^[|｜·]$/,
  1090→                    /^在线|^离线|^活跃/,
  1091→                ];
  1092→
  1093→                const isSkippable = (text) => {
  1094→                    return skipPatterns.some(p => p.test(text));
  1095→                };
  1096→
  1097→                for (let i = 0; i < textParts.length; i++) {
  1098→                    const part = textParts[i];
  1099→                    if (isSkippable(part)) continue;
  1100→
  1101→                    if (!name && part.length >= 2 && part.length <= 15) {
  1102→                        if (/^[\\u4e00-\\u9fa5a-zA-Z·\\s]+$/.test(part)) {
  1103→                            name = part;
  1104→                            continue;
  1105→                        }
  1106→                    }
  1107→
  1108→                    if (name && !company && part.length > 1) {
  1109→                        company = part;
  1110→                        continue;
  1111→                    }
  1112→
  1113→                    if (name && company && !title && part.length > 1) {
  1114→                        title = part;
  1115→                        break;
  1116→                    }
  1117→                }
