     1→"""Random delay generators for human-like behavior simulation.
     2→
     3→Uses log-normal distribution instead of uniform for more realistic timing.
     4→Includes fatigue simulation (delays increase over session duration) and
     5→random "distraction" pauses to mimic real human behavior.
     6→"""
     7→
     8→import datetime as dt_module
     9→import math
    10→import random
    11→import asyncio
    12→import time
    13→from loguru import logger
    14→from src.utils.config import get_config
    15→
    16→
    17→# Session start time for fatigue calculation
    18→_session_start: float = time.monotonic()
    19→
    20→# Action counter for distraction triggers
    21→_action_count: int = 0
    22→
    23→
    24→def reset_session():
    25→    """Reset fatigue/distraction tracking for a new command invocation."""
    26→    global _session_start, _action_count
    27→    _session_start = time.monotonic()
    28→    _action_count = 0
    29→
    30→
    31→def _lognormal_delay(min_val: float, max_val: float) -> float:
    32→    """Generate a log-normal distributed delay between min and max.
    33→
    34→    Log-normal produces a right-skewed distribution: most delays are
    35→    near the low end, with occasional longer pauses — matching human behavior.
    36→    """
    37→    # Map min/max to log-normal mu/sigma
    38→    mid = (min_val + max_val) / 2
    39→    mu = math.log(mid)
    40→    sigma = 0.4  # Controls spread; 0.4 gives moderate variance
    41→    delay = random.lognormvariate(mu, sigma)
    42→    return max(min_val, min(delay, max_val * 1.5))
    43→
    44→
    45→def _fatigue_multiplier() -> float:
    46→    """Calculate fatigue multiplier based on session duration.
    47→
    48→    Humans slow down over time. After 30 minutes, delays increase by ~20%.
    49→    After 1 hour, by ~40%. Caps at 1.6x.
    50→    """
    51→    elapsed_minutes = (time.monotonic() - _session_start) / 60.0
    52→    if elapsed_minutes < 10:
    53→        return 1.0
    54→    # Gradual ramp: 1.0 at 10min → ~1.6 at 90min
    55→    factor = 1.0 + 0.006 * min(elapsed_minutes - 10, 100)
    56→    return min(factor, 1.6)
    57→
    58→
    59→def _maybe_distraction_pause() -> float:
    60→    """Occasionally return a long "distraction" pause.
    61→
    62→    About 3% chance of a 5-30 second pause (simulating checking phone,
    63→    reading something, etc.). Returns 0 most of the time.
    64→    """
    65→    global _action_count
    66→    _action_count += 1
    67→
    68→    # Don't trigger on first few actions
    69→    if _action_count < 5:
    70→        return 0.0
    71→
    72→    if random.random() < 0.03:
    73→        pause = random.uniform(5.0, 30.0)
    74→        logger.debug(f"Simulating distraction pause: {pause:.1f}s")
    75→        return pause
    76→    return 0.0
    77→
    78→
    79→def _time_of_day_multiplier() -> float:
    80→    """Adjust delays based on time of day.
    81→
    82→    Late night (23:00-07:00): slower (1.3x) — fewer people browse late.
    83→    Peak hours (10:00-12:00, 14:00-17:00): normal (1.0x).
    84→    Other times: slightly slower (1.1x).
    85→    """
    86→    hour = dt_module.datetime.now().hour
    87→
    88→    if 23 <= hour or hour < 7:
    89→        return 1.3
    90→    elif (10 <= hour <= 12) or (14 <= hour <= 17):
    91→        return 1.0
    92→    else:
    93→        return 1.1
    94→
    95→
    96→def get_random_delay(delay_type: str = 'action') -> float:
    97→    """
    98→    Get a random delay based on configured ranges.
    99→
   100→    Uses log-normal distribution with fatigue, distraction, and
   101→    time-of-day adjustments for realistic human behavior.
   102→
   103→    Args:
   104→        delay_type: Type of delay - 'page_load', 'action', 'typing', 'scroll', 'message'
   105→
   106→    Returns:
   107→        Random delay in seconds
   108→    """
   109→    config = get_config()
   110→
   111→    delay_ranges = {
   112→        'page_load': (
   113→            config.get('delays.page_load_min', 2),
   114→            config.get('delays.page_load_max', 5)
   115→        ),
   116→        'action': (
   117→            config.get('delays.action_min', 0.5),
   118→            config.get('delays.action_max', 2.0)
   119→        ),
   120→        'typing': (
   121→            config.get('delays.typing_min', 0.05),
   122→            config.get('delays.typing_max', 0.15)
   123→        ),
   124→        'scroll': (
   125→            config.get('delays.scroll_min', 0.5),
   126→            config.get('delays.scroll_max', 1.5)
   127→        ),
   128→        'message': (
   129→            config.get('delays.action_min', 0.5),
   130→            config.get('delays.action_max', 2.0)
   131→        ),
   132→    }
   133→
   134→    min_delay, max_delay = delay_ranges.get(delay_type, (0.5, 2.0))
   135→
   136→    # Use log-normal instead of uniform
   137→    base_delay = _lognormal_delay(min_delay, max_delay)
   138→
   139→    # Apply fatigue (typing is exempt — fatigue doesn't affect keystroke speed much)
   140→    if delay_type != 'typing':
   141→        base_delay *= _fatigue_multiplier()
   142→        base_delay *= _time_of_day_multiplier()
   143→
   144→    # Add possible distraction pause for non-typing delays
   145→    if delay_type in ('action', 'page_load', 'message'):
   146→        base_delay += _maybe_distraction_pause()
   147→
   148→    return base_delay
   149→
   150→
   151→async def async_random_delay(delay_type: str = 'action'):
   152→    """
   153→    Asynchronous version of get_random_delay.
   154→
   155→    Args:
   156→        delay_type: Type of delay - 'page_load', 'action', 'typing', 'scroll', 'message'
   157→    """
   158→    delay = get_random_delay(delay_type)
   159→    await asyncio.sleep(delay)
   160→
   161→
   162→
   163→def get_retry_delay(attempt: int, base_delay: float = 1.0, max_delay: float = 60.0) -> float:
   164→    """
   165→    Calculate exponential backoff delay for retries.
   166→
   167→    Args:
   168→        attempt: Current retry attempt number (0-indexed)
   169→        base_delay: Base delay in seconds
   170→        max_delay: Maximum delay cap
   171→
   172→    Returns:
   173→        Delay in seconds with jitter
   174→    """
   175→    # Exponential backoff: base * 2^attempt
   176→    delay = min(base_delay * (2 ** attempt), max_delay)
   177→
   178→    # Add jitter (±25%)
   179→    jitter = delay * random.uniform(-0.25, 0.25)
   180→    return max(0, delay + jitter)
   181→
