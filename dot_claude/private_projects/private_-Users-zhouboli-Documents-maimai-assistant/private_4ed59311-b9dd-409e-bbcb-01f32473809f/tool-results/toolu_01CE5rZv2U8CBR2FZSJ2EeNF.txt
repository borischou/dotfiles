    97→@cli.command()
    98→@click.option('--keywords', '-k', required=True, multiple=True, help='Search keywords (can specify multiple times)')
    99→@click.option('--company', '-c', multiple=True, help='Filter by company name (can specify multiple)')
   100→@click.option('--location', '-l', help='Filter by location')
   101→@click.option('--max-pages', '-p', type=int, default=1, help='Maximum pages to scrape per keyword')
   102→@click.option('--mode', '-m', type=click.Choice(['search_center', 'enterprise']), default='search_center',
   103→              help='Search mode: search_center (default) or enterprise')
   104→@click.option('--export', '-e', help='Export results to file (Excel/CSV)')
   105→@click.pass_context
   106→def search(ctx, keywords, company, location, max_pages, mode, export):
   107→    """
   108→    Search for candidates with specified criteria.
   109→
   110→    Supports multiple keywords - each will be searched separately and results merged.
   111→
   112→    Use --mode enterprise to search via the enterprise recruitment tab.
   113→
   114→    Examples:\n
   115→        maimai search -k "高级前端工程师" -c "阿里巴巴" --max-pages 3\n
   116→        maimai search -k "前端工程师" -k "React开发" -k "Vue工程师" -p 2\n
   117→        maimai search -k "研发" -m enterprise
   118→    """
   119→    from src.modules.export import export_candidates
   120→    from src.core.browser import get_browser_manager
   121→    from src.services.pipeline import search_all_keywords, search_all_keywords_enterprise
   122→
   123→    _apply_cdp_endpoint(ctx)
   124→
   125→    async def _search():
   126→        reset_delay_session()
   127→        browser = get_browser_manager()
   128→        try:
   129→            if mode == 'enterprise':
   130→                all_candidates, seen_ids = await search_all_keywords_enterprise(
   131→                    browser, keywords, company, max_pages
   132→                )
   133→            else:
   134→                all_candidates, seen_ids = await search_all_keywords(
   135→                    browser, keywords, company, location, max_pages
   136→                )
   137→
   138→            # Display combined results
   139→            if all_candidates:
   140→                table = Table(title=f"Search Results ({len(all_candidates)} candidates total)")
   141→                table.add_column("Name", style="cyan")
   142→                table.add_column("Title", style="green")
   143→                table.add_column("Company", style="yellow")
   144→                table.add_column("Location", style="blue")
   145→                table.add_column("ID", style="dim")
   146→
   147→                for c in all_candidates[:30]:  # Show first 30
   148→                    table.add_row(c.name, c.title, c.company, c.location or '', c.maimai_id)
   149→
   150→                console.print(table)
   151→
   152→                if len(all_candidates) > 30:
   153→                    console.print(f"[dim]... and {len(all_candidates) - 30} more[/dim]")
   154→            else:
   155→                console.print("[yellow]No candidates found for any keyword.[/yellow]")
   156→
   157→            # Export if requested
   158→            if export and all_candidates:
   159→                fmt = 'csv' if export.endswith('.csv') else 'excel'
   160→                export_candidates(format=fmt, output_path=export)
   161→
   162→        finally:
   163→            await browser.close()
   164→
   165→    run_async(_search())

   193→@cli.command()
   194→@click.argument('maimai_id')
   195→@click.option('--template', '-t', required=True, help='Message template name')
   196→@click.option('--preview', is_flag=True, help='Preview only, do not send')
   197→@click.option('--var', '-v', multiple=True, help='Template variable (format: key=value)')
   198→@click.option('--interactive', '-i', is_flag=True, help='Enable interactive selector discovery when auto-detection fails')
   199→@click.pass_context
   200→def message(ctx, maimai_id, template, preview, var, interactive):
   201→    """
   202→    Send message to a candidate (requires confirmation).
   203→
   204→    Example:
   205→        maimai message abc123 -t greeting -v name=张三 -v recruiter_name=李四
   206→        maimai message abc123 -t follow_up --preview
   207→        maimai message abc123 -t greeting -i
   208→    """
   209→    from src.modules.message import send_message, preview_message
   210→    from src.core.browser import get_browser_manager
   211→
   212→    # Parse variables
   213→    variables = _parse_variables(var)
   214→
   215→    if preview:
   216→        # Just preview, no send
   217→        preview_message(template, variables)
   218→        return
   219→

   221→
   222→    async def _message():
   223→        browser = get_browser_manager()
   224→        try:
   225→            discovered_selectors = None
   226→            if interactive:
   227→                from src.modules.selector_discovery import DiscoveredSelectors
   228→                discovered_selectors = DiscoveredSelectors()
   229→
   230→            success = await send_message(
   231→                maimai_id=maimai_id,
   232→                template_name=template,
   233→                variables=variables,
   234→                browser=browser,
   235→                interactive=interactive,
   236→                discovered_selectors=discovered_selectors,
   237→            )
   238→            if not success:
   239→                sys.exit(1)
   240→        finally:
   241→            await browser.close()
   242→

   246→@cli.command()
   247→@click.option('--keywords', '-k', multiple=True, help='Search keywords (can specify multiple)')
   248→@click.option('--template', '-t', help='Message template name')
   249→@click.option('--var', '-v', multiple=True, help='Template variable (format: key=value)')
   250→@click.option('--company', '-c', multiple=True, help='Filter by company name (can specify multiple)')
   251→@click.option('--location', '-l', help='Filter by location')
   252→@click.option('--max-pages', '-p', type=int, default=1, help='Max pages to scrape per keyword')
   253→@click.option('--mode', '-m', type=click.Choice(['search_center', 'enterprise']), default='search_center',
   254→              help='Search mode: search_center (default) or enterprise')
   255→@click.option('--dry-run', is_flag=True, help='Search only, do not send messages')
   256→@click.option('--skip-messaged/--no-skip-messaged', default=True, help='Skip already messaged candidates')
   257→@click.option('--batch-pages', '-b', type=int, default=0, help='Send messages every N pages (0 = send after all pages)')
   258→@click.option('--interactive', '-i', is_flag=True, help='Enable interactive selector discovery when auto-detection fails')
   259→@click.option('--resume', is_flag=True, help='Resume from last checkpoint')
   260→@click.pass_context
   261→def pipeline(ctx, keywords, template, var, company, location, max_pages, mode, dry_run, skip_messaged, batch_pages, interactive, resume):
   262→    """
   263→    Search candidates and batch send messages in one command.
   264→
   265→    Combines multi-keyword search with batch messaging:
   266→    search → preview → confirm → send.
   267→
   268→    Use --mode enterprise for the enterprise recruitment path with inline messaging.
   269→
   270→    Use --batch-pages to interleave search and messaging every N pages.
   271→    Example: -p 10 --batch-pages 2 searches pages 1-2, sends messages,
   272→    then pages 3-4, sends messages, etc.
   273→
   274→    Use --resume to continue from the last interrupted pipeline run.
   275→
   276→    Examples:\n
   277→        maimai pipeline -k "前端工程师" -t greeting -v "recruiter_name=张三" -p 2\n
   278→        maimai pipeline -k "前端" -k "后端" -t greeting -v "recruiter_name=张三" --dry-run\n
   279→        maimai pipeline -k "前端" -t greeting -p 10 --batch-pages 2\n
   280→        maimai pipeline -k "前端" -t greeting -p 1 -i\n
   281→        maimai pipeline -k "研发" -t greeting -m enterprise -p 3\n
   282→        maimai pipeline --resume
   283→    """
   284→    from src.core.browser import get_browser_manager
   285→    from src.utils.checkpoint import CheckpointManager
   286→    from src.services.pipeline import run_pipeline_default, run_pipeline_batched, SearchMode
   287→
   288→    # Handle resume mode
   289→    if resume:
   290→        ckpt = CheckpointManager.load_latest()
   291→        if ckpt is None:
   292→            console.print("[red]No checkpoint found to resume from.[/red]")
   293→            return
   294→
   295→        from rich.panel import Panel
   296→        console.print(Panel(ckpt.summary, title="Resuming Pipeline", border_style="cyan"))
   297→
   298→        p = ckpt.params
   299→        keywords = tuple(p.get('keywords', []))
   300→        template = p.get('template', template)
   301→        variables = p.get('variables', {})
   302→        company = tuple(p.get('company', []))
   303→        location = p.get('location', location)
   304→        max_pages = p.get('max_pages', max_pages)
   305→        batch_pages = p.get('batch_pages', batch_pages)
   306→        skip_messaged = p.get('skip_messaged', skip_messaged)
   307→        interactive = p.get('interactive', interactive)
   308→        mode = p.get('mode', mode)
   309→        dry_run = p.get('dry_run', dry_run)
   310→    else:
   311→        ckpt = None
   312→        variables = _parse_variables(var)
   313→
   314→    if not keywords or not template:
   315→        console.print("[red]--keywords (-k) and --template (-t) are required (or use --resume).[/red]")
   316→        return
   317→
   318→    # Resolve SearchMode enum
   319→    search_mode = SearchMode(mode)
   320→
   321→    if search_mode == SearchMode.ENTERPRISE and batch_pages > 0:
   322→        console.print("[yellow]Note: --batch-pages with enterprise mode uses infinite scroll; "
   323→                      "each batch opens a fresh enterprise search. Deduplication prevents re-messaging.[/yellow]")
   324→
   325→    _apply_cdp_endpoint(ctx)
   326→
   327→    async def _pipeline():
   328→        from src.modules.auth import ensure_logged_in
   329→
   330→        reset_delay_session()
   331→        browser = get_browser_manager()
   332→        try:
   333→            await ensure_logged_in(browser)
   334→
   335→            if batch_pages > 0:
   336→                await run_pipeline_batched(
   337→                    browser, keywords, company, location,
   338→                    max_pages, batch_pages, template, variables,
   339→                    dry_run, skip_messaged, interactive, ckpt,
   340→                    mode=search_mode,
   341→                )
   342→            else:
   343→                await run_pipeline_default(
   344→                    browser, keywords, company, location,
   345→                    max_pages, template, variables,
   346→                    dry_run, skip_messaged, interactive,
   347→                    mode=search_mode,
   348→                )
   349→        finally:
   350→            await browser.close()
   351→
   352→    run_async(_pipeline())

   407→@cli.command()
   408→def status():
   409→    """Show current rate limit status and quota."""
   410→    try:
   411→        rate_limiter = get_rate_limiter()
   412→
   413→        table = Table(title="Rate Limit Status")
   414→        table.add_column("Action", style="cyan")
   415→        table.add_column("Hourly Used", justify="right")
   416→        table.add_column("Hourly Limit", justify="right")
   417→        table.add_column("Daily Used", justify="right")
   418→        table.add_column("Daily Limit", justify="right")
   419→        table.add_column("Remaining", justify="right", style="green")
   420→
   421→        action_types = ['search', 'view_profile', 'send_message']
   422→        action_names = {
   423→            'search': 'Search',
   424→            'view_profile': 'Profile View',
   425→            'send_message': 'Send Message'
   426→        }
   427→
   428→        for action_type in action_types:
   429→            quota = rate_limiter.get_quota_status(action_type)
   430→            hourly = quota['hourly']
   431→            daily = quota['daily']
   432→
   433→            # Use minimum remaining
   434→            remaining = min(hourly['remaining'], daily['remaining'])
   435→
   436→            table.add_row(
   437→                action_names[action_type],
   438→                str(hourly['used']),
   439→                str(hourly['limit']),
   440→                str(daily['used']),
   441→                str(daily['limit']),
   442→                str(remaining)
   443→            )
   444→
   445→        console.print(table)
   446→
   447→        # Show adaptive throttle status
   448→        throttle = rate_limiter.throttle_factor
   449→        if throttle < 1.0:
   450→            console.print(f"\n[yellow]Adaptive throttle active: {throttle:.0%} capacity (captcha/error detected recently)[/yellow]")
   451→
   452→        # Show session info
   453→        from src.core.session import get_session_manager
   454→        session_manager = get_session_manager()
   455→        session_info = session_manager.get_session_info()
   456→
   457→        if session_info:
   458→            console.print(f"\n[dim]Session age: {session_info['age']} | Cookies: {session_info['cookie_count']}[/dim]")
   459→        else:
   460→            console.print("\n[yellow]No saved session[/yellow]")
   461→
   462→    except Exception as e:
   463→        console.print(f"[red]Error getting status: {e}[/red]")

   479→def _parse_variables(var: tuple) -> dict:
   480→    """Parse key=value variable pairs from CLI option."""
   481→    variables = {}
   482→    for v in var:
   483→        if '=' in v:
   484→            key, value = v.split('=', 1)
   485→            variables[key.strip()] = value.strip()
   486→        else:
   487→            console.print(f"[yellow]Warning: ignoring malformed variable '{v}' (expected key=value format)[/yellow]")
   488→    return variables
