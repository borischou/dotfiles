     1→"""Browser manager for Playwright automation.
     2→
     3→Provides a singleton browser manager that connects to an existing Chrome
     4→instance via CDP (Chrome DevTools Protocol). Supports multi-tab management
     5→for workflows that open new tabs (e.g., enterprise recruitment).
     6→"""
     7→
     8→import asyncio
     9→from pathlib import Path
    10→from typing import List, Optional
    11→from datetime import datetime
    12→
    13→from patchright.async_api import async_playwright, Browser, BrowserContext, Page, Playwright
    14→from loguru import logger
    15→
    16→from src.utils.config import get_config, Config
    17→from src.core.session import SessionManager, get_session_manager
    18→from src.exceptions.errors import BrowserError
    19→
    20→
    21→class BrowserManager:
    22→    """
    23→    Singleton browser manager for Maimai automation.
    24→
    25→    Connects to a user-managed Chrome instance via CDP endpoint.
    26→    The user starts Chrome with --remote-debugging-port=9222 and logs in
    27→    manually. This manager connects/disconnects without closing Chrome.
    28→    """
    29→
    30→    _instance: Optional['BrowserManager'] = None
    31→    _lock: Optional[asyncio.Lock] = None
    32→    _lock_loop: Optional[asyncio.AbstractEventLoop] = None  # track which loop owns _lock
    33→
    34→    def __new__(cls, **kwargs):
    35→        if cls._instance is None:
    36→            cls._instance = super().__new__(cls)
    37→            cls._instance._initialized = False
    38→        return cls._instance
    39→
    40→    def __init__(
    41→        self,
    42→        config: Optional[Config] = None,
    43→        session_manager: Optional[SessionManager] = None,
    44→    ):
    45→        if self._initialized:
    46→            return
    47→
    48→        self._playwright: Optional[Playwright] = None
    49→        self._browser: Optional[Browser] = None
    50→        self._context: Optional[BrowserContext] = None
    51→        self._page: Optional[Page] = None
    52→        self._primary_page: Optional[Page] = None  # Original tab to return to
    53→        self._opened_tabs: List[Page] = []  # Tabs opened during our session
    54→        self._session_manager: SessionManager = session_manager or get_session_manager()
    55→        self._config: Config = config or get_config()
    56→        self._cdp_endpoint: Optional[str] = None
    57→        self._owned_page: bool = False  # True if we created the page (vs reusing existing)
    58→        self._initialized = True
    59→
    60→    @property
    61→    def is_running(self) -> bool:
    62→        """Check if browser is currently connected."""
    63→        return self._browser is not None and self._browser.is_connected()
    64→
    65→    @property
    66→    def page(self) -> Optional[Page]:
    67→        """Get current page instance."""
    68→        return self._page
    69→
    70→    @property
    71→    def context(self) -> Optional[BrowserContext]:
    72→        """Get current browser context."""
    73→        return self._context
    74→
    75→    @classmethod
    76→    def _ensure_lock(cls):
    77→        """Lazily create the asyncio.Lock inside a running event loop.
    78→
    79→        Recreates the lock if the event loop has changed (e.g. between
    80→        separate asyncio.run() calls) to avoid 'attached to a different
    81→        event loop' RuntimeError.
    82→        """
    83→        current_loop = asyncio.get_running_loop()
    84→        if cls._lock is None or cls._lock_loop is not current_loop:
    85→            cls._lock = asyncio.Lock()
    86→            cls._lock_loop = current_loop
    87→
    88→    def set_cdp_endpoint(self, endpoint: str) -> None:
    89→        """Override CDP endpoint (e.g. from CLI --cdp flag)."""
    90→        self._cdp_endpoint = endpoint
    91→
    92→    async def start(self) -> Page:
    93→        """
    94→        Connect to Chrome via CDP and return a page.
    95→
    96→        Returns:
    97→            Configured Playwright page instance
    98→
    99→        Raises:
   100→            BrowserError: If connection fails
   101→        """
   102→        self._ensure_lock()
   103→        async with self._lock:
   104→            if self.is_running:
   105→                logger.debug("Browser already connected, returning existing page")
   106→                return self._page
   107→
   108→            try:
   109→                # Determine CDP endpoint
   110→                endpoint = self._cdp_endpoint or self._config.get(
   111→                    'browser.cdp_endpoint', 'http://localhost:9222'
   112→                )
   113→                timeout = self._config.get('browser.timeout', 30000)
   114→
   115→                logger.info(f"Connecting to Chrome via CDP: {endpoint}")
   116→
   117→                # Start Playwright
   118→                self._playwright = await async_playwright().start()
   119→
   120→                # Connect to existing Chrome via CDP
   121→                self._browser = await self._playwright.chromium.connect_over_cdp(endpoint)
   122→
   123→                # Get the default context (the user's browser context with cookies)
   124→                contexts = self._browser.contexts
   125→                if contexts:
   126→                    self._context = contexts[0]
   127→                else:
   128→                    # Fallback: create a new context (shouldn't happen with CDP)
   129→                    logger.warning("No existing context found, creating new one")
   130→                    self._context = await self._browser.new_context()
   131→
   132→                # Set default timeout
   133→                self._context.set_default_timeout(timeout)
   134→
   135→                # Reuse existing page or create one
   136→                pages = self._context.pages
   137→                if pages:
   138→                    self._page = pages[0]
   139→                    self._owned_page = False
   140→                    logger.info(f"Reusing existing tab: {self._page.url}")
   141→                else:
   142→                    self._page = await self._context.new_page()
   143→                    self._owned_page = True
   144→                    logger.info("Created new tab")
   145→
   146→                # Remember the primary tab
   147→                self._primary_page = self._page
   148→                self._opened_tabs = []
   149→
   150→                logger.info("Connected to Chrome via CDP successfully")
   151→                return self._page
   152→
   153→            except Exception as e:
   154→                logger.error(f"Failed to connect to Chrome via CDP: {e}")
   155→                await self._close_internal()
   156→                raise BrowserError(
   157→                    f"Failed to connect to Chrome via CDP at {endpoint}. "
   158→                    f"Make sure Chrome is running with --remote-debugging-port=9222. "
   159→                    f"Error: {e}"
   160→                )
   161→
   162→    async def close(self) -> None:
   163→        """Disconnect from Chrome via CDP. Does NOT close the user's Chrome."""
   164→        self._ensure_lock()
   165→        async with self._lock:
   166→            await self._close_internal()
   167→
   168→    async def _close_internal(self) -> None:
   169→        """Actual cleanup logic — callable without acquiring _lock.
   170→
   171→        Must only be called while the lock is already held, or from
   172→        start()'s except block (which already holds the lock).
   173→        """
   174→        try:
   175→            # Close tabs we opened during the session
   176→            for tab in self._opened_tabs:
   177→                try:
   178→                    if not tab.is_closed():
   179→                        await tab.close()
   180→                except Exception:
   181→                    pass
   182→            self._opened_tabs = []
   183→
   184→            # Only close pages we created, not the user's existing tabs
   185→            if self._page and self._owned_page:
   186→                try:
   187→                    await self._page.close()
   188→                except Exception:
   189→                    pass
   190→            self._page = None
   191→            self._primary_page = None
   192→            self._owned_page = False
   193→
   194→            # Don't close the context — it belongs to the user's Chrome
   195→            self._context = None
   196→
   197→            # browser.close() in CDP mode just disconnects, doesn't kill Chrome
   198→            if self._browser:
   199→                try:
   200→                    await self._browser.close()
   201→                except Exception as e:
   202→                    logger.error(f"Error closing browser connection: {e}")
   203→                finally:
   204→                    self._browser = None
   205→
   206→            if self._playwright:
   207→                try:
   208→                    await self._playwright.stop()
   209→                except Exception as e:
   210→                    logger.error(f"Error stopping Playwright: {e}")
   211→                finally:
   212→                    self._playwright = None
   213→
   214→            logger.info("Disconnected from Chrome (browser stays open)")
   215→
   216→        except Exception as e:
   217→            logger.error(f"Error disconnecting from Chrome: {e}")
   218→
   219→    async def wait_for_new_tab(self, timeout: int = 10000) -> Page:
   220→        """Wait for a click-triggered new tab, apply stealth, and switch to it.
   221→
   222→        Call this right after performing an action that opens a new tab
   223→        (e.g., clicking a link with target=_blank).
   224→
   225→        Args:
   226→            timeout: Max wait time in milliseconds
   227→
   228→        Returns:
   229→            The new Page instance (also set as the active page)
   230→
   231→        Raises:
   232→            BrowserError: If no new tab appears within timeout
   233→        """
   234→        if not self._context:
   235→            raise BrowserError("No active context for wait_for_new_tab")
   236→
   237→        try:
   238→            new_page = await self._context.wait_for_event(
   239→                "page", timeout=timeout
   240→            )
   241→            await new_page.wait_for_load_state("domcontentloaded")
   242→            self._opened_tabs.append(new_page)
   243→            self._page = new_page
   244→            logger.info(f"New tab captured: {new_page.url}")
   245→            return new_page
   246→        except Exception as e:
   247→            raise BrowserError(f"No new tab opened within {timeout}ms: {e}")
   248→
   249→    def switch_to_primary_tab(self) -> Page:
   250→        """Switch back to the original (primary) tab.
   251→
   252→        Returns:
   253→            The primary Page instance
   254→        """
   255→        if not self._primary_page or self._primary_page.is_closed():
   256→            raise BrowserError("Primary tab is not available")
   257→
   258→        self._page = self._primary_page
   259→        logger.debug(f"Switched to primary tab: {self._page.url}")
   260→        return self._page
   261→
   262→    async def close_current_tab(self) -> Page:
   263→        """Close the current active tab and switch to the previous one.
   264→
   265→        Will not close the primary tab. After closing, switches to the
   266→        most recent remaining tab or falls back to the primary tab.
   267→
   268→        Returns:
   269→            The page that became active after closing
   270→
   271→        Raises:
   272→            BrowserError: If trying to close the primary tab
   273→        """
   274→        if not self._page:
   275→            raise BrowserError("No active page to close")
   276→
   277→        if self._page == self._primary_page:
   278→            raise BrowserError("Cannot close the primary tab")
   279→
   280→        closing = self._page
   281→        try:
   282→            # Remove from opened_tabs tracking
   283→            if closing in self._opened_tabs:
   284→                self._opened_tabs.remove(closing)
   285→            await closing.close()
   286→        except Exception as e:
   287→            logger.warning(f"Error closing tab: {e}")
   288→
   289→        # Switch to previous tab
   290→        if self._context and self._context.pages:
   291→            self._page = self._context.pages[-1]
   292→        else:
   293→            self._page = self._primary_page
   294→
   295→        logger.debug(f"Closed tab, switched to: {self._page.url if self._page else 'none'}")
   296→        return self._page
   297→
   298→    async def save_session(self) -> None:
   299→        """Save current session to disk (backup of cookies)."""
   300→        if self._context:
   301→            await self._session_manager.save(self._context)
   302→        else:
   303→            logger.warning("No active context to save session from")
   304→
   305→    async def screenshot(self, name: str = None) -> str:
   306→        """
   307→        Take a screenshot of current page.
   308→
   309→        Args:
   310→            name: Screenshot name (timestamp used if None)
   311→
   312→        Returns:
   313→            Path to saved screenshot
   314→        """
   315→        if not self._page:
   316→            raise BrowserError("No active page for screenshot")
   317→
   318→        screenshot_dir = Path(self._config.get('data.screenshot_dir', 'data/screenshots'))
   319→        screenshot_dir.mkdir(parents=True, exist_ok=True)
   320→
   321→        if name is None:
   322→            name = datetime.now().strftime('%Y%m%d_%H%M%S')
   323→
   324→        filepath = screenshot_dir / f"{name}.png"
   325→
   326→        await self._page.screenshot(path=str(filepath), full_page=True)
   327→        logger.info(f"Screenshot saved: {filepath}")
   328→
   329→        return str(filepath)
   330→
   331→
   332→
   333→# Global browser manager instance
   334→_browser_manager: Optional[BrowserManager] = None
   335→
   336→
   337→def get_browser_manager() -> BrowserManager:
   338→    """Get global browser manager instance."""
   339→    global _browser_manager
   340→    if _browser_manager is None:
   341→        _browser_manager = BrowserManager()
   342→    return _browser_manager
