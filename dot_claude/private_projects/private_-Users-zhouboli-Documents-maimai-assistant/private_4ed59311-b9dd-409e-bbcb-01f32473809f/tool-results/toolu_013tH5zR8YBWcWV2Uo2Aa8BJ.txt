     1→"""Pipeline checkpoint manager for resume capability.
     2→
     3→Saves pipeline progress to a JSON file so that interrupted pipelines
     4→can be resumed from where they left off.
     5→"""
     6→
     7→import json
     8→import os
     9→import tempfile
    10→from datetime import datetime, timezone
    11→from pathlib import Path
    12→from typing import Any, Dict, List, Optional, Set
    13→
    14→from loguru import logger
    15→
    16→
    17→def _get_checkpoint_dir() -> Path:
    18→    """Resolve checkpoint directory from config or use default."""
    19→    try:
    20→        from src.utils.config import get_config
    21→        config = get_config()
    22→        return Path(config.get('data.checkpoint_dir', 'data/.checkpoints'))
    23→    except Exception:
    24→        return Path("data/.checkpoints")
    25→
    26→
    27→class CheckpointManager:
    28→    """Manages pipeline checkpoint save/load for resume support."""
    29→
    30→    def __init__(self, checkpoint_id: Optional[str] = None):
    31→        checkpoint_dir = _get_checkpoint_dir()
    32→        checkpoint_dir.mkdir(parents=True, exist_ok=True)
    33→
    34→        if checkpoint_id:
    35→            self.checkpoint_path = checkpoint_dir / f"{checkpoint_id}.json"
    36→        else:
    37→            # Generate ID from timestamp
    38→            ts = datetime.now(timezone.utc).strftime("%Y%m%d_%H%M%S")
    39→            self.checkpoint_path = checkpoint_dir / f"pipeline_{ts}.json"
    40→
    41→        self._data: Dict[str, Any] = {}
    42→
    43→    def save(
    44→        self,
    45→        keywords: tuple,
    46→        template: str,
    47→        variables: dict,
    48→        company: tuple,
    49→        location: Optional[str],
    50→        max_pages: int,
    51→        batch_pages: int,
    52→        skip_messaged: bool,
    53→        interactive: bool,
    54→        completed_pages: int,
    55→        sent_ids: List[str],
    56→        seen_ids: List[str],
    57→        total_results: Optional[dict] = None,
    58→        mode: str = "search_center",
    59→        dry_run: bool = False,
    60→    ) -> None:
    61→        """Save current pipeline state to checkpoint file."""
    62→        self._data = {
    63→            "created_at": self._data.get("created_at", datetime.now(timezone.utc).isoformat()),
    64→            "updated_at": datetime.now(timezone.utc).isoformat(),
    65→            "params": {
    66→                "keywords": list(keywords),
    67→                "template": template,
    68→                "variables": variables,
    69→                "company": list(company) if company else [],
    70→                "location": location,
    71→                "max_pages": max_pages,
    72→                "batch_pages": batch_pages,
    73→                "skip_messaged": skip_messaged,
    74→                "interactive": interactive,
    75→                "mode": mode,
    76→                "dry_run": dry_run,
    77→            },
    78→            "progress": {
    79→                "completed_pages": completed_pages,
    80→                "sent_ids": sent_ids,
    81→                "seen_ids": seen_ids,
    82→                "total_results": total_results or {},
    83→            },
    84→        }
    85→
    86→        try:
    87→            tmp_fd, tmp_path = tempfile.mkstemp(
    88→                dir=self.checkpoint_path.parent, suffix=".tmp"
    89→            )
    90→            try:
    91→                with os.fdopen(tmp_fd, "w", encoding="utf-8") as f:
    92→                    json.dump(self._data, f, ensure_ascii=False, indent=2)
    93→                os.replace(tmp_path, self.checkpoint_path)
    94→            except BaseException:
    95→                try:
    96→                    os.unlink(tmp_path)
    97→                except OSError:
    98→                    pass
    99→                raise
   100→            logger.debug(f"Checkpoint saved: {self.checkpoint_path}")
   101→        except Exception as e:
   102→            logger.error(f"Failed to save checkpoint: {e}")
   103→            raise
   104→
   105→    @classmethod
   106→    def load_latest(cls) -> Optional["CheckpointManager"]:
   107→        """Load the most recent checkpoint file.
   108→
   109→        Returns:
   110→            CheckpointManager with loaded data, or None if no checkpoint exists.
   111→        """
   112→        checkpoint_dir = _get_checkpoint_dir()
   113→        checkpoint_dir.mkdir(parents=True, exist_ok=True)
   114→        files = sorted(checkpoint_dir.glob("pipeline_*.json"), key=lambda p: p.stat().st_mtime, reverse=True)
   115→
   116→        if not files:
   117→            return None
   118→
   119→        mgr = cls(checkpoint_id="__loading__")
   120→        mgr.checkpoint_path = files[0]
   121→
   122→        try:
   123→            with open(files[0], "r", encoding="utf-8") as f:
   124→                mgr._data = json.load(f)
   125→            logger.info(f"Loaded checkpoint: {files[0]}")
   126→            return mgr
   127→        except Exception as e:
   128→            logger.error(f"Failed to load checkpoint: {e}")
   129→            return None
   130→
   131→    @property
   132→    def params(self) -> Dict[str, Any]:
   133→        return self._data.get("params", {})
   134→
   135→    @property
   136→    def progress(self) -> Dict[str, Any]:
   137→        return self._data.get("progress", {})
   138→
   139→    @property
   140→    def completed_pages(self) -> int:
   141→        return self.progress.get("completed_pages", 0)
   142→
   143→    @property
   144→    def sent_ids(self) -> List[str]:
   145→        return list(self.progress.get("sent_ids", []))
   146→
   147→    @property
   148→    def seen_ids(self) -> Set[str]:
   149→        return set(self.progress.get("seen_ids", []))
   150→
   151→    @property
   152→    def total_results(self) -> Dict[str, int]:
   153→        return self.progress.get("total_results", {})
   154→
   155→    @property
   156→    def summary(self) -> str:
   157→        """Human-readable summary of checkpoint state."""
   158→        p = self.params
   159→        pr = self.progress
   160→        created = self._data.get("created_at", "unknown")
   161→        updated = self._data.get("updated_at", "unknown")
   162→        mode = p.get("mode", "search_center")
   163→        return (
   164→            f"Keywords: {p.get('keywords', [])}\n"
   165→            f"Template: {p.get('template', '')}\n"
   166→            f"Mode: {mode}\n"
   167→            f"Pages completed: {pr.get('completed_pages', 0)} / {p.get('max_pages', '?')}\n"
   168→            f"Candidates seen: {len(pr.get('seen_ids', []))}\n"
   169→            f"Messages sent: {len(pr.get('sent_ids', []))}\n"
   170→            f"Created: {created}\n"
   171→            f"Last updated: {updated}"
   172→        )
   173→
   174→    def delete(self) -> None:
   175→        """Remove checkpoint file after successful completion."""
   176→        try:
   177→            if self.checkpoint_path.exists():
   178→                self.checkpoint_path.unlink()
   179→                logger.debug(f"Checkpoint deleted: {self.checkpoint_path}")
   180→        except Exception as e:
   181→            logger.error(f"Failed to delete checkpoint: {e}")
   182→
