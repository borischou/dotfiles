     1→"""Rate limiter for preventing account bans."""
     2→
     3→import asyncio
     4→from datetime import datetime, timedelta, timezone
     5→from typing import Tuple, Optional
     6→
     7→from loguru import logger
     8→from sqlalchemy import func
     9→
    10→from src.models.database import ActionLog, get_db_manager, DatabaseManager
    11→from src.utils.config import get_config, Config
    12→from src.exceptions.errors import RateLimitExceededError
    13→
    14→
    15→class RateLimiter:
    16→    """Rate limiter for controlling action frequency.
    17→
    18→    Supports adaptive throttling: when captchas or errors are reported,
    19→    limits are temporarily tightened to reduce detection risk.
    20→    """
    21→
    22→    def __init__(self, config: Optional[Config] = None, db_manager: Optional[DatabaseManager] = None):
    23→        """Initialize rate limiter with configuration.
    24→
    25→        Args:
    26→            config: Config instance (uses global singleton if None)
    27→            db_manager: DatabaseManager instance (uses global singleton if None)
    28→        """
    29→        self.config = config or get_config()
    30→        self.db_manager = db_manager or get_db_manager()
    31→
    32→        # Adaptive throttle state
    33→        self._throttle_factor: float = 1.0  # 1.0 = normal, <1.0 = stricter
    34→        self._captcha_timestamps: list = []
    35→        self._pruned: bool = False  # one-time prune flag
    36→
    37→    def report_captcha(self) -> None:
    38→        """Report a captcha occurrence to trigger adaptive throttling."""
    39→        now = datetime.now(timezone.utc)
    40→        self._captcha_timestamps.append(now)
    41→        # Keep only last 30 minutes of captcha events
    42→        cutoff = now - timedelta(minutes=30)
    43→        self._captcha_timestamps = [t for t in self._captcha_timestamps if t >= cutoff]
    44→        self._update_throttle_factor()
    45→
    46→    def _update_throttle_factor(self) -> None:
    47→        """Recalculate throttle factor based on recent captcha frequency.
    48→
    49→        Throttle factor ranges from 1.0 (normal) to 0.3 (heavily throttled).
    50→        - 1 captcha in 30min: reduce to 0.7x (30% fewer operations allowed)
    51→        - 2 captchas: reduce to 0.5x
    52→        - 3+ captchas: reduce to 0.3x
    53→        """
    54→        captcha_count = len(self._captcha_timestamps)
    55→
    56→        if captcha_count >= 3:
    57→            self._throttle_factor = 0.3
    58→        elif captcha_count >= 2:
    59→            self._throttle_factor = 0.5
    60→        elif captcha_count >= 1:
    61→            self._throttle_factor = 0.7
    62→        else:
    63→            self._throttle_factor = 1.0
    64→
    65→        if self._throttle_factor < 1.0:
    66→            logger.warning(
    67→                f"Adaptive throttle active: factor={self._throttle_factor:.1f} "
    68→                f"(captchas={captcha_count} in last 30min)"
    69→            )
    70→
    71→    def _refresh_throttle(self) -> None:
    72→        """Prune expired timestamps and recalculate throttle factor."""
    73→        cutoff = datetime.now(timezone.utc) - timedelta(minutes=30)
    74→        self._captcha_timestamps = [t for t in self._captcha_timestamps if t >= cutoff]
    75→        self._update_throttle_factor()
    76→
    77→    @property
    78→    def throttle_factor(self) -> float:
    79→        """Current adaptive throttle factor (1.0 = normal, <1.0 = throttled)."""
    80→        return self._throttle_factor
    81→
    82→    def _get_limits(self, action_type: str) -> Tuple[int, int, int]:
    83→        """
    84→        Get rate limits for an action type, adjusted by adaptive throttle.
    85→
    86→        Args:
    87→            action_type: Type of action ('search', 'view_profile', 'send_message')
    88→
    89→        Returns:
    90→            Tuple of (per_hour_limit, per_day_limit, min_interval_seconds)
    91→        """
    92→        action_map = {
    93→            'search': ('search_per_hour', 'search_per_day'),
    94→            'view_profile': ('profile_view_per_hour', 'profile_view_per_day'),
    95→            'send_message': ('message_per_hour', 'message_per_day'),
    96→        }
    97→
    98→        if action_type not in action_map:
    99→            raise ValueError(f"Unknown action type: {action_type}")
   100→
   101→        hour_key, day_key = action_map[action_type]
   102→
   103→        per_hour = self.config.get(f'rate_limits.{hour_key}', 20)
   104→        per_day = self.config.get(f'rate_limits.{day_key}', 100)
   105→        min_interval = self.config.get('rate_limits.min_interval_seconds', 30)
   106→
   107→        # Apply adaptive throttle: reduce limits and increase interval
   108→        per_hour = max(1, int(per_hour * self._throttle_factor))
   109→        per_day = max(1, int(per_day * self._throttle_factor))
   110→        if self._throttle_factor < 1.0:
   111→            min_interval = int(min_interval / self._throttle_factor)
   112→
   113→        return per_hour, per_day, min_interval
   114→
   115→    def _count_actions(self, action_type: str, since: datetime) -> int:
   116→        """
   117→        Count actions of a type since a given time.
   118→
   119→        Args:
   120→            action_type: Type of action
   121→            since: Count actions since this time
   122→
   123→        Returns:
   124→            Number of actions
   125→        """
   126→        with self.db_manager.session_scope() as session:
   127→            count = session.query(func.count(ActionLog.id)).filter(
   128→                ActionLog.action_type == action_type,
   129→                ActionLog.timestamp >= since
   130→            ).scalar()
   131→
   132→            return count or 0
   133→
   134→    def _get_last_action_time(self, action_type: Optional[str] = None) -> Optional[datetime]:
   135→        """
   136→        Get timestamp of last action.
   137→
   138→        Args:
   139→            action_type: Type of action, or None for any action
   140→
   141→        Returns:
   142→            Timestamp of last action, or None if no actions
   143→        """
   144→        with self.db_manager.session_scope() as session:
   145→            query = session.query(func.max(ActionLog.timestamp))
   146→
   147→            if action_type:
   148→                query = query.filter(ActionLog.action_type == action_type)
   149→
   150→            last_time = query.scalar()
   151→            return last_time
   152→
   153→    def check_rate_limit(self, action_type: str) -> None:
   154→        """
   155→        Check if action is allowed under rate limits.
   156→
   157→        Args:
   158→            action_type: Type of action to check
   159→
   160→        Raises:
   161→            RateLimitExceededError: If rate limit is exceeded
   162→        """
   163→        # Prune stale entries once per session to keep counts accurate
   164→        if not self._pruned:
   165→            self.prune_old_entries()
   166→            self._pruned = True
   167→
   168→        self._refresh_throttle()
   169→        per_hour, per_day, min_interval = self._get_limits(action_type)
   170→
   171→        now = datetime.now(timezone.utc)
   172→        one_hour_ago = now - timedelta(hours=1)
   173→        one_day_ago = now - timedelta(days=1)
   174→
   175→        # Check hourly limit
   176→        hourly_count = self._count_actions(action_type, one_hour_ago)
   177→        if hourly_count >= per_hour:
   178→            # Calculate wait time until oldest action expires
   179→            with self.db_manager.session_scope() as session:
   180→                oldest_in_window = session.query(ActionLog.timestamp).filter(
   181→                    ActionLog.action_type == action_type,
   182→                    ActionLog.timestamp >= one_hour_ago
   183→                ).order_by(ActionLog.timestamp.asc()).first()
   184→
   185→                if oldest_in_window:
   186→                    wait_until = oldest_in_window[0] + timedelta(hours=1)
   187→                    wait_seconds = max(1, int((wait_until - now).total_seconds()))
   188→                else:
   189→                    wait_seconds = 3600
   190→
   191→            raise RateLimitExceededError(
   192→                action_type=action_type,
   193→                wait_time=wait_seconds,
   194→                message=f"Hourly rate limit exceeded for {action_type}. "
   195→                        f"Limit: {per_hour}/hour. Wait {wait_seconds // 60} minutes."
   196→            )
   197→
   198→        # Check daily limit
   199→        daily_count = self._count_actions(action_type, one_day_ago)
   200→        if daily_count >= per_day:
   201→            with self.db_manager.session_scope() as session:
   202→                oldest_in_window = session.query(ActionLog.timestamp).filter(
   203→                    ActionLog.action_type == action_type,
   204→                    ActionLog.timestamp >= one_day_ago
   205→                ).order_by(ActionLog.timestamp.asc()).first()
   206→
   207→                if oldest_in_window:
   208→                    wait_until = oldest_in_window[0] + timedelta(days=1)
   209→                    wait_seconds = max(1, int((wait_until - now).total_seconds()))
   210→                else:
   211→                    wait_seconds = 86400
   212→
   213→            raise RateLimitExceededError(
   214→                action_type=action_type,
   215→                wait_time=wait_seconds,
   216→                message=f"Daily rate limit exceeded for {action_type}. "
   217→                        f"Limit: {per_day}/day. Wait {wait_seconds // 3600} hours."
   218→            )
   219→
   220→        # Check minimum interval (per-action-type to avoid cross-type blocking)
   221→        last_action = self._get_last_action_time(action_type)
   222→        if last_action:
   223→            elapsed = (now - last_action).total_seconds()
   224→            if elapsed < min_interval:
   225→                wait_seconds = max(1, int(min_interval - elapsed))
   226→                raise RateLimitExceededError(
   227→                    action_type='any',
   228→                    wait_time=wait_seconds,
   229→                    message=f"Minimum interval not met. Wait {wait_seconds} seconds."
   230→                )
   231→
   232→        logger.debug(f"Rate limit check passed for {action_type}")
   233→
   234→    def log_action(self, action_type: str, details: Optional[dict] = None) -> None:
   235→        """
   236→        Log an action to the database.
   237→
   238→        Args:
   239→            action_type: Type of action
   240→            details: Optional action details
   241→        """
   242→        with self.db_manager.session_scope() as session:
   243→            action_log = ActionLog(
   244→                action_type=action_type,
   245→                timestamp=datetime.now(timezone.utc),
   246→                details=details or {}
   247→            )
   248→            session.add(action_log)
   249→
   250→        logger.debug(f"Logged action: {action_type}")
   251→
   252→    def prune_old_entries(self, retention_hours: int = 48) -> int:
   253→        """Remove ActionLog entries older than the retention window.
   254→
   255→        This prevents stale counts from accumulating across sessions and
   256→        keeps the ActionLog table from growing indefinitely.
   257→
   258→        Args:
   259→            retention_hours: Delete entries older than this many hours (default 48)
   260→
   261→        Returns:
   262→            Number of entries deleted
   263→        """
   264→        cutoff = datetime.now(timezone.utc) - timedelta(hours=retention_hours)
   265→        try:
   266→            with self.db_manager.session_scope() as session:
   267→                count = session.query(ActionLog).filter(
   268→                    ActionLog.timestamp < cutoff
   269→                ).delete()
   270→                logger.debug(f"Pruned {count} old ActionLog entries (older than {retention_hours}h)")
   271→                return count
   272→        except Exception as e:
   273→            logger.error(f"Failed to prune ActionLog entries: {e}")
   274→            return 0
   275→
   276→    def get_quota_status(self, action_type: str) -> dict:
   277→        """
   278→        Get current quota status for an action type.
   279→
   280→        Args:
   281→            action_type: Type of action
   282→
   283→        Returns:
   284→            Dictionary with quota information
   285→        """
   286→        per_hour, per_day, _ = self._get_limits(action_type)
   287→
   288→        now = datetime.now(timezone.utc)
   289→        one_hour_ago = now - timedelta(hours=1)
   290→        one_day_ago = now - timedelta(days=1)
   291→
   292→        hourly_used = self._count_actions(action_type, one_hour_ago)
   293→        daily_used = self._count_actions(action_type, one_day_ago)
   294→
   295→        return {
   296→            'action_type': action_type,
   297→            'hourly': {
   298→                'used': hourly_used,
   299→                'limit': per_hour,
   300→                'remaining': max(0, per_hour - hourly_used)
   301→            },
   302→            'daily': {
   303→                'used': daily_used,
   304→                'limit': per_day,
   305→                'remaining': max(0, per_day - daily_used)
   306→            }
   307→        }

   310→async def async_check_rate_limit(action_type: str, max_retries: int = 60) -> None:
   311→    """
   312→    Check rate limit, waiting through any violations until the action is allowed.
   313→
   314→    Blocks until the rate limit clears. Raises RateLimitExceededError if
   315→    still blocked after max_retries attempts (prevents infinite loops from
   316→    persistent failures such as corrupt DB data).
   317→
   318→    Args:
   319→        action_type: Type of action to check
   320→        max_retries: Maximum wait-retry cycles before giving up
   321→    """
   322→    rate_limiter = get_rate_limiter()
   323→    for attempt in range(max_retries):
   324→        try:
   325→            rate_limiter.check_rate_limit(action_type)
   326→            return
   327→        except RateLimitExceededError as e:
   328→            if attempt >= max_retries - 1:
   329→                raise
   330→            wait = e.wait_time + 1  # +1s buffer
   331→            logger.info(f"Rate limit hit: {e}. Waiting {wait}s before retrying (attempt {attempt + 1}/{max_retries})...")
   332→            await asyncio.sleep(wait)
   333→
   334→
   335→# Global rate limiter instance
   336→_rate_limiter = None
   337→
   338→
   339→def get_rate_limiter() -> RateLimiter:
   340→    """Get global rate limiter instance."""
   341→    global _rate_limiter
   342→    if _rate_limiter is None:
   343→        _rate_limiter = RateLimiter()
   344→    return _rate_limiter
