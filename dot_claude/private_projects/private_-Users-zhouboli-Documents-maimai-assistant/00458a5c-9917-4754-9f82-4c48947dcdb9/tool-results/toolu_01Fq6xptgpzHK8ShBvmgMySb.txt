     1→"""Browser manager for Playwright automation.
     2→
     3→Provides a singleton browser manager that connects to an existing Chrome
     4→instance via CDP (Chrome DevTools Protocol). Supports multi-tab management
     5→for workflows that open new tabs (e.g., enterprise recruitment).
     6→"""
     7→
     8→import asyncio
     9→from pathlib import Path
    10→from typing import List, Optional
    11→from datetime import datetime
    12→
    13→from patchright.async_api import async_playwright, Browser, BrowserContext, Page, Playwright
    14→from loguru import logger
    15→
    16→from src.utils.config import get_config, Config
    17→from src.core.session import SessionManager, get_session_manager
    18→from src.exceptions.errors import BrowserError
    19→
    20→
    21→class BrowserManager:
    22→    """
    23→    Singleton browser manager for Maimai automation.
    24→
    25→    Connects to a user-managed Chrome instance via CDP endpoint.
    26→    The user starts Chrome with --remote-debugging-port=9222 and logs in
    27→    manually. This manager connects/disconnects without closing Chrome.
    28→    """
    29→
    30→    _instance: Optional['BrowserManager'] = None
    31→    _lock: Optional[asyncio.Lock] = None
    32→    _lock_loop: Optional[asyncio.AbstractEventLoop] = None  # track which loop owns _lock
    33→
    34→    def __new__(cls, **kwargs):
    35→        if cls._instance is None:
    36→            cls._instance = super().__new__(cls)
    37→            cls._instance._initialized = False
    38→        return cls._instance
    39→
    40→    def __init__(
    41→        self,
    42→        config: Optional[Config] = None,
    43→        session_manager: Optional[SessionManager] = None,
    44→    ):
    45→        if self._initialized:
    46→            return
    47→
    48→        self._playwright: Optional[Playwright] = None
    49→        self._browser: Optional[Browser] = None
    50→        self._context: Optional[BrowserContext] = None
    51→        self._page: Optional[Page] = None
    52→        self._primary_page: Optional[Page] = None  # Original tab to return to
    53→        self._opened_tabs: List[Page] = []  # Tabs opened during our session
    54→        self._session_manager: SessionManager = session_manager or get_session_manager()
    55→        self._config: Config = config or get_config()
    56→        self._cdp_endpoint: Optional[str] = None
    57→        self._owned_page: bool = False  # True if we created the page (vs reusing existing)
    58→        self._initialized = True
    59→
    60→    @property
    61→    def is_running(self) -> bool:
    62→        """Check if browser is currently connected."""
    63→        return self._browser is not None and self._browser.is_connected()
    64→
    65→    @property
    66→    def is_connected(self) -> bool:
    67→        """Read-only connection status check for API endpoints."""
    68→        return self.is_running
    69→
    70→    @property
    71→    def page(self) -> Optional[Page]:
    72→        """Get current page instance."""
    73→        return self._page
    74→
    75→    @property
    76→    def context(self) -> Optional[BrowserContext]:
    77→        """Get current browser context."""
    78→        return self._context
    79→
    80→    @classmethod
    81→    def _ensure_lock(cls):
    82→        """Lazily create the asyncio.Lock inside a running event loop.
    83→
    84→        Recreates the lock if the event loop has changed (e.g. between
    85→        separate asyncio.run() calls) to avoid 'attached to a different
    86→        event loop' RuntimeError.
    87→        """
    88→        current_loop = asyncio.get_running_loop()
    89→        if cls._lock is None or cls._lock_loop is not current_loop:
    90→            cls._lock = asyncio.Lock()
    91→            cls._lock_loop = current_loop
    92→
    93→    def set_cdp_endpoint(self, endpoint: str) -> None:
    94→        """Override CDP endpoint (e.g. from CLI --cdp flag)."""
    95→        self._cdp_endpoint = endpoint
    96→
    97→    async def start(self) -> Page:
    98→        """
    99→        Connect to Chrome via CDP and return a page.
   100→
   101→        Returns:
   102→            Configured Playwright page instance
   103→
   104→        Raises:
   105→            BrowserError: If connection fails
   106→        """
   107→        self._ensure_lock()
   108→        async with self._lock:
   109→            if self.is_running:
   110→                logger.debug("Browser already connected, returning existing page")
   111→                return self._page
   112→
   113→            try:
   114→                # Determine CDP endpoint
   115→                endpoint = self._cdp_endpoint or self._config.get(
   116→                    'browser.cdp_endpoint', 'http://localhost:9222'
   117→                )
   118→                timeout = self._config.get('browser.timeout', 30000)
   119→
   120→                logger.info(f"Connecting to Chrome via CDP: {endpoint}")
   121→
   122→                # Start Playwright
   123→                self._playwright = await async_playwright().start()
   124→
   125→                # Connect to existing Chrome via CDP
   126→                self._browser = await self._playwright.chromium.connect_over_cdp(endpoint)
   127→
   128→                # Get the default context (the user's browser context with cookies)
   129→                contexts = self._browser.contexts
   130→                if contexts:
   131→                    self._context = contexts[0]
   132→                else:
   133→                    # Fallback: create a new context (shouldn't happen with CDP)
   134→                    logger.warning("No existing context found, creating new one")
   135→                    self._context = await self._browser.new_context()
   136→
   137→                # Set default timeout
   138→                self._context.set_default_timeout(timeout)
   139→
   140→                # Reuse existing page or create one
   141→                pages = self._context.pages
   142→                if pages:
   143→                    self._page = pages[0]
   144→                    self._owned_page = False
   145→                    logger.info(f"Reusing existing tab: {self._page.url}")
   146→                else:
   147→                    self._page = await self._context.new_page()
   148→                    self._owned_page = True
   149→                    logger.info("Created new tab")
   150→
   151→                # Remember the primary tab
   152→                self._primary_page = self._page
   153→                self._opened_tabs = []
   154→
   155→                logger.info("Connected to Chrome via CDP successfully")
   156→                return self._page
   157→
   158→            except Exception as e:
   159→                logger.error(f"Failed to connect to Chrome via CDP: {e}")
   160→                await self._close_internal()
   161→                raise BrowserError(
   162→                    f"Failed to connect to Chrome via CDP at {endpoint}. "
   163→                    f"Make sure Chrome is running with --remote-debugging-port=9222. "
   164→                    f"Error: {e}"
   165→                )
   166→
   167→    async def close(self) -> None:
   168→        """Disconnect from Chrome via CDP. Does NOT close the user's Chrome."""
   169→        self._ensure_lock()
   170→        async with self._lock:
   171→            await self._close_internal()
   172→
   173→    async def _close_internal(self) -> None:
   174→        """Actual cleanup logic — callable without acquiring _lock.
   175→
   176→        Must only be called while the lock is already held, or from
   177→        start()'s except block (which already holds the lock).
   178→        """
   179→        try:
   180→            # Close tabs we opened during the session
   181→            for tab in self._opened_tabs:
   182→                try:
   183→                    if not tab.is_closed():
   184→                        await tab.close()
   185→                except Exception:
   186→                    pass  # Tab may already be closed by user
   187→            self._opened_tabs = []
   188→
   189→            # Only close pages we created, not the user's existing tabs
   190→            if self._page and self._owned_page:
   191→                try:
   192→                    await self._page.close()
   193→                except Exception:
   194→                    pass  # Page may already be closed
   195→            self._page = None
   196→            self._primary_page = None
   197→            self._owned_page = False
   198→
   199→            # Don't close the context — it belongs to the user's Chrome
   200→            self._context = None
   201→
   202→            # browser.close() in CDP mode just disconnects, doesn't kill Chrome
   203→            if self._browser:
   204→                try:
   205→                    await self._browser.close()
   206→                except Exception as e:
   207→                    logger.error(f"Error closing browser connection: {e}")
   208→                finally:
   209→                    self._browser = None
   210→
   211→            if self._playwright:
   212→                try:
   213→                    await self._playwright.stop()
   214→                except Exception as e:
   215→                    logger.error(f"Error stopping Playwright: {e}")
   216→                finally:
   217→                    self._playwright = None
   218→
   219→            logger.info("Disconnected from Chrome (browser stays open)")
   220→
   221→        except Exception as e:
   222→            logger.error(f"Error disconnecting from Chrome: {e}")
   223→
   224→    async def wait_for_new_tab(self, timeout: int = 10000) -> Page:
   225→        """Wait for a click-triggered new tab, apply stealth, and switch to it.
   226→
   227→        Call this right after performing an action that opens a new tab
   228→        (e.g., clicking a link with target=_blank).
   229→
   230→        Args:
   231→            timeout: Max wait time in milliseconds
   232→
   233→        Returns:
   234→            The new Page instance (also set as the active page)
   235→
   236→        Raises:
   237→            BrowserError: If no new tab appears within timeout
   238→        """
   239→        if not self._context:
   240→            raise BrowserError("No active context for wait_for_new_tab")
   241→
   242→        try:
   243→            new_page = await self._context.wait_for_event(
   244→                "page", timeout=timeout
   245→            )
   246→            await new_page.wait_for_load_state("domcontentloaded")
   247→            self._opened_tabs.append(new_page)
   248→            self._page = new_page
   249→            logger.info(f"New tab captured: {new_page.url}")
   250→            return new_page
   251→        except Exception as e:
   252→            raise BrowserError(f"No new tab opened within {timeout}ms: {e}")
   253→
   254→    def switch_to_primary_tab(self) -> Page:
   255→        """Switch back to the original (primary) tab.
   256→
   257→        Returns:
   258→            The primary Page instance
   259→        """
   260→        if not self._primary_page or self._primary_page.is_closed():
   261→            raise BrowserError("Primary tab is not available")
   262→
   263→        self._page = self._primary_page
   264→        logger.debug(f"Switched to primary tab: {self._page.url}")
   265→        return self._page
   266→
   267→    async def close_current_tab(self) -> Page:
   268→        """Close the current active tab and switch to the previous one.
   269→
   270→        Will not close the primary tab. After closing, switches to the
   271→        most recent remaining tab or falls back to the primary tab.
   272→
   273→        Returns:
   274→            The page that became active after closing
   275→
   276→        Raises:
   277→            BrowserError: If trying to close the primary tab
   278→        """
   279→        if not self._page:
   280→            raise BrowserError("No active page to close")
   281→
   282→        if self._page == self._primary_page:
   283→            raise BrowserError("Cannot close the primary tab")
   284→
   285→        closing = self._page
   286→        try:
   287→            # Remove from opened_tabs tracking
   288→            if closing in self._opened_tabs:
   289→                self._opened_tabs.remove(closing)
   290→            await closing.close()
   291→        except Exception as e:
   292→            logger.warning(f"Error closing tab: {e}")
   293→
   294→        # Switch to previous tab
   295→        if self._context and self._context.pages:
   296→            self._page = self._context.pages[-1]
   297→        else:
   298→            self._page = self._primary_page
   299→
   300→        logger.debug(f"Closed tab, switched to: {self._page.url if self._page else 'none'}")
   301→        return self._page
   302→
   303→    async def save_session(self) -> None:
   304→        """Save current session to disk (backup of cookies)."""
   305→        if self._context:
   306→            await self._session_manager.save(self._context)
   307→        else:
   308→            logger.warning("No active context to save session from")
   309→
   310→    async def screenshot(self, name: str = None) -> str:
   311→        """
   312→        Take a screenshot of current page.
   313→
   314→        Args:
   315→            name: Screenshot name (timestamp used if None)
   316→
   317→        Returns:
   318→            Path to saved screenshot
   319→        """
   320→        if not self._page:
   321→            raise BrowserError("No active page for screenshot")
   322→
   323→        screenshot_dir = Path(self._config.get('data.screenshot_dir', 'data/screenshots'))
   324→        screenshot_dir.mkdir(parents=True, exist_ok=True)
   325→
   326→        if name is None:
   327→            name = datetime.now().strftime('%Y%m%d_%H%M%S')
   328→
   329→        filepath = screenshot_dir / f"{name}.png"
   330→
   331→        await self._page.screenshot(path=str(filepath), full_page=True)
   332→        logger.info(f"Screenshot saved: {filepath}")
   333→
   334→        return str(filepath)
   335→
   336→
   337→
   338→# Global browser manager instance
   339→_browser_manager: Optional[BrowserManager] = None
   340→
   341→
   342→def get_browser_manager() -> BrowserManager:
   343→    """Get global browser manager instance."""
   344→    global _browser_manager
   345→    if _browser_manager is None:
   346→        _browser_manager = BrowserManager()
   347→    return _browser_manager
