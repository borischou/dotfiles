    53→def _is_non_rd_title(title: str) -> bool:
    54→    """Check if a title indicates a non-R&D role.
    55→
    56→    Strict mode: if any non-R&D keyword appears, filter out regardless of
    57→    whether R&D keywords are also present (e.g. "技术产品经理" is filtered).
    58→    Returns False (keep) only if title is empty or contains no non-R&D keywords.
    59→    """
    60→    if not title:
    61→        return False
    62→
    63→    # Strict: any non-R&D keyword present → filter out
    64→    if any(kw in title for kw in NON_RD_KEYWORDS):
    65→        return True
    66→
    67→    return False

    82→async def search_candidates(
    83→    keywords: str,
    84→    company: Optional[tuple] = None,
    85→    location: Optional[str] = None,
    86→    pages: int = 1,
    87→    start_page: int = 1,
    88→    browser: Optional[BrowserManager] = None
    89→) -> List[CandidateBasic]:
    90→    """
    91→    Search for candidates on Maimai.
    92→
    93→    Args:
    94→        keywords: Search keywords (e.g., "高级前端工程师")
    95→        company: Filter by company names, tuple of strings (optional)
    96→        location: Filter by location (optional)
    97→        pages: Number of pages to scrape (default 1)
    98→        start_page: First page number to scrape (default 1)
    99→        browser: Browser manager instance
   100→
   101→    Returns:
   102→        List of CandidateBasic objects
   103→
   104→    Raises:
   105→        RateLimitExceededError: If rate limit exceeded
   106→        SelectorNotFoundError: If page elements not found
   107→    """
   108→    # Validate inputs
   109→    validate_keywords(keywords)
   110→
   111→    if browser is None:
   112→        browser = get_browser_manager()
   113→
   114→    config = get_config()
   115→    rate_limiter = get_rate_limiter()
   116→    db_manager = get_db_manager()
   117→
   118→    # Ensure logged in
   119→    await ensure_logged_in(browser)
   120→
   121→    all_candidates: List[CandidateBasic] = []
   122→
   123→    with Progress(
   124→        SpinnerColumn(),
   125→        TextColumn("[progress.description]{task.description}"),
   126→        console=console
   127→    ) as progress:
   128→
   129→        end_page = start_page + pages
   130→        for page_num in range(start_page, end_page):
   131→            task = progress.add_task(f"Searching page {page_num}/{end_page - 1}...", total=None)
   132→
   133→            try:
   134→                # Check rate limit (waits through min-interval)
   135→                await async_check_rate_limit('search')
   136→
   137→                # Build search URL
   138→                params = {
   139→                    'type': 'contact',
   140→                    'query': keywords,
   141→                    'highlight': 'true'
   142→                }
   143→                if page_num > 1:
   144→                    params['page'] = str(page_num)
   145→
   146→                search_url = f"{MAIMAI_SEARCH_URL}?{urlencode(params)}"
   147→
   148→                page = browser.page
   149→
   150→                # Navigate to search page
   151→                await retry_navigation(page, search_url, wait_until='domcontentloaded')
   152→
   153→                # Wait for page load
   154→                await async_random_delay('page_load')
   155→
   156→                # Check for captcha
   157→                await check_captcha(page, report_to_rate_limiter=True)
   158→
   159→                # Click "人脉" tab to show contact results
   160→                try:
   161→                    renmai_tab = await page.query_selector('text="人脉"')
   162→                    if renmai_tab:
   163→                        await renmai_tab.click()
   164→                        logger.info("Clicked '人脉' tab")
   165→                        await async_random_delay('action')
   166→                        # Wait for content to load
   167→                        await page.wait_for_timeout(2000)
   168→                except Exception as e:
   169→                    logger.debug(f"Could not click 人脉 tab: {e}")
   170→
   171→                # Simulate human behavior
   172→                await human_scroll(page, 'down', distance=300)
   173→                await async_random_delay('action')
   174→
   175→                # Extract candidates from this page
   176→                candidates = await _extract_search_results(page, config)
   177→
   178→                if not candidates:
   179→                    logger.info(f"No results found on page {page_num}")
   180→                    await browser.screenshot(f"search_debug_page{page_num}")
   181→                    progress.update(task, description=f"Page {page_num}: No results")
   182→                    # Allow one empty page (could be transient), break on second consecutive
   183→                    if page_num > start_page and not all_candidates:
   184→                        # First page had no results either — genuinely empty
   185→                        break
   186→                    continue
   187→
   188→                # Filter out non-R&D roles
   189→                before_rd_filter = len(candidates)
   190→                candidates = [c for c in candidates if not _is_non_rd_title(c.title)]
   191→                rd_filtered = before_rd_filter - len(candidates)
   192→                if rd_filtered > 0:
   193→                    logger.info(f"Filtered out {rd_filtered} non-R&D candidates")
   194→
   195→                # Apply filters if specified
   196→                if company:
   197→                    company_lower = [cn.lower() for cn in company]
   198→                    candidates = [c for c in candidates if any(cn in c.company.lower() for cn in company_lower)]
   199→
   200→                if location:
   201→                    candidates = [c for c in candidates if c.location and location.lower() in c.location.lower()]
   202→
   203→                all_candidates.extend(candidates)
   204→                progress.update(task, description=f"Page {page_num}: Found {len(candidates)} candidates")
   205→
   206→                # Log the action
   207→                rate_limiter.log_action('search', {
   208→                    'keywords': keywords,
   209→                    'page': page_num,
   210→                    'results': len(candidates)
   211→                })
   212→
   213→                # Save candidates to database
   214→                try:
   215→                    _save_candidates_to_db(db_manager, candidates, keywords)
   216→                except Exception as e:
   217→                    console.print(f"[yellow]Warning: Failed to save candidates to DB: {e}[/yellow]")
   218→
   219→                # Delay before next page
   220→                if page_num < end_page - 1:
   221→                    await async_random_delay('page_load')
   222→
   223→            except (BrowserError, SessionExpiredError):
   224→                raise  # Critical errors must propagate
   225→            except Exception as e:
   226→                logger.error(f"Error on page {page_num}: {e}")
   227→                progress.update(task, description=f"Page {page_num}: Error - {str(e)[:30]}")
   228→                # Take screenshot for debugging
   229→                await browser.screenshot(f"search_error_page{page_num}")
   230→                continue
   231→
   232→    # Log search to history
   233→    _log_search_history(db_manager, keywords, company, location, pages, len(all_candidates))
   234→
   235→    console.print(f"[green]Search complete. Found {len(all_candidates)} candidates.[/green]")
   236→    return all_candidates

   239→async def _extract_search_results(page: Page, config) -> List[CandidateBasic]:
   240→    """
   241→    Extract candidate information from search results page.
   242→
   243→    Args:
   244→        page: Playwright page instance
   245→        config: Config instance
   246→
   247→    Returns:
   248→        List of CandidateBasic objects
   249→    """
   250→    candidates = []
   251→
   252→    try:
   253→        # Wait for page to be fully loaded
   254→        await page.wait_for_load_state('networkidle', timeout=10000)
   255→    except Exception:
   256→        pass  # Continue even if networkidle times out
   257→
   258→    # Wait for actual result cards to render (avoid "搜索数据中..." race condition)
   259→    try:
   260→        await page.wait_for_function(
   261→            '''() => {
   262→                const body = document.body ? document.body.innerText : '';
   263→                if (body.includes('搜索数据中') || body.includes('加载中')) return false;
   264→                // Check for avatar images indicating result cards are rendered
   265→                const imgs = document.querySelectorAll('img');
   266→                let avatarCount = 0;
   267→                for (const img of imgs) {
   268→                    const rect = img.getBoundingClientRect();
   269→                    if (rect.width >= 20 && rect.width <= 100 && rect.height >= 20 && rect.height <= 100) {
   270→                        avatarCount++;
   271→                    }
   272→                }
   273→                return avatarCount >= 1;
   274→            }''',
   275→            timeout=8000,
   276→        )
   277→    except Exception:
   278→        logger.debug("Timed out waiting for result cards to render, proceeding with extraction")
   279→
   280→    try:
   281→        # Strategy: Extract candidates by finding avatar images and their associated text
   282→        candidates_data = await page.evaluate('''() => {
   283→            const candidates = [];
   284→            const seenNames = new Set();
   285→
   286→            // Find all images that look like avatars
   287→            const images = document.querySelectorAll('img');
   288→
   289→            for (const img of images) {
   290→                // Skip large images (probably not avatars)
   291→                const rect = img.getBoundingClientRect();
   292→                if (rect.width > 100 || rect.height > 100) continue;
   293→                if (rect.width < 20 || rect.height < 20) continue;
   294→
   295→                // Find parent container (the card)
   296→                let card = img.parentElement;
   297→                for (let i = 0; i < 8 && card; i++) {
   298→                    const cardRect = card.getBoundingClientRect();
   299→                    // Look for a reasonable card size
   300→                    if (cardRect.width > 200 && cardRect.height > 50 && cardRect.height < 200) {
   301→                        break;
   302→                    }
   303→                    card = card.parentElement;
   304→                }
   305→
   306→                if (!card) continue;
   307→
   308→                // Get all text nodes from card, preserving order
   309→                const textParts = [];
   310→                const walker = document.createTreeWalker(card, NodeFilter.SHOW_TEXT, null, false);
   311→                while (walker.nextNode()) {
   312→                    const text = walker.currentNode.textContent.trim();
   313→                    if (text && text.length > 0 && text.length < 100) {
   314→                        textParts.push(text);
   315→                    }
   316→                }
   317→
   318→                // Parse candidate info from text parts
   319→                // Pattern: Name, Activity time, Company+Position, Industry|Department
   320→                let name = '';
   321→                let company = '';
   322→                let title = '';
   323→                let industry = '';
   324→
   325→                // Compound job title patterns (longer patterns first for greedy match)
   326→                const jobPatterns = [
   327→                    // Compound titles with prefix modifiers
   328→                    /(?:高级|资深|首席|初级|中级|senior|junior|staff|principal|lead)?\s*(?:前端|后端|全栈|web|Web|iOS|Android|Java|Python|Go|C\+\+|PHP|.NET|算法|数据|AI|机器学习|深度学习|NLP|大数据|云计算|DevOps|SRE|安全|测试|嵌入式|移动端|客户端|服务端|区块链|具身智能|自动驾驶|机器人)?\s*(?:工程师|开发|架构师|专家|技术经理|技术总监|技术负责人|Tech\s*Lead)/i,
   329→                    // Management titles
   330→                    /(?:技术|产品|运营|市场|销售|人力|财务|行政|项目|研发)?\s*(?:总监|经理|主管|负责人|VP|总裁|副总|合伙人|创始人|董事|CEO|CTO|COO|CFO|CMO|CHO|CIO)/i,
   331→                    // Other roles
   332→                    /(?:产品|UI|UX|交互|视觉)?\s*(?:设计师|分析师|顾问|运营|助理|实习生?)/i,
   333→                ];
   334→
   335→                // Patterns for text that should be skipped
   336→                const skipPatterns = [
   337→                    /^[+＋]\s*好友$/,          // "+ 好友" button
   338→                    /^\d+个共同好友$/,           // "1个共同好友" social label
   339→                    /^共同好友/,                 // "共同好友..." variants
   340→                    /^已关注$/,                  // "已关注" button
   341→                    /^添加好友$/,                // "添加好友" button
   342→                    /^关注$/,                    // "关注" button
   343→                    /^好友$/,                    // "好友" label
   344→                    /^[|｜·]$/,                  // separators
   345→                ];
   346→
   347→                const isSkippable = (text) => {
   348→                    if (skipPatterns.some(p => p.test(text))) return true;
   349→                    // Activity time
   350→                    if (/天前|小时前|分钟前|刚刚|活跃|在线/.test(text)) return true;
   351→                    return false;
   352→                };
   353→
   354→                for (let i = 0; i < textParts.length; i++) {
   355→                    const part = textParts[i];
   356→
   357→                    if (isSkippable(part)) continue;
   358→
   359→                    // First short text (2-15 chars, pure CJK/letters) is likely the name
   360→                    if (!name && part.length >= 2 && part.length <= 15) {
   361→                        if (/^[\u4e00-\u9fa5a-zA-Z·\s]+$/.test(part)) {
   362→                            name = part;
   363→                            continue;
   364→                        }
   365→                    }
   366→
   367→                    // After name, next substantial text is company+position
   368→                    if (name && !company && part.length > 2) {
   369→                        // Try to split company and position using regex patterns
   370→                        let splitDone = false;
   371→                        for (const pattern of jobPatterns) {
   372→                            const match = part.match(pattern);
   373→                            if (match && match.index > 0) {
   374→                                company = part.substring(0, match.index).trim();
   375→                                title = match[0].trim();
   376→                                splitDone = true;
   377→                                break;
   378→                            }
   379→                        }
   380→                        if (!splitDone) {
   381→                            company = part;
   382→                        }
   383→                        continue;
   384→                    }
   385→
   386→                    // Next part is industry|department info
   387→                    if (name && company && !industry && part.length > 2) {
   388→                        industry = part;
   389→                        // If title wasn't split from company, use industry as fallback context
   390→                        if (!title) {
   391→                            title = industry;
   392→                        }
   393→                        break;
   394→                    }
   395→                }
   396→
   397→                // Skip if no name found or premium placeholder
   398→                if (!name) continue;
   399→                if (name === '脉脉高价值人脉') continue;
   400→
   401→                // Extract user ID from DOM patterns on card and ancestor elements.
   402→                // Two patterns exist:
   403→                //   1. contactcard_{numeric_id} — on ancestors (colleagues layout)
   404→                //   2. member_entry_{numeric_id} — on descendants (standard layout)
   405→                let userId = '';
   406→
   407→                // Check ancestors for numeric contactcard_{id}
   408→                let el = card;
   409→                while (el) {
   410→                    const elId = el.id || '';
   411→                    const ccMatch = elId.match(/^contactcard_(\d+)$/);
   412→                    if (ccMatch) {
   413→                        userId = ccMatch[1];
   414→                        break;
   415→                    }
   416→                    el = el.parentElement;
   417→                }
   418→
   419→                // Check descendants for member_entry_{id} (most common pattern)
   420→                if (!userId) {
   421→                    const meEl = card.querySelector('[id^="member_entry_"]');
   422→                    if (meEl) {
   423→                        const meMatch = meEl.id.match(/^member_entry_(\d+)$/);
   424→                        if (meMatch) {
   425→                            userId = meMatch[1];
   426→                        }
   427→                    }
   428→                }
   429→
   430→                // Also check ancestor descendants for contactcard with numeric ID
   431→                if (!userId) {
   432→                    const ccEl = card.querySelector('[id^="contactcard_"]');
   433→                    if (ccEl) {
   434→                        const ccMatch = ccEl.id.match(/^contactcard_(\d+)$/);
   435→                        if (ccMatch) {
   436→                            userId = ccMatch[1];
   437→                        }
   438→                    }
   439→                }
   440→
   441→                // Use real user_id as maimai_id when available, fall back to hash
   442→                let maimaiId;
   443→                if (userId) {
   444→                    maimaiId = 'mm_' + userId;
   445→                } else {
   446→                    const idSource = (name || '') + '|' + (company || '') + '|' + (title || '');
   447→                    let hash = 0;
   448→                    for (let j = 0; j < idSource.length; j++) {
   449→                        const ch = idSource.charCodeAt(j);
   450→                        hash = ((hash << 5) - hash) + ch;
   451→                        hash = hash & hash; // Convert to 32bit integer
   452→                    }
   453→                    maimaiId = 'mm_' + Math.abs(hash).toString(16).padStart(8, '0');
   454→                }
   455→
   456→                // Skip duplicates based on the computed ID
   457→                if (seenNames.has(maimaiId)) continue;
   458→                seenNames.add(maimaiId);
   459→
   460→                // Build profile URL from user ID when available
   461→                let profileUrl = '';
   462→                if (userId) {
   463→                    profileUrl = 'https://maimai.cn/contact/detail/' + userId + '?dstu=' + userId;
   464→                }
   465→
   466→                // Fallback: extract profile URL from <a> tags inside the card
   467→                if (!profileUrl) {
   468→                    let fallbackUrl = '';
   469→
   470→                    // First: check <a> ancestors between img and card
   471→                    let ancestor = img.parentElement;
   472→                    while (ancestor && ancestor !== card) {
   473→                        if (ancestor.tagName === 'A' && ancestor.href) {
   474→                            const href = ancestor.getAttribute('href') || '';
   475→                            if (href.includes('/profile/detail') || href.includes('/contact/detail/') || href.includes('/u/') || href.includes('egid=') || href.includes('dstu=')) {
   476→                                profileUrl = href.startsWith('http') ? href : 'https://maimai.cn' + href;
   477→                                break;
   478→                            }
   479→                        }
   480→                        ancestor = ancestor.parentElement;
   481→                    }
   482→
   483→                    // Second: search all <a> descendants inside the card
   484→                    if (!profileUrl) {
   485→                        const links = card.querySelectorAll('a[href]');
   486→                        for (const link of links) {
   487→                            const href = link.getAttribute('href') || '';
   488→                            if (href.includes('/profile/detail') || href.includes('/contact/detail/') || href.includes('/u/') || href.includes('egid=') || href.includes('dstu=')) {
   489→                                profileUrl = href.startsWith('http') ? href : 'https://maimai.cn' + href;
   490→                                break;
   491→                            }
   492→                            if (!fallbackUrl && href && !href.startsWith('javascript:') && (href.startsWith('/') || href.includes('maimai.cn'))) {
   493→                                fallbackUrl = href.startsWith('http') ? href : 'https://maimai.cn' + href;
   494→                            }
   495→                        }
   496→                    }
   497→                    if (!profileUrl && fallbackUrl) {
   498→                        profileUrl = fallbackUrl;
   499→                    }
   500→                }
   501→
   502→                // Extract location: look for remaining short text parts after name/company/title
   503→                // Common patterns: city names like "北京", "上海", "深圳" etc.
   504→                let location = '';
   505→                const locationPatterns = /^(北京|上海|广州|深圳|杭州|成都|南京|武汉|西安|苏州|长沙|天津|重庆|郑州|东莞|青岛|大连|宁波|厦门|合肥|佛山|福州|昆明|济南|珠海|无锡|常州|海外|远程|香港|台北|新加坡|美国|日本|韩国|英国|全国|其他)/;
   506→                for (let i = 0; i < textParts.length; i++) {
   507→                    const part = textParts[i];
   508→                    if (part === name || part === company || part === title || part === industry) continue;
   509→                    if (isSkippable(part)) continue;
   510→                    if (locationPatterns.test(part) && part.length <= 10) {
   511→                        location = part;
   512→                        break;
   513→                    }
   514→                }
   515→
   516→                candidates.push({
   517→                    name: name,
   518→                    title: title || '',
   519→                    company: company || '',
   520→                    maimai_id: maimaiId,
   521→                    profile_url: profileUrl,
   522→                    location: location,
   523→                });
   524→            }
   525→
   526→            return candidates;
   527→        }''')
   528→
   529→        logger.info(f"JavaScript extraction found {len(candidates_data)} candidates")
   530→
   531→        for data in candidates_data:
   532→            if data.get('name'):
   533→                candidates.append(CandidateBasic(
   534→                    maimai_id=data.get('maimai_id', ''),
   535→                    name=data.get('name', ''),
   536→                    title=data.get('title', ''),
   537→                    company=data.get('company', ''),
   538→                    location=data.get('location', ''),
   539→                    profile_url=data.get('profile_url', '')
   540→                ))
   541→
   542→        # For candidates missing profile_url, try extracting from contactcard DOM
   543→        missing_url = [c for c in candidates if not c.profile_url]
   544→        if missing_url:
   545→            logger.info(f"{len(missing_url)} candidates missing profile_url, searching contactcard DOM")
   546→            for c in missing_url:
   547→                try:
   548→                    profile_url = await _click_avatar_for_profile_url(page, c.name, c.company)
   549→                    if profile_url:
   550→                        c.profile_url = profile_url
   551→                except Exception as e:
   552→                    logger.debug(f"Contactcard lookup failed for '{c.name}': {e}")
   553→
   554→        # Fallback: Strategy 1 - Find all profile links directly
   555→        if not candidates:
   556→            profile_links = await page.query_selector_all('a[href*="/contact/"], a[href*="/u/"], a[href*="egid="], a[href*="profile"]')
   557→        else:
   558→            profile_links = []
   559→
   560→        if profile_links and not candidates:
   561→            logger.debug(f"Found {len(profile_links)} profile links")
   562→
   563→            seen_ids = set()
   564→            for link in profile_links:
   565→                try:
   566→                    href = await link.get_attribute('href')
   567→                    if not href:
   568→                        continue
   569→
   570→                    # Extract maimai_id from URL
   571→                    maimai_id = None
   572→                    if '/contact/detail/' in href:
   573→                        maimai_id = href.split('/contact/detail/')[-1].split('?')[0].split('/')[0]
   574→                    elif '/u/' in href:
   575→                        maimai_id = href.split('/u/')[-1].split('?')[0].split('/')[0]
   576→
   577→                    if not maimai_id or maimai_id in seen_ids:
   578→                        continue
   579→                    seen_ids.add(maimai_id)
   580→
   581→                    # Get name from link text
   582→                    name = await link.inner_text()
   583→                    name = name.strip()
   584→                    if not name or len(name) > 50:  # Skip if no name or too long (probably not a name)
   585→                        continue
   586→
   587→                    # Try to get parent container for more info
   588→                    parent = await link.evaluate_handle('el => el.closest("div, li, article") || el.parentElement.parentElement')
   589→
   590→                    title = ""
   591→                    company = ""
   592→
   593→                    if parent:
   594→                        # Get all text content and try to parse
   595→                        try:
   596→                            full_text = await parent.inner_text()
   597→                            lines = [l.strip() for l in full_text.split('\n') if l.strip()]
   598→
   599→                            # Usually: name, then title/company info
   600→                            for line in lines:
   601→                                if line == name:
   602→                                    continue
   603→                                if '好友' in line or '关注' in line:
   604→                                    continue
   605→                                if not title and len(line) < 50:
   606→                                    title = line
   607→                                elif not company and len(line) < 50:
   608→                                    company = line
   609→                                    break
   610→                        except Exception:
   611→                            pass  # Best-effort text parsing; proceed with partial data
   612→
   613→                    profile_url = href if href.startswith('http') else f"https://maimai.cn{href}"
   614→
   615→                    candidates.append(CandidateBasic(
   616→                        maimai_id=maimai_id,
   617→                        name=name,
   618→                        title=title,
   619→                        company=company,
   620→                        profile_url=profile_url
   621→                    ))
   622→
   623→                except Exception as e:
   624→                    logger.debug(f"Failed to extract from link: {e}")
   625→                    continue
   626→
   627→        # Strategy 2: If no links found, try button-based detection
   628→        if not candidates:
   629→            logger.debug("Trying button-based detection")
   630→            buttons = await page.query_selector_all('button:has-text("好友"), button:has-text("关注")')
   631→
   632→            for button in buttons:
   633→                try:
   634→                    parent = await button.evaluate_handle('el => el.closest("div, li, article")')
   635→                    if not parent:
   636→                        continue
   637→
   638→                    # Find link in parent
   639→                    link = await parent.query_selector('a[href*="/contact"], a[href*="/u/"]')
   640→                    if link:
   641→                        href = await link.get_attribute('href')
   642→                        name = await link.inner_text()
   643→
   644→                        maimai_id = None
   645→                        if '/contact/detail/' in href:
   646→                            maimai_id = href.split('/contact/detail/')[-1].split('?')[0]
   647→                        elif '/u/' in href:
   648→                            maimai_id = href.split('/u/')[-1].split('?')[0]
   649→
   650→                        if maimai_id and name:
   651→                            candidates.append(CandidateBasic(
   652→                                maimai_id=maimai_id,
   653→                                name=name.strip(),
   654→                                title="",
   655→                                company="",
   656→                                profile_url=href
   657→                            ))
   658→                except Exception:
   659→                    continue
   660→
   661→        # Filter out premium placeholder profiles
   662→        before_filter = len(candidates)
   663→        candidates = [c for c in candidates if c.name not in PREMIUM_PLACEHOLDER_NAMES]
   664→        filtered_count = before_filter - len(candidates)
   665→        if filtered_count > 0:
   666→            logger.info(f"Filtered out {filtered_count} premium placeholder profiles")
   667→
   668→        logger.info(f"Extracted {len(candidates)} candidates from page")
   669→
   670→    except Exception as e:
   671→        logger.error(f"Error extracting search results: {e}")
   672→        console.print(f"[yellow]Warning: Search result extraction failed: {e}[/yellow]")
   673→
   674→    return candidates

   677→def _save_candidates_to_db(db_manager, candidates: List[CandidateBasic], source_search: str) -> None:
   678→    """
   679→    Save or update candidates in database.
   680→
   681→    Args:
   682→        db_manager: Database manager instance
   683→        candidates: List of candidates to save
   684→        source_search: Search keywords that found these candidates
   685→    """
   686→    try:
   687→        with db_manager.session_scope() as session:
   688→            # Batch pre-fetch existing candidates to avoid N+1 queries
   689→            maimai_ids = [c.maimai_id for c in candidates]
   690→            existing_by_id = {
   691→                row.maimai_id: row
   692→                for row in session.query(Candidate).filter(
   693→                    Candidate.maimai_id.in_(maimai_ids)
   694→                ).all()
   695→            }
   696→
   697→            # Also batch-fetch by name+company for dedup
   698→            name_company_pairs = [(c.name, c.company) for c in candidates]
   699→            from sqlalchemy import tuple_
   700→            existing_by_name = {}
   701→            if name_company_pairs:
   702→                for row in session.query(Candidate).filter(
   703→                    tuple_(Candidate.name, Candidate.company).in_(name_company_pairs)
   704→                ).all():
   705→                    existing_by_name[(row.name, row.company)] = row
   706→
   707→            for c in candidates:
   708→                existing = existing_by_id.get(c.maimai_id)
   709→                if not existing:
   710→                    existing = existing_by_name.get((c.name, c.company))
   711→
   712→                if existing:
   713→                    existing.name = c.name
   714→                    existing.title = c.title
   715→                    existing.company = c.company
   716→                    existing.location = c.location
   717→                    existing.profile_url = c.profile_url
   718→                else:
   719→                    candidate = Candidate(
   720→                        maimai_id=c.maimai_id,
   721→                        name=c.name,
   722→                        title=c.title,
   723→                        company=c.company,
   724→                        location=c.location,
   725→                        profile_url=c.profile_url,
   726→                        source_search=source_search
   727→                    )
   728→                    session.add(candidate)
   729→
   730→            logger.debug(f"Saved {len(candidates)} candidates to database")
   731→
   732→    except Exception as e:
   733→        logger.error(f"Failed to save candidates to database: {e}")
   734→        raise

   762→async def _click_avatar_for_profile_url(page: Page, name: str, company: str) -> Optional[str]:
   763→    """Try to extract profile URL for a candidate from the DOM contactcard pattern.
   764→
   765→    Searches the current page for a contactcard element whose parent card
   766→    contains the candidate's name (and optionally company), then constructs
   767→    the profile URL from the extracted user ID.
   768→
   769→    Args:
   770→        page: Playwright page on the search results
   771→        name: Candidate name to match
   772→        company: Candidate company to help disambiguate
   773→
   774→    Returns:
   775→        Profile URL string, or None if not found
   776→    """
   777→    try:
   778→        user_id = await page.evaluate('''(args) => {
   779→            const { name, company } = args;
   780→            // Check contactcard_{numeric_id} ancestors
   781→            const ccElements = document.querySelectorAll('[id^="contactcard_"]');
   782→            for (const el of ccElements) {
   783→                const text = el.textContent || '';
   784→                if (!text.includes(name)) continue;
   785→                if (company && !text.includes(company)) continue;
   786→                const match = el.id.match(/^contactcard_(\\d+)$/);
   787→                if (match) return match[1];
   788→            }
   789→            // Check member_entry_{numeric_id} descendants
   790→            const meElements = document.querySelectorAll('[id^="member_entry_"]');
   791→            for (const el of meElements) {
   792→                // Walk up to find the card containing this member_entry
   793→                let card = el.parentElement;
   794→                for (let i = 0; i < 8 && card; i++) {
   795→                    const text = card.textContent || '';
   796→                    if (text.includes(name) && (!company || text.includes(company))) {
   797→                        const match = el.id.match(/^member_entry_(\\d+)$/);
   798→                        if (match) return match[1];
   799→                    }
   800→                    card = card.parentElement;
   801→                }
   802→            }
   803→            return null;
   804→        }''', {'name': name, 'company': company})
   805→
   806→        if user_id:
   807→            profile_url = f"https://maimai.cn/contact/detail/{user_id}?dstu={user_id}"
   808→            logger.info(f"Resolved profile_url for '{name}' via contactcard: {profile_url}")
   809→            return profile_url
   810→
   811→        logger.debug(f"No contactcard found for '{name}'")
   812→        return None
   813→
   814→    except Exception as e:
   815→        logger.debug(f"_click_avatar_for_profile_url error for '{name}': {e}")
   816→        return None

  1015→async def _extract_enterprise_results(page: Page) -> List[CandidateBasic]:
  1016→    """Extract candidate information from enterprise recruitment results page.
  1017→
  1018→    The enterprise UI has a different card layout from search_center.
  1019→    Uses JS-based extraction tailored to the enterprise talent cards.
  1020→
  1021→    Args:
  1022→        page: Playwright page instance on the enterprise results page
  1023→
  1024→    Returns:
  1025→        List of CandidateBasic objects
  1026→    """
  1027→    candidates = []
  1028→
  1029→    try:
  1030→        await page.wait_for_load_state("networkidle", timeout=10000)
  1031→    except Exception:
  1032→        pass  # Continue even if networkidle times out
  1033→
  1034→    try:
  1035→        candidates_data = await page.evaluate('''() => {
  1036→            const candidates = [];
  1037→            const seenIds = new Set();
  1038→
  1039→            // Enterprise talent cards typically have a structured layout
  1040→            // Try multiple selectors for the card containers
  1041→            const cardSelectors = [
  1042→                '[class*="talent-card"]',
  1043→                '[class*="candidate-card"]',
  1044→                '[class*="recommend-card"]',
  1045→                '[class*="talent-item"]',
  1046→                '[class*="card-wrap"]',
  1047→            ];
  1048→
  1049→            let cards = [];
  1050→            for (const sel of cardSelectors) {
  1051→                cards = document.querySelectorAll(sel);
  1052→                if (cards.length > 0) break;
  1053→            }
  1054→
  1055→            // Fallback: find cards by avatar image pattern (same as search_center)
  1056→            if (cards.length === 0) {
  1057→                const images = document.querySelectorAll('img');
  1058→                const foundCards = [];
  1059→
  1060→                for (const img of images) {
  1061→                    const rect = img.getBoundingClientRect();
  1062→                    if (rect.width > 100 || rect.height > 100) continue;
  1063→                    if (rect.width < 30 || rect.height < 30) continue;
  1064→
  1065→                    let card = img.parentElement;
  1066→                    for (let i = 0; i < 8 && card; i++) {
  1067→                        const cardRect = card.getBoundingClientRect();
  1068→                        if (cardRect.width > 200 && cardRect.height > 80) {
  1069→                            break;
  1070→                        }
  1071→                        card = card.parentElement;
  1072→                    }
  1073→                    if (card) foundCards.push(card);
  1074→                }
  1075→                cards = foundCards;
  1076→            }
  1077→
  1078→            for (const card of cards) {
  1079→                const textParts = [];
  1080→                const walker = document.createTreeWalker(card, NodeFilter.SHOW_TEXT, null, false);
  1081→                while (walker.nextNode()) {
  1082→                    const text = walker.currentNode.textContent.trim();
  1083→                    if (text && text.length > 0 && text.length < 100) {
  1084→                        textParts.push(text);
  1085→                    }
  1086→                }
  1087→
  1088→                let name = '';
  1089→                let company = '';
  1090→                let title = '';
  1091→
  1092→                const skipPatterns = [
  1093→                    /^立即沟通$/,
  1094→                    /^不合适$/,
  1095→                    /^感兴趣$/,
  1096→                    /^不感兴趣$/,
  1097→                    /^(\\d+\\s*)?年/,
  1098→                    /^[|｜·]$/,
  1099→                    /^在线|^离线|^活跃/,
  1100→                ];
  1101→
  1102→                const isSkippable = (text) => {
  1103→                    return skipPatterns.some(p => p.test(text));
  1104→                };
  1105→
  1106→                for (let i = 0; i < textParts.length; i++) {
  1107→                    const part = textParts[i];
  1108→                    if (isSkippable(part)) continue;
